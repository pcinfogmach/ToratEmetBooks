פרק א - משנה א

שַׁמַּאי אוֹמֵר כָּל הַנָּשִׁים דַּיָּן שְׁעָתָן. מִשּׁוּם דְּרַבִּי אֱלִיעֶזֶר אוֹמֵר אַרְבַּע נָשִׁים דַּיָּן שְׁעָתָן, תָּנֵי נַמִּי שַׁמַּאי אוֹמֵר כָּל הַנָּשִׁים, לַאֲפוֹקֵי מִדְּרַבִּי אֱלִיעֶזֶר: דַּיָּן שְׁעָתָן. נָשִׁים הָרוֹאוֹת דָּם דַּיָּן שְׁעָתָן לְטַמֵּא טָהֳרוֹת שֶׁנָּגְעוּ בָּהֶן מִשְּׁעַת רְאִיָּתָן וָאֵילָךְ, וְלֹא אָמְרִינַן מִקַּמֵּי הָכִי נַמִּי הָיָה דָּם וְכָתְלֵי בֵּית הָרֶחֶם הֶעֱמִידוּהוּ (א) וְהָיְתָה טְמֵאָה מִקֹּדֶם. דְּכָל הַנָּשִׁים מְטַמְּאוֹת בַּבַּיִת הַחִיצוֹן וְאַף עַל פִּי שֶׁלֹּא יָצָא הַדָּם לַחוּץ, דִּכְתִיב (וַיִּקְרָא טו) דָּם יִהְיֶה זֹבָהּ בִּבְשָׂרָהּ, אַף עַל פִּי שֶׁהוּא בַּבָּשָׂר וְלֹא יָצָא לַחוּץ, טְמֵאָה. וְטַעֲמָא דְּשַׁמַּאי דְּלֹא חָיֵשׁ דִּלְמָא מִקַּמֵּי הָכִי הֲוָה דָּם, שֶׁאִם אַתָּה אוֹמֵר כֵּן, לִבּוֹ שֶׁל אָדָם נוֹקְפוֹ תָּמִיד בִּשְׁעַת בִּיאָה וּפוֹרֵשׁ מֵאִשְׁתּוֹ וְנִמְצֵאתָ מְבַטֵּל בְּנוֹת יִשְׂרָאֵל מִפְּרִיָּה וּרְבִיָּה: מִפְּקִידָה לִפְקִידָה. בָּדְקָה הַיּוֹם וּמָצְאָה טְהוֹרָה, וּבָדְקָה לְסוֹף שָׁבוּעַ וּמָצְאָה טְמֵאָה, חוֹשְׁשִׁין לְמַגָּעָהּ מִבְּדִיקָה רִאשׁוֹנָה וָאֵילָךְ, שֶׁמָּא עִם סִלּוּק יָדֶיהָ רָאֲתָה וְכָתְלֵי בֵּית הָרֶחֶם הֶעֱמִידוּהוּ. וּלְבִטּוּל פְּרִיָּה וּרְבִיָּה לֹא חָיְשִׁינַן, דְּדַוְקָא לְטָהֳרוֹת הוּא דִּמְטַמִּינַן לְהוּ מִפְּקִידָה לִפְקִידָה, וְלֹא לְבַעְלָהּ. וְשַׁמַּאי אוֹמֵר, לְטָהֳרוֹת נַמִּי לֹא מְטַמִּינַן, דְּאִם כֵּן לִבּוֹ נוֹקְפוֹ וּפוֹרֵשׁ נַמִּי מִתַּשְׁמִישׁ: לֹא כְדִבְרֵי זֶה. שַׁמַּאי דְּמֵקֵל טְפֵי וְאֵינוֹ עוֹשֶׂה סְיָג לִדְבָרָיו: וְלֹא כְדִבְרֵי זֶה. הִלֵּל דְּמַחְמִיר יוֹתֵר מִדַּאי. דְּוַדַּאי כֻּלֵּי הַאי יָמִים רַבִּים לֹא מוֹקְמִי כָּתְלֵי בֵּית הָרֶחֶם דָּם: מֵעֵת לְעֵת מְמַעֵט עַל יַד מִפְּקִידָה לִפְקִידָה. שְׁנֵי זְמַנִּים הֻזְכְּרוּ בְּאִשָּׁה לְטַמֵּא טָהֳרוֹת לְמַפְרֵעַ, וְהַלֵּךְ אַחַר הַקַּל שֶׁבִּשְׁנֵיהֶן. אִם מִפְּקִידָה לִפְקִידָה יוֹתֵר עַל מֵעֵת לְעֵת, הַלֵּךְ אַחַר מֵעֵת לְעֵת וְלֹא תְּטַמֵּא אֶלָּא טָהֳרוֹת שֶׁנָּגְעָה מֵאֶתְמוֹל בַּשָּׁעָה הַזֹּאת. וְאִם מֵעֵת לְעֵת יוֹתֵר עַל מִפְּקִידָה לִפְקִידָה, כְּגוֹן שֶׁבָּדְקָה עַצְמָהּ שַׁחֲרִית וּמָצְאָה טְהוֹרָה וְעַרְבִית מָצְאָה טְמֵאָה, לֹא מְטַמְּאִינַן אֶלָּא הָנָךְ טָהֳרוֹת דְּמִבְּדִיקַת שַׁחֲרִית וָאֵילָךְ. וְטַעֲמַיְהוּ דְּרַבָּנָן, לְפִי שֶׁחֲכָמִים תִּקְּנוּ לִבְנוֹת יִשְׂרָאֵל שֶׁיִּהְיוּ בּוֹדְקוֹת שַׁחֲרִית וְעַרְבִית, שַׁחֲרִית לְהַכְשִׁיר טָהֳרוֹת שֶׁל לַיְלָה, וְעַרְבִית לְהַכְשִׁיר טָהֳרוֹת שֶׁל יוֹם. וְאִם בָּדְקָה שַׁחֲרִית וּמָצְאָה טְהוֹרָה, עַרְבִית וּמָצְאָה טְמֵאָה, הִפְסִידָה טָהֳרוֹת שֶׁל יוֹם, וְזוֹ שֶׁיָּשְׁבָה יָמִים רַבִּים וְלֹא בָּדְקָה כְּתַקָּנַת חֲכָמִים, הִפְסִידָה עוֹנָה אַחַת יְתֵרָה, מִשּׁוּם קְנָס. וּכְדֵי לְהַשְׁווֹת מִדּוֹתֵיהֶם שֶׁל חֲכָמִים שֶׁלֹּא תַּחֲלֹק בֵּין הֵיכָא דְּבָדְקָה שַׁחֲרִית לְהֵיכָא דְּבָדְקָה בַּצָהֳרָיִם (ב), לֹא רָצוּ לְחַלֵּק וּלְמַעֵט וּלְהוֹסִיף בַּזְּמַן, וְנָתְנוּ זְמַן קָבוּעַ מֵעֵת לְעֵת, שֶׁלֹּא לִתֵּן דִּבְרֵיהֶם לְשִׁעוּרִין. וַהֲלָכָה כַּחֲכָמִים: עַל יַד. אַחַר. כְּמוֹ וְעַל יָדוֹ הֶחֱזִיק, שֶׁהוּא כְּמוֹ וְאַחֲרָיו, בְּסֵפֶר עֶזְרָא [נְחֶמְיָה ג]: וֶסֶת. קָבוּעַ, שֶׁנִּקְבַּע לָהּ זְמַן הָאֹרַח שָׁלֹשׁ פְּעָמִים וּבָדְקָה בִּשְׁעַת וֶסְתָּהּ (ד) וּמָצְאָה טְמֵאָה: דַּיָּהּ שְׁעָתָהּ. וְלֹא חָיְשִׁינַן דִּלְמָא מִקַּמֵּי הָכִי הֲוָה, דְּוַדַּאי אֹרַח בִּזְמַנּוֹ בָּא (ה): וְהַמְשַׁמֶּשֶׁת בְּעִדִּים. שֶׁבָּדְקָה עַצְמָהּ בְּעִדִּים, דְּהַיְנוּ בְּסָדִין, קֹדֶם תַּשְׁמִישׁ וּלְאַחַר תַּשְׁמִישׁ: הֲרֵי זוֹ כִפְקִידָה. בַּגְּמָרָא מוֹקֵי לְמַתְנִיתִין דְּתַרְתֵּי קָתָנֵי וְהָכִי קָאָמַר, וּמְשַׁמֶּשֶׁת בְּעִדִּים כְּלוֹמַר מִצְוָה עַל אִשָּׁה שֶׁתְּשַׁמֵּשׁ בִּשְׁנֵי עִדִּים, אֶחָד לִפְנֵי תַּשְׁמִישׁ וְאֶחָד לְאַחַר תַּשְׁמִישׁ. וַהֲרֵי זוֹ כִּפְקִידָה, הָעֵד שֶׁל אַחַר תַּשְׁמִישׁ הֲרֵי הוּא כִּפְקִידָה. וּמְמַעֶטֶת עַל יַד מֵעֵת לְעֵת וְעַל יַד מִפְּקִידָה לִפְקִידָה, דְּמַהוּ דְתֵימָא שֶׁמָּא תִּרְאֶה טִפַּת דָּם כְּחַרְדָּל וּתְחַפֶּנָּה שִׁכְבַת זֶרַע וְלֹא לֶהֱוֵי כִּפְקִידָה, קָמַשְׁמַע לָן. וְאִם שִׁמְּשָׁה בַּצָהֳרַיִם בְּעִדִּים, וּבָדְקָה אַחֲרֵי כֵן בָּעֶרֶב וּמָצְאָה טְמֵאָה, לֹא מְטַמְּאִינַן טָהֳרוֹת דְּמִבְּדִיקַת שַׁחֲרִית עַד תַּשְׁמִישׁ, שֶׁהֲרֵי בִּשְׁעַת תַּשְׁמִישׁ בָּדְקָה וּמָצְאָה טְהוֹרָה. וְאַף עַל גַּב דִּבְדִיקָה שֶׁקֹּדֶם תַּשְׁמִישׁ לָא חֲשִׁיבָא בְּדִיקָה מְעַלְּיָתָא, שֶׁמִּתּוֹךְ שֶׁהִיא בְּהוּלָה לְשַׁמֵּשׁ אֵינָהּ מַכְנֶסֶת הָעֵד לַחוֹרִים וְלַסְּדָקִין, בְּדִיקָה בָּעֵד שֶׁלְאַחַר תַּשְׁמִישׁ בְּדִיקָה מְעַלְּיָתָא הִיא וַחֲשִׁיבָא כִּפְקִידָה כְּדַאֲמַרַן:

פרק א - משנה ב

הָיְתָה יוֹשֶׁבֶת בַּמִּטָּה וַעֲסוּקָה בְטָהֳרוֹת. הָא דְּלֹא תָּנֵי הָיְתָה עֲסוּקָה בְּטָהֳרוֹת וּפָרְשָׁה וְרָאֲתָה, וְקָתָנֵי הָיְתָה יוֹשֶׁבֶת בַּמִּטָּה, לְאַשְׁמוֹעִינַן דְּמִשּׁוּם דְּיֵשׁ לָהּ וֶסֶת וְדַיָּהּ שַׁעְתָּהּ, הַמִּטָּה נַמִּי טְהוֹרָה, כִּדְּקָתָנֵי וְכֻלָּן טְהוֹרוֹת, הָא אִם אֵין לָהּ וֶסֶת, דִּמְטַמִּינַן מֵעֵת לְעֵת, הַמִּטָּה נַמִּי טְמֵאָה (ז), כְּמִשְׁכַּב נִדָּה עַצְמָהּ שֶׁמְּטַמְּאָה טֻמְאָה חֲמוּרָה לְטַמֵּא אָדָם לְטַמֵּא בְּגָדִים שֶׁעָלָיו: אַף עַל פִּי שֶׁאָמְרוּ. אִשָּׁה שֶׁאֵין לָהּ וֶסֶת מְטַמְּאָה מֵעֵת לְעֵת: אֵינָהּ מוֹנָה. יְמֵי נִדָּה אֶלָּא מִשָּׁעָה שֶׁרָאֲתָה:

פרק א - משנה ג

אַרְבַּע נָשִׁים דַּיָּן שְׁעָתָן. דְּאֵינָן רְגִילוֹת לִרְאוֹת דָּם: אֲבָל הֲלָכָה כְּרַבִּי אֱלִיעֶזֶר. וְהָכִי מַסִּיק בַּגְּמָרָא (ח) דַּהֲלָכָה כְּרַבִּי אֱלִיעֶזֶר:

פרק א - משנה ד

מִשֶּׁיִּוָּדַע עֻבָּרָהּ. לְסוֹף שְׁלֹשָׁה חֳדָשִׁים מִיּוֹם שֶׁנִּתְעַבְּרָה: עַד שֶׁתִּגְמֹל. עַד שֶׁיַּשְׁלִים לִבְנָהּ עֶשְׂרִים וְאַרְבָּעָה חֹדֶשׁ. וַאֲפִלּוּ מֵת בְּתוֹךְ עֶשְׂרִים וְאַרְבָּעָה חֹדֶשׁ, דַּיָּהּ שַׁעְתָּהּ (ט), שֶׁהַיּוֹלֶדֶת אֵבָרֶיהָ מִתְפָּרְקִים וְאֵין דָּמָהּ חוֹזֵר עַד אַחַר עֶשְׂרִים וְאַרְבָּעָה חֹדֶשׁ. וּמִשָּׁם וָאֵילָךְ הֲרֵי הִיא כְּכָל הַנָּשִׁים וּמְטַמְּאָה מֵעֵת לְעֵת וּמִפְּקִידָה לִפְקִידָה, וְאַף עַל פִּי שֶׁמֵּנִיקָה וְהוֹלֶכֶת אַרְבַּע וְחָמֵשׁ שָׁנִים: גְּמָלַתּוּ אוֹ מֵת רַבִּי מֵאִיר אוֹמֵר מְטַמְּאָה מֵעֵת לְעֵת. דְּסָבַר רַבִּי מֵאִיר דָּם נֶעְכָּר וְנַעֲשָׂה חָלָב (י). וּכְשֶׁאֵינָהּ מֵנִיקָה, אֲפִלּוּ קֹדֶם עֶשְׂרִים וְאַרְבָּעָה חֹדֶשׁ חוֹזֵר הַדָּם לִמְקוֹמוֹ וְחוֹזֶרֶת לִהְיוֹת רוֹאָה. וְאֵין הֲלָכָה כְּרַבִּי מֵאִיר:

וְתָדִיחַ אֶת פָּנֶיהָ. קִנּוּחַ שֶׁלְּמַטָּה: וְאַחַר כָּךְ. כְּלוֹמַר אִם שָׁהֲתָה יוֹתֵר מִשִּׁעוּר זֶה: מְטַמְּאָה מֵעֵת לְעֵת. טָהֳרוֹת שֶׁנִּתְעַסְּקָה בָּהֶן. כְּדִין מֵעֵת לְעֵת דְּאָמְרִינַן בְּפֶרֶק קַמָּא: וְאֵינָהּ מְטַמְּאָה אֶת בּוֹעֲלָהּ. טֻמְאַת שִׁבְעָה, אֶלָּא טֻמְאַת עֶרֶב כְּדִין כָּל נוֹגֵעַ בְּנִדָּה (ז): רַבִּי עֲקִיבָא אוֹמֵר אַף מְטַמְּאָה אֶת בּוֹעֲלָהּ. דְּסָבַר רַבִּי עֲקִיבָא כָּל מֵעֵת לְעֵת מְטַמְּאָה אֶת בּוֹעֲלָהּ טֻמְאַת שִׁבְעָה מִסָּפֵק. וְאֵין הֲלָכָה כְּרַבִּי עֲקִיבָא: וּמוֹדִים חֲכָמִים לְרַבִּי עֲקִיבָא בְּרוֹאָה כֶתֶם. וְאַף עַל גַּב דִּבְטֻמְאַת מֵעֵת לְעֵת אָמְרוּ חֲכָמִים שֶׁאֵינָהּ מְטַמְּאָה אֶת בּוֹעֲלָהּ לְפִי שֶׁטֻּמְאַת מֵעֵת לְעֵת אֵינָהּ אֶלָּא מִדְּרַבָּנָן, מוֹדִים הֵם בְּרוֹאָה כֶּתֶם, דְּאַף עַל גַּב דְּטֻמְאַת כְּתָמִים דְּרַבָּנָן, מְטַמְּאָה אֶת בּוֹעֲלָהּ טֻמְאַת שִׁבְעָה:

כָּל הַנָּשִׁים בְּחֶזְקַת טָהֳרָה לְבַעֲלֵיהֶן. וְאֵין צָרִיךְ לִשְׁאֹל לָהּ אִם הִיא טְהוֹרָה. וְהָנֵי מִילֵי, כְּשֶׁהִנִּיחָהּ בְּחֶזְקַת טְהוֹרָה: הַבָּאִין מִן הַדֶּרֶךְ נְשֵׁיהֶן לָהֶן בְּחֶזְקַת טָהֳרָה. דְּסָלְקָא דַעְתָּךְ אֲמֵינָא הָנֵי מִילֵי הֵיכָא דְּאִיתֵיהּ בְּמָתָא, דְּרַמְיָא אַנַּפְשַׁהּ וּבָדְקָה, אֲבָל הֵיכָא דְּלֵיתֵיהּ בְּמָתָא, דְּלֹא רַמְיָא אַנַּפְשַׁהּ, לֹא, קָמַשְׁמַע לָן: שְׁנֵי עִדִּים עַל כָּל תַּשְׁמִישׁ. אֶחָד לְפָנָיו וְאֶחָד לְאַחֲרָיו, וּלְמָחָר תְּעַיֵּן בָּהֶם: אוֹ תְשַׁמֵּשׁ לְאוֹר הַנֵּר. וְתִבְדֹּק לְאַחַר תַּשְׁמִישׁ (יא) עֵד שֶׁקִּנְּחָה בּוֹ: דַּיָּהּ בִּשְׁנֵי עִדִּים כָּל הַלָּיְלָה. אֶחָד לִפְנֵי תַּשְׁמִישׁ רִאשׁוֹן, וְאֶחָד לְאַחַר תַּשְׁמִישׁ אַחֲרוֹן (יב). וְאַף עַל גַּב דְּמִשְׁמְשָׁה וְלֹא יָדְעָה אִי רָאֲתָה דָּם בֵּינֵי וּבֵינֵי, בֵּין לְבֵית שַׁמַּאי בֵּין לְבֵית הִלֵּל לֹא אִכְפַּת לָן, דְּלֹא הֻצְרְכוּ אֶלָּא לְחֹמֶר טָהֳרוֹת:

פרק ב - משנה ה

הַחֶדֶר וְהָעֲלִיָּה וְהַפְּרוֹזְדוֹר. חֶדֶר מִבִּפְנִים, וּפְרוֹזְדוֹר מִבַּחוּץ, שְׁנֵיהֶם זֶה אֵצֶל זֶה, חֶדֶר לְצַד אֲחוֹרֶיהָ, וּפְרוֹזְדוֹר לְפָנֶיהָ, וַעֲלִיָּה בְּנוּיָה עַל שְׁתֵּיהֶן, וְנֶקֶב יֵשׁ בֵּין עֲלִיָּה לַפְּרוֹזְדוֹר וְנִקְרָא לוּל. וּפְעָמִים יוֹרֵד דָּם מִן הָעֲלִיָּה לַפְּרוֹזְדוֹר דֶּרֶךְ הַלּוּל. וְכָתְלֵי רֶחֶם, לְמַטָּה בְּאֶמְצַע פְּרוֹזְדוֹר, וְדֶרֶךְ שָׁם דָּמִים יוֹצְאִים: דַּם הַחֶדֶר טָמֵא. דְּהַיְנוּ דַּם הַמָּקוֹר: סְפֵקוֹ טָמֵא. מִסָּפֵק טָמֵא טֻמְאָה וַדָּאִית: שֶׁחֶזְקָתוֹ מִן הַמָּקוֹר. וְלֹא אָמְרִינַן שֶׁמָּא מִן הָעֲלִיָּה בָּא, וּדְמֵי הָעֲלִיָּה טְהוֹרִים הֵם, אֶלָּא עֲשָׂאוּהוּ כְּאִלּוּ וַדַּאי בָּא מִן הַמָּקוֹר. וְהָנֵי מִילֵי, כְּשֶׁנִּמְצָא מִן הַלּוּל וְלִפְנִים לְצַד הַחֶדֶר, אֲבָל נִמְצָא מִן הַלּוּל וְלַחוּץ, אֵינוֹ טָמֵא טֻמְאָה וַדָּאִית לִשְׂרוֹף, אֶלָּא סְפֵקוֹ טָמֵא לִתְלוֹת, שֶׁהֲרֵי שְׁנֵי דָּמִים יוֹצְאִין דֶּרֶךְ שָׁם, דְּמֵי הַחֶדֶר וּדְמֵי הָעֲלִיָּה, וְלֹא יָדְעִינַן מֵהֵי אֲתָא:

פרק ב - משנה ו

חֲמִשָּׁה דָמִים טְמֵאִים. דִּשְׁתֵּי פְּעָמִים כְּתִיב דָּמֶיהָ בַּקְּרָא בְּטֻמְאַת דָּמִים שֶׁבָּאִשָּׁה, (וַיִּקְרָא כ) וְהִיא גִּלְּתָה אֶת מְקוֹר דָּמֶיהָ, (שָׁם יב) וְטָהֲרָה מִמְּקֹר דָּמֶיהָ, וּמִעוּט דָּמֶיהָ שְׁנַיִם, הֲרֵי אַרְבָּעָה דָמִים. אֹדֶם, וְקֶרֶן כַּרְכֹּם, וּמֵימֵי אֲדָמָה, וּמָזוּג. וְהַשָּׁחֹר בִּכְלַל אָדֹם הוּא, וְהָכִי אָמְרִינַן בַּגְּמָרָא, הַאי שָׁחֹר אָדֹם הוּא, אֶלָּא שֶׁלָּקָה: וּכְקֶרֶן כַּרְכֹּם. כְּמַרְאֵה זִיו הַכַּרְכֹּם. לְשׁוֹן כִּי קָרַן עוֹר פָּנָיו (שְׁמוֹת לד): וּכְמֵימֵי אֲדָמָה. כֻּלָּן מְפָרֵשׁ לָהֶן לְקַמָּן בְּמַתְנִיתִין: וּכְמָזוּג. כְּיַיִן אָדֹם הַמָּזוּג בְּמַיִם: כְּמֵימֵי תִלְתָּן. מַיִם שֶׁשּׁוֹרִין בָּהֶן תִּלְתָּן. פינקיריק"ו בְּלַעַ"ז: וּכְמֵימֵי בָשָׂר צָלִי. מֹחַל הַיּוֹצֵא מִבָּשָׂר צָלוּי: וּבֵית הִלֵּל מְטַהֲרִין. וְשָׁלֹשׁ מַחֲלוֹקוֹת בַּדָּבָר. תַּנָּא קַמָּא דְּאָמַר חֲמִשָּׁה דָּמִים טְמֵאִים וְתוּ לֹא, סְבִירָא לֵיהּ דְּמֵימֵי תִּלְתָּן וּמֵי בָּשָׂר צָלִי תּוֹלִין, לֹא אוֹכְלִין וְלֹא שׂוֹרְפִין. וּבֵית שַׁמַּאי מְטַמְּאִין, וְשׂוֹרְפִים עֲלֵיהֶם תְּרוּמָה וְקָדָשִׁים. וּבֵית הִלֵּל מְטַהֲרִין לְגַמְרֵי. וַהֲלָכָה כְּתַנָּא קַמָּא, דְּתוֹלִין: עֲקַבְיָא בֶּן מַהֲלַלְאֵל מְטַמֵּא. דִּסְבִירָא לֵיהּ הַאי יָרֹק כְּקֶרֶן כַּרְכֹּם הוּא, אֶלָּא שֶׁלָּקָה. וְיָרֹק דִּמְטַמֵּא עֲקַבְיָא בֶּן מַהֲלַלְאֵל, דְּהַיְנוּ כְּמַרְאֵה אֶתְרוֹג, וְלֹא כְּמַרְאֵה כַּרְתִּי שֶׁאֵינוֹ נוֹטֶה לְאַדְמִימוּת כְּלָל: וַחֲכָמִים מְטַהֲרִים. וְשָׁלֹשׁ מַחֲלוֹקוֹת נַמִּי בְּיָרֹק, תַּנָּא קַמָּא סָבַר תּוֹלִין לֹא אוֹכְלִים וְלֹא שׂוֹרְפִים, וַעֲקַבְיָא בֶּן מַהֲלַלְאֵל מְטַמֵּא וְשׂוֹרְפִים, וַחֲכָמִים מְטַהֲרִין וְאוֹכְלִים. וַהֲלָכָה כְּתַנָּא קַמָּא: אִם אֵינוֹ מְטַמֵּא מִשּׁוּם כֶּתֶם. לִהְיוֹתוֹ חָשׁוּב מִן הַדָּמִים הַמְטַמְּאִין בָּאִשָּׁה: מְטַמֵּא מִשּׁוּם מַשְׁקֶה. וְחָשׁוּב הוּא דָּם לְהַכְשִׁיר הַזְּרָעִים לְקַבֵּל טֻמְאָה כְּכָל שְׁאָר דָּם (יד), שֶׁהוּא אֶחָד מִן הַשִּׁבְעָה מַשְׁקִין שֶׁמַּכְשִׁירִין אֶת הַזְּרָעִים: לֹא כָךְ וְלֹא כָךְ. לֹא מְטַמֵּא מִשּׁוּם כֶּתֶם, וְלֹא מַכְשִׁיר אֶת בְּגָדָיו שֶׁל חָבֵר שֶׁדָּרַס כּוּתִי עֲלֵיהֶן בַּתְּרוּמָה (ו), לֹא שָׂרְפִינַן תְּרוּמָה עֲלֵיהֶן. דְּאִי מִשּׁוּם טֻמְאַת עַם הָאָרֶץ, הָא טָבֵיל לֵיהּ. וְאִי מִשּׁוּם בּוֹעֵל נִדָּה, שֶׁמָּא לֹא בָּעַל בְּקָרוֹב וּכְבָר עָבְרוּ יְמֵי טֻמְאָתוֹ וְסָלְקָא לֵיהּ טְבִילָה. וְאִם תִּמְצָא לוֹמַר בָּעַל בְּקָרוֹב, שֶׁמָּא לֹא הָיְתָה אִשְׁתּוֹ יוֹשֶׁבֶת עַל דָּם טָמֵא אֶלָּא עַל דָּם טָהוֹר, וְהָוֵי סְפֵק סְפֵקָא, וְאַסְּפֵק סְפֵקָא לֹא שָׂרְפִינַן תְּרוּמָה:

פֵּרְשׁוּ לָלֶכֶת בְּדַרְכֵי יִשְׂרָאֵל הֲרֵי הֵן כְּיִשְׂרְאֵלִיּוֹת. אֲבָל סְתָמָא, הֲרֵי הֵן כַּכּוּתִיּוֹת (ז): רַבִּי יוֹסֵי אוֹמֵר לְעוֹלָם הֵן כְּיִשְׂרָאֵל. וּסְתָמָא הֵן כְּיִשְׂרְאֵלִיּוֹת. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי:

פרק ד - משנה ג

דַּם נָכְרִית בֵּית שַׁמַּאי מְטַהֲרִין. וְאַף עַל גַּב דִּבְרֻקָּהּ וּבְמֵימֵי רַגְלֶיהָ מוֹדוּ בֵּית שַׁמַּאי דְּרַבָּנָן גָּזְרוּ (ח) עֲלֵיהֶן לִהְיוֹת כְּזָבִין לְכָל דִּבְרֵיהֶן (ט), אֲפִלּוּ הָכִי דָּמָהּ טָהוֹר, דְּשִׁיְּרוּ בָּהּ רַבָּנָן לְמִהְוֵי הֶכֵּירָא (י) דְּטֻמְאָתָן דְּרַבָּנָן, כִּי הֵיכִי דְּלֹא לִשְׂרְפוּ עֲלַיְהוּ תְּרוּמָה וְקָדָשִׁים: וְדַם טָהוֹר שֶׁל מְצֹרַעַת. וְאַף עַל גַּב דְּמַעְיָנוֹת שֶׁל מְצֹרַעַת כְּגוֹן רֻקָּהּ וּמֵימֵי רַגְלֶיהָ הֲווּ אַב הַטֻּמְאָה כְּשֶׁל זָב, אֲפִלּוּ הָכִי דַּם טֹהַר שֶׁלָּהּ טָהוֹר, דְּלָאו מַעְיָן הוּא, דְּדוּמְיָא דְּרֹק בָּעִינַן שֶׁמִּתְעַגֵּל וְיוֹצֵא (יא): כְּרֻקָּהּ וּכְמֵימֵי רַגְלֶיהָ. דִּמְטַמְּאִין לַחִין וְאֵין מְטַמְּאִין יְבֵשִׁין. וְהַךְ הֶיכֵּרָא הוּא דְּעַבוּד גַּבֵּי דַּם הַנָּכְרִית דִּמְטַמֵּא מִדְּרַבָּנָן שֶׁגָּזְרוּ עֲלֵיהֶם שֶׁיִּהְיוּ כַּזָּבִים לְכָל דִּבְרֵיהֶן, דְּאִלּוּ דְּיִשְׂרְאֵלִית מְטַמֵּא לַח וְיָבֵשׁ, וּדְנָכְרִית לַח וְלֹא יָבֵשׁ, כְּרֻקָּהּ וּכְמֵימֵי רַגְלֶיהָ דְּאֵין מְטַמְּאִין יְבֵשִׁין. וְדָם טָהוֹר שֶׁל מְצֹרַעַת נַמִּי, אַף עַל גַּב דְּדָם לָאו מַעְיָן הוּא, דִּין מַעְיָן אִית לֵיהּ, דְּרַחֲמָנָא רַבְּיֵהּ מִקְּרָא דִּכְתִיב לְזָכָר, לְרַבּוֹת מְצֹרָע לְמַעְיְנוֹתָיו, וְלִנְקֵבָה, לְרַבּוֹת מְצֹרַעַת לְמַעְיְנוֹתֶיהָ, וְאָמְרִינַן מַעְיְנוֹתֶיהָ דְּמַאי, אִילֵימָא שְׁאָר מַעְיְנוֹתֶיהָ כְּגוֹן רֻקָּהּ וּמֵימֵי רַגְלֶיהָ דְּאָתוּ מִזָּב דִּכְתִיב בֵּיהּ וְכִי יָרֹק הַזָּב, וּמִנֵּיהּ נָפְקֵי שְׁאָר מַעְיָנוֹת, מִזָּכָר נַפְקָא, דְּכֵיוָן דְּאִתְרַבֵּי מְצֹרָע לְמַעְיְנוֹתָיו, אִתְרַבֵּי מְצֹרַעַת לְכָל מַעְיְנוֹתֶיהָ דְּאִיכָּא דִּכְוָתַיְהוּ בְּזָכָר, אֶלָּא לְדָמָהּ דְּלֵיתֵיהּ בְּזָכָר, וְכֵיוָן דְּרַבְּיֵהּ רַחֲמָנָא לְדָמָהּ שֶׁיְּהֵא כְּרֻקָּהּ וּכְמֵימֵי רַגְלֶיהָ, אִם כֵּן מָה רֻקָּהּ וּמֵימֵי רַגְלֶיהָ מְטַמְּאִין לַחִין וְאֵין מְטַמְּאִין יְבֵשִׁים, אַף דָּמָהּ כֵּן: דַּם הַיּוֹלֶדֶת. שֶׁעָבְרוּ שִׁבְעַת יְמֵי טֻמְאַת לֵידָה וְלֹא טָבְלָה: בֵּית שַׁמַּאי אוֹמְרִים כְּרֻקָּהּ וּכְמֵימֵי רַגְלֶיהָ. מְטַמֵּא לַח וְאֵינוֹ מְטַמֵּא יָבֵשׁ. וְאַף עַל גַּב דְּדַם טֹהַר הוּא, דִּבְיוֹמֵי תָּלָה רַחֲמָנָא וְהָא מָלוּ, וְלֵיכָּא לְטַמּוֹיֵהּ נַמִּי מִשּׁוּם מַעְיָן דְּהָא אֵין מִתְעַגֵּל וְיוֹצֵא, אֲפִלּוּ הָכִי רַבָּנָן גַּזּוּר טֻמְאָה עֲלֵיהּ, הוֹאִיל וְלֹא טָבְלָה וּטְמֵאָה הִיא, גְּזֵרָה אַחַר שִׁבְעָה אַטּוּ תּוֹךְ שִׁבְעָה. וּמִיהוּ לֹא גַּזּוּר שֶׁיְּטַמֵּא לַח וְיָבֵשׁ כְּלִפְנֵי שִׁבְעָה, כִּי הֵיכִי דְּלַהֲוֵי לֵיהּ הֶכֵּירָא דְּלֵידַע דְּטֻמְאָתוֹ דְּרַבָּנָן (יב): וּבֵית הִלֵּל אוֹמְרִים מְטַמֵּא לַח וְיָבֵשׁ. דְּטֻמְאָתוֹ מִן הַתּוֹרָה כָּל זְמַן שֶׁלֹּא טָבְלָה. דְּסָבְרֵי בֵּית הִלֵּל בְּיוֹמֵי וּטְבִילָה תָּלָה רַחֲמָנָא, וְכָל זְמַן שֶׁלֹּא טָבְלָה, הֲרֵי דָּמָהּ לְאַחַר שִׁבְעָה כְּלִפְנֵי שִׁבְעָה: וּמוֹדִים בְּיוֹלֶדֶת בְּזוֹב. אִם בְּעֵת לֵדָתָהּ הָיְתָה זָבָה שֶׁהִיא צְרִיכָה לִסְפֹּר שִׁבְעָה נְקִיִּים קֹדֶם שֶׁתִּטְהַר, וְיָלְדָה וְלֹא סָפְרָה, מוֹדִים בֵּית שַׁמַּאי לְבֵית הִלֵּל שֶׁדָּמָהּ טָמֵא לַח וְיָבֵשׁ אַף עַל פִּי שֶׁעָבְרוּ שִׁבְעַת יְמֵי לֵידָה לְזָכָר וּשְׁבוּעַיִם לִנְקֵבָה, שֶׁדָּמָהּ שֶׁל זָבָה לְעוֹלָם הוּא בְּטֻמְאָתוֹ מִן הַתּוֹרָה עַד שֶׁתִּסְפֹּר שִׁבְעָה נְקִיִּים (יג):

הַמַּקְשָׁה נִדָּה. הָכִי קָאָמַר, הַמַּקְשָׁה בְּלֵידָתָהּ בִּימֵי נִדָּתָהּ וְרָאֲתָה דָּם בְּהַקְשׁוֹתָהּ, הֲרֵי הִיא נִדָּה, דְּלֹא טִהֲרָה תּוֹרָה לְדַם קֹשִׁי אֶלָּא בְּאַחַד עָשָׂר יוֹם שֶׁל יְמֵי זִיבָה, אֲבָל בְּיָמִים שֶׁל נִדָּה לֹא טַהֲרֵיהּ רַחֲמָנָא: קִשְּׁתָה. מֵחֲמַת וָלָד שְׁלֹשָׁה יָמִים בְּתוֹךְ אַחַד עָשָׂר יוֹם שֶׁל יְמֵי זִיבָה וְרָאֲתָה כָּל שְׁלֹשָׁה יָמִים רְצוּפִים: וְשָׁפְתָה. שֶׁעָמְדָה בְּשֹׁפִי וּבְנַחַת מֵחֲבָלֶיהָ מֵעֵת לְעֵת בָּרְבִיעִי: וְיָלְדָה. אִגַּלַּאי מִלְּתָא דְּדָמִים שֶׁרָאֲתָה לֹא בָּאוּ מֵחֲמַת וָלָד, הִלְכָּךְ הֲרֵי זוֹ יוֹלֶדֶת בְּזוֹב. אֲבָל אִם לֹא שָׁפְתָה סָמוּךְ לְלֵדָתָהּ, כָּל דָּמִים שֶׁרָאֲתָה, מֵחֲמַת וָלָד הוּא, וְרַחֲמָנָא טַהֲרִינְהוּ מִזִּיבָה, דִּכְתִיב זוֹב דָּמָהּ, דָּמָהּ מֵחֲמַת עַצְמָהּ וְלֹא דָּמָהּ מֵחֲמַת וָלָד: לַיְלָה וָיוֹם. דְּאִם שָׁפְתָה בַּחֲצִי יוֹם שְׁלִישִׁי וְיָלְדָה בַּחֲצִי יוֹם רְבִיעִי, דְּשָׁפְתָה מֵעֵת לְעֵת, אֵין זֶה שֹׁפִי, אֶלָּא אִם כֵּן שָׁפְתָה לַיְלָה וְיוֹם שֶׁל אַחַר הַלַּיְלָה: כְּלֵילֵי שַׁבָּת וְיוֹמוֹ. וַהֲלָכָה כְּרַבִּי אֱלִיעֶזֶר: שֶׁשָּׁפְתָה מִן הַצַּעַר. כְּלוֹמַר וְשָׁפְתָה זוֹ שֶׁאָמַרְנוּ, אִם שָׁפְתָה מִן הַצַּעַר אַף עַל פִּי שֶׁלֹּא שָׁפְתָה מִן הַדָּם, הָוֵי שֹׁפִי:

כַּמָּה הוּא קִשּׁוּיָהּ. דְּלֹא אָתְיָא לִידֵי זִיבָה בְּכָל דָּמִים שֶׁתִּרְאֶה: אֲפִלּוּ אַרְבָּעִים וַחֲמִשִּׁים יוֹם. אִם לֹא שָׁפְתָה יוֹם אֶחָד סָמוּךְ לַלֵּידָה, לֹא אָתְיָא לִידֵי זִיבָה וְלֹא הַוְיָא יוֹלֶדֶת בְּזוֹב (טז): דַּיָּהּ חָדְשָׁהּ. דַּיָּהּ אִם נְטַהֵר אֶת הַקֹּשִׁי חֹדֶשׁ אֶחָד, דְּהַיְנוּ חֹדֶשׁ תְּשִׁיעִי. אֲבָל קָשְׁתָה שְׁלֹשָׁה בְּתוֹךְ אַחַד עָשָׂר יְמֵי זִיבָה בְּחֹדֶשׁ שְׁמִינִי וַאֲפִלּוּ קָשְׁתָה וְהוֹלֶכֶת כָּל תְּשִׁיעִי, הֲרֵי זוֹ יוֹלֶדֶת בְּזוֹב מֵחֲמַת שְׁלֹשָׁה יָמִים שֶׁל שְׁמִינִי: אֵין קִשּׁוּי יוֹתֵר מִשְּׁתֵּי שַׁבָּתוֹת. שֶׁאִם קָשְׁתָה וְרָאֲתָה שְׁלֹשָׁה יָמִים קֹדֶם שְׁתֵּי שַׁבָּתוֹת וְהָיְתָה מַקְשָׁה וְהוֹלֶכֶת כָּל הַשַּׁבָּתוֹת אַחֲרֵי כֵן, הֲרֵי זוֹ יוֹלֶדֶת בְּזוֹב מֵחֲמַת שְׁלֹשָׁה יָמִים שֶׁהָיוּ קֹדֶם הַשְּׁתֵּי שַׁבָּתוֹת. וַהֲלָכָה כְּרַבִּי יוֹסֵי וְרַבִּי שִׁמְעוֹן:

הַמַּקְשָׁה בְתוֹךְ שְׁמוֹנִים שֶׁל נְקֵבָה. כְּגוֹן שֶׁשִּׁמְּשָׁה לְאַחַר אַרְבָּעָה עָשָׂר שֶׁל יְמֵי לֵידָה וְנִתְעַבְּרָה וְהִפִּילָה בְּתוֹךְ שְׁמוֹנִים: כָּל דָּמִים שֶׁהִיא רוֹאָה טְהוֹרִים. דְּכָל תּוֹךְ מְלֹאת הָוֵי דָּם טָהוֹר: וְרַבִּי אֱלִיעֶזֶר מְטַמֵּא. נִדָּה, דְּכֵיוָן דִּבְקֹשִׁי בָּא לָאו דַּם טֹהַר הוּא אֶלָּא מֵחֲמַת וָלָד שֶׁבְּמֵעֶיהָ, וּמִזִּיבָה הוּא דְּטַהֲרֵיהּ רַחֲמָנָא לַקִּשּׁוּי, וְלֹא מִנִּדָּה: שֶׁהֶחְמִיר בְּדַם הַשֹּׁפִי. כְּגוֹן שְׁאָר יוֹלֶדֶת שֶׁרוֹאָה שְׁלֹשָׁה בְּשֹׁפִי וְיָלְדָה, הֲרֵי זוֹ יוֹלֶדֶת בְּזוֹב: הֵקֵל בְּדַם הַקֹּשִׁי. לְטַהֲרָהּ מִזִּיבָה: מְקוֹם שֶׁהֵקֵל בְּדַם הַשֹּׁפִי. לְטַהֲרָהּ מִכָּל טֻמְאָה, כְּגוֹן תּוֹךְ מְלֹאת: אֵינוֹ דִּין שֶׁנָּקֵל בְּדַם הַקֹּשִׁי. לְטַהֲרוֹ מִכָּל טֻמְאוֹת וַאֲפִלּוּ מִנִּדָּה: דַּיּוֹ לַבָּא מִן הַדִּין. כְּגוֹן קֹשִׁי שֶׁל תּוֹךְ מְלֹאת שֶׁאַתֶּם לְמֵדִים מִקֹּשִׁי שֶׁל עָלְמָא, דַּיּוֹ לִהְיוֹת כַּנִּדּוֹן כְּקֹשִׁי דְּעָלְמָא, מַה קֹּשִׁי דְּעָלְמָא טָמֵא בְּנִדָּה, אַף קֹשִׁי דְּתוֹךְ מְלֹאת טָמֵא בְּנִדָּה:

כָּל אַחַד עָשָׂר יוֹם. שֶׁאַחַר שִׁבְעַת יְמֵי נִדָּה: בְּחֶזְקַת טָהֳרָה. וְאֵינָהּ צְרִיכָה לִבְדֹּק עַצְמָהּ (כא). דַּהֲלָכָה לְמֹשֶׁה מִסִּינַי שֶׁאֵין בֵּין נִדָּה לְנִדָּה פָּחוֹת מֵאַחַד עָשָׂר יוֹם: יָשְׁבָה לָהּ וְלֹא בָדְקָה. מַתְנִיתִין חַסּוֹרֵי מִחַסְּרָא וְהָכִי קָתָנֵי, בִּימֵי נִדָּתָהּ בַּעְיָא בְּדִיקָה. יָשְׁבָה וְלֹא בָּדְקָה, שָׁגְגָה, נֶאֶנְסָה, הֵזִידָה וְלֹא בָּדְקָה, טְהוֹרָה: הִגִּיעַ שְׁעַת וִסְתָּהּ וְלֹא בָדְקָה. אַף עַל פִּי שֶׁבָּדְקָה אַחַר כָּךְ וּמָצְאָה טְהוֹרָה, הוֹאִיל וְלֹא בָּדְקָה בִּשְׁעַת וֶסְתָּהּ, טְמֵאָה (כב), דְּאֹרַח בִּזְמַנּוֹ בָּא: בְּמַחֲבֵא. מֵחֲמַת לִסְטִים אוֹ חַיִל שֶׁבָּא לָעִיר. וְאֵין הֲלָכָה כְּרַבִּי מֵאִיר: אֲבָל יְמֵי הַזָּב וְהַזָּבָה. שִׁבְעַת יָמִים נְקִיִּים שֶׁצְּרִיכִים לִסְפֹּר: וְשׁוֹמֶרֶת יוֹם. כְּגוֹן הָרוֹאָה יוֹם אֶחָד אוֹ שְׁנֵי יָמִים בִּימֵי זִיבָתָהּ, אֵינָהּ סוֹפֶרֶת אֶלָּא יוֹם אֶחָד בִּנְקִיּוּת, וְטוֹבֶלֶת לָעֶרֶב (כג) וּטְהוֹרָה לְבַעְלָהּ: הֲרֵי אֵלּוּ בְּחֶזְקַת טֻמְאָה. כֵּיוָן שֶׁנִּפְתַּח מַעְיָנָהּ וְרָאֲתָה הַיּוֹם, אַף עַל גַּב דְּתוֹךְ אַחַד עָשָׂר הִיא, הַוְיָא לְמָחָר בְּחֶזְקַת טֻמְאָה:

פרק ה - משנה ג

מִטַּמְּאָה בְנִדָּה. אִם רָאֲתָה דָּם: בַּת עֲשָׂרָה יָמִים מִטַּמְּאָה בְזִיבָה. אִם רָאֲתָה שְׁלֹשָׁה יָמִים רְצוּפִים לְאַחַר שִׁבְעָה. דְּכָל דָּם שֶׁרוֹאָה בְּתוֹךְ שִׁבְעָה דַּם נִדָּה הוּא, הִלְכָּךְ אִי אֶפְשָׁר לָהּ לִטָּמֵא בַּזִּיבָה דְּהַיְנוּ לִהְיוֹת זָבָה גְּדוֹלָה, עַד עֲשָׂרָה יָמִים: תִּינוֹק בֶּן יוֹם אֶחָד מִטַּמֵּא בְזִיבָה. דִּכְתִיב (שָׁם טו) אִישׁ אִישׁ כִּי יִהְיֶה זָב, לִמֵּד עַל תִּינוֹק בֶּן יוֹמוֹ שֶׁמִּטַּמֵּא בְּזִיבָה: וּמִטַּמֵּא בִנְגָעִים. דְּגַבֵּי נְגָעִים כְּתִיב (שָׁם יג) אָדָם כִּי יִהְיֶה בְעוֹר בְּשָׂרוֹ, וְתִינוֹק בֶּן יוֹמוֹ אָדָם הוּא, דְּאָדָם כָּל שֶׁהוּא מַשְׁמַע: וּמִטַּמֵּא בִטְמֵא מֵת. דִּכְתִיב (בְּמִדְבַּר יט) וְעַל הַנְּפָשׁוֹת אֲשֶׁר הָיוּ שָׁם, נֶפֶשׁ כָּל שֶׁהוּא (ג): וְזוֹקֵק לְיִבּוּם. אִם נוֹלַד יוֹם אֶחָד קֹדֶם מִיתַת אָחִיו. אֲבָל נוֹלַד לְאַחַר מִיתַת אָחִיו, אֵינוֹ זוֹקֵק, דִּכְתִיב (דְּבָרִים כח) כִּי יֵשְׁבוּ אַחִים יַחְדָּו, שֶׁהָיְתָה לָהֶם יְשִׁיבָה אַחַת בָּעוֹלָם (ד): וּפוֹטֵר מִן הַיִּבּוּם. אִם נוֹלַד לְאַחַר מִיתַת אָבִיו וְחָיָה שָׁעָה אַחַת וּמֵת (ה). אֲבָל כָּל זְמַן שֶׁמְּעֻבֶּרֶת, אֵינוֹ מַתִּירָהּ לִנָּשֵׂא: וּמַאֲכִיל בַּתְּרוּמָה. בַּת יִשְׂרָאֵל שֶׁנִּסֵּת לְכֹהֵן וּמֵת, וְנוֹלַד בֵּן לְאַחַר מִיתָתוֹ, אוֹכֶלֶת בִּשְׁבִילוֹ בּוֹ בַּיּוֹם שֶׁנּוֹלַד, כְּדִכְתִיב (וַיִּקְרָא כב) וִילִיד בֵּיתוֹ הֵם יֹאכְלוּ בְלַחְמוֹ, וְדָרְשִׁינַן יַאֲכִילוּ אֶת אִמָּם. אֲבָל הִנִּיחָהּ מְעֻבֶּרֶת, אֵינָהּ אוֹכֶלֶת בִּשְׁבִילוֹ עַד שֶׁיִּוָּלֵד: וּפוֹסֵל מִן הַתְּרוּמָה. כְּגוֹן כֹּהֵן שֶׁהָיוּ לוֹ שְׁתֵּי נָשִׁים, אַחַת גְּרוּשָׁה שֶׁעָבַר וּנְשָׂאָהּ כְּשֶׁהִיא גְרוּשָׁה, וְאַחַת שֶׁאֵינָהּ גְּרוּשָׁה, וְיֵשׁ לוֹ בָּנִים מִשֶּׁאֵינָהּ גְּרוּשָׁה, וַעֲבָדִים שֶׁנָּפְלוּ לָהֶן בִּירֻשָּׁה אוֹכְלִים בַּתְּרוּמָה בִּשְׁבִילָן, דִּכְתִיב (שָׁם) וְכֹהֵן כִּי יִקְנֶה נֶפֶשׁ קִנְיַן כַּסְפּוֹ, וְיֵשׁ לוֹ בֶּן יוֹם אֶחָד מִן הַגְּרוּשָׁה וְהוּא חָלָל, וְיֵשׁ לוֹ חֵלֶק בָּעֲבָדִים, וּפוֹסֵל אֶת כֻּלָּן מִן הַתְּרוּמָה (ו), דְּאֵין בְּרֵרָה לָדַעַת מִי מֵהֶן מַגִּיעַ לְחֶלְקוֹ. וְדַוְקָא בֶּן יוֹם אֶחָד דְּרָאוּי לִירֻשָּׁה. אֲבָל כָּל זְמַן שֶׁלֹּא נוֹלַד, אוֹכְלִים כֻּלָּן, דְּאֵין זְכִיָּה לְעֻבָּר (ז): וְנוֹחֵל וּמַנְחִיל. נוֹחֵל אֶת הָאֵם. אִם מֵתָה אִמּוֹ בַּיּוֹם שֶׁנּוֹלַד, הֲרֵי הוּא נוֹחֲלָהּ. וּכְשֶׁמֵּת הוּא, בָּאִין אֶחָיו מֵאָבִיו וְיוֹרְשִׁים הֵימֶנּוּ נִכְסֵי אִמּוֹ. אֲבָל עֻבָּר שֶׁבְּמֵעֶיהָ, אִם מֵתָה מְעֻבֶּרֶת אֵינוֹ נוֹחֵל אֶת אִמּוֹ לְהַנְחִיל לְאֶחָיו מֵאָבִיו, שֶׁהֲרֵי מֵת הוּא קֹדֶם אִמּוֹ, וְאֵין הַבֵּן יוֹרֵשׁ אֶת אִמּוֹ בַּקֶּבֶר לְהַנְחִיל לְאֶחָיו מֵאָבִיו: וְהַהוֹרְגוֹ חַיָּב. דִּכְתִיב (וַיִּקְרָא כד) וְאִישׁ כִּי יַכֶּה כָּל נֶפֶשׁ, מִכָּל מָקוֹם: כְּחָתָן שָׁלֵם. שֶׁחַיָּבִים לְהִתְאַבֵּל עָלָיו. וְהָנֵי מִלֵּי, דְּקִים לְהוּ בְּגַוֵּיהּ שֶׁכָּלוּ לוֹ חֳדָשָׁיו, אֲבָל לֹא קִים לְהוּ שֶׁכָּלוּ לוֹ חֳדָשָׁיו, אֵין חַיָּבִין לְהִתְאַבֵּל עָלָיו (י) עַד שֶׁיִּהְיֶה בֶּן שְׁלֹשִׁים וְאֶחָד יוֹם:

פרק ה - משנה ד

מִתְקַדֶּשֶׁת בְּבִיאָה. אִם מְסָרָהּ אָבִיהָ לְכָךְ: וְאִם בָּא עָלֶיהָ יָבָם קְנָאָהּ. לִהְיוֹת אִשְׁתּוֹ לְכָל דָּבָר, וְיוֹרְשָׁהּ וּמִטַּמֵּא לָהּ וְזָכָה בְּנִכְסֵי אָחִיו: וְחַיָּבִין עָלֶיהָ מִשּׁוּם אֵשֶׁת אִישׁ. אִם קִבֵּל בָּהּ אָבִיהָ קִדּוּשִׁין מֵאֶחָד, וּבָא עָלֶיהָ אַחֵר, נֶהֱרָג עָלֶיהָ (יא), דְּבִיאָתָהּ בִּיאָה: וּמְטַמְּאָה אֶת בּוֹעֲלָהּ. אִם הִיא נִדָּה. אֲבָל פָּחוֹת מִכָּן, אַף עַל פִּי שֶׁהִיא מְטַמְּאָה בְּמַגָּע, אֵינָהּ מְטַמְּאָה אֶת בּוֹעֲלָהּ מִשּׁוּם בּוֹעֵל נִדָּה לְטֻמְאַת שִׁבְעָה, אֶלָּא מִשּׁוּם נוֹגֵעַ וּלְטֻמְאַת עֶרֶב: תַּחְתּוֹן. שֶׁל בּוֹעֵל נִדָּה: כְּעֶלְיוֹנוֹ. שֶׁל זָב, שֶׁמְּטַמֵּא אֳכָלִין וּמַשְׁקִין וְלֹא אָדָם וְכֵלִים: מִן הַפְּסוּלִין. כְּגוֹן נָכְרִי, וְעֶבֶד, וְנָתִין, מַמְזֵר, אוֹ חָלָל: פְּסָלָהּ. מִתְּרוּמַת אָבִיהָ (יב), אִם הִיא בַּת כֹּהֵן: מִכָּל הָעֲרָיוֹת. כְּגוֹן אָבִיהָ אוֹ חָמִיהָ (יג): וְהִיא פְטוּרָה. מִפְּנֵי שֶׁהִיא קְטַנָּה וְאֵינָהּ בַּת עֳנָשִׁים: כְּנוֹתֵן אֶצְבַּע בָּעָיִן. שֶׁדּוֹמֵעַ וְחוֹזֵר לִכְמוֹת שֶׁהָיָה, אַף כָּאן חוֹזֶרֶת בְּתוּלָה כְּמוֹ שֶׁהָיְתָה (יד):

פרק ה - משנה ה

שֶׁבָּא עַל יְבִמְתּוֹ קְנָאָהּ. וְזָכָה בְּנִכְסֵי אָחִיו (טז). וְאַף עַל פִּי שֶׁאֵין קִנְיָן לְקָטָן, שָׁאנֵי הָכָא דְּמִן שְׁמַיָּא אַקְנוֹיַהּ נִיהָלֵיהּ: וְאֵינוֹ נוֹתֵן גֵּט עַד שֶׁיַּגְדִּיל. אִם בָּא לְגָרְשָׁהּ. דְּקִדּוּשֵׁי אָחִיו קִדּוּשִׁין גְּמוּרִים, וְגֵרוּשִׁין שֶׁל זֶה אֵינָן גֵּרוּשִׁין גְּמוּרִין. וְדַוְקָא כְּשֶׁבָּא עָלֶיהָ לְאַחַר שֶׁהִגְדִּיל, דַּי לָהּ בְּגֵט. אֲבָל אִם לֹא בָּא עָלֶיהָ לְאַחַר שֶׁהִגְדִּיל, צְרִיכָה גֵּט וַחֲלִיצָה: וּמִטַּמֵּא בְנִדָּה. אִם בָּא עָלֶיהָ בְּנִדָּתָהּ. דְּבִיאָתוֹ בִּיאָה לְכָל דָּבָר: וּפוֹסֵל. כְּגוֹן נָכְרִי וְעֶבֶד אוֹ חָלָל (יז) בֶּן תֵּשַׁע שָׁנִים שֶׁבָּא עַל הַכֹּהֶנֶת, פְּסָלָהּ מִתְּרוּמַת אָבִיהָ: וְאֵינוֹ מַאֲכִיל. אִם הוּא כֹּהֵן וְנָשָׂא יִשְׂרְאֵלִית, אֵינוֹ מַאֲכִילָהּ בַּתְּרוּמָה, לְפִי שֶׁאֵין קִנְיָנוֹ קִנְיָן גָּמוּר. וַאֲפִלּוּ הוּא יָבָם דְּקַנְיַהּ בְּהַךְ בִּיאָה כְּדַאֲמַרַן, אֲפִלּוּ הָכִי אֵינוֹ מַאֲכִילָהּ בַּתְּרוּמָה: וּפוֹסֵל אֶת הַבְּהֵמָה. אִם רְבָעָהּ עַל פִּי עֵד אֶחָד (יח): וְנִסְקֶלֶת עַל יָדוֹ. אִם יֵשׁ שְׁנֵי עֵדִים:

פרק ה - משנה ו

נְדָרֶיהָ נִבְדָּקִים. אִם יָדְעָה לְשֵׁם מִי נָדְרָה, לְשֵׁם מִי הִקְדִּישָׁה, נָדְרָה נֶדֶר: נְדָרֶיהָ קַיָּמִין. וְהוּא שֶׁהֵבִיאָה שְׁתֵּי שְׂעָרוֹת. וְכֵן בֶּן שְׁלֹשׁ עֶשְׂרֵה שָׁנָה וְיוֹם אֶחָד נְדָרָיו קַיָּמִים, דַּוְקָא כְּשֶׁהֵבִיא שְׁתֵּי שְׂעָרוֹת: וּבוֹדְקִין כָּל שְׁתֵּים עֶשְׂרֵה. מַהוּ דְּתֵימָא הוֹאִיל וְאָמַר מַר שְׁלֹשִׁים יוֹם בַּשָּׁנָה חֲשׁוּבִים שָׁנָה, הֵיכָא דִּבְדָקְנוּהָ שְׁלֹשִׁים יוֹם קַמָּאֵי דִּשְׁתֵּים עֶשְׂרֵה וְלֹא יָדְעָה לְפָרֵשׁ לְשֵׁם מִי נָדְרָה, תּוּ לֹא נִבְדֹּק, וּלְחַזְקָהּ כִּקְטַנָּה, קָמַשְׁמַע לָן: קֹדֶם לַזְּמַן הַזֶּה. קֹדֶם רֹאשׁ הַשָּׁנָה (כא) שֶׁל שְׁתֵּים עֶשְׂרֵה לְתִינוֹקֶת וְשֶׁל שְׁלֹשׁ עֶשְׂרֵה לְתִינוֹק: אֵין נִדְרֵיהֶן נֶדֶר. דִּקְטַנִּים הֵן:

פרק ה - משנה ז

אֵיזוֹ הִיא סִימָנֶיהָ. אַצֶּמֶל קָאֵי: קֶמֶט. שֶׁגָּדְלוּ דַּדֶּיהָ עַד שֶׁכְּפוּלִים מְעַט עַל הֶחָזֶה וְיֵרָאֶה כְּקֶמֶט: מִשֶּׁיַּטּוּ הַדַּדִּין. גְּדוֹלִים יוֹתֵר מִקֶּמֶט: הַפִּטֹּמֶת. רֹאשׁ הַדַּד. כְּמוֹ נִטְּלָה פִּטְמָתוֹ [סֻכָּה פ"ג מ"ו]: עַל הָעֹקֶץ. חֻדּוֹ שֶׁל דַּד שֶׁתִּינוֹק יוֹנֵק. וּפִטֹּמֶת הַיְנוּ סְבִיבוֹת הָעֹקֶץ. וְכָל הַסִּימָנִים הַלָּלוּ כְּשֶׁנִּרְאוּ בְּאִשָּׁה הֲרֵי הִיא בּוֹגֶרֶת, לְמַר כִּדְאִית לֵיהּ וּלְמַר כִּדְאִית לֵיהּ, וְשׁוּב אֵין לְאָבִיהָ רְשׁוּת בָּהּ. וַהֲלָכָה כְּדִבְרֵי כֻּלָּם לְהַחְמִיר. הִלְכָּךְ אִם נִרְאָה בָּהּ אֶחָד מִן הַסִּימָנִים הַלָּלוּ, דָּיְנִינַן בָּהּ דִּין בּוֹגֶרֶת לְחֻמְרָא. אֲבָל לֹא לְקֻלָּא (כג), שֶׁאִם קִדְּשָׁה אֶת עַצְמָהּ, צְרִיכָה גֵּט שֶׁמָּא בּוֹגֶרֶת הִיא. וְאִם אָבִיהָ קִדְּשָׁהּ, צְרִיכָה נַמִּי גֵּט, דְּשֶׁמָּא עֲדַיִן נַעֲרָה הִיא, עַד שֶׁיִּהְיוּ בָּהּ כָּל הַסִּימָנִים הַלָּלוּ כֻּלָּם, וְאָז תֵּחָשֵׁב בּוֹגֶרֶת בֵּין לְקֻלָּא בֵּין לְחֻמְרָא:

וְהִיא אַיְלוֹנִית. כְּלוֹמַר וְצָרִיךְ שֶׁיֵּרָאוּ בָּהּ נַמִּי סִימָנֵי אַיְלוֹנִית. שֶׁאִם לֹא נִרְאוּ בָּהּ סִימָנֵי אַיְלוֹנִית, אַף עַל פִּי שֶׁהִיא בַּת עֶשְׂרִים וְלֹא הֵבִיאָה שְׁתֵּי שְׂעָרוֹת, עֲדַיִן קְטַנָּה הִיא, עַד שֶׁתִּהְיֶה בַּת שְׁלֹשִׁים וְחָמֵשׁ שָׁנָה וְיוֹם אֶחָד. וְאִם לֹא הֵבִיאָה שְׁתֵּי שְׂעָרוֹת אָז, אַף עַל פִּי שֶׁלֹּא נִרְאוּ בָּהּ סִימָנֵי אַיְלוֹנִית, הֲרֵי זוֹ אַיְלוֹנִית. וְסִימָנֵי אַיְלוֹנִית פֵּרַשְׁנוּ [רֵישׁ] פֶּרֶק קַמָּא דִּיבָמוֹת: לֹא חוֹלֶצֶת וְלֹא מִתְיַבֶּמֶת. וּמֻתֶּרֶת לִנָּשֵׂא לַשּׁוּק. דְּרַחֲמָנָא פַּטְרַהּ, דִּכְתִיב (דְּבָרִים כה) וְהָיָה הַבְּכוֹר אֲשֶׁר תֵּלֵד, פְּרָט לְאַיְלוֹנִית, שֶׁאֵינָהּ יוֹלֶדֶת: וְהוּא סָרִיס. כְּלוֹמַר שֶׁנִּרְאוּ בּוֹ סִימָנֵי סָרִיס. דְּאִי לֹא נִרְאוּ בּוֹ סִימָנֵי סָרִיס, עֲדַיִן הוּא קָטָן עַד שֶׁיִּהְיֶה בֶּן שְׁלֹשִׁים וְחָמֵשׁ שָׁנָה וְיוֹם אֶחָד, דְּאָז אֲפִלּוּ לֹא נִרְאוּ בּוֹ סִימָנֵי סָרִיס, בְּיָדוּעַ שֶׁהוּא סָרִיס, כֵּיוָן שֶׁלֹּא הֵבִיא שְׁתֵּי שְׂעָרוֹת. וְסָרִיס לָאו בַּר חֲלִיצָה [וְיִבּוּם] הוּא, דִּכְתִיב (שָׁם) לְהָקִים לְאָחִיו שֵׁם בְּיִשְׂרָאֵל, וְהַאי לָאו בַּר הָכִי הוּא: רַבִּי אֱלִיעֶזֶר אוֹמֵר הַזָּכָר כְּדִבְרֵי בֵית הִלֵּל. בֶּן עֶשְׂרִים, אִם נִרְאוּ בּוֹ סִימָנֵי סָרִיס, הֲרֵי זֶה סָרִיס: וְהַנְּקֵבָה. בַּת שְׁמֹנֶה עֶשְׂרֵה, אִם נִרְאוּ בָּהּ סִימָנֵי אַיְלוֹנִית, הֲרֵי זוֹ אַיְלוֹנִית כְּדִבְרֵי בֵּית שַׁמַּאי. וְאֵין הֲלָכָה כְּרַבִּי אֱלִיעֶזֶר:

כָּל כְּלִי חֶרֶס שֶׁהוּא מַכְנִיס. מַשְׁקֶה. כְּשֶׁמַּנִּיחוֹ עַל הַמַּיִם, יֵשׁ בּוֹ נֶקֶב גָּדוֹל שֶׁהַמַּיִם נִכְנָסִים לְתוֹכוֹ, כָּל שֶׁכֵּן שֶׁאִם נוֹתֵן הַמַּיִם לְתוֹכוֹ יוֹצֵא הַמַּשְׁקֶה לַחוּץ. אֲבָל יֵשׁ נֶקֶב קָטָן שֶׁמּוֹצִיא וְאֵינוֹ מַכְנִיס. וְנַפְקָא מִנַּהּ לְהָא דְּקַיְמָא לָן דִּכְלִי חֶרֶס שָׁלֵם שֶׁנִּקַּב, אֵינוֹ פָּסוּל לְקַדֵּשׁ בּוֹ מֵי חַטָּאת, דְּהַיְנוּ לָתֵת בְּתוֹכוֹ מַיִם שֶׁל אֵפֶר פָּרָה (ב), עַד שֶׁיִּנָּקֵב נֶקֶב גָּדוֹל בְּכוֹנֵס מַשְׁקֶה. אֲבָל נִקַּב נֶקֶב קָטָן כְּמוֹצִיא מַשְׁקֶה, לֹא נָפֵיק מִתּוֹרַת כְּלִי וְלֹא מִפְּסִיל בְּהָכִי. וּלְעוֹלָם אֵין נֶקֶב קָטָן בְּמוֹצִיא מַשְׁקֶה מוֹצִיא מִתּוֹרַת כְּלִי אֶלָּא הַגִּסְטְרָא בִּלְבַד, דְּהַיְנוּ שֶׁבֶר כְּלִי שֶׁיִּחֲדוֹ לְמַלְּאוֹת בּוֹ מַיִם וּלְהִשְׁתַּמֵּשׁ בּוֹ, אִם נִקַּב בְּמוֹצִיא מַשְׁקֶה, בָּטַל מִתּוֹרַת כְּלִי, שֶׁאֵין אוֹמְרִים הָבֵא גִסְטְרָא אַחֶרֶת וְנָשִׂים תַּחַת גִּסְטְרָא זוֹ לְקַבֵּל הַמַּשְׁקֶה הַיּוֹצֵא. אֲבָל כְּלִי שָׁלֵם שֶׁנִּקַּב בְּמוֹצִיא מַשְׁקֶה, לֹא בָּטַל מִכְּלוּם, לְפִי שֶׁאָדָם חָס עָלָיו וְאֵינוֹ שׁוֹבְרוֹ, וּמִשְׁתַּמֵּשׁ בּוֹ, וּמֵבִיא שֶׁבֶר כְּלִי וּמַנִּיחַ תַּחְתָּיו לְקַבֵּל הַמַּשְׁקֶה הַיּוֹצֵא מִמֶּנּוּ. וּמֵעַתָּה הָכִי פֵּרוּשָׁא דְּמַתְנִיתִין, כָּל כְּלִי חֶרֶס הַפָּסוּל לְמֵי חַטָּאת פָּסוּל לְגִסְטְרָא, וְיֵשׁ פָּסוּל לְגִסְטְרָא וְכָשֵׁר לְמֵי חַטָּאת: כָּל אֵבָר שֶׁיֶּשׁ בּוֹ צִפֹּרֶן. כְּגוֹן אֶצְבַּע יְתֵרָה שֶׁבְּאֶצְבְּעוֹת הַיָּד, אִם יֵשׁ בָּהּ צִפֹּרֶן, חֲשׁוּבָה אֵבָר, וּמְטַמֵּא בְּמַגָּע וּבְמַשָּׂא וּבְאֹהֶל, כְּאֵבָר שֶׁיֵּשׁ בּוֹ עֶצֶם שֶׁמְּטַמֵּא בְּאֹהֶל וַאֲפִלּוּ אִי לֵיכָּא כַּזַּיִת בָּשָׂר, דְּקַיְמָא לָן הָאֵבָרִים אֵין לָהֶם שִׁעוּר: וְיֵשׁ שֶׁיֶּשׁ בּוֹ עֶצֶם. וְלֹא חֲשׁוּבָה אֵבָר אִם יוֹתֶרֶת הִיא, כֵּיוָן שֶׁאֵין בּוֹ צִפֹּרֶן, וּמְטַמְּאָה בְּמַגָּע וּבְמַשָּׂא, וְאֵינָהּ מְטַמְּאָה בְּאֹהֶל כָּל זְמַן שֶׁאֵין בָּהּ כַּזַּיִת בָּשָׂר. וְאִי לָאו יוֹתֵרֶת הִיא, וַדַּאי מְטַמְּאָה בְּאֹהֶל מִשּׁוּם אֵבָר, אַף עַל גַּב דְּאֵין בָּהּ צִפֹּרֶן וְאַף עַל גַּב דְּלֵיכָּא כַּזַּיִת בָּשָׂר:

כֹּל הַמִּטַּמֵּא בְּמִדְרָס. הַזָּב, כָּל כְּלִי הָרָאוּי לִהְיוֹת אַב הַטֻּמְאָה (ג) בְּמִדְרַס הַזָּב, כְּגוֹן כְּלִי הַמְיֻחָד לְמִשְׁכָּב וּמוֹשָׁב, רָאוּי לִהְיוֹת אַב הַטֻּמְאָה אִם נָגַע בְּמֵת אוֹ נִטְמָא בְּאֹהֶל הַמֵּת. וְיֵשׁ שֶׁהוּא כְּלִי גָּמוּר לְקַבֵּל כָּל טֻמְאוֹת וְנַעֲשֶׂה אַב הַטֻּמְאָה עַל יְדֵי מֵת, וְאֵינוֹ נַעֲשֶׂה אַב הַטֻּמְאָה עַל יְדֵי הַזָּב בְּמִדְרָס, כְּגוֹן אִם כָּפָה הַזָּב סְאָה וְיָשַׁב עָלֶיהָ, אוֹ תַּרְקַב וְיָשַׁב עָלֶיהָ, אֵין אוֹתוֹ כְּלִי טָמֵא מִדְרָס לִהְיוֹת נַעֲשֶׂה אַב הַטֻּמְאָה, אֶלָּא רִאשׁוֹן לַטֻּמְאָה בְּמַגָּעוֹ שֶׁל זָב, דִּכְתִיב (וַיִּקְרָא טו) אֲשֶׁר יֵשֵׁב עָלָיו הַזָּב, מִי שֶׁמְּיֻחָד לִישִׁיבָה, יָצָא זֶה שֶׁאוֹמְרִים לוֹ עֲמֹד וְנַעֲשֶׂה מְלַאכְתֵּנוּ. אֲבָל אִם נָגַע בְּמֵת, נַעֲשֶׂה אַב הַטֻּמְאָה, שֶׁאֵין אוֹמְרִים בִּטְמֵא מֵת (ד) עֲמֹד וְנַעֲשֶׂה מְלַאכְתֵּנוּ (ה):

כֹּל שֶׁחַיָּב בְּרֵאשִׁית הַגֵּז. רְחֵלִים וְאֵילִים בִּלְבַד: חַיָּב בַּמַּתָּנוֹת. הַזְּרוֹעַ וְהַלְּחָיַיִם וְהַקֵּבָה: וְיֵשׁ שֶׁחַיָּב בַּמַּתָּנוֹת. שֶׁחִיּוּב הַמַּתָּנוֹת לְשׁוֹר וְעִזִּים וּכְבָשִׂים:

פרק ו - משנה ח

וְיֵשׁ שֶׁיֶּשׁ לוֹ שְׁבִיעִית וְאֵין לוֹ בִעוּר. כְּגוֹן הַדְּבָרִים שֶׁשָּׁרְשָׁם נִשְׁאָר בַּקַּרְקַע בֵּין בִּימוֹת הַחַמָּה בֵּין בִּימוֹת הַגְּשָׁמִים, כְּמוֹ הַלּוּף שׁוֹטֶה וְהַדַּנְדַּנָּה, שֶׁקּוֹרִים בַּעֲרָבִי נַעְנַ"ע, וּבְלַעַ"ז מֵינְטַ"א, דְּכֵיוָן שֶׁאֵין כָּלִין לְעוֹלָם מִן הַשָּׂדֶה, אֵין חַיָּב לְבַעֲרָם מִן הַבַּיִת, דִּכְתִיב (וַיִּקְרָא כה) וְלִבְהֶמְתְּךָ וְלַחַיָּה אֲשֶׁר בְּאַרְצֶךָ, כָּל זְמַן שֶׁחַיָּה אוֹכֶלֶת בַּשָּׂדֶה, הַאֲכֵל לִבְהֶמְתְּךָ מִן הַבַּיִת, כָּלָה לַחַיָּה מִן הַשָּׂדֶה כַּלֵּה לִבְהֶמְתְּךָ מִן הַבַּיִת, וְכָל הָנָךְ דְּלֹא כָּלוּ לַחַיָּה מִן הַשָּׂדֶה, אֵין צָרִיךְ לְבַעֲרָם מִן הַבַּיִת (י):

פרק ו - משנה ט

קַשְׂקֶשֶׂת. מַלְבּוּשׁ שֶׁל דָּג, הַקָּבוּעַ בּוֹ: סְנַפִּיר. כְּנָפַיִם שֶׁפּוֹרֵחַ בָּהֶן. וְכָל שֶׁיֵּשׁ לוֹ קַשְׂקֶשֶׂת בְּיָדוּעַ שֶׁיֵּשׁ לוֹ סְנַפִּיר וְאֵין צָרִיךְ לִבְדֹּק אַחֲרָיו עוֹד, שֶׁדָּג טָהוֹר הוּא: וְיֵשׁ שֶׁיֶּשׁ לוֹ סְנַפִּיר. וְהוּא דָּג טָמֵא, שֶׁאֵין לוֹ קַשְׂקֶשֶׂת: יֶשׁ לוֹ טְלָפַיִם. פַּרְסוֹתָיו סְדוּקוֹת. שֶׁאֵין קַרְנַיִם אֶלָּא לִבְהֵמָה וּלְחַיָּה טְהוֹרָה: וְאֵין לוֹ קַרְנָיִם. כְּגוֹן חֲזִיר (יב), שֶׁפַּרְסוֹתָיו סְדוּקוֹת, וְאֵין לוֹ קַרְנַיִם:

פרק ו - משנה י

וְיֵשׁ טָעוּן בְּרָכָה לְפָנָיו וְאֵין טָעוּן בְּרָכָה לְאַחֲרָיו. כְּגוֹן צִיצִית, וּתְפִלִּין, וּמְזוּזָה, שׁוֹפָר, סֻכָּה, וְלוּלָב (יג), שֶׁמְּבָרְכִין עֲלֵיהֶן עוֹבֵר לַעֲשִׂיָּתָן, וְאֵין טְעוּנִין בְּרָכָה לְאַחֲרֵיהֶן:

פרק ו - משנה יא

תִּינוֹקֶת שֶׁהֵבִיאָה שְׁתֵּי שְׂעָרוֹת. לְאַחַר שְׁתֵּים עֶשְׂרֵה שָׁנָה וְיוֹם אֶחָד: וְחוֹלֶצֶת אוֹ מִתְיַבֶּמֶת. אַף עַל גַּב דְּתָנָא חַיֶּבֶת בְּכָל מִצְוֹת הָאֲמוּרוֹת בַּתּוֹרָה, אִצְטְרִיךְ לְמִתְנֵי חוֹלֶצֶת אוֹ מִתְיַבֶּמֶת, דְּמַהוּ דְּתֵימָא אִישׁ כְּתִיב בַּפָּרָשָׁה (דְּבָרִים כה), וְאִם לֹא יַחְפֹּץ הָאִישׁ, הִלְכָּךְ קָטָן לֹא חָלֵיץ. אֲבָל אִשָּׁה דְּלֹא כְּתִיב בַּפָּרָשָׁה אֶלָּא יְבִמְתּוֹ, בֵּין גְּדוֹלָה בֵּין קְטַנָּה חוֹלֶצֶת אוֹ מִתְיַבֶּמֶת, קָא מַשְׁמַע לָן דְּאִי אַיְיתִּיאַת שְׁתֵּי שְׂעָרוֹת, אִין. אִי לֹא, לֹא. מַאי טַעֲמָא, דְּמַקְשִׁינַן אִשָּׁה לְאִישׁ: וְכֵן הַתִּינוֹק שֶׁהֵבִיא שְׁתֵּי שְׂעָרוֹת. לְאַחַר שְׁלֹשׁ עֶשְׂרֵה שָׁנָה וְיוֹם אֶחָד. וְקַמֵּי שְׁתֵּים עֶשְׂרֵה לִנְקֵבָה וּשְׁלֹשׁ עֶשְׂרֵה לְזָכָר, אֵינוֹ סִימָן אֶלָּא שׁוּמָא: וְרָאוּי לִהְיוֹת בֵּן סוֹרֵר וּמוֹרֶה מִשֶּׁיָּבִיא שְׁתֵּי שְׂעָרוֹת עַד שֶׁיַּקִּיף זָקָן הַתַּחְתּוֹן. אֲבָל קֹדֶם שֶׁיָּבִיא שְׁתֵּי שְׂעָרוֹת לָאו בַּר עוֹנְשִׁים הוּא. וּלְאַחַר שֶׁהִקִּיף זָקָן הַתַּחְתּוֹן, רָאוּי לְהוֹלִיד (יד), וְרַחֲמָנָא אָמַר בֵּן, וְלֹא הָרָאוּי לִהְיוֹת אָב: הַתַּחְתּוֹן וְלֹא הָעֶלְיוֹן. כְּלוֹמַר, בְּאֵיזוֹ הַקָּפַת זָקָן אָמְרוּ, בְּתַחְתּוֹן שֶׁל מַטָּה, וְלֹא בָּעֶלְיוֹן: עַד שֶׁיִּרְבֶּה הַשָּׁחֹר. שֶׁיַּשְׁחִיר אוֹתוֹ מָקוֹם שְׂעָרוֹת הַרְבֵּה (טו). וּפְסַק הֲלָכָה, שֶׁהַבַּת מְמָאֶנֶת עַד שֶׁתְּהֵא בַּת שְׁתֵּים עֶשְׂרֵה שָׁנָה וְיוֹם אֶחָד. וּלְאַחַר זְמַן זֶה, אִם הֵבִיאָה סִימָנִים, אֵינָהּ יְכוֹלָה לְמָאֵן. וְאִם לֹא הֵבִיאָה סִימָנִים, מְמָאֶנֶת וְהוֹלֶכֶת וַאֲפִלּוּ עַד שֶׁתְּהֵא בַּת עֶשְׂרִים שָׁנָה וְיֵרָאוּ בָּהּ סִימָנֵי אַיְלוֹנִית (טז). וְהָנֵי מִילֵי, כְּשֶׁלֹּא בָּא עָלֶיהָ לְאַחַר שֶׁהָיוּ לָהּ שְׁתֵּים עֶשְׂרֵה שָׁנָה וְיוֹם אֶחָד. אֲבָל אִם בָּא עָלֶיהָ לְאַחַר זְמַן זֶה, שׁוּב אֵינָהּ יְכוֹלָה לְמָאֵן. וְאַף עַל פִּי שֶׁלֹּא נִרְאוּ בָּהּ סִימָנִים, חָיְשִׁינַן שֶׁמָּא הֵבִיאָה שְׁתֵּי שְׂעָרוֹת (יז) וְנָשְׁרוּ:

הָרוֹאָה כֶתֶם. שֶׁמָּצְאָה טִפַּת דָּם בַּחֲלוּקָהּ: הֲרֵי זוֹ מְקֻלְקֶלֶת. אֵינָהּ יוֹדַעַת פֶּתַח נִדּוֹתֶיהָ לֵידַע מָתַי יַתְחִילוּ אַחַד עָשָׂר יוֹם שֶׁבֵּין נִדָּה לְנִדָּה, דְּהָא לֹא יָדְעָה אֵימָת חֲזַאי: וְחוֹשֶׁשֶׁת מִשּׁוּם זוֹב. אִם עָבְרוּ עָלֶיהָ שְׁלֹשָׁה יְמֵי זִיבָה מִשֶּׁלָּבְשָׁה חָלוּק זֶה, וּמָצְאָה עָלָיו כֶּתֶם גָּדוֹל שִׁעוּר שְׁלֹשָׁה גְּרִיסִין וָעוֹד (יט), כְּלוֹמַר מְעַט יוֹתֵר מִשְּׁלֹשָׁה גְּרִיסִין, מְסַפְּקִינַן לְהוּ דִּלְמָא כָּל חֲדָא וַחֲדָא בְּחַד יוֹמָא, וְהַוְיָא זָבָה גְּדוֹלָה מִסָּפֵק, וְאַף עַל גַּב דִּבְמָקוֹם אֶחָד נִמְצְאוּ: וַחֲכָמִים אוֹמְרִים אֵין בַּכְּתָמִים מִשּׁוּם זוֹב. בַּגְּמָרָא מוֹקֵי מִלְּתַיְהוּ דַּחֲכָמִים דְּהָכָא, כִּיחִידָאָה, וְאָמַר דְּרַבִּי חֲנִינָא בֶּן אַנְטִיגְנוֹס הִיא. וּמוֹדֶה רַבִּי חֲנִינָא בֶּן אַנְטִיגְנוֹס שֶׁאִם לָבְשָׁה שָׁלֹשׁ חֲלוּקוֹת לְבָנוֹת בִּשְׁלֹשָׁה יָמִים בִּימֵי זִיבָה, חָלוּק לְיוֹם, וּמָצְאָה בְּכָל אַחַת מֵהֶן כֶּתֶם, דְּחוֹשֶׁשֶׁת לְזִיבוּת. וְאֵין צָרִיךְ לוֹמַר אִם רָאֲתָה שְׁנֵי יָמִים דָּם מִגּוּפָהּ, וּבַשְּׁלִישִׁי לָבְשָׁה חָלוּק הַבָּדוּק לָהּ וּמָצְאָה בּוֹ כֶּתֶם, דִּפְשִׁיטָא דְּחוֹשֶׁשֶׁת לְזִיבוּת. וְלֹא נֶחֱלַק רַבִּי חֲנִינָא בֶּן אַנְטִיגְנוֹס דְּהוּא חֲכָמִים דְּמַתְנִיתִין, אֶלָּא עַל הַלּוֹבֶשֶׁת חָלוּק אֶחָד בִּשְׁלֹשָׁה יָמִים מִימֵי זִיבָתָהּ וּלְבַסּוֹף מָצְאָה כֶּתֶם גָּדוֹל שִׁעוּר שְׁלֹשָׁה גְּרִיסִין וָעוֹד בְּמָקוֹם אֶחָד, אוֹ אֲפִלּוּ בִּשְׁלֹשָׁה מְקוֹמוֹת, דְּאִיהוּ סָבַר בִּכְהַאי גַוְנָא אֵינָהּ חוֹשֶׁשֶׁת לְזִיבוּת, הוֹאִיל וְלֹא הָיוּ הַכְּתָמִים בִּשְׁלֹשָׁה חֲלוּקוֹת. וְרַבִּי מֵאִיר וְרַבָּנָן דְּאִיפְּלִיגוּ עֲלֵיהּ סָבְרֵי, כְּשֶׁמָּצְאָה שִׁעוּר שְׁלֹשָׁה גְּרִיסִין וָעוֹד בְּחָלוּק אֶחָד, בֵּין בְּמָקוֹם אֶחָד בֵּין בִּשְׁלֹשָׁה מְקוֹמוֹת, חוֹשֶׁשֶׁת לְזִיבוּת. וְאֵין הֲלָכָה כְּרַבִּי חֲנִינָא בֶּן אַנְטִיגְנוֹס:

הָרוֹאָה יוֹם אַחַד עָשָׂר. שֶׁהוּא סוֹף יְמֵי זִיבָה. וּלְמָחָר מַתְחִילִין יְמֵי נִדָּה: בֵּין הַשְּׁמָשׁוֹת. סְפֵק יוֹם הוּא וְהָוֵי דַּם זִיבָה, סְפֵק לַיְלָה וּתְחִלַּת נִדָּה: תְּחִלַּת נִדָּה וְסוֹף נִדָּה. בַּגְּמָרָא מְפָרֵשׁ דְּהָכִי קָתָנֵי, תְּחִלַּת נִדָּה וְסוֹף זִיבָה. וּבַשְּׁבִיעִי לְנִדָּתָהּ, סוֹף נִדָּה וּתְחִלַּת זִיבָה. כְּלוֹמַר, בֵּין הַשְּׁמָשׁוֹת שֶׁל אַחַד עָשָׂר, אִם לַיְלָה הוּא, הֲרֵי רְאִיָּתָהּ תְּחִלַּת נִדָּה. וְאִם יוֹם, הֲרֵי הַדָּם דָּם שֶׁל סוֹף זִיבָה. וְאִם רָאֲתָה שְׁנֵי יָמִים קֹדֶם לָכֵן, חוֹשֶׁשֶׁת מִשּׁוּם זִיבָה. וְכֵן אִם רָאֲתָה בִּשְׁבִיעִי לְנִדָּתָהּ בֵּין הַשְּׁמָשׁוֹת, מְסַפְּקִינַן נַמִּי, אִם לַיְלָה הוּא, הֲרֵי הַדָּם דָּם שֶׁל תְּחִלַּת זִיבָה, וְאִם תִּרְאֶה שְׁנֵי יָמִים אַחֲרֵי כֵן, חוֹשֶׁשֶׁת מִשּׁוּם זִיבָה. וְאִם יוֹם הוּא, הֲרֵי רְאִיָּתָהּ סוֹף יְמֵי נִדָּה, וְאַף עַל פִּי שֶׁתִּרְאֶה שְׁנֵי יָמִים אַחַר כָּךְ אֵינָהּ זָבָה: יוֹם אַרְבָּעִים לַזָּכָר וְיוֹם שְׁמוֹנִים לַנְּקֵבָה בֵּין הַשְּׁמָשׁוֹת לְכֻלָּן. כְּלוֹמַר בְּאֵיזוֹ מֵאֵלּוּ שֶׁתִּרְאֶה בֵּין הַשְּׁמָשׁוֹת, סָפֵק דָּם טָמֵא סָפֵק דָּם טָהוֹר. וְאִם תִּרְאֶה שְׁנֵי יָמִים אֲחֵרִים חוּץ מִזּוֹ, הֲרֵי הִיא זָבָה (כ) מִסָּפֵק, וּמְבִיאָה קָרְבָּן וְאֵינוֹ נֶאֱכָל, שֶׁכָּל הַטּוֹעוֹת מְבִיאוֹת קָרְבָּן וְאֵינוֹ נֶאֱכָל: הַשּׁוֹטוֹת. לַטּוֹעוֹת שֶׁאֵינָן יוֹדְעוֹת מָתַי יְמֵי נִדָּה וִימֵי זִיבָה, קָרֵי שׁוֹטוֹת: עַד שֶׁאַתֶּם מְתַקְּנִים אֶת הַשּׁוֹטוֹת. אֶת הַטּוֹעוֹת הַלָּלוּ שֶׁרָאוּ בִּשְׁעַת הַסָּפֵק: בֹּאוּ וְתַקְּנוּ אֶת הַפִּקְחוֹת. שֶׁרוֹאוֹת בְּשָׁעָה וַדָּאִית וְהֵן צְרִיכוֹת תִּקּוּן וּלְפָרֵשׁ יְמֵי שִׁמּוּרָן וְתַשְׁמִישָׁן. כְּגוֹן הָנָךְ דְּמַיְתֵי בַּבְּרַיְתָא, מִי שֶׁרוֹאָה יוֹם אֶחָד טָמֵא וָיוֹם אֶחָד טָהוֹר כָּל יָמֶיהָ, וְתָמִיד הִיא רוֹאָה בַּיּוֹם וְלֹא בַּלַּיְלָה, מְשַׁמֶּשֶׁת שְׁמִינִי לְיוֹם שֶׁרָאֲתָה בּוֹ רְאִיָּה רִאשׁוֹנָה, שֶׁהֲרֵי טְהוֹרָה הִיא, שֶׁהֲרֵי בְּשִׁבְעָה לָעֶרֶב טָבְלָה וְלֹא תִּרְאֶה עַד תְּשִׁיעִי, הִלְכָּךְ מְשַׁמֶּשֶׁת שְׁמִינִי שָׁלֵם לַיְלָה וְיוֹם, וְלַיְלָה שֶׁלְּאַחֲרָיו שֶׁהוּא נַגְהֵי תְּשִׁיעִי. וּמְשַׁמֶּשֶׁת אַרְבָּעָה לֵילוֹת מִתּוֹךְ שְׁמֹנָה עָשָׂר יוֹם לִרְאִיָּה רִאשׁוֹנָה, שֶׁהֲרֵי אִשָּׁה זוֹ לֹא תְּהֵא זָבָה לְעוֹלָם, שֶׁאֵינָהּ רוֹאָה שְׁלֹשָׁה רְצוּפִים. וּכְשֶׁהִיא רוֹאָה תְּשִׁיעִי, תְּהֵא שׁוֹמֶרֶת עֲשִׂירִי וּתְשַׁמֵּשׁ בַּלַּיְלָה. וְתִרְאֶה בְּאַחַד עָשָׂר. וְתִשְׁמֹר שְׁנֵים עָשָׂר וּתְשַׁמֵּשׁ בַּלַּיְלָה. הֲרֵי שְׁנַיִם. וְתִרְאֶה בִּשְׁלֹשָׁה עָשָׂר, וְתִשְׁמֹר אַרְבָּעָה עָשָׂר וּתְשַׁמֵּשׁ בַּלַּיְלָה, הֲרֵי שְׁלֹשָׁה. וְתִרְאֶה חֲמִשָּׁה עָשָׂר, וְתִשְׁמֹר שִׁשָּׁה עָשָׂר וּתְשַׁמֵּשׁ בָּעֶרֶב, הֲרֵי אַרְבָּעָה. וּבְשִׁבְעָה עָשָׂר תִּרְאֶה, וְתִשְׁמֹר שְׁמֹנָה עָשָׂר. הֲרֵי כָּלוּ שְׁמֹנָה עָשָׂר וְלֹא שִׁמְּשָׁה אֶלָּא אַרְבָּעָה לֵילוֹת. לְבַד שְׁמִינִי וְלֵילוֹ. וְתַשְׁמִישׁ אַחֲרוֹן אֵינוֹ בְּתוֹךְ טוּבָא שֶׁצְּרִיכוֹת תִּקּוּן לִימֵי תַּשְׁמִישָׁן. וְכֻלְּהוּ מַיְתֵי לַהּ בַּבְּרַיְתָא בַּגְּמָרָא בְּפִרְקִין:

פרק ז - משנה א

דַּם הַנִּדָּה. מְטַמְּאִין לַחִין. דִּכְתִיב (וַיִּקְרָא טו) וְהַדָּוָה בְּנִדָּתָהּ, מַה שֶּׁזָּב מִמֶּנָּה טָמֵא כְּמוֹתָהּ (א), וּבְשָׁעָה שֶׁזָּב לַח הוּא: וּבְשַׂר הַמֵּת. מְטַמֵּא יָבֵשׁ, דִּכְתִיב (בְּמִדְבַּר יב) אוֹ בְּעֶצֶם אָדָם, אָדָם דּוּמְיָא דְּעֶצֶם, מָה עֶצֶם יָבֵשׁ אַף בָּשָׂר יָבֵשׁ: וּמְטַמְּאִים יְבֵשִׁים. דִּכְתִיב (שָׁם) דָּם יִהְיֶה זֹבָהּ, בַּהֲוָיָתוֹ יְהֵא, וְאַף עַל פִּי שֶׁנִּתְיַבֵּשׁ (ב): וְהַנִּיעַ. רֹק רַךְ שֶׁיּוֹצֵא עַל יְדֵי נִעְנוּעַ. וַאֲנִי שָׁמַעְתִּי, לֵחָה הַיּוֹרֶדֶת מִן הַחֹטֶם. וּמִמַּעְיְנוֹת הַזָּב הֵן וַהֲווּ אַב הַטֻּמְאָה: וְהַשֶּׁרֶץ. דִּכְתִיב (וַיִּקְרָא יא) הַנֹּגֵעַ בָּהֶם בְּמֹתָם, כְּעֵין מוֹתָם, דְּהַיְנוּ כְּשֶׁהֵם לַחִים כְּמוֹ שֶׁהָיוּ בְּעֵת מוֹתָם. אֲבָל כָּל זְמַן שֶׁהַשִּׁדְרָה קַיֶּמֶת וְהָעֲצָמוֹת מְחֻבָּרוֹת בַּשִּׁדְרָה, הוֹאִיל וְצוּרָתוֹ נִכֶּרֶת, כְּלַח הוּא נֶחֱשָׁב: וְהַנְּבֵלָה. דִּכְתִיב (שָׁם) כִּי יָמוּת, כְּעֵין מִיתָה: וְהַשִּׁכְבַת זֶרַע. שֶׁצָּרִיךְ שֶׁיְּהֵא רָאוּי לְהַזְרִיעַ: מְטַמְּאִין לַחִין וְאֵין מְטַמְּאִין יְבֵשִׁין. הַזּוֹב מְטַמֵּא לַח וְלֹא יָבֵשׁ, דִּכְתִיב (שָׁם טו) רָר בְּשָׂרוֹ אֶת זוֹבוֹ, כְּמִין רִיר שֶׁהוּא לַח. כְּמוֹ וַיּוֹרֶד רִירוֹ עַל זְקָנוֹ (שְׁמוּאֵל א כא): וְהַנִּיעַ וְהָרֹק. דִּכְתִיב (וַיִּקְרָא טו) וְכִי יָרֹק, וּבָעֵת שֶׁיָּרֹק לַח הוּא: וְכַמָּה הִיא שְׁרִיָּתָן. דְּנֵימָא אִי הָדְרָא בְּכִי הַאי שִׁעוּרָא לַחִין הֵן: בְּפוֹשְׁרִים מֵעֵת לְעֵת. אֲבָל אִי צְרִיכֵי טְפֵי מַיִם חַמִּים, אוֹ יוֹתֵר מִזְּמַן מֵעֵת לְעֵת, יְבֵשִׁין הֵן: רַבִּי יוֹסֵי אוֹמֵר בְּשַׂר הַמֵּת יָבֵשׁ, טָהוֹר. מִלְּטַמֵּא בְּכַזַּיִת. אֲבָל מְטַמֵּא טֻמְאַת רָקָב כִּמְלֹא תַּרְוָד. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי:

פרק ז - משנה ב

מְטַמֵּא לְמַפְרֵעַ. כָּל טָהֳרוֹת שֶׁנִּמְצְאוּ בַּמָּבוֹי. דְּמָבוֹי רְשׁוּת הַיָּחִיד הוּא לְעִנְיַן טֻמְאָה, וּסְפֵקוֹ טָמֵא: אוֹ עַד שְׁעַת כִּבּוּד. מִפְּנֵי שֶׁחֶזְקַת בְּנֵי יִשְׂרָאֵל בּוֹדְקִין מְבוֹאוֹתֵיהֶן בִּשְׁעַת כִּבּוּדֵיהֶן (ג), וְאִם הָיָה שָׁם שֶׁרֶץ הָיָה רוֹאֶה אוֹתוֹ: וְכֵן כֶּתֶם שֶׁנִּמְצָא בְחָלוּק מְטַמֵּא לְמַפְרֵעַ. כָּל טָהֳרוֹת שֶׁעָשְׂתָה הָאִשָּׁה מִיּוֹם כִּבּוּס. אֲבָל אוֹתָן שֶׁל קֹדֶם כִּבּוּס, לֹא, מִפְּנֵי שֶׁחֶזְקַת בְּנוֹת יִשְׂרָאֵל בּוֹדְקוֹת חֲלוּקֵיהֶם בִּשְׁעַת כִּבּוּסֵיהֶן (ד): הַיָּבֵשׁ מְטַמֵּא לְמַפְרֵעַ. עַד שְׁעַת כִּבּוּד. דְּאִיכָּא לְמֵימַר לְאַחַר כִּבּוּד מִיָּד נָפַל, אֲבָל הַלַּח אֵינוֹ מְטַמֵּא לְמַפְרֵעַ, אֶלָּא עַד שָׁעָה שֶׁנּוּכַל לוֹמַר אִם נָפַל בְּאוֹתוֹ יוֹם עֲדַיִן יָכוֹל לִהְיוֹת עַכְשָׁיו לַח. אֲבָל עַד שְׁעַת הַכִּבּוּד לִזְמַן מְרֻבֶּה, לֹא, דְּהָא לַח הוּא, וְאִי מֵהַהִיא שַׁעְתָּהּ נָפַל, הֲוָה מִתְעֲבֵיד יָבֵשׁ. וְדַוְקָא בְּשֶׁרֶץ הוּא דְּפָלֵיג רַבִּי שִׁמְעוֹן בֵּין לַח לְיָבֵשׁ. אֲבָל בְּכֶתֶם, אֲפִלּוּ לַח מְטַמֵּא לְמַפְרֵעַ עַד שְׁעַת כִּבּוּס, דְּאֵימַר יָבֵשׁ הָיָה וּמַיָּא הוּא דְּנָפַל עֲלֵיהּ. אֲבָל בְּשֶׁרֶץ לֵיכָּא לְמֵימַר הָכִי, דְּאִי אִיתָא דְּמַיִם נָפוּל עֲלֵיהּ, אִמַּרְטוֹטֵי אִמַּרְטוּט (ה). וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן:

פרק ז - משנה ג

הַבָּאִים מֵרְקָם. שֶׁהֵן נָכְרִים וְדָמָן טָהוֹר (ז). וְאַף עַל גַּב דְּרַבָּנָן גָּזְרוּ עַל הַנָּכְרִים שֶׁיִּהְיוּ כַּזָּבִים לְכָל דִּבְרֵיהֶם, עַל כִּתְמֵיהֶם לֹא גָּזְרוּ: מֵרֶקֶם. בֵּין קָדֵשׁ וּבֵין שׁוּר, מְתַרְגְּמִינַן בֵּין רֶקֶם וּבֵין חַגְּרָא: שֶׁהֵן גֵּרִים. וְדָמָן טָמֵא: וְטוֹעִים. כְּלוֹמַר, לֹא צְנִיעֵי כֻּלֵּי הַאי, וְאֵין מַצְנִיעִים אֶת כִּתְמֵיהֶן, הִלְכָּךְ חָיְישִׁינַן שֶׁמָּא כְּתָמִים שֶׁל אִשָּׁה נִדָּה הֵן: הַבָּאִים מִבֵּין הַגּוֹיִם טְהוֹרִים. דְּלֹא גָּזְרוּ רַבָּנָן עַל כִּתְמֵיהֶן, כֵּיוָן דְּדָמָן וַדַּאי טָהוֹר מִן הַתּוֹרָה: מִבֵּין יִשְׂרָאֵל וּמִבֵּין הַכּוּתִים רַבִּי מֵאִיר מְטַמֵּא. מַתְנִיתִין חַסּוֹרֵי מִחַסְּרָא וְהָכִי קָתָנֵי, מִבֵּין יִשְׂרָאֵל וּמִבֵּין הַכּוּתִים, טְמֵאִים, דְּכוּתִים גֵּרֵי אֱמֶת הֵן וְדָמָן טָמֵא. כְּתָמִים הַנִּמְצָאִים בְּעָרֵי יִשְׂרָאֵל דְּהַיְנוּ בְּמָקוֹם גָּלוּי, טְהוֹרִים, דְּלֹא נֶחְשְׁדוּ עַל כִּתְמֵיהֶן וְאַצְנוֹעֵי מַצְנְעֵי לְהוּ. הַנִּמְצָאִין בְּעָרֵי כּוּתִים, רַבִּי מֵאִיר מְטַמֵּא, דְּכוּתִים נֶחְשְׁדוּ עַל כִּתְמֵיהֶן. וַחֲכָמִים מְטַהֲרִים, שֶׁלֹּא נֶחְשְׁדוּ עַל כִּתְמֵיהֶן וּמַצְנְעֵי לְהוּ, וְתָלִינַן לְהַאי כֶּתֶם בְּדַם חַיָּה אוֹ בְּהֵמָה, דְּאִלּוּ דַּם אִשָּׁה נִדָּה, לֹא הָיָה נִמְצָא בְּמָקוֹם גָּלוּי, דְּאִינְהוּ נַמִּי מַצְנְעוּ לְהוּ כְּיִשְׂרָאֵל. וּכְבָר נִפְסְקָה הַהֲלָכָה דְּבַזְמַן הַזֶּה גָּזְרוּ עַל כּוּתִים לִהְיוֹת כַּנָּכְרִים לְכָל דִּבְרֵיהֶם, וְכִתְמֵיהֶן טְהוֹרִים (ט) כְּכִתְמֵי הַנָּכְרִים:

פרק ז - משנה ד

בְּכָל מָקוֹם. בִּמְקוֹם יִשְׂרָאֵל: טְהוֹרִים. שֶׁחֶזְקָתָן אֵינָן מִדַּם נִדָּה, דְּאַצְנוֹעֵי מַצְנְעֵי לְהוּ: בֵּית הַטֻּמְאוֹת. חֶדֶר שֶׁהַנָּשִׁים מִשְׁתַּמְּשׁוֹת בּוֹ בִּימֵי נִדּוּתָן: קוֹבְרִין שָׁם. לְפִי שָׁעָה (י), כְּדֵי שֶׁיְּסַלְּקוּ אוֹתָן לְאַחַר זְמַן. וּלְפִי שֶׁאֵין אָנוּ יוֹדְעִין אִם הֵסִירוּ אוֹתָן אִם לָאו, מְטַמְּאִים בְּאֹהֶל:

פרק ז - משנה ה

נֶאֱמָנִים לוֹמַר קָבַרְנוּ. דְּטֻמְאַת מֵת מִידִי דְּאוֹרַיְתָא הִיא. וּכְגוֹן שֶׁכֹּהֵן כּוּתִי עוֹמֵד שָׁם וּתְרוּמָה בְּיָדוֹ וְאוֹכְלָהּ (יא), דְּלֵיכָּא לְמֵיחַשׁ שֶׁמָּא תְּרוּמָה טְמֵאָה הִיא. וּבְלָאו הָכִי אֵין נֶאֱמָנִין, דְּלֵית לְהוּ לִפְנֵי עִוֵּר לֹא תִתֵּן מִכְשׁוֹל, וְלֹא אִכְפַּת לָהֶם אִם אָנוּ חוֹטְאִין עַל פִּיהֶם: אִם בִּכְּרָה. וְהוּא שֶׁרְאִינוּהוּ עוֹבֵד וְגוֹזֵז בּוֹ (יב). דְּאִי לָאו דְּיָדַע דְּלָאו בְּכוֹר הוּא, לֹא עָבַר אַדְּאוֹרַיְתָא, דִּכְתִיב (דְּבָרִים טו) לֹא תַעֲבֹד בִּבְכֹר שׁוֹרֶךָ וְלֹא תָגֹז בְּכוֹר צֹאנֶךָ: נֶאֱמָנִים עַל צִיּוּן קְבָרוֹת. לְצַיֵּן מְקוֹם הַקֶּבֶר. וְסוֹמְכִין עֲלֵיהֶן (יג). וְאַף עַל גַּב דְּמִדְּרַבָּנָן הוּא, כֵּיוָן דִּכְתִיבָא, זְהִירִין בֵּיהּ. דִּכְתִיב (יְחֶזְקֵאל לט) וְרָאָה עֶצֶם אָדָם וּבָנָה אֶצְלוֹ צִיּוּן: לֹא עַל הַסְּכָכוֹת. אִילָן הַמֵּסֵךְ עַל הָאָרֶץ וַעֲנָפָיו מֻבְדָּלִים וְיָדוּעַ וַדַּאי שֶׁיֵּשׁ קֶבֶר תַּחַת אַחַת הָאֳהָלוֹת וְאֵין יָדוּעַ אֵיזֶה הוּא, וּבָא כּוּתִי וְהֵעִיד עַל אֶחָד שֶׁזֶּה טָהוֹר, הִשְׁאִילָה חֲלוּקָהּ. וְאַחַר כָּךְ לְבָשַׁתָּהּ הִיא וּמָצְאָה עָלָיו כֶּתֶם: לְנָכְרִית אוֹ לְנִדָּה. לְנָכְרִית שֶׁהִיא רוֹאָה דָּם, שֶׁהִיא גְּדוֹלָה וְיָדַעְנוּ בָּהּ שֶׁכְּבָר רָאֲתָה (ה), דּוּמְיָא דְּנִדָּה: הֲרֵי זוֹ תּוֹלָה בָהּ. וּטְהוֹרָה. דְּכֵיוָן דְּנִדָּה הִיא, לֹא מְקַלְקְלָה לַהּ מִידִי. וְנָכְרִית נַמִּי טְמֵאָה וְתָלְיָא בָּהּ. אֲבָל יִשְׂרְאֵלִית טְהוֹרָה שֶׁשָּׁאֲלָה חָלוּק מֵחֲבֶרְתָּהּ טְהוֹרָה, וְאַחַר כָּךְ לְבָשַׁתּוּ חֲבֶרְתָּהּ, שְׁתֵּיהֶן טְמֵאוֹת, כִּדְקָתָנֵי שָׁלֹשׁ נָשִׁים וְכוּ', דְּכֵיוָן דְּאִי תָּלְיָא בָּהּ, מְקַלְקְלָא לַהּ, מַאי חָזֵית דִּמְקַלְקֶלֶת לְהַךְ טְפֵי מֵהַךְ: שֶׁל אֶבֶן. דְּאֵינוֹ מְקַבֵּל טֻמְאָה. וְסַפְסָל דְּרֵישָׁא [אַיְרֵי] בִּכְלִי עֵץ דְּרָאוּי לִטָּמֵא בְּמוֹשַׁב הַזָּב וְהַנִּדָּה: אֵינוֹ מְקַבֵּל כְּתָמִים. כְּלוֹמַר לֹא גָּזְרוּ עַל כְּתָמִים הַנִּמְצָאִים בּוֹ. וְטַעֲמָא דְּרַבִּי נֶחֶמְיָה, דִּכְתִיב (יְשַׁעְיָה ג) וְנִקָּתָה לָאָרֶץ תֵּשֵׁב, כֵּיוָן שֶׁיָּשְׁבָה לָאָרֶץ שֶׁאֵינָהּ מְקַבֶּלֶת טֻמְאָה, נִקָּתָה. וְהוּא הַדִּין לְכָל דָּבָר שֶׁאֵינוֹ מְקַבֵּל טֻמְאָה. וְכֵן אִם יָשְׁבָה עַל גַּבֵּי כְּלִי חֶרֶס, שֶׁאֵין מְקַבֵּל טֻמְאָה מִגַּבּוֹ, אוֹ עַל גַּבֵּי מַטְלָנִיּוֹת שֶׁל בֶּגֶד שֶׁאֵין בָּהֶן שָׁלֹשׁ עַל שָׁלֹשׁ שֶׁאֵינָן מְקַבְּלִין טֻמְאָה, בְּכָל אֵלּוּ לֹא גָּזְרוּ עַל כְּתָמִים הַנִּמְצָאִים בָּהֶם. וַהֲלָכָה כְּרַבִּי נֶחֶמְיָה:

פרק ט - משנה ד

וְתוֹלוֹת זוֹ בָזוֹ. שֶׁאִם הָיְתָה הָאַחַת מְעֻבֶּרֶת (י), תּוֹלָה הַדָּם בְּשֶׁאֵינָהּ מְעֻבֶּרֶת, וְהִיא טְהוֹרָה: וְאִם לֹא הָיוּ רְאוּיוֹת. כְּגוֹן שֶׁהָיוּ כֻּלָּן מְעֻבָּרוֹת אוֹ כֻּלָּן מֵנִיקוֹת, כֻּלָּן טְמֵאוֹת כְּאִלּוּ רְאוּיוֹת לִרְאוֹת, דְּמִכָּל מָקוֹם דָּם מִבֵּינַיְהוּ נְפַק:

פרק ט - משנה ה

תַּחַת הַפְּנִימִית. אוֹתָהּ שֶׁל צַד הַקִּיר קְרוּיָה פְּנִימִית, וְחִיצוֹנָה קְרוּיָה אוֹתָהּ שֶׁל צַד דֶּרֶךְ עֲלִיַּת הַמִּטָּה. וְהָא דִּמְפַלֵּיג תַּנָּא הָכָא בֵּין כְּשֶׁנִּמְצָא תַּחַת הַפְּנִימִית לְנִמְצָא תַּחַת הַחִיצוֹנָה, וּלְעֵיל קָתָנֵי וְנִמְצָא דָּם תַּחַת אַחַת מֵהֶן כֻּלָּן טְמֵאוֹת, וְלֹא מְפַלֵּיג בֵּין נִמְצָא תַּחַת פְּנִימִית לְנִמְצָא תַּחַת חִיצוֹנָה, מְפָרֵשׁ בַּגְּמָרָא דִּלְעֵיל אַיְרֵי כְּשֶׁהֵן תְּכוּפוֹת וּדְבוּקוֹת יַחַד, הִלְכָּךְ אִיכָּא לְסַפּוֹקֵי בַּחִיצוֹנָה אֲפִלּוּ כְּשֶׁנִּמְצָא הַדָּם תַּחַת הַפְּנִימִית: אֵימָתַי, אָמְרוּ נִמְצָא תַּחַת הַחִיצוֹנָה פְּנִימִית טְהוֹרָה: בִּזְמַן שֶׁעָבְרוּ דֶרֶךְ מַרְגְּלוֹת הַמִּטָּה. דְּלֹא עָבְרָה פְּנִימִית עַל מְקוֹם מְצִיאַת הַדָּם: אֲבָל עָבְרוּ שְׁלָשְׁתָּן דֶּרֶךְ עָלֶיהָ. שֶׁהֵן דּוֹרְסוֹת וְעוֹבְרוֹת עַל הַחִיצוֹנָה: כֻּלָּן טְמֵאוֹת. וַאֲפִלּוּ פְּנִימִית, דְּשֶׁמָּא כְּשֶׁעָבְרָה פְּנִימִית דֶּרֶךְ שָׁם נָפַל מִמֶּנָּה: לְגַּל טָמֵא. שֶׁהָיָה שָׁם כַּזַּיִת מִן הַמֵּת: עַד שֶׁמַּגִּיעַ לְסֶלַע אוֹ לִבְתוּלָה. קַרְקַע בְּתוּלָה. וְאִם לֹא מָצָא, הַגַּל טָהוֹר, דְּאֵמוּר בָּא עוֹרֵב וְנָטַל אֶת הַטֻּמְאָה. אֲבָל הָכָא כְּשֶׁבָּדְקוּ שְׁלָשְׁתָּן וּמָצְאוּ טְהוֹרוֹת, מוֹדוּ רַבָּנָן לְרַבִּי מֵאִיר דְּכֻלָּן טְמֵאוֹת, דְּהַאי דָּם מֵהֵיכָא אֲתָא, עַל כָּרְחָךְ מֵחֲדָא מִנַּיְהוּ הוּא. וַהֲלָכָה כַּחֲכָמִים:

פרק ט - משנה ו

שִׁבְעָה סַמְמָנִים מַעֲבִירִין עַל הַכֶּתֶם. שֶׁאֵין הַדָּם בָּטֵל עַד שֶׁיַּעֲבִיר כֻּלָּן עָלָיו: רֹק תָּפֵל וּמֵי גְרִיסִין. מְפָרֵשׁ לְקַמָּן בְּמַתְנִיתִין: נֶתֶר. אלו"ם בְּלַעַ"ז. וּמִין קַרְקַע הוּא וּמַזְהִיר. וּבַעֲרָבִי קוֹרִין לוֹ שי"ב: בּוֹרִית. מִין צֶמַח הוּא שֶׁמְּנַקֶּה וּמְטַהֵר: קְמוּנְיָא. עֵשֶׂב שֶׁמְּיַבְּשִׁים אוֹתוֹ וְטוֹחֲנִים אוֹתוֹ וּמְנַקִּין בַּעֲפָרוֹ אֶת הַיָּדַיִם לְהַעֲבִיר אֶת הַזֻּהֲמָא. וּבִלְשׁוֹן גְּמָרָא קָרוּי שְׁלוֹף דּוֹץ: אֶשְׁלָג. לֹא אִתְפָּרַשׁ לִי: הִטְבִּילוֹ. לַבֶּגֶד שֶׁיֵּשׁ בּוֹ כֶּתֶם קֹדֶם שֶׁהֶעֱבִיר עָלָיו הַסַּמְמָנִים הַלָּלוּ: הֲרֵי זֶה צֶבַע. דְּאִי הֲוָה דָּם הֲוָה עָבַר: שֶׁדֵּהָה. נִשְׁתַּנָּה (טו) מֵאַדְמִימוּתוֹ:

פרק ט - משנה ז

שֶׁלֹּא טָעַם כְּלוּם מִבָּעֶרֶב. כָּל אוֹתָהּ הַלַּיְלָה, וְהַיְנוּ תָּפֵל בְּלֹא טַעַם. וְהוּא שֶׁיָּשַׁן (יח) מֵחֲצוֹת אַחֲרוֹן שֶׁל לַיְלָה עַד הַבֹּקֶר. וְעַד שָׁלֹשׁ שָׁעוֹת בַּיּוֹם נִקְרָא רֹק תָּפֵל לְעִנְיָן זֶה. וְאִם הִשְׁכִּים בַּבֹּקֶר וְשָׁנָה פִּרְקוֹ, לֹא הָוֵי רֹק תָּפֵל (יט): לְעִסַּת גְּרִיסִין שֶׁל פּוֹל חֲלוּקַת נָפֶשׁ. שֶׁנֶּחֱלַק הָאֹכֶל מִן הַקְּלִפָּה. לוֹעֲסָה לִגְרִיסַת הַפּוֹל וּמַעֲבִירוֹ עַל הַכֶּתֶם: שֶׁהֶחֱמִיצוּ. שֶׁהִסְרִיחוּ. וְכַמָּה חִמּוּצָן, כָּל שֶׁשָּׁהוּ שְׁלֹשָׁה יָמִים: וְצָרִיךְ לְכַסְכֵּס. לְשַׁפְשֵׁף שָׁלֹשׁ פְּעָמִים לְכָל סַם וְסַם, כְּדֶרֶךְ כִּבּוּס בְּגָדִים שֶׁכּוֹפֵל צַד עִם צִדּוֹ וּמְשַׁפְשֵׁף: כְּסִדְרָן. שֶׁשְּׁנוּיִים כָּאן: לֹא עָשָׂה וְלֹא כְלוּם. לֹא לִבְדֹּק וְלֹא לְבַטֵּל (כ):

תִּינוֹקֶת. נוֹתְנִין לָהּ אַרְבָּעָה לֵילוֹת. וַאֲפִלּוּ רוֹאָה, אָמְרִינַן דַּם בְּתוּלִים הוּא וְטָהוֹר: עַד שֶׁתִּחְיֶה הַמַּכָּה. שֶׁתִּתְרַפֵּא הַמַּכָּה שֶׁל בְּעִילָה. וְכָל זְמַן שֶׁהַדָּם שׁוֹתֵת מִמֶּנָּה כְּשֶׁהִיא עוֹמֶדֶת, וּכְשֶׁהִיא יוֹשֶׁבֶת אֵינוֹ שׁוֹתֵת, אוֹ שֶׁשּׁוֹתֵת כְּשֶׁהִיא יוֹשֶׁבֶת עַל גַּבֵּי דָּבָר קָשֶׁה, וּכְשֶׁיּוֹשֶׁבֶת עַל גַּבֵּי דָּבָר רַךְ כְּגוֹן עַל גַּבֵּי כָּרִים וּכְסָתוֹת אֵינוֹ שׁוֹתֵת, בְּיָדוּעַ שֶׁלֹּא חָיְתָה הַמַּכָּה, וּמֵחֲמַת הַמַּכָּה שֶׁל בְּתוּלִים בָּא הַדָּם. אֲבָל אִם הַדָּם שׁוֹתֵת בֵּין כְּשֶׁהִיא עוֹמֶדֶת בֵּין כְּשֶׁהִיא יוֹשֶׁבֶת וּבֵין כְּשֶׁיּוֹשֶׁבֶת עַל דָּבָר רַךְ וּבֵין כְּשֶׁיּוֹשֶׁבֶת עַל דָּבָר קָשֶׁה, אֵין זֶה דַּם מַכָּה אֶלָּא דַּם נִדָּה: הִגִּיעַ זְמַנָּהּ לִרְאוֹת. וְלֹא רָאֲתָה, וְנִשֵּׂאת: עַד מוֹצָאֵי שַׁבָּת אַרְבָּעָה לֵילוֹת. שֶׁהֲרֵי הִיא נִשֵּׂאת בִּרְבִיעִי. דִּבְתוּלָה נִשֵּׂאת לְיוֹם רְבִיעִי וְנִבְעֶלֶת לֵיל חֲמִישִׁי: בְּעִילַת מִצְוָה. בְּעִילָה רִאשׁוֹנָה: כָּל הַלַּיְלָה כֻּלָּהּ. וַאֲפִלּוּ בּוֹעֵל בְּעִילוֹת הַרְבֵּה. וְכָל הַדְּבָרִים הַלָּלוּ, בַּדּוֹרוֹת הָרִאשׁוֹנִים בִּלְבַד, אֲבָל אַחַר שֶׁפָּשְׁטָה הַחֻמְרָא שֶׁהֶחְמִירוּ בְּנוֹת יִשְׂרָאֵל עַל עַצְמָן שֶׁאֲפִלּוּ רוֹאוֹת דָּם טִפָּה כְּחַרְדָּל יוֹשְׁבוֹת עָלֶיהָ שִׁבְעָה נְקִיִּים, כָּל הַנּוֹשֵׂא בְּתוּלָה אֲפִלּוּ קְטַנָּה שֶׁלֹּא הִגִּיעַ זְמַנָּהּ לִרְאוֹת וְלֹא רָאֲתָה מֵעוֹלָם, לְאַחַר שֶׁבָּעַל בְּעִילַת מִצְוָה (ד) אֵין מֻתָּר לוֹ לָבֹא עָלֶיהָ עַד שֶׁתִּסְפֹּר שִׁבְעָה נְקִיִּים (ה) אַחַר שֶׁיִּפְסֹק הַדָּם, כְּדִין שְׁאָר נָשִׁים שֶׁרוֹאוֹת דָּם:

לֹא הִפְרִישָׁה. לֹא בָּדְקָה עַצְמָהּ לְהַפְרִישׁ בְּטָהֳרָה: וּלְאַחַר יָמִים. לְאַחַר שֶׁטָּבְלָה בְּלֵיל שְׁמִינִי: הֲרֵי זוֹ בְחֶזְקַת טָהֳרָה. בַּיָּמִים שֶׁבֵּין טְבִילָתָהּ לִמְצִיאַת טֻמְאָה. וְטָהֳרוֹתֶיהָ שֶׁנָּגְעָה בָּהֶן טְהוֹרוֹת, דְּאֵמוּר הַשְׁתָּא הוּא דַּחֲזַאת: וּמְטַמְּאָה מֵעֵת לְעֵת. אָרֵישָׁא קָאֵי, דְּקָתָנֵי וּלְאַחַר יָמִים בָּדְקָה וּמָצְאָה טְמֵאָה הֲרֵי הִיא בְּחֶזְקַת טָהֳרָה, אֲבָל מְטַמְּאָה מֵעֵת לְעֵת שֶׁל רְאִיָּה זוֹ אוֹ מִפְּקִידָה לִפְקִידָה: אֲפִלּוּ בַשְּׁנִיָּה לְנִדָּתָהּ. בָּדְקָה שַׁחֲרִית וּמָצְאָה טְהוֹרָה, וּבֵין הַשְּׁמָשׁוֹת לֹא הִפְרִישָׁה, וְלִבְסוֹף שִׁבְעָה טָבְלָה, וּלְאַחַר יָמִים בָּדְקָה וּמָצְאָה טְמֵאָה, הֲרֵי זוֹ עַד עַכְשָׁיו בְּחֶזְקַת טָהֳרָה. וְשָׁלֹשׁ מַחֲלֹקוֹת בַּדָּבָר, לְתַנָּא קַמָּא בְּדִיקַת שַׁחֲרִית דִּשְׁבִיעִי הוּא דִּמְטַהַרְתָּהּ, אֲבָל שֵׁנִי לְנִדָּתָהּ, לֹא. וּלְרַבִּי יְהוּדָה אֲפִלּוּ בְּדִיקַת שַׁחֲרִית דִּשְׁבִיעִי לֹא מְטַהֲרָהּ עַד שֶׁתִּפְרֹשׁ בֵּין הַשְּׁמָשׁוֹת. וּלְרַבָּנָן אֲפִלּוּ בַּשֵּׁנִי, דְּכֵיוָן דְּפָסַק פָּסַק. וְדַוְקָא בַּשֵּׁנִי הוּא דִּמְטַהֲרִי רַבָּנָן, אֲבָל בָּרִאשׁוֹן מָצְאָה טָהוֹר וְשׁוּב לֹא בָּדְקָה, וְטָבְלָה בְּלֵיל שְׁמִינִי, וְעָשְׂתָה טָהֳרוֹת, וְאַחַר כָּךְ מָצְאָה טָמֵא, לֹא מְטַהֲרִי רַבָּנָן, לְפִי שֶׁבְּיוֹם רִאשׁוֹן הֻחְזְקָה מַעְיַן פָּתוּחַ. וּמִשּׁוּם הָכִי קָתָנֵי וַאֲפִלּוּ בַּשְּׁנִיָּה לְנִדָּתָהּ דַּוְקָא, אֲבָל רִאשׁוֹן לֹא. וַהֲלָכָה כַּחֲכָמִים:

פרק י - משנה ג

הַזָּב וְהַזָּבָה. שֶׁצְּרִיכִין לִסְפֹּר שִׁבְעָה נְקִיִּים, וּפָסְקוּ מֵרְאִיָּתָן וְהִתְחִילוּ לִסְפֹּר, וּבָדְקוּ יוֹם רִאשׁוֹן וּשְׁבִיעִי כוּ': אֵין לָהֶם אֶלָּא יוֹם רִאשׁוֹן וּשְׁבִיעִי בִלְבָד. וּצְרִיכִין לִסְפֹּר עוֹד חֲמִשָּׁה, יָמִים נְקִיִּים, כְּדֵי לְהַשְׁלִים לְשִׁבְעָה: רַבִּי עֲקִיבָא אוֹמֵר אֵין לָהֶם אֶלָּא שְׁבִיעִי. שֶׁמָּא רָאוּ בֵּינְתַיִם וְסָתְרוּ סְפִירָתָן הָרִאשׁוֹנָה. וַהֲלָכָה כְּרַבִּי אֱלִיעֶזֶר:

פרק י - משנה ד

מְטַמְּאִין בְּמַשָּׂא. בַּגְּמָרָא פָּרֵיךְ, אַטּוּ כָּל מֵת מִי לֹא מְטַמֵּא בְּמַשָּׂא. אֶלָּא מַאי בְּמַשָּׂא, בְּאֶבֶן מְסָמָא (ז). כְּלוֹמַר אֶבֶן גְּדוֹלָה וּכְבֵדָה שֶׁאֵינָהּ מִטַּלְטֶלֶת. וְאִם הִיא מֻנַּחַת עַל גַּבֵּי כֵּלִים וּמֵת נָתוּן עָלֶיהָ אֵין זֶה מַשָּׂא, הוֹאִיל וְאֵינָהּ רְאוּיָה לְהִטַּלְטֵל עִמּוֹ, וּשְׁאָר מֵתִים לֹא מְטַמּוּ, וְהָנֵי מְטַמּוּ מִדְּרַבָּנָן, הוֹאִיל וּמֵחַיִּים מְטַמּוּ בְּאֶבֶן מְסָמָא מִדְּאוֹרַיְתָא שֶׁהֲרֵי מְטַמְּאִים מוֹשָׁב כָּל שֶׁתַּחְתֵּיהֶן, גַּזּוּר בְּהוּ רַבָּנָן לְאַחַר מִיתָה שֶׁמָּא יִתְעַלְּפוּ וְדוֹמֶה לְמֵת, וְאִי אָמַרַת זָב מֵת אֵינוֹ מְטַמֵּא, בְּזָב חַי שֶׁנִּתְעַלֵּף אָתֵי לְטַהוֹרֵי: עַד שֶׁיִּמֹּק הַבָּשָׂר. דִּמֵהַשְׁתָּא הָוֵי מֵת וַדַּאי: נָכְרִי שֶׁמֵּת. אַף עַל גַּב דְּמֵחַיִּים הֲרֵי הוּא כַּזָּב לְכָל דְּבָרָיו וּמְטַמֵּא בְּאֶבֶן מְסָמָא, מִשֶּׁמֵּת, טָהוֹר מִלְּטַמֵּא בְּאֶבֶן מְסָמָא, הוֹאִיל וְטֻמְאַת זִיבָה שֶׁעָלָיו מֵחַיִּים אֵינָהּ אֶלָּא מִדִּבְרֵי סוֹפְרִים: כָּל הַנָּשִׁים. מַחְזְקִינַן לְהוּ כְּנִדּוֹת, וּמַטְבִּילִין כֵּלִים שֶׁהָיוּ עֲלֵיהֶן סָמוּךְ לִיצִיאַת נְשָׁמָה אַף עַל פִּי שֶׁלֹּא נָגְעוּ בָּהֶן לְאַחַר מִיתָה. לְפִי שֶׁבָּרִאשׁוֹנָה הָיוּ מַטְבִּילִין כֵּלִים עַל גַּבֵּי נִדּוֹת מֵתוֹת, וְהָיוּ נִדּוֹת חַיּוֹת מִתְבַּיְּשׁוֹת, שֶׁאֲפִלּוּ בְּמִיתָתָן הֵן מְשֻׁנּוֹת מִשְּׁאָר נָשִׁים, הִתְקִינוּ שֶׁיִּהְיוּ מַטְבִּילִין עַל גַּבֵּי כָּל הַנָּשִׁים, מִפְּנֵי כְּבוֹדָן שֶׁל נִדּוֹת חַיּוֹת: וּבֵית הִלֵּל אוֹמְרִים אֵין נִדָּה אֶלָּא שֶׁמֵּתָה נִדָּה. דְּלֵית לְהוּ לְבֵית הִלֵּל הַאי תַּקַּנְתָּא:

פרק י - משנה ה

מְטַמְּאָה מִשּׁוּם כֶּתֶם. שֶׁל נִדָּה, בְּמַשֶּׁהוּ (ח), דְּאִי מִשּׁוּם טֻמְאַת מֵת, לֹא מְטַמְּאָה אֶלָּא בִּרְבִיעִית: וּמְטַמְּאָה בְּאֹהֶל. אִם כָּל הָרְבִיעִית יַחַד: שֶׁמְּטַמָּא מִשּׁוּם כָּתֶם. בְּמַשֶּׁהוּ. שֶׁנֶּעֱקַר מֵחַיִּים. וְאֵין הֲלָכָה לֹא כְּרַבִּי יְהוּדָה וְלֹא כְּרַבִּי יוֹסֵי. הִלְכָּךְ, לֹא שְׁנָא יוֹשֶׁבֶת עַל מַשְׁבֵּר וְלֹא שְׁנָא שְׁאָר מֵתוֹת, כֻּלְּהוּ מְטַמְּאוֹת מִשּׁוּם כֶּתֶם, וּמְטַמְּאוֹת בְּאֹהֶל (ט):

פרק י - משנה ו

מְעָרָה מַיִם לַפֶּסַח. מִכְּלִי אֶל כְּלִי לִרְחֹץ בָּם אֶת בְּשַׂר הַפֶּסַח. אֲבָל בַּמַּיִם אֵינָהּ נוֹגַעַת (י), שֶׁהִיא טְבוּלַת יוֹם, שֶׁטָּבְלָה לְסוֹף שְׁבוּעַיִם וְאֵין לָהּ הַעֲרֵב שֶׁמֶשׁ עַד יוֹם שְׁמוֹנִים (יא) שֶׁהוּא תֵּכֶף לַהֲבָאַת כַּפָּרָתָהּ, וְהַכָּתוּב עֲשָׂאָהּ טְבוּלַת יוֹם, דִּכְתִיב (וַיִּקְרָא יב) בְּכָל קֹדֶשׁ לֹא תִגָּע, וְקַיְמָא לָן בְּכָל קֹדֶשׁ, לְרַבּוֹת אֶת הַתְּרוּמָה, וּטְבוּל יוֹם, שֵׁנִי הוּא, וּכְשֶׁהִיא מְעָרָה, אֵינָהּ נוֹגַעַת בַּמַּיִם אֶלָּא בַּכְּלִי, וְשֵׁנִי אֵינוֹ מְטַמֵּא כְּלִי. אֲבָל בַּמַּיִם לֹא תִגַּע, הוֹאִיל וְהֵם עֲשׂוּיִין לִרְחֹץ בָּהֶם אֶת הַפֶּסַח, דְּחֻלִּין שֶׁנַּעֲשׂוּ עַל טָהֳרַת הַקֹּדֶשׁ כְּקֹדֶשׁ דָּמֵי: חָזְרוּ לוֹמַר הֲרֵי הִיא כְמַגָּע טְמֵא מֵת. טְבוּל יוֹם דִּינוֹ כְּנוֹגֵעַ בִּטְמֵא מֵת, שֶׁהוּא רִאשׁוֹן, דַּוְקָא לַקָּדָשִׁים, אֲבָל לְחֻלִּין, לֹא הָוֵי טְבוּל יוֹם כְּנוֹגֵעַ בִּטְמֵא מֵת לִהְיוֹת רִאשׁוֹן, אֶלָּא שֵׁנִי, הִלְכָּךְ יְכוֹלָה הִיא לִגַּע אֲפִלּוּ בַּמַּיִם, שֶׁהַמַּיִם חֻלִּין הֵן, וּטְבוּל יוֹם, שֵׁנִי הוּא, וְאֵין שֵׁנִי עוֹשֶׂה שְׁלִישִׁי בְּחֻלִּין. וְאַף עַל פִּי שֶׁהַמַּיִם הַלָּלוּ נַעֲשִׂים עַל טָהֳרַת הַקֹּדֶשׁ שֶׁהֲרֵי לִרְחִיצַת פֶּסַח עֲשׂוּיוֹת, לֹא מַהֲנֵי שֵׁנִי בְּהוּ, דְּחֻלִּין שֶׁנַּעֲשׂוּ עַל טָהֳרַת הַקֹּדֶשׁ לָאו כְּקֹדֶשׁ דָּמוּ (יג): בֵּית שַׁמַּאי אוֹמְרִים אַף כִּטְמֵא מֵת. שֶׁהוּא אַב הַטֻּמְאָה וְעוֹשֶׂה מַגָּעוֹ רִאשׁוֹן. כָּךְ מַגָּעוֹ שֶׁל טְבוּל יוֹם, רִאשׁוֹן הוּא:

פרק י - משנה ז

וּמוֹדִים שֶׁאוֹכֶלֶת בַּמַּעֲשֵׂר. כְּדִין טְבוּל יוֹם, דְּטָבַל וְעָלָה אוֹכֵל בַּמַּעֲשֵׂר: וְקוֹצָה לָהּ חַלָּה. וְאֵינָהּ קוֹרְאָה לָהּ שֵׁם, וּמַנִּיחָתָהּ בִּכְלִי. וְאַף עַל פִּי שֶׁהִיא נוֹגַעַת בְּחֻלִּין הַטְּבוּלִין לְחַלָּה, לָאו כַּחַלָּה דָּמֵי, וְאֵין שֵׁנִי פּוֹסֵל בָּהֶן: וּמַקֶּפֶת. מְקָרֶבֶת הַכְּלִי שֶׁהִנִּיחָה בּוֹ הַחַלָּה אֵצֶל הָעִסָּה. לְפִי שֶׁמִּצְוָה לִתְרֹם מִן הַמֻּקָּף, מִן הַקָּרוֹב לַדָּבָר שֶׁהוּא תּוֹרֵם עָלָיו: וְאִם נָפַל מֵרֻקָּהּ וּמִדַּם טָהֳרָהּ. דַּהֲווּ לְהוּ מַשְׁקִין הַיּוֹצְאִין מִן הַטְּבוּל יוֹם, וּטְהוֹרִין הֵן: צְרִיכָה טְבִילָה בָּאַחֲרוֹנָה. לֵיל שְׁמוֹנִים צְרִיכָה טְבִילָה לֶאֱכֹל בַּתְּרוּמָה אִם כֹּהֶנֶת הִיא. וְאִם יִשְׂרְאֵלִית הִיא, צְרִיכָה טְבִילָה לְבִיאַת מִקְדָּשׁ, מִפְּנֵי שֶׁטְּבוּלַת יוֹם אָרֹךְ הִיא וְהִסִּיחָה דַּעְתָּהּ מִן הַתְּרוּמָה וּמִן הַמִּקְדָּשׁ: אֵינָהּ צְרִיכָה טְבִילָה בָּאַחֲרוֹנָה. וְסוֹמֶכֶת עַל הַטְּבִילָה שֶׁל סוֹף שְׁבוּעַיִם. וְדַוְקָא לִתְרוּמָה וּלְבִיאַת מִקְדָּשׁ. אֲבָל לַאֲכִילַת קָדָשִׁים, מוֹדוּ בֵּית הִלֵּל שֶׁהִיא צְרִיכָה טְבִילָה אַחֶרֶת לְסוֹף שְׁמוֹנִים, דְּקַיְמָא לָן (טז) הָאוֹנֵן וּמְחֻסַּר כִּפּוּרִים צְרִיכִים טְבִילָה לַקֹּדֶשׁ, הִלְכָּךְ, לְאַחַר שֶׁהֵבִיאָה כַּפָּרָתָהּ, טוֹבֶלֶת לַקָּדָשִׁים:

פרק י - משנה ח

הָרוֹאָה יוֹם אַחַד עָשָׂר. שֶׁהוּא סוֹף יְמֵי זִיבָה, וְאֵין יוֹם שֶׁלְּאַחֲרָיו מִצְטָרֵף עִמּוֹ לְזִיבָה. שֶׁאִם רָאֲתָה יוֹם שְׁנֵים עָשָׂר וּשְׁלֹשָׁה עָשָׂר, אֵינָהּ זָבָה גְּדוֹלָה בִּשְׁבִיל כֵּן, לְפִי שֶׁהֵן תְּחִלַּת יְמֵי נִדָּה וְאֵינָן מִצְטָרְפִין עִם יְמֵי זִיבָה: בֵּית שַׁמַּאי אוֹמְרִים מְטַמְּאִין מִשְׁכָּב וּמוֹשָׁב. וַאֲפִלּוּ הַבּוֹעֵל מְטַמֵּא מִשְׁכָּב וּמוֹשָׁב לְטַמֵּא אֳכָלִין וּמַשְׁקִין, מִפְּנֵי שֶׁהוּא כְּבָא עַל שׁוֹמֶרֶת יוֹם כְּנֶגֶד יוֹם, דְּסָבְרֵי בֵּית שַׁמַּאי יוֹם אַחַד עָשָׂר בָּעֵי שִׁמּוּר (יח). וּבֵית הִלֵּל פּוֹטְרִים מִן הַקָּרְבָּן, דְּסָבְרֵי, יוֹם אַחַד עָשָׂר הוֹאִיל וְאֵין יוֹם שֶׁלְּאַחֲרָיו מִצְטָרֵף עִמּוֹ לְזִיבָה, לֹא בָּעֵי שִׁמּוּר. מִיהוּ לְעִנְיַן טֻמְאָה מוֹדוּ דְּבוֹעֲלָהּ מְטַמֵּא מִדְּרַבָּנָן, גְּזֵרָה יוֹם אַחַד עָשָׂר אַטּוּ תּוֹךְ אַחַד עָשָׂר, דִּמְטַמְּאָה בּוֹעֲלָהּ מִדְּאוֹרַיְתָא, אֲבָל לְעִנְיַן קָרְבָּן, לְהָבִיא חֻלִּין לָעֲזָרָה, לֹא: טָבְלָה בַיּוֹם שֶׁלְּאַחֲרָיו. דְּעַבְדָה שִׁמּוּר קְצָת. וְשִׁמּוּר מְעַלְּיָא הוּא, אֶלָּא שֶׁתּוֹךְ אַחַד עָשָׂר יוֹם אָסוּר לַעֲשׂוֹת כֵּן, שֶׁלֹּא תָּבֹא לִידֵי סָפֵק, שֶׁמָּא תִּרְאֶה לְאַחַר תַּשְׁמִישׁ וּמִצְטָרְפִין הַיָּמִים לְזִיבָה: מְטַמְּאִין מִשְׁכָּב וּמוֹשָׁב. מִדְּרַבָּנָן מְטַמֵּא אֲפִלּוּ בּוֹעֲלָהּ מִשְׁכָּב וּמוֹשָׁב. גְּזֵרָה אַטּוּ תּוֹךְ אַחַד עָשָׂר: וּפְטוּרִין מִן הַקָּרְבָּן. דְּהָא עַבְדָה שִׁמּוּר. וְאַף עַל פִּי שֶׁחָזְרָה וְרָאֲתָה, אֵין מִצְטָרְפִין לְזִיבָה, שֶׁהֲרֵי תְּחִלַּת נִדָּה הִיא: גַּרְגְּרָן. מְמַהֵר לַחֲטֹא. שֶׁמָּא יַרְגִּיל בְּכָךְ תּוֹךְ אַחַד עָשָׂר. אֲבָל לְעִנְיַן טֻמְאָה, טָהוֹר: וְחַיָּבִין בַּקָּרְבָּן. אֲפִלּוּ לֹא תִּרְאֶה לְמָחָר, דְּהָא בַּעְיָא שִׁמּוּר, וְזֹאת לֹא שָׁמְרָה כְּלָל, וְנִמְצָא שֶׁבָּעַל זָבָה קְטַנָּה: טָבְלָה בַיּוֹם שֶׁלְּאַחֲרָיו. דְּעַבְדָא קְצָת שִׁמּוּר, תְּלוּיִין. שֶׁאִם תִּרְאֶה, מַגָּעָן מַגַּע זָבָה וְחַיָּבִין קָרְבָּן בִּבְעִילָתָן. וְאִם לֹא תִרְאֶה, מַגָּעָן טָהוֹר, וּפְטוּרִין מִן הַקָּרְבָּן עַל בְּעִילָתָן:
