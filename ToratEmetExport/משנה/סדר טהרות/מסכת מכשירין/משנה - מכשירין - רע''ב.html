פרק א - משנה א

כָּל מַשְׁקֶה שֶׁתְּחִלָּתוֹ לְרָצוֹן. דְּנִיחָא לֵיהּ בְּהָנָךְ מַשְׁקִין, בִּשְׁבִיל שֶׁתּוּדַח הַקְּעָרָה אוֹ לִשְׁאָר תַּשְׁמִישִׁין (א): אַף עַל פִּי שֶׁאֵין סוֹפוֹ לְרָצוֹן. כְּשֶׁנָּפְלוּ עַל הַפֵּרוֹת לֹא נִיחָא לֵיהּ, אֲפִלּוּ הָכִי הָוֵי הֶכְשֵׁר. אִי נַמִּי, סוֹפוֹ לְרָצוֹן, כְּשֶׁנָּפְלוּ עַל הַפֵּרוֹת. אֲבָל בִּתְחִלַּת בִּיאָתָן הָיָה הַדָּבָר קָשֶׁה בְּעֵינָיו, הַכֹּל הָוֵי הֶכְשֵׁר: הֲרֵי זֶה בְכִי יֻתַּן. כְּלוֹמַר, הָאֳכָלִין שֶׁנָּפְלוּ עֲלֵיהֶן מַשְׁקִין הַלָּלוּ, מֻכְשָׁרִין לְקַבֵּל טֻמְאָה מִכָּאן וָאֵילָךְ. שֶׁאֵין שׁוּם אֹכֶל מְקַבֵּל טֻמְאָה עַד שֶׁיָּבֹא עָלָיו מַיִם (ב) אוֹ אֶחָד מִשִּׁבְעָה מַשְׁקִין, שֶׁהֵם, מַיִם, יַיִן, שֶׁמֶן, חָלָב, דְּבַשׁ, דָּם, טַל. דִּכְתִיב (וַיִּקְרָא יא) וְכִי יֻתַּן מַיִם עַל זֶרַע וְנָפַל מִנִּבְלָתָם עָלָיו טָמֵא הוּא, כְּתִיב יִתֵּן וְקָרִינַן יֻתַּן, מַה יִתֵּן דְּנִיחָא לֵיהּ, אַף יֻתַּן דְּנִיחָא לֵיהּ: מְטַמְּאִין לְרָצוֹן וְשֶׁלֹּא לְרָצוֹן. כֵּיוָן דִּטְמֵאִין הֵן, טֻמְאָתָן וְהֶכְשֵׁרָן בָּאִים כְּאֶחָד, וְאִם נָפְלוּ עַל הַפֵּרוֹת אֲפִלּוּ שֶׁלֹּא לְרָצוֹן, מְטַמְּאִין אוֹתָם, דְּלֹא בָעִינַן דְּנִיחָא לֵיהּ אֶלָּא כְשֶׁבָּא הַמַּשְׁקֶה קֹדֶם לַטֻּמְאָה, דִּכְתִיב וְכִי יֻתַּן מַיִם עַל זֶרַע, וַהֲדַר וְנָפַל מִנִּבְלָתָם עָלָיו. אֲבָל כְּשֶׁהַמַּשְׁקֶה וְהַטֻּמְאָה בָּאִין כְּאֶחָד כְּגוֹן שֶׁהַמַּשְׁקִין טְמֵאִין, לֹא בָעִינַן דְּנִיחָא לֵיהּ:

פרק א - משנה ב

הַמַּרְעִיד אֶת הָאִילָן. מְנַעְנְעוֹ: אוֹ אֶת הַטֻּמְאָה. כְּגוֹן שֶׁהֵבִיא שָׁם עוֹרֵב שֶׁרֶץ אוֹ כַזַּיִת מִן הַמֵּת אוֹ נְבֵלָה וְנִשְׁאַר בֵּין עַנְפֵי הָאִילָן: אֵינָן בְּכִי יֻתַּן. אִם מֵחֲמַת רְעָדָה זוֹ נָפְלוּ מֵי גְשָׁמִים שֶׁבָּאִילָן עַל פֵּרוֹת שֶׁתַּחְתָּיו, לֹא הֻכְשְׁרוּ: הַיּוֹצְאִים וְאֶת שֶׁבּוֹ. מַשְׁקִים הַיּוֹצְאִים מֵחֲמַת הָרְעָדָה, וּמַשְׁקִים הַנִּשְׁאָרִים בָּאִילָן וְנָפְלוּ אַחַר כָּךְ עַל הַפֵּרוֹת, הֻכְשְׁרוּ, דְּאַחְשׁוֹבֵי אַחְשְׁבִינְהוּ וְכִתְלוּשִׁים דָּמוּ (ג): מִפְּנֵי שֶׁהוּא מִתְכַּוֵּן שֶׁיֵּצְאוּ מִכֻּלּוֹ. נִתְכַּוֵּן לְהוֹצִיא כָל הַמַּיִם מִן הָאִילָן, וְהַנִּשְׁאָרִים לָאו דַּעְתֵּיהּ עֲלַיְהוּ וְלֹא מַכְשְׁרֵי:

פרק א - משנה ג

הַמַּרְעִיד אֶת הָאִילָן. לְהַפִּיל פֵּרוֹתָיו: וְנָפְלוּ עַל חֲבֵרוֹ. עַל אִילָן אַחֵר, וּמֵאוֹתוֹ אִילָן נָפְלוּ עַל הַזְּרָעִים וְעַל הַיְרָקוֹת הַמְחֻבָּרִים לַקַּרְקַע, וּמַשְׁקֶה טוֹפֵחַ עַל אוֹתָן הַזְּרָעִים מִן הַמַּיִם שֶׁרְגִילִין לְהַשְׁקוֹת אוֹתָן אוֹ מִגְּשָׁמִים שֶׁנָּפְלוּ עֲלֵיהֶן: בֵּית שַׁמַּאי אוֹמְרִים הֲרֵי הֵן בְּכִי יֻתַּן. דְּכֵיוָן דְּיוֹדֵעַ הָיָה שֶׁיִּפְּלוּ פֵרוֹת הַלָּלוּ עַל אוֹתָן זְרָעִים שֶׁיֵּשׁ עֲלֵיהֶן מַשְׁקֶה טוֹפֵחַ, גַּלֵּי דַעְתֵּיהּ דְּנִיחָא לֵיהּ, וְהֻכְשְׁרוּ הַפֵּרוֹת שֶׁנָּפְלוּ עַל גַּבֵּי אוֹתָן זְרָעִים, לְקַבֵּל טֻמְאָה: וּבֵית הִלֵּל אוֹמְרִים אֵינָן בְּכִי יֻתַּן. כֵּיוָן דְּלֹא נָפְלוּ לְהֶדְיָא מִן הָאִילָן עַל הַזְּרָעִים אֶלָּא מֵאִילָן לְאִילָן אַחֵר: חֲלֵי קוּפְרִי. כָּךְ הָיָה כִנּוּיוֹ: טִבְעוֹן. שֵׁם מָקוֹם: עַד שֶׁיִּתְכַּוֵּן. וְהָנֵי הוֹאִיל וּמֵאִילָן לְאִילָן אוֹ מִסּוֹכָה לְסוֹכָה נָפְלוּ, אֵין זוֹ כַוָּנָה. וְטַעֲמַיְהוּ דְבֵית הִלֵּל קָאָמַר:

פרק א - משנה ד

הַנּוֹעֵר אֲגֻדָּה שֶׁל יָרָק. כְּגוֹן יָרָק שֶׁיָּרְדוּ עָלָיו גְּשָׁמִים אוֹ שֶׁנָּפַל בַּמַּיִם שֶׁלֹּא לְרָצוֹן, וּמְנַעֲרוֹ כְּדֵי שֶׁיֵּצְאוּ הַמַּיִם (ז): חוֹשְׁשִׁים אָנוּ. בִּתְמִיהָה. כְּלוֹמַר לְדִבְרֵיכֶם הָיָה לָנוּ לָחוּשׁ גַּם לָזֶה: הַמַּעֲלֶה שַׂק מָלֵא פֵרוֹת. שֶׁנָּפַל בְּמַיִם שֶׁלֹּא לְרָצוֹן, וְהֶעֱלָהוּ וְהִנִּיחוֹ עַל שְׂפַת הַנָּהָר: הַתַּחְתּוֹן טָהוֹר. כְּלוֹמַר, אַף הַתַּחְתּוֹן טָהוֹר, וְכָל שֶׁכֵּן הָעֶלְיוֹן:

פרק א - משנה ה

הַמְמַחֵק אֶת הַכְּרֵישָׁה. דּוֹחֵק אֶת הַכְּרֵישָׁה כְּדֵי שֶׁיֵּצֵא מִמֶּנָּה הַטַּל: כְּרֵישָׁה. חָצִיר בִּלְשׁוֹן מִקְרָא. פורו"ש בְּלַעַ"ז וּבַעֲרָבִי כורא"ת: וְהַסּוֹחֵט בִּשְׂעָרוֹ וּבִכְסוּתוֹ (יא). שֶׁהָיָה בָא בַדֶּרֶךְ וְיָרְדוּ גְשָׁמִים בִּשְׂעָרוֹ וְעַל כְּסוּתוֹ, וְסוֹחֵט בָּהֶן כְּדֵי שֶׁיֵּצְאוּ הַמַּיִם: וְאֶת שֶׁבּוֹ. מַשְׁקִין הַנִּשְׁאָרִים וְנָפְלוּ אַחַר כָּךְ עַל הַפֵּרוֹת: מִתְכַּוֵּן שֶׁיֵּצְאוּ מִכֻּלּוֹ. דְּנִתְכַּוֵּן לְהוֹצִיא מֵהֶן כָּל הַמַּיִם וְלָאו דַּעְתֵּיהּ עַל הַנִּשְׁאָרִים, הִלְכָּךְ לֹא מַכְשִׁירִים. כְּבֵית הִלֵּל דְּרֵישׁ פִּרְקִין:

פרק א - משנה ו

הַנּוֹפֵחַ. בְּפִיו: בַּעֲדָשִׁים. וְיָצָא עֲלֵיהֶן רֹק מִפִּיו בִּנְפִיחָתוֹ, וְהָרֹק הַיּוֹצֵא מִן הַפֶּה מִתּוֹלְדוֹת הַמַּיִם הוּא חָשׁוּב: אֵינָן בְּכִי יֻתַּן. כֵּיוָן דְּלֹא נִתְכַּוֵּן לְכָךְ: וַחֲכָמִים אוֹמְרִים בְּכִי יֻתַּן. כֵּיוָן דְּנִתְכַּוֵּן לְנַפֵּחַ, וְדֶרֶךְ הָרֹק לָצֵאת עַל יְדֵי נְפִיחָה. וַהֲלָכָה כַּחֲכָמִים: הָאוֹכֵל שֻׁמְשְׁמִין בְּאֶצְבָּעוֹ. כְּגוֹן שֶׁהִנִּיחַ שֻׁמְשְׁמִין בְּיָדוֹ אַחַת, וְאֶצְבָּעוֹ שֶׁבְּיָדוֹ הַשֵּׁנִית מַכְנִיסָהּ לְתוֹךְ פִּיו לְלַחְלְחָהּ בָּרֹק וְנוֹתְנָהּ בַּשֻּׁמְשְׁמִין שֶׁבְּיָדוֹ הַשֵּׁנִית וְהֵן נִדְבָּקִין בָּהּ וּמַכְנִיס בְּפִיו וְאוֹכֵל, וּמֵחֲמַת כֵּן נִתְלַחְלְחָה הַיָּד שֶׁהַשֻּׁמְשְׁמִין בָּהּ, וְיֵשׁ בָּהּ מַשְׁקֶה טוֹפֵחַ: רַבִּי שִׁמְעוֹן אוֹמֵר אֵינָן בְּכִי יֻתַּן. דִּלְלַחְלֵחַ אֶצְבָּעוֹ נִתְכַּוֵּן, וְלֹא לְלַחְלֵחַ הַיָּד שֶׁהַשֻּׁמְשְׁמִין בָּהּ. וְרַבָּנָן סָבְרֵי, כֵּיוָן דְּאַחְשְׁבֵיהּ בְּאֶצְבָּעוֹ, מַכְשִׁיר אֲפִלּוּ שֶׁלֹּא לְרָצוֹן, כְּדִתְנַן בְּרֵישׁ פִּרְקִין כָּל מַשְׁקֶה שֶׁתְּחִלָּתוֹ לְרָצוֹן, אַף עַל פִּי שֶׁאֵין סוֹפוֹ לְרָצוֹן, הֲרֵי זֶה בְּכִי יֻתַּן. וְרַבִּי שִׁמְעוֹן סָבַר, דְּלֹא הָיָה מִתְּחִלָּתוֹ לְרָצוֹן אֶלָּא הַמַּשְׁקֶה הַמְלַחְלֵחַ הָאֶצְבַּע בִּלְבַד. וְהַנִּשְׁאָר שֶׁנִּתְלַחְלְחָה בוֹ הַיָּד, לֹא הָיָה לְעוֹלָם לְרָצוֹן. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן בְּכֻלַּהּ מַתְנִיתִין: הַסְּרִיקִין. וַיִּתְלַקְּטוּ אֶל יִפְתָּח אֲנָשִׁים רֵיקִים (שׁוֹפְטִים יא) מְתַרְגְּמִינַן גֻּבְרִין סְרִיקִין. לְשׁוֹן אִילָן סְרָק שֶׁאֵינוֹ עוֹשֶׂה פֵרוֹת. פֵּרוּשׁ אַחֵר, סְרִיקִין, גַּנָּבִים. שֶׁכֵּן בִּלְשׁוֹן יִשְׁמָעֵאל קוֹרִין לְגַנָּבֵי סְרָאקִין. וְיֵשׁ גּוֹרְסִין סִיקָרִין, לְשׁוֹן לֹא הָיָה סִקְרִיקוֹן בִּיהוּדָה בַּהֲרוּגֵי הַמִּלְחָמָה, בְּפֶרֶק הַנִּזָּקִין, פֵּרוּשׁ גַּזְלָנִין וְאַנָּסִין: בְּשִׁבֹּלֶת הַנָּהָר. שְׁבִיל הַנָּהָר וּמְרוּצַת מֵימָיו. לְשׁוֹן וְשִׁבֹּלֶת שְׁטָפָתְנִי (תְּהִלִּים סט): לַהֲבִיאָן עִמּוֹ. שֶׁלֹּא הָיָה יָכוֹל לְנָשְׂאָן, וּמְשִׁיטָן בַּנָּהָר:

זֵעַת בָּתִּים טְהוֹרָה. אֵינָהּ מַכְשֶׁרֶת הַזְּרָעִים לְקַבֵּל טֻמְאָה: זֵעָתוֹ טְמֵאָה. דְּמֵחֲמַת הַמַּיִם הִיא. וְדַוְקָא בָא בְּמַיִם שְׁאוּבִין. אֲבָל בָּא בְּמַיִם מְחֻבָּרִים דְּלֹא מַכְשְׁרֵי, לֹא, דְּאַף עַל גַּב דְּהַשְׁתָּא מִיהָא תְלוּשִׁים נִינְהוּ, לֹא לְרָצוֹן נִתְלְשׁוּ (ב):

פרק ב - משנה ב

מֶרְחָץ טְמֵאָה. שֶׁל מַיִם שְׁאוּבִין: זֵעָתָהּ טְמֵאָה. מַכְשֶׁרֶת אֶת הַפֵּרוֹת וּמְטַמֵּאתָן (ג): וּטְהוֹרָה. כְּלוֹמַר, וְזֵעַת מֶרְחָץ טְהוֹרָה, כְּגוֹן מֶרְחָץ שֶׁל מֵי מַעְיָן דְּלָאו מַיִם שְׁאוּבִים נִינְהוּ, הֲרֵי הוּא בְּכִי יֻתַּן, שֶׁהַזֵּעָה מַכְשֶׁרֶת אֶת הַפֵּרוֹת: אִם טְמֵאָה. הַבְּרֵכָה. כָּל זֵעָה שֶׁיַּזִיעַ הַבַּיִת מֵחֲמָתָהּ, מַכְשֶׁרֶת וּמְטַמְּאָה. אֲבָל שֶׁלֹּא מֵחֲמָתָהּ, הַוְיָא לַהּ זֵעַת הַבַּיִת וּטְהוֹרָה:

פרק ב - משנה ג

בַּרְזֶל טָמֵא. שֶׁבָּא מִשִּׁבְרֵי כֵלִים טְמֵאִין: שֶׁבְּלָלוֹ. שֶׁטְּרָפוֹ וְעֵרְבוֹ עִם בַּרְזֶל טָהוֹר שֶׁבָּא מִן הָעֶשֶׁת: אִם רֹב מִן הַטָּמֵא טָמֵא. חֲכָמִים גָּזְרוּ עַל כְּלֵי מַתָּכוֹת שֶׁנִּטְמְאוּ וּשְׁבָרָן וְהִתִּיכָן וְעָשָׂה מֵהֶן כֵּלִים, שֶׁיַּחְזְרוּ לְטֻמְאָתָן הַיְשָׁנָה עַד שֶׁיַּזֶּה עֲלֵיהֶן שְׁלִישִׁי וּשְׁבִיעִי וְיַטְבִּילֵם וְיַעֲרִיב שִׁמְשָׁן: גִּסְטְרָיוֹת. כְּמִין קְדֵרוֹת עֲשׂוּיוֹת לְהַשְׁתִּין בָּהֶן. וּבְלָשׁוֹן עֲרָבִי קוֹרִין לָהֶן בסריא"ש: שֶׁיִּשְׂרָאֵל וְנָכְרִים מַטִּילִין לְתוֹכָן. מַשְׁתִּינִים לְתוֹכָן. וְשֶׁתֶן שֶׁל נָכְרִי טָמֵא, דְּגָזְרוּ עֲלֵיהֶן שֶׁיִּהְיוּ כְזָבִין לְכָל דִּבְרֵיהֶן: מֵי שְׁפִיכוּת. מַיִם שֶׁרָחֲצוּ בָהֶן וְשׁוֹפְכִין אוֹתָן לַחוּץ, וּבְחֶזְקַת טְמֵאִין הֵן: אֵימָתַי, מִתְבַּטְּלִין מֵי הַשְּׁפִיכוּת בְּרֹב: בִּזְמַן שֶׁקָּדְמוּ. הֵן, וְאַחַר כָּךְ רַבּוּ עֲלֵיהֶן מֵי גְשָׁמִים וְטִהֲרוּם: אֲבָל אִם קָדְמוּ מֵי גְשָׁמִים. אֲפִלּוּ נָפְלוּ עֲלֵיהֶן מַשֶּׁהוּ מֵי שְׁפִיכוּת, טְמֵאִים:

פרק ב - משנה ד

הַמְטַהֵר אֶת גַּגּוֹ. גָּרְסִינַן. וְאִית דְּגָרַס הַטּוֹרֵף אֶת גַּגּוֹ, פֵּרוּשׁ שֶׁטָּח גַּגּוֹ טִיט אוֹ סִיד לְהַשְׁווֹתוֹ. וְהַמַּיִם שֶׁנּוֹתְנִין עָלָיו כְּשֶׁמְּטַהֲרוֹ אוֹ כְשֶׁטָּחוֹ בְטִיט, סְתָמָן שׁוֹפְכִין הֵן וּמְטַמְּאִין. וְכֵן הַמַּיִם שֶׁמְּכַבֵּס בָּהֶן אֶת כְּסוּתוֹ: וְיָרְדוּ עֲלֵיהֶן גְּשָׁמִין. שֶׁהֵן מַיִם טְהוֹרִים. וְהַגַּג אוֹ הַכְּסוּת מְנַטְּפִים מַיִם: אִם רֹב מִן הַטָּהוֹר טָהוֹר. כְּגוֹן שֶׁבַּתְּחִלָּה הָיָה מְנַטֵּף טִפִּים דַּקּוֹת וּלְאַחַר כֵּן מְנַטֵּף טִפִּים עָבוֹת, בְּיָדוּעַ שֶׁרֹב מִן הַטָּהוֹר: אִם הוֹסִיפוּ לְנַטֵּף. שֶׁמְּמַהֲרִים הַטִּפִּים לֵירֵד זוֹ אַחַר זוֹ יוֹתֵר מִבַּתְּחִלָּה, טָהוֹר, דְּוַדַּאי רַבּוּ. וְאַף עַל פִּי שֶׁאֵין הַטִּפּוֹת עַצְמָן יוֹתֵר גְּדוֹלוֹת מִמַּה שֶּׁהָיוּ בַתְּחִלָּה. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה:

פרק ב - משנה ה

רוֹחֵץ בָּהּ מִיָּד. לְמוֹצָאֵי שַׁבָּת, וְאֵין צָרִיךְ לְהַמְתִּין כְּדֵי שֶׁיֵּחַמּוּ. דְּכֵיוָן דְּרֹב נָכְרִים, אַדַּעְתָּא דְּנָכְרִים הוּחֲמוּ: אִם יֵשׁ בָּהּ רָשׁוּת. הַמַּלְכוּת קְרוּיָה רָשׁוּת, מִפְּנֵי שֶׁהָרְשׁוּת בְּיָדָהּ לַעֲשׂוֹת כִּרְצוֹנָהּ. וְלָאו דַּוְקָא מֶלֶךְ, אֶלָּא אִם יֵשׁ שָׁם אָדָם חָשׁוּב שֶׁיֵּשׁ לוֹ עֲשָׂרָה עֲבָדִים שֶׁמְּחַמְּמִין לוֹ עֲשָׂרָה קֻמְקוּמִיּוֹת מַיִם בְּבַת אַחַת, מֻתָּר לִרְחֹץ בָּהּ מִיָּד, וְאַף עַל פִּי שֶׁרֹב הָעִיר יִשְׂרָאֵל. דְּאָמְרִינַן לִשְׁמוֹ שֶׁל אוֹתוֹ אָדָם חָשׁוּב הוּחֲמוּ:

פרק ב - משנה ו

הָיָה בָהּ יָרָק נִמְכָּר. יָרָק שֶׁנִּתְלַשׁ בְּשַׁבָּת, נִמְכָּר לְמוֹצָאֵי שַׁבָּת: אִם רֹב נָכְרִים לוֹקֵחַ מִיָּד. דְּכֵיוָן דְּרֹב נָכְרִים, אַדַּעְתָּא דְנָכְרִים נִתְלָשׁ: כְּדֵי שֶׁיָּבֹאוּ מִמָּקוֹם קָרוֹב. שֶׁיֵּשׁ שָׁם יָרָק. וְאַף עַל גַּב דְּשֶׁמָּא הוּבָא מִמָּקוֹם רָחוֹק, אֵין צָרִיךְ לְהַמְתִּין אֶלָּא כְדֵי שֶׁיִּלְקְטוּ יָרָק מִמָּקוֹם קָרוֹב שֶׁיֵּשׁ בּוֹ יָרָק (ו) וִיבִיאוּהוּ מִשָּׁם. שֶׁלֹּא אָמְרוּ חֲכָמִים וְלָעֶרֶב יַמְתִּין בִּכְדֵי שֶׁיַּעֲשׂוּ, אֶלָּא גְזֵרָה שֶׁמָּא יֹאמַר לְנָכְרִי בְּשַׁבָּת לֵךְ וְהָבֵא (ז), וּכְשֶׁרוֹאֶה שֶׁצָּרִיךְ לְהַמְתִּין לָעֶרֶב כְּדֵי שֶׁיָּבֹא הַדָּבָר הַהוּא וַאֲפִלּוּ מִמָּקוֹם קָרוֹב, לֹא יֹאמַר לַנָּכְרִי לַהֲבִיאוֹ בְּשַׁבָּת: וְאִם יֵשׁ שָׁם רָשׁוּת לוֹקֵחַ מִיָּד. דְּאַדַּעְתָּא דְאָדָם חָשׁוּב רְגִילִין תָּמִיד לְהָבִיא, וְלֹא אַדַּעְתָּא דְיִשְׂרָאֵל הֱבִיאוֹ:

פרק ב - משנה ז

אִם רֹב נָכְרִים נָכְרִי. וּמֻתָּר לְהַאֲכִילוֹ נְבֵלוֹת וּטְרֵפוֹת שְׁקָצִים וּרְמָשִׂים בַּיָּדַיִם (ח): וְאִם רֹב יִשְׂרָאֵל יִשְׂרָאֵל. וְחַיָּבִים לְהַחֲזִיר לוֹ אֲבֵדָתוֹ (ט) כְּיִשְׂרָאֵל. וּרְבוּתָא קָא מַשְׁמַע לָן, דְּמוֹצִיאִין מָמוֹן מִיַּד יִשְׂרָאֵל שֶׁזָּכָה בוֹ, וְלֹא אָמְרִינַן אוֹקֵי מָמוֹנָא אַחַזְקָתֵיהּ שֶׁזָּכָה בוֹ עַד דְּמַיְתֵי הִיאַךְ רְאָיָה שֶׁהוּא יִשְׂרָאֵל: מֶחֱצָה עַל מֶחֱצָה יִשְׂרָאֵל. לְעִנְיַן נְזָקִין (י), שֶׁאִם שׁוֹרוֹ נָגַח שׁוֹר שֶׁל יִשְׂרָאֵל, מְשַׁלֵּם חֲצִי נֶזֶק בִּלְבָד. וְאֵינוֹ מְשַׁלֵּם נֶזֶק שָׁלֵם כְּדִין שׁוֹר שֶׁל נָכְרִי שֶׁנָּגַח שׁוֹר שֶׁל יִשְׂרָאֵל (יא), דְּבֵין תָּם בֵּין מוּעָד מְשַׁלֵּם נֶזֶק שָׁלֵם. דְּמָצֵי אָמַר לֵיהּ, אַיְתֵּי רְאָיָה דְּלָאו יִשְׂרָאֵל אֲנָא וּשְׁקוֹל. וְלִשְׁאָר כָּל הַדְּבָרִים הָוֵי סָפֵק, וְדָנִין בּוֹ לְהַחְמִיר. מִי שֶׁהֲרָגוֹ, אֵינוֹ נֶהֱרָג עָלָיו. קִדֵּשׁ אִשָּׁה, צְרִיכָה גֵט מִסָּפֵק. וְאֵינוֹ כְיִשְׂרָאֵל גָּמוּר עַד שֶׁיִּטְבֹּל לְשֵׁם גֵּרוּת: אַחַר רֹב הַמַּשְׁלִיכִין. הַיְנוּ אַחַר הַנָּכְרִים וְאַף עַל פִּי שֶׁהֵן מֻעָט. וַאֲפִלּוּ לֹא הָיְתָה שָׁם אֶלָּא נָכְרִית אַחַת, הִיא חֲשׁוּדָה לְהַשְׁלִיךְ. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה:

פרק ב - משנה ח

רֹב נַחְתּוֹמִים. אִם רֹב נַחְתּוֹמִים נָכְרִים, הַפַּת אֲסוּרָה, דְּפַת שֶׁל נָכְרִים הוּא מִשְּׁמֹנָה עָשָׂר דָּבָר שֶׁגָּזְרוּ בּוֹ בַיּוֹם: פַּת עִסָּה. פַּת נְקִיָּה: פַּת קִבָּר. פַּת שֶׁאֵינָהּ נְקִיָּה: אַחַר רֹב אוֹכְלֵי פַת קִבָּר. אִם רֻבָּן נָכְרִים, הַפַּת אֲסוּרָה. אִי נַמִּי יֵשׁ לְפָרֵשׁ, אַחַר רֹב נַחְתּוֹמִין, אַחַר רֹב אוֹכְלֵי פַת עִסָּה וְכוּ', אִם רֻבָּן חֲבֵרִים, הַפַּת טְהוֹרָה וְאֵין צָרִיךְ לְעַשֵּׂר. וְאִם רֻבָּן עַמֵּי הָאָרֶץ, הַפַּת טְמֵאָה וְצָרִיךְ לְעַשֵּׂר:

פרק ב - משנה ט

הוֹלְכִין אַחַר רֹב טַבָּחִים. אִם רֹב טַבָּחֵי יִשְׂרָאֵל, הַבָּשָׂר מֻתָּר. דְּאֵין הֲלָכָה כְּרַב דְּאָמַר בָּשָׂר שֶׁנִּתְעַלֵּם מִן הָעַיִן אָסוּר (יג). וְרַב מוֹקֵי מַתְנִיתִין בְּעוֹמֵד וְרוֹאֶה מִשָּׁעָה שֶׁנִּשְׁחַט עַד שֶׁנָּפְלָה, אֲבָל לֹא יָדַע מִי הַשּׁוֹחֵט:

פרק ב - משנה י

חָבִית. שֶׁל חֶרֶס, אוֹ שֶׁל כְּלֵי אֶבֶן הָרַכִּים שֶׁהֵן כְּחֶרֶס. אֲבָל שְׁאָר כֵּלִים לֹא בָלְעֵי כָּל כָּךְ עַד שֶׁיִּהְיוּ הַפֵּרוֹת שֶׁבְּתוֹכָן שׁוֹאֲבִים: שָׁאֲבוּ. נִתְלַחְלְחוּ הַפֵּרוֹת וּמַשְׁקֶה טוֹפֵחַ עֲלֵיהֶן: הָכִי גָרְסִינַן, וּשְׁאָר כָּל הַמַּשְׁקִים טְהוֹרִים. שְׁאָר שִׁבְעָה מַשְׁקִים הַמַּכְשִׁירִים לְקַבֵּל טֻמְאָה, טְהוֹרִים הַפֵּרוֹת שֶׁבְּתוֹךְ הֶחָבִית אִם נִתְלַחְלְחוּ מֵהֶן (ב). דְּדַוְקָא מַיִם וְיַיִן וְחֹמֶץ שֶׁהֵן דַּקִּים וּצְלוּלִים, הֵן נִבְלָעִין בֶּחָבִית מִפְּנֵי דַקּוּתָן, וְהַפֵּרוֹת שׁוֹאֲבִים מֵהֶן. אֲבָל שֶׁמֶן דְּבַשׁ חָלָב דָּם טַל, הֵן עָבִין, וּמִפְּנֵי עָבְיָן אֵינָן יְכוֹלִין לְהִבָּלַע בְּתוֹךְ הַכְּלִי כָּל כָּךְ עַד שֶׁיִּהְיוּ הַפֵּרוֹת שֶׁבְּתוֹךְ הֶחָבִית שׁוֹאֲבִים מֵהֶן, הִלְכָּךְ אֵינָן בְּכִי יֻתַּן: שֶׁאֵין הַקִּטְנִית שׁוֹאָבֶת. כְּגוֹן פּוֹלִים וַאֲפוּנִים וְכַיּוֹצֵא בָהֶן. וְאֵין הֲלָכָה כְּרַבִּי נְחֶמְיָה:

הַנִּתָּזִין. מִמֶּנָּה לַחוּץ: וְהַצָּפִין. לְמַעְלָה וְיָצְאוּ לַחוּץ. כֻּלְּהוּ אֵינָן בְּכִי יֻתַּן, לְפִי שֶׁהֵן שֶׁלֹּא לְרָצוֹן, וְכֵן הַמַּיִם שֶׁבְּתוֹכָהּ: נְטָלָהּ לְשָׁפְכָן. שֶׁלֹּא שְׁפָכָן בִּמְקוֹמָן, אֶלָּא נָטַל הָעֲרֵבָה לִשְׁפֹּךְ הַמַּיִם בְּמָקוֹם אַחֵר, הֲרֵי הֵן בְּכִי יֻתַּן, דִּלְרָצוֹן הֵן הוֹאִיל וְלֹא שְׁפָכָן מִיָּד שָׁם בִּמְקוֹמָן: וּבֵית הִלֵּל אוֹמְרִים. כֵּיוָן שֶׁלֹּא הִנִּיחָן שָׁם אֶלָּא נְטָלָהּ לְשָׁפְכָן, אֵינָן בְּכִי יֻתַּן: הִנִּיחָהּ. אִם הִנִּיחַ אֶת הָעֲרֵבָה מִתְּחִלָּה כְּדֵי שֶׁיֵּרֵד הַדֶּלֶף לְתוֹכָהּ: אֵלּוּ וָאֵלּוּ. בֵּית שַׁמַּאי וּבֵית הִלֵּל מוֹדִים שֶׁאַף עַל פִּי שֶׁנְּטָלָהּ אַחַר כָּךְ וּשְׁפָכָהּ, הֲרֵי הֵן בְּכִי יֻתַּן, הוֹאִיל וּכְבָר חִשַּׁב עֲלֵיהֶן: מַיִם הָעוֹלִים בְּיָדָיו. שֶׁמְּכַבֵּס בָּהֶן, נִיחָא לֵיהּ: בְּרַגְלָיו. לֹא נִיחָא לֵיהּ: אֶלָּא אִם כֵּן נִטַּנְּפוּ רַגְלָיו. הָווּ בְּכִי יֻתַּן. מִידִי דַהֲוָה אַמְמַלֵּא בֶחָבִית דִּלְעֵיל (ג), דְּמַיִם הָעוֹלִים אַחֲרֶיהָ בְּכִי יֻתַּן, כִּדְפָרִישְׁנָא לְעֵיל, לְפִי שֶׁאִי אֶפְשָׁר לַמַּיִם לִכָּנֵס לְתוֹכָהּ עַד שֶׁיִּגְּעוּ לַאֲחוֹרֶיהָ:

יָדָיו טְהוֹרוֹת. דְּלֹא בָעִינַן כַּוָּנָה לְחֻלִּין. אֲבָל אִם הָיוּ פֵרוֹת מַעֲשֵׂר (ז) וְכָל שֶׁכֵּן שֶׁל תְּרוּמָה, יָדָיו טְמֵאוֹת כְּשֶׁהָיוּ, עַד שֶׁיְּכַוֵּן לְהַטְבִּיל יָדָיו: וְהַפֵּרוֹת טְהוֹרִים. וְלֹא הֻכְשְׁרוּ, דְּלֹא אַחְשְׁבַהּ לְהַךְ נְפִילָה: וְאִם בִּשְׁבִיל שֶׁיּוּדְחוּ יָדָיו. נִתְכַּוֵּן לִטֹּל אֶת הַפֵּרוֹת מִן הַמַּיִם, הַפֵּרוֹת הֲרֵי הֵן בְּכִי יֻתַּן וּמֻכְשָׁרִים לְקַבֵּל טֻמְאָה מִכָּאן וּלְהַבָּא (ח). דְּאַחְשְׁבַהּ לְהַךְ נְפִילַת פֵּרוֹת לִטֹּל יָדָיו בְּהַגְבָּהָתָן, וְהָוֵי לֵיהּ מַשְׁקֶה שֶׁסּוֹפוֹ לְרָצוֹן, דְּאַף עַל פִּי שֶׁאֵין תְּחִלָּתוֹ לְרָצוֹן הֲרֵי זֶה בְּכִי יֻתַּן:

וּפָשַׁט אַב הַטֻּמְאָה אֶת יָדָיו לְתוֹכָהּ טְמֵאָה. דִּכְלִי חֶרֶס מְקַבֵּל טֻמְאָה מֵאַב הַטֻּמְאָה הַנּוֹגֵעַ בְּתוֹכוֹ, וְלֹא עָלְתָה לוֹ טְבִילָה, וְאַף עַל פִּי שֶׁהוּא בְתוֹךְ הַמִּקְוֶה, לְפִי שֶׁהַמַּיִם שֶׁבְּתוֹכוֹ חוֹצְצִים בֵּינוֹ וּבֵין מֵי הַמִּקְוֶה (ט): מַגַּע טֻמְאוֹת טְהוֹרָה. אִם לֹא הָיָה אַב הַטֻּמְאָה שֶׁנָּגַע בָּהּ, אֶלָּא רִאשׁוֹן לַטֻּמְאָה וְהוּא הַנִּקְרָא מַגַּע טֻמְאוֹת, כְּלוֹמַר הַנּוֹגֵעַ בַּאֲבוֹת הַטֻּמְאָה, אָז טְהוֹרָה הַקְּדֵרָה, לְפִי שֶׁאֵין כְּלִי חֶרֶס מְקַבֵּל טֻמְאָה אֶלָּא מֵאַב הַטֻּמְאָה, וְנִמְצָא שֶׁלֹּא נִטְמֵאת הַקְּדֵרָה אֶלָּא מֵחֲמַת מַשְׁקִים טְמֵאִים שֶׁבְּתוֹכָהּ שֶׁנִּטְמְאוּ וְחוֹזְרִים וּמְטַמְּאִים אוֹתָהּ, וּמִשּׁוּם הָכִי אֵינָהּ מְטַמְּאָה, שֶׁהֲרֵי הֻשְּׁקוּ הַמַּיִם שֶׁבְּתוֹכָהּ בְּמֵי מִקְוֶה וְנִטְהֲרוּ. אֲבָל שְׁאָר מַשְׁקִים כְּגוֹן יַיִן שֶׁמֶן וְחָלָב, אֵין הַמִּקְוֶה מְטַהֵר אוֹתָם, דְּלֹא סָלְקָא לְהוּ הַשָּׁקָה, הִלְכָּךְ חוֹזְרִים וּמְטַמְּאִים אֶת הַקְּדֵרָה, דְּרַבָּנָן גָּזְרוּ עַל הַמַּשְׁקִים שֶׁיִּהְיוּ מְטַמְּאִין כֵּלִים (י), גְּזֵרָה מִשּׁוּם מַשְׁקֶה זָב וְזָבָה:

פרק ד - משנה ט

הַמְמַלֵּא בְקִילוֹן. כְּשֶׁהַמַּיִם מְכֻנָּסִים בְּמָקוֹם אֶחָד וּמְבַקְּשִׁים לַהֲבִיאָן אֶל שָׂדֶה אַחֵר, עוֹשִׂים לָהֶן דֶּרֶךְ שֶׁיָּבֹאוּ בוֹ אוֹתָן הַמַּיִם, וְשֵׁם אוֹתוֹ דֶרֶךְ שֶׁאוֹתָן הַמַּיִם נִמְשָׁכִים בּוֹ נִקְרָא קִילוֹן, וְעַד שְׁלֹשָׁה יָמִים אֵין אוֹתוֹ קִילוֹן מִתְנַגֵּב מִן הַמַּיִם שֶׁהִמְשִׁיכוּ לְתוֹכוֹ, וּמַכְשִׁירִין אֶת הַפֵּרוֹת שֶׁנָּפְלוּ לְתוֹכָן, שֶׁהֲרֵי הֵן לְרָצוֹן: רַבִּי עֲקִיבָא אוֹמֵר אִם נִגְּבוּ מִיָּד. רַבִּי עֲקִיבָא לֹא פָלֵיג אַתַּנָּא קַמָּא. דִּסְתָם קִילוֹן אֵינוֹ מִתְנַגֵּב עַד שְׁלֹשָׁה יָמִים, וְתַנָּא קַמָּא בִּסְתָם אַיְרֵי (יא). וְהָכִי מוּכָח בְּהֶדְיָא בַּתּוֹסֶפְתָּא, דְּמוֹדֶה רַבִּי עֲקִיבָא בִּסְתָם שֶׁהוּא שְׁלֹשָׁה יָמִים:

פרק ד - משנה י

שֶׁנָּפְלוּ עֲלֵיהֶם מַשְׁקִים. טְמֵאִים: אִם רַבּוּ. מֵי גְשָׁמִים עַל הַמַּשְׁקִים הָהֵם, בִּטְּלוּם וְטִהֲרוּם: הוֹצִיאָן כְּדֵי שֶׁיֵּרְדוּ עֲלֵיהֶם גְּשָׁמִים. דְּאַחְשְׁבִינְהוּ לַגְּשָׁמִים: אַף עַל פִּי שֶׁרַבּוּ טְמֵאִין. דְּנִטְמְאוּ הַגְּשָׁמִים עַצְמָן: בָּלְעוּ מַשְׁקִין טְמֵאִים. כְּגוֹן שֶׁהָעֵצִים נְגוּבִים מִבַּחוּץ וּבְתוֹכָן מַשְׁקִין בְּלוּעִין, אַף עַל פִּי שֶׁהוֹצִיאָן כְּדֵי שֶׁיֵּרְדוּ עֲלֵיהֶן גְּשָׁמִים דְּאַחְשְׁבִינְהוּ לַגְּשָׁמִים, אֲפִלּוּ הָכִי טְהוֹרִין, דְּלֹא נָגְעוּ בְמַשְׁקִין טְמֵאִין (יב), שֶׁהֲרֵי בָלְעוּ: לֹא יַסִּיקֵם אֶלָּא בְיָדַיִם טְהוֹרוֹת. שֶׁלֹּא יְטַמְּאוּ הַיָּדַיִם אֶת מֵי הַגְּשָׁמִים שֶׁהֻחְשְׁבוּ וְיַחְזְרוּ וִיטַמְּאוּ אֶת הַתַּנּוּר, דְּמַשְׁקֶה מְטַמֵּא כְלִי. אֲבָל בְּיָדַיִם טְהוֹרוֹת יָכוֹל לְהַסִּיקָם בַּתַּנּוּר, וְהַתַּנּוּר טָהוֹר, וְלֹא אָמְרִינַן דְּסוֹף מַשְׁקֶה טָמֵא הַבָּלוּעַ לָצֵאת מֵחֲמַת הַהֶסֵּק, דִּבְעֵצִים שֶׁהָאֵשׁ שׁוֹלֶטֶת בָּהֶן וְנֶאֱחָז בָּהֶן הָאוּר (יג) כָּלִין הַמַּשְׁקִים לְתוֹכָן, וְאֵין יוֹצְאִים לַחוּץ כְּשֶׁמַּסִּיקָן בַּתַּנּוּר: אִם הָיוּ לַחִין. עֵצִים הַלָּלוּ שֶׁבָּלְעוּ מַשְׁקִין טְמֵאִין, וּמֵחֲמַת לַחוּתָן יָצְאוּ מַשְׁקִין עַצְמָן מֵהֶן, וְאִם אוֹתָן מַשְׁקִים פָּלְטוּ מַשְׁקֶה הַטָּמֵא הַבָּלוּעַ, וְרַבּוּ מַשְׁקִין שֶׁמֵּעַצְמָן עַל מַשְׁקֶה הַבָּלוּעַ: טְהוֹרִין. שֶׁבִּטְּלוּ אוֹתָן. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן:

פרק ה - משנה א

מִי שֶׁטָּבַל בַּנָּהָר. וּמַיִם שֶׁעָלָיו מַכְשִׁירִין, שֶׁהֲרֵי לְרָצוֹן הֵן: וְהָיָה לְפָנָיו נָהָר אַחֵר וְעָבַר בּוֹ. וּמַיִם שֶׁעָלָיו מִן הַנָּהָר הַשֵּׁנִי אֵינָן לְרָצוֹן וְאֵינָן מַכְשִׁירִין: טִהֲרוּ. מֵי הַנָּהָר הַשֵּׁנִי, מַיִם שֶׁהֶעֱלָן עִמּוֹ מִן הַטְּבִילָה, וְאֵין מַכְשִׁירִין לֹא אֵלּוּ וְלֹא אֵלּוּ: דָּחָיוֹ. דָּחָהוּ: לְשָׁכְרוֹ. שֶׁהָיָה חֲבֵרוֹ שִׁכּוֹר וּמִתּוֹךְ שִׁכְרוּתוֹ דָּחָהוּ וְהִפִּילוֹ בַנָּהָר: וְכֵן לִבְהֶמְתּוֹ. שֶׁהִפִּיל אֶת הַבְּהֵמָה בְּשִׁכְרוּתוֹ בַּנָּהָר, וְהָיוּ עָלָיו מֵי טְבִילָה, וְעַל הַבְּהֵמָה מַיִם לְרָצוֹן: טִהֲרוּ שְׁנִיִּים. שֶׁהָיוּ עַל יְדֵי שִׁכְרוּת וְלֹא לְרָצוֹן: אֶת הָרִאשׁוֹנִים. שֶׁהָיוּ לְרָצוֹן: וְאִם כִּמְשַׂחֵק. דָּחָה אֶת חֲבֵרוֹ בַּנָּהָר אוֹ אֶת הַבְּהֵמָה, גַּם הַשְּׁנִיִּים בְּכִי יֻתַּן, דְּכֻלְּהוּ לְרָצוֹן נִינְהוּ:

פרק ה - משנה ב

הַנִּתָּזִין. שֶׁלֹּא בְכַוָּנָה: הָעוֹשֶׂה צִפּוֹר בַּמַּיִם. שֶׁמַּעֲלֶה עַל פְּנֵי הַמַּיִם רְתִיחוֹת כְּמִין אֲבַעְבּוּעוֹת: הַנִּתָּזִין. בַּעֲשִׂיַּת אֲבַעְבּוּעוֹת: וְאֶת שֶׁבָּהּ. דְּהַיְנוּ מֵי אֲבַעְבּוּעוֹת עַצְמָן. וְאִית דְּגָרְסֵי, הָעוֹשֶׂה צִנּוֹר בַּמַּיִם. כְּמוֹ שֶׁעוֹשִׂים הַנְּעָרִים כְּמִין צִנּוֹר שֶׁל קָנֶה אוֹ שֶׁל עֵץ חָלוּל, לְנַתֵּז מִמֶּנּוּ הַמַּיִם בִּצְחוֹק (ב): וְאֶת שֶׁבָּהּ. הַמַּיִם הַנִּשְׁאָרִים בַּצִּנּוֹר:

פרק ה - משנה ג

וּבְלָלָן שֶׁיְּנֻגְּבוּ. בָּלַל כָּל הַפֵּרוֹת בְּמַיִם, שֶׁכְּשֶׁיִּהְיוּ הַמַּיִם מְפֻזָּרִים בְּכָל הַפֵּרוֹת, יִתְנַגְּבוּ יוֹתֵר מְהֵרָה: וַחֲכָמִים אוֹמְרִים אֵינָן בְּכִי יֻתַּן. וַהֲלָכָה כַּחֲכָמִים:

פרק ה - משנה ד

לְעָמְקוֹ בְּכִי יֻתַּן וּלְרָחְבּוֹ אֵינוֹ בְכִי יֻתַּן. כְּשֶׁאָדָם מוֹדֵד עָמְקוֹ שֶׁל בּוֹר בְּקָנֶה, נִכָּר רֹשֶׁם הַמַּיִם בַּקָּנֶה עַד כַּמָּה הֵן מַגִּיעִים, וּמִתּוֹךְ כָּךְ יוֹדֵעַ עָמְקוֹ, לְפִיכָךְ רוֹצֶה הוּא בַּמַּשְׁקֶה שֶׁבָּרֹשֶׁם (ד). אֲבָל לְרָחְבּוֹ אֵינוֹ רוֹצֶה בְּמַשְׁקֶה שֶׁבָּרֹשֶׁם. וַהֲלָכָה כְּרַבִּי עֲקִיבָא (ה), שֶׁחָזַר רַבִּי טַרְפוֹן לִהְיוֹת שׁוֹנֶה כִדְבָרָיו:

פרק ה - משנה ה

לֵידַע כַּמָּה מַיִם יֶשׁ בּוֹ. רוֹצֶה בַּמַּשְׁקֶה שֶׁבָּרֹשֶׁם, כְּדַאֲמַרַן, לְפִיכָךְ הָוֵי בְּכִי יֻתַּן. אֲבָל לֵידַע אִם יֶשׁ בּוֹ מַיִם, אֵין צָרִיךְ רֹשֶׁם שֶׁל מַשְׁקֶה (ו): וְשֶׁבָּאֶבֶן. כְּלוֹמַר, שֶׁבָּאֶבֶן נַמִּי טְהוֹרִים, שֶׁאֵין מַכְשִׁירִים (ז):

פרק ה - משנה ו

הַחוֹבֵט עַל הַשֶּׁלַח. מַכֶּה בָעוֹר אַחַר רְחִיצָתוֹ כְּדֵי לְנַעֲרוֹ מִן הַמַּיִם: הַשֶּׁלַח. הָעוֹר כְּמוֹת שֶׁהוּא כְּשֶׁמַּפְשִׁיטִים אוֹתוֹ מִן הַבְּהֵמָה [בְּרֵאשִׁית לז] נִקְרָא שֶׁלַח. תַּרְגּוּם וַיַּפְשִׁיטוּ, וְאַשְׁלָחוּ: שֶׁהוּא מִתְכַּוֵּן שֶׁיֵּצְאוּ עִם הַצּוֹאָה. מִתְכַּוֵּן שֶׁיֵּצְאוּ הַמַּיִם עִם הַלִּכְלוּךְ שֶׁנִּתְחַבֵּר בָּהֶן, לְפִיכָךְ מַכְשִׁיר מַה שֶּׁנִּתָּז מִמֶּנּוּ וַאֲפִלּוּ שֶׁהוּא בְתוֹךְ הַמַּיִם. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי:

פרק ה - משנה ז

הֶעֵקֶל. מָקוֹם בְּשׁוּלֵי הַסְּפִינָה שֶׁמִּתְקַבְּצִים שָׁם הַמַּיִם הַנִּכְנָסִים בַּסְּדָקִים שֶׁל סְפִינָה: אִם נִעֵר. לְהָסִיר הַמַּיִם שֶׁעֲלֵיהֶן. כְּמוֹ הַמְנַעֵר טַלִּיתוֹ: לְצָרְפָהּ. לְבָדְקָהּ אִם הַמַּיִם נִכְנָסִים לְתוֹכָהּ אִם לָאו. פֵּרוּשׁ אַחֵר, לְחַזְּקָהּ וּלְחָסְמָהּ. לְפִי שֶׁכְּשֶׁהִיא בַיַּבָּשָׁה, נִבְקַעַת מֵאֵלֶיהָ, וּמַכְנִיסִים אוֹתָהּ בַּיָּם כְּדֵי שֶׁתִּשְׁרֶה בַמַּיִם וְיִסָּתְמוּ בִקּוּעֶיהָ וּסְדָקֶיהָ: מַסְמֵר לַגְּשָׁמִים לְצָרְפוֹ. לְחַזְּקוֹ וּלְחָסְמוֹ, כְּשֶׁיָּבֹאוּ עָלָיו מַיִם לְאַחַר שֶׁנִּתְלַבֵּן בָּאֵשׁ: אוּד. עֵץ שֶׁמִּקְצָתוֹ אָכְלָה הָאֵשׁ:

פרק ה - משנה ח

קַסְיָא שֶׁל שֻׁלְחָנוֹת. כִּסּוּי שֶׁמְּכַסִּין עַל הַשֻּׁלְחָן שֶׁלֹּא יִפֹּל עַל הַמָּזוֹן שֶׁעָלָיו מַיִם אוֹ דָבָר שֶׁל טִנּוּף וְלִכְלוּךְ: וְהַשִּׁיפָא שֶׁל לְבֵנִים. מַחְצֶלֶת שֶׁל שִׁיפָא וְשֶׁל גֶּמִי שֶׁמְּכַסִּים עַל הַלְּבֵנִים לְהָגֵן מִן הַגְּשָׁמִים. וַהֲרֵי אֵלּוּ כְּמוֹ בִשְׁבִיל שֶׁלֹּא יִלְקֶה הַכֹּתֶל שֶׁאֵינָן בְּכִי יֻתַּן: אֲבָל אִם נִעֵר. שֶׁמִּתְכַּוֵּן לַהֲדִיחָם, הֲרֵי זֶה בְּכִי יֻתַּן:

פרק ה - משנה ט

כָּל הַנִּצּוֹק טָהוֹר. אִם מְעָרֶה מִכְּלִי טָהוֹר לִכְלִי טָמֵא, מַה שֶּׁבָּעֶלְיוֹן טָהוֹר, דְּנִצּוֹק אֵינוֹ חִבּוּר. חוּץ מֵהָנֵי דַּחֲשִׁיב, מִפְּנֵי שֶׁסּוֹלְדִים לַאֲחוֹרֵיהֶם, כְּשֶׁהַקִּלּוּחַ נִפְסָק, חוֹזֵר הַנִּשְׁאָר לְמַעְלָה: מִדְּבַשׁ הַזִּיפִים. אִית דִּמְפָרְשֵׁי דְּבַשׁ טוֹב וְחָשׁוּב, שֶׁמִּתּוֹךְ חֲשִׁיבוּתוֹ מְזַיְּפִין אוֹתוֹ (יב) וּמְעָרְבִין בּוֹ פְסֹלֶת. וְאִית דְּאָמְרֵי, עַל שֵׁם מְקוֹמוֹ שֶׁבָּא מִזִּיף. כְּמוֹ בְּבֹא הַזִּיפִים וַיֹּאמְרוּ לְשָׁאוּל (תְּהִלִּים נד): וְהַצַּפָּחַת. מַאֲכָל עָשׂוּי מִקֶּמַח וּדְבַשׁ וְהוּא רַךְ כָּל כָּךְ שֶׁיְּכוֹלִים לְהַצִּיקוֹ. וְדוֹמֶה לוֹ כְּצַפִּיחִית בִּדְבַשׁ [שְׁמוֹת טז]: הַמִּקְפָּה. תַּבְשִׁיל שֶׁאֵינוֹ לֹא רַךְ וְלֹא קָשֶׁה אֶלָּא קָפוּי, קָרוּי מִקְפָּה: שֶׁל גְּרִיסִין. פּוֹלִים טְחוּנוֹת וּגְרוּסוֹת בְּרֵחַיִם: וְשֶׁל פּוֹלִים. שְׁלֵמוֹת שֶׁאֵינָן גְּרוּסוֹת בְּרֵחַיִם: מִפְּנֵי שֶׁהִיא סוֹלֶדֶת לַאֲחוֹרֶיהָ. כְּשֶׁהַקִּלּוּחַ נִפְסָק, חוֹזֵר אוֹתוֹ עַמּוּד הַשּׁוֹתֵת שֶׁהָיָה קְצָתוֹ בַמַּשְׁקִין טְמֵאִין שֶׁבַּכְּלִי הַתַּחְתּוֹן וְנִמְשָׁךְ לְמַעְלָה לַמַּשְׁקִין שֶׁבַּכְּלִי הָעֶלְיוֹן הַטָּהוֹר, וּמְטַמֵּא כָל מַה שֶּׁבַּכְּלִי הָעֶלְיוֹן:

פרק ה - משנה י

הַמְעָרֶה מֵחַם לְחַם. הָכִי נַמִּי אַיְרֵי שֶׁמַּשְׁקִין שֶׁבַּכְּלִי הָעֶלְיוֹן שֶׁמְּעָרֶה מִמֶּנּוּ, טְהוֹרִים. וְשֶׁבִּכְלִי תַחְתּוֹן שֶׁמְּעָרֶה לְתוֹכוֹ, הֵן טְמֵאִין: מִצּוֹנֵן לְחַם טָמֵא. שֶׁתַּחְתּוֹן הַחַם מַעֲלֶה הֶבֶל לָעֶלְיוֹן הַצּוֹנֵן וּמְטַמְּאוֹ (יד), שֶׁהֶעָנָן הָעוֹלֶה מִן הַחַם כְּמַשְׁקֶה הוּא חָשׁוּב: וְכֹחוֹ שֶׁל תַּחְתּוֹן יָפֶה. שֶׁחֻמּוֹ גָדוֹל מִשֶּׁל עֶלְיוֹן. וְאֵין הֲלָכָה כְּרַבִּי שִׁמְעוֹן, שֶׁכֵּיוָן שֶׁהָעֶלְיוֹן חַם, אַף עַל פִּי שֶׁהַתַּחְתּוֹן יוֹתֵר חַם מִמֶּנּוּ, אֵין פְּעֻלָּתוֹ שֶׁל תַּחְתּוֹן נִכֶּרֶת בָּעֶלְיוֹן וְאֵינוֹ מְטַמְּאוֹ:

פרק ה - משנה יא

וּמְגִיסָה בִקְדֵרָה. מְהַפֶּכֶת בְּכַף שֶׁבְּיָדָהּ לַתַּבְשִׁיל שֶׁבַּקְּדֵרָה: אִם הִזִּיעוּ יָדֶיהָ. מֵהֶבֶל הַקְּדֵרָה, נִטְמְאוּ יָדֶיהָ כְּאִלּוּ נָגְעָה בַמַּשְׁקֶה שֶׁבַּקְּדֵרָה: הָיוּ יָדֶיהָ טְמֵאוֹת. אִם הִזִּיעוּ יָדֶיהָ, נִטְמֵאת הַזֵּעָה מֵחֲמַת יָדֶיהָ, וְנִטְמֵאת הַקְּדֵרָה, לְפִי שֶׁהַהֶבֶל מְחַבֵּר בֵּין הַזֵּעָה שֶׁבְּיָדֶיהָ לַמַּשְׁקִין שֶׁבַּקְּדֵרָה (טו): רַבִּי יוֹסֵי אוֹמֵר אִם נָטְפוּ. אֵין הַקְּדֵרָה טְמֵאָה אֶלָּא אִם כֵּן נָטְפוּ טִפִּין מִן הַזֵּעָה שֶׁבְּיָדֶיהָ לְתוֹךְ הַקְּדֵרָה. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי: הַיַּיִן שֶׁבַּכַּף טָהוֹר. וְאֵינוֹ מַכְשִׁיר, דְּלֹא הֻחְשַׁב לִהְיוֹת מַשְׁקֶה עַד שֶׁיְּעָרֶה אוֹתוֹ לְתוֹךְ הַכְּלִי. וּכְמוֹ שֶׁאֵין לוֹ תּוֹרַת מַשְׁקֶה לְעִנְיַן הֶכְשֵׁר, כָּךְ אֵין לוֹ תּוֹרַת מַשְׁקֶה לְעִנְיַן טֻמְאָה וּלְעִנְיַן יֵין נֶסֶךְ:

פרק ו - משנה א

הַמַּעֲלֶה פֵרוֹתָיו לַגַּג מִפְּנֵי הַכְּנִימָה. תּוֹלָעִים הַגְּדֵלִים בַּפֵּרוֹת. לְשׁוֹן וַתְּהִי הַכִּנָּם (שְׁמוֹת ח): אִם נִתְכַּוֵּן לְכָךְ. שֶׁחָשַׁב שֶׁיֵּרֵד עֲלֵיהֶן טַל (א): שֶׁיֶּשׁ לָהֶן מַעֲשֶׂה. כְּגוֹן אִם הָיוּ שׁוֹפְכִין עֲלֵיהֶן טַל בְּכַוָּנָה, הָיוּ מֻכְשָׁרִין:

הַמַּעֲלֶה אֶת הָאֲגֻדּוֹת. שֶׁל יָרָק: וְאֶת הַקְּצִיעוֹת. כּוֹתְשִׁים הַתְּאֵנִים בְּמַכְתֶּשֶׁת וְעוֹשִׂים אוֹתָן כְּעֵין כִּכָּרוֹת גְּדוֹלוֹת, וְהֵן נִקְרָאוֹת קְצִיעוֹת: יַמְתִּינוּ. לְשׁוֹן לִחְלוּחַ (ב). וְדוֹמֶה לוֹ בַגְּמָרָא פֶּרֶק לֹא יַחְפֹּר [בָּבָא בַתְרָא דַּף יח] מִיתְנָא קָשֶׁה לַכְּתָלִים: אֵינָן בְּכִי יֻתַּן. אִם יָרַד עֲלֵיהֶן טַל: כָּל הָאֲגֻדּוֹת. שֶׁל יָרָק: שֶׁל בֵּית הַשְּׁוָקִים טְמֵאִין. לְפִי שֶׁהַמּוֹכְרִים רְגִילִים לְזַלֵּף עֲלֵיהֶן מַיִם כְּדֵי לְלַחְלְחָן, וְהֵן מֻכְשָׁרִין לְקַבֵּל טֻמְאָה, וְהַיָּדַיִם מְמַשְׁמְשִׁים בָּהֶן תָּמִיד, הִלְכָּךְ הֵן בְּחֶזְקַת טֻמְאָה: מִפְּנֵי מַשְׁקֵה הַפֶּה. אַף עַל פִּי שֶׁאֵין מְזַלְּפִין עֲלֵיהֶן מַיִם כְּשֶׁהֵן לַחִין, מִכָּל מָקוֹם רְגִילִים לָתֵת עֲלֵיהֶן מַשְׁקֶה הַיּוֹצֵא מִן הַפֶּה כְּדֵי לְהָסִיר הָאָבָק שֶׁעֲלֵיהֶן, וּבוֹ הֵן מֻכְשָׁרִין. וְאִיכָּא בֵּין תַּנָּא קַמָּא לְרַבִּי מֵאִיר מָקוֹם שֶׁאֵין רְגִילִין לְלַחְלְחָם בְּמַשְׁקֵה הַפֶּה, לְרַבִּי מֵאִיר טְהוֹרִים. וּלְתַנָּא קַמָּא בְּכָל עִנְיָן טְמֵאִים, דְּאַף בְּלַחִין נוֹתְנִין בָּהֶן מַיִם. וַהֲלָכָה כְּתַנָּא קַמָּא: הַקְּמָחִין וְהַסְּלָתוֹת שֶׁל בֵּית הַשְּׁוָקִים טְמֵאִין. לְפִי שֶׁלּוֹתְתִים הַחִטִּים שֶׁלָּהֶם קֹדֶם שֶׁיִּטְחֲנוּ אוֹתָן וְהֵן מֻכְשָׁרוֹת, וְהַיָּדַיִם מְמַשְׁמְשִׁים בָּהֶן תָּמִיד, לְפִיכָךְ הֵן בְּחֶזְקַת טֻמְאָה: הַחִילְקָה. חִטָּה כְתוּשָׁה וְנֶחְלַק כָּל גַּרְעִין לִשְׁנַיִם: טְרָגִיס. שֶׁנֶּחְלַק כָּל גַּרְעִין לִשְׁלֹשָׁה: טִסְגִי. שֶׁנֶּחֱלַק כָּל גַּרְעִין לְאַרְבָּעָה: בְּכָל מָקוֹם. אֲפִלּוּ שֶׁלֹּא בְבֵית הַשְּׁוָקִים, שֶׁהַכֹּל מְזַלְּפִין עֲלֵיהֶם מַיִם:

חוּץ מִשֶּׁל מוֹכְרֵי מַשְׁקֶה. מִפְּנֵי שֶׁנּוֹגְעִים בָּהֶם בְּעוֹד שֶׁמַּשְׁקֶה טוֹפֵחַ עַל יְדֵיהֶם, וְאֵין מַקְפִּידִין עַל הַלִּכְלוּךְ, שֶׁקְּלִפָּתָם מְגִנָּה עֲלֵיהֶן (ד): וְאִם הָיוּ מוֹכְרִים עִמָּהֶם פֵּרוֹת יְבֵשִׁים. מִתּוֹךְ שֶׁמְּנַגְּבִים יְדֵיהֶם כְּדֵי שֶׁלֹּא יְטַנְּפוּ אֶת הַפֵּרוֹת, מְנַגְּבִים יְדֵיהֶם נַמִּי קֹדֶם שֶׁיִּגְּעוּ בַבֵּיצִים: כָּל הַדָּגִים בְּחֶזְקַת טֻמְאָה. לְפִי שֶׁהֵם מֻכְשָׁרִים בַּמַּיִם, וְהַטָּמֵא וְהַטָּהוֹר מְמַשְׁמְשִׁים בָּהּ: חֲתִיכַת הָאִלְתּוֹת. מִין דָּג שֶׁרְגִילִין לְמָכְרוֹ חֲתִיכוֹת חֲתִיכוֹת. וּמָקוֹם יֵשׁ שֶׁשְּׁמוֹ אִלְתּוֹת וְשָׁם צָדִין אוֹתוֹ וּמוֹכְרִין אוֹתוֹ: וְדָג הַמִּצְרִי. דָּג קָטָן מְאֹד שֶׁמְּבִיאִין מִמֶּנּוּ קֻפּוֹת מְלֵאוֹת בְּמִצְרַיִם: וְקוּלְיָס הָאִסְפָּנִין. מִין דָּג שֶׁעוֹרוֹ רַךְ מְאֹד וְקוֹרִין לוֹ אלשבי"ט. וּשְׁלֹשָׁה מִינִים הַלָּלוּ אֵין הַמַּיִם מַכְשִׁירִין אוֹתָם כִּשְׁאָר דָּגִים, לְפִי שֶׁהֵן חַיִּין זְמַן מְרֻבֶּה אַחַר צֵידָתָן, וּלְאַחַר שֶׁהֵן נַעֲשִׂין אֹכֶל דְּהַיְנוּ לְאַחַר מִיתָתָן אֵין נוֹתְנִין עֲלֵיהֶם מַיִם, לְפִי שֶׁהַמַּיִם מַפְסִידִין אוֹתָן. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: כָּל הַצִּיר בְּחֶזְקַת טֻמְאָה. שֶׁסְּתָם צִיר נָתְנוּ בוֹ מַיִם (ו), וְהַיָּדַיִם מְמַשְׁמְשִׁים בּוֹ: וְעַל כֻּלָּם. בֵּין בֵּיצִים בֵּין פֵּרוֹת בֵּין צִיר (ז): נֶאֱמָן. עַם הָאָרֶץ לוֹמַר לֹא הֻכְשְׁרוּ: מִפְּנֵי שֶׁמַּפְקִידִים אוֹתָן. דְּכָל הַפֵּרוֹת מַפְקִידִים אוֹתָן אֵצֶל עַם הָאָרֶץ, וְאִי לָאו דְּנֶאֱמָן הֵיאַךְ מַפְקִידִים. אֲבָל דָּגָה אֵין מַפְקִידִים בְּיָדוֹ, לְפִיכָךְ אֵינוֹ נֶאֱמָן: צִיר טָהוֹר. שֶׁיָּדוּעַ בּוֹ שֶׁלֹּא הֻכְשַׁר עַל יְדֵי מַשְׁקֶה (ח): שֶׁנָּפַל לְתוֹכוֹ מַיִם כָּל שֶׁהוּ טָמֵא. דְּמַיִם כָּל שֶׁהוּ מַכְשִׁירִים אוֹתוֹ, וּמִשֶּׁהֻכְשַׁר חֶזְקָתוֹ טָמֵא, שֶׁהַיָּדַיִם מְמַשְׁמְשִׁים בּוֹ. וַהֲלָכָה כְּרַבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב:

בֵּין גְּדוֹלִים. מַשְׁקִין הַיּוֹצְאִין מִפִּי טַבַּעַת (יב): בֵּין קְטַנִּים. מֵי רַגְלַיִם: וְדַם הַקָּזָה לִשְׁתִיָּה. דַּם הַהַקָּזָה שֶׁנִּתְכַּוֵּן עָלָיו לְהַשְׁקוֹתוֹ לִבְהֵמָה וּלְחַיָּה אוֹ לְנָכְרִי: מֵי הֶחָלָב. הוּא הַקְּרוּם הַנִּפְרָשׁ מִן הַגְּבִינָה כְּשֶׁמַּעֲמִידִים הֶחָלָב וּמַקְפִּיאִין אוֹתוֹ בַּקֵּבָה, מַכְשִׁיר אֶת הַזְּרָעִים: וְהַמֹּחַל כַּשֶּׁמֶן. מַיִם שְׁחוֹרִים הַיּוֹצְאִים מִן הַזֵּיתִים: שֶׁאֵין מֹחַל יוֹצֵא מִידֵי שֶׁמֶן. וּמֹחַל עַצְמוֹ, לְרַבִּי שִׁמְעוֹן מֵי פֵרוֹת הוּא וְאֵינוֹ חָשׁוּב מַשְׁקֶה, אֲבָל מַכְשִׁיר מִפְּנֵי שְׁיוּרֵי שֶׁמֶן שֶׁבּוֹ: רַבִּי מֵאִיר אוֹמֵר אַף עַל פִּי שֶׁאֵין בּוֹ שָׁמֶן. מֹחַל עַצְמוֹ חָשׁוּב מַשְׁקֶה וּמַכְשִׁיר וּמְטַמֵּא. וַהֲלָכָה כְּרַבִּי מֵאִיר (טז): דַּם הַשֶּׁרֶץ כִּבְשָׂרוֹ. וּמִצְטָרְפִין לְכָעֲדָשָׁה. דִּכְתִיב בִּשְׁמֹנָה שְׁרָצִים (וַיִּקְרָא יא) וְזֶה לָכֶם הַטָּמֵא, לְהָבִיא דָמוֹ שֶׁמְּטַמֵּא כִבְשָׂרוֹ. יָכוֹל יַכְשִׁיר אֶת הַזְּרָעִים, תַּלְמוּד לוֹמַר הַטָּמֵא, טָמֵא הוּא וְאֵינוֹ מַכְשִׁיר אֶת הַזְּרָעִים:

פרק ו - משנה ו

וְאֵלּוּ מְטַמְּאִים וּמַכְשִׁירִים. שֶׁהֵן עַצְמָן מַשְׁקִין טְמֵאִים, וּמְטַמְּאִים וּמַכְשִׁירִים כְּאֶחָד (יז): זוֹבוֹ שֶׁל זָב. דִּכְתִיב (שָׁם טו) זוֹבוֹ טָמֵא הוּא: וְרֻקּוֹ. וְכִי יָרֹק וְגוֹ': וּמֵימֵי רַגְלָיו. וְזֹאת תִּהְיֶה טֻמְאָתוֹ, וְזֹאת, לְרַבּוֹת מֵימֵי רַגְלָיו: וּרְבִיעִית דָּם מִן הַמֵּת. דִּכְתִיב (בְּמִדְבַּר יט) כָּל הַנּוֹגֵעַ בְּמֵת בְּנֶפֶשׁ הָאָדָם, וְנֶפֶשׁ הָאָדָם הַיְנוּ רְבִיעִית דָּם, דִּכְתִיב (וַיִּקְרָא יז) כִּי הַדָּם הוּא הַנֶּפֶשׁ: וְדַם הַנִּדָּה. דִּכְתִיב (שָׁם טו) וְהַדָּוָה בְּנִדָּתָהּ, מַה שֶּׁזָּב מִמֶּנָּה טָמֵא כְמוֹתָהּ: שִׁכְבַת זֶרַע אֵינָהּ מַכְשֶׁרֶת. שֶׁאֵינָהּ חֲשׁוּבָה מַשְׁקֶה, דְּיֵשׁ לָהּ שֵׁם לְוַי: דַּם הַנִּדָּה אֵינוֹ מַכְשִׁיר. מִידִי דַּהֲוֵי אַדָּם שֶׁל הַתַּחְתּוֹנִיּוֹת שֶׁגַּם הוּא אֵינוֹ מַכְשִׁיר: דַּם הַמֵּת אֵינוֹ מַכְשִׁיר. דְּכֵיוָן שֶׁמֵּת יָצָא דָמוֹ מִתּוֹרַת מַשְׁקֶה. וְלֵית הֲלָכָה כְּחַד מֵהָנָךְ תַּנָּאֵי, אֶלָּא כֻלְּהוּ מְטַמְּאִים וּמַכְשִׁירִים כְּתַנָּא קַמָּא:

פרק ו - משנה ז

וְאֵלּוּ לֹא מְטַמְּאִים. אֲפִלּוּ יָצְאוּ מִן הַטָּמֵא, כְּגוֹן מִן הַזָּב וְכַיּוֹצֵא בוֹ. דִּכְתִיב בְּזוֹבוֹ טָמֵא הוּא, הוּא טָמֵא וְאֵין זֵעָה וְלֵחָה סְרוּחָה וּרְעִי שֶׁלּוֹ טְמֵאִים: וְהַלֵּחָה סְרוּחָה. הַיּוֹצֵאת מִן הַמֻּרְסָה. וְהוּא הַדִּין בְּרוֹקֵק לֵחָה סְרוּחָה: וְהָרְאִי. צוֹאָה. כְּמוֹ וְהָרְעִי: וְדָם הַיּוֹצֵא עִמָּהֶם. עִם הַלֵּחָה וְהָרְעִי: וּמַשְׁקֶה בֶן שְׁמֹנָה. כָּל מַשְׁקֶה שֶׁיָּצָא מִנֵּפֶל שֶׁנּוֹלַד בֶּן שְׁמֹנָה חֳדָשִׁים, כְּגוֹן דָּמוֹ וְרֻקּוֹ וּמֵימֵי רַגְלָיו. לְפִי שֶׁבֶּן שְׁמֹנָה חַי הֲרֵי הוּא כְאֶבֶן: חוּץ מִדָּמוֹ. שֶׁמְּטַמֵּא כְדָם שֶׁל בֶּן קְיָמָא. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי: אַף עַל פִּי שֶׁיּוֹצְאִין נְקִיִּים. שֶׁמּוֹצִיא אוֹתָן דֶּרֶךְ בֵּית הָרְעִי צְלוּלִים כְּמוֹ שֶׁהָיוּ כְּשֶׁשְּׁתָאָן, אֲפִלּוּ הָכִי אֵין מַכְשִׁירִין, הוֹאִיל וְיָצְאוּ דֶרֶךְ בֵּית הָרְעִי: רַבִּי אֱלִיעֶזֶר מְטַמֵּא בָאֵלּוּ. דְּמַשְׁקִין הֵן חֲשׁוּבִין, וּמְטַמְּאִין וּמַכְשִׁירִין. וְאֵין הֲלָכָה כְּרַבִּי אֱלִיעֶזֶר: חֲלֵב הַזָּכָר טָהוֹר. דְּלָא חֲשִׁיב מַשְׁקֶה (כב). וְכֵן הֲלָכָה:

פרק ו - משנה ח

חֲלֵב הָאִשָּׁה מְטַמֵּא. מַכְשִׁיר לְקַבֵּל טֻמְאָה. וְאִם נִטְמָא אוֹ שֶׁהָאִשָּׁה טְמֵאָה, מְטַמֵּא כְדִין מַשְׁקִין טְמֵאִים, דְּתוֹרַת מַשְׁקֶה יֶשׁ בּוֹ: לְרָצוֹן. הַנֶּחְלָב בְּכַוָּנָה (כג): וְשֶׁלֹּא לְרָצוֹן. כְּגוֹן חָלָב הַנּוֹטֵף מִדַּדֶּיהָ: שֶׁאֵינוֹ מְיֻחָד אֶלָּא לִקְטַנִּים. וּסְתָמֵיהּ לָא חֲשִׁיב כֻּלֵּי הַאי: שֶׁדַּם מַגֵּפָתָהּ טָמֵא. דְּהַיּוֹצֵא מִמַּכָּתָהּ טָמֵא, דְּדָמֵי לְדַם חֲלָלִים שֶׁקָּרוּי מַשְׁקֶה, דְּמַה לִּי קַטְלַהּ כֻּלַּהּ, מַה לִּי קַטְלַהּ פַּלְגָּא: שֶׁדַּם מַגֵּפָתָהּ טָהוֹר. דְּדַם בְּהֵמָה לֹא אִקְּרֵי דַם חֲלָלִים, דְּגַבֵּי אָדָם כְּתִיב: סַלֵּי זֵיתִים וַעֲנָבִים יוֹכִיחוּ. שֶׁהַסּוֹחֲטָן לִרְפוּאָה טָמֵא, וְאַף עַל פִּי כֵן לְרָצוֹן טְמֵאִים, שֶׁלֹּא לְרָצוֹן טְהוֹרִים. אַף הֶחָלָב, אַף עַל פִּי שֶׁהַחוֹלֵב לִרְפוּאָה טָמֵא, כְּשֶׁנֶּחְלַב שֶׁלֹּא לְרָצוֹן דִּין הוּא שֶׁיִּהְיֶה טָהוֹר: וְשֶׁלֹּא לְרָצוֹן טְהוֹרִים. שֶׁאֵין הַיּוֹצֵא מִן הַזֵּיתִים וַעֲנָבִים חָשׁוּב מַשְׁקֶה עַד שֶׁיֵּצֵא לְרָצוֹן:
