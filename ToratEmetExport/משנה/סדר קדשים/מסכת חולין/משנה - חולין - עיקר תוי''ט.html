(א) דְּוַדַּאי לֹא אָתָא לְאַשְׁמְעִינַן אֶלָּא אַמַּאי דִמְרַבִּינַן. דִּבְמִי דִפְשִׁיטָא לָן שֶׁשּׁוֹחֵט, מַאי אָתֵי לְמֵימְרָא, קְרָא כְּתִיב וְזָבַחְתָּ: (ב) הַנְהוּ דְכָתַב בְּרֵישׁ פֶּרֶק ב' דְּהִלְכָתָא גְּמִירֵי לְהוּ: (ג) לְשׁוֹן עֲיֵפוּת וְחַלִּישׁוּת הַלֵּב, שֶׁאֵין יָכוֹל לִרְאוֹת מַכַּת חֶרֶב וְסַכִּין אֲפִלּוּ בִּבְהֵמָה. כְּמוֹ וְתִתְעַלַּפְנָה הַבְּתוּלֹת [עָמוֹס ח]. רַשִׁ"י: (ד) וּמִיהוּ אַדִּידְהוּ נַמִּי קָאֵי, דְּדַוְקָא בְּדִיעֲבַד מַכְשִׁירִין שְׁחִיטָתָן, אֲבָל לְכַתְּחִלָּה לֹא אֲפִלּוּ בַאֲחֵרִים עוֹמְדִים עַל גַּבֵּיהֶם. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (ה) רַשִׁ"י. וּמְסַיֵּם הָרַ"ן, שֶׁאֵין שְׁחִיטָה אֶלָּא בְּיִשְׂרָאֵל. אֲבָל נָכְרִי כֵּיוָן דְּלָאו בְּתוֹרַת שְׁחִיטָה הוּא, כְּשֶׁשְּׁחָטָהּ הַוְיָא לָהּ כְּמֵתָה מֵאֵלֶיהָ. הָרַ"ן. וּבְבֵית יוֹסֵף בְּשֵׁם רִ"י, מִשּׁוּם דִּכְתִיב וְזָבַחְתָּ וְאָכַלְתָּ אוֹתוֹ, שֶׁהוּא בַּר זְבִיחָה אֱכֹל מִזִּבְחוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (ו) כְּלוֹמַר כְּתִקְרֹבֶת עֲבוֹדָה זָרָה דַּאֲסוּרָה בַהֲנָאָה, כְּדִתְנַן בְּמִשְׁנָה ג' פֶּרֶק ב' דַּעֲבוֹדָה זָרָה: (ז) וְזֶה לְשׁוֹן הָרַמְבַּ"ם, לְפִי שֶׁעוֹבְדֵי עֲבוֹדָה זָרָה חֲלוּקִים לִשְׁנֵי חֲלָקִים, הָאֶחָד מֵהֶן הַיּוֹדְעִין לַעֲשׂוֹת אוֹתָהּ, רְצוֹנוֹ לוֹמַר לְהַבִּיט אֶל הַמַּזָּלוֹת הַצּוֹמְחִים לְצֹרֶךְ מְלַאכְתָּם, וּלְהוֹרִיד הָרוּחָנִיּוּת בָּהֶן וּשְׁאָר אוֹתָן שְׁטֻיּוֹת הַמְטַנְּפִים הַשֵּׂכֶל כוּ'. וְהַחֵלֶק הַשֵּׁנִי, הֵן הָעוֹבְדִים לְאוֹתָן הַצּוּרוֹת הַיְדוּעוֹת כְּפִי מַה שֶּׁלָּמְדוּ מֵחַכְמֵיהֶם בִּלְבָד. וְכֵן הֵם רֹב עוֹבְדֵי עֲבוֹדָה זָרָה. וְהַחֵלֶק הַזֶּה הָאַחֲרוֹן עָלָיו אָמְרוּ חֲכָמִים זַ"ל בַּלָּשׁוֹן הַזֶּה, נָכְרִים שֶׁבְּחוּץ לָאָרֶץ לָאו עוֹבְדֵי עֲבוֹדָה זָרָה אֶלָּא מַעֲשֵׂה אֲבוֹתֵיהֶם בִּידֵיהֶם. הָרַמְבַּ"ם: (ח) וּכְרַבִּי יְהוּדָה בֶּן בְּתֵירָא דְּתַנְיָא כוּ' מִנַיִן לְתִקְרֹבֶת עֲבוֹדָה זָרָה שֶׁהִיא מְטַמְּאָה בְאֹהֶל, שֶׁנֶאֱמַר [תְּהִלִּים קו] וַיִּצָּמְדוּ לְבַעַל פְּעוֹר וַיֹּאכְלוּ זִבְחֵי מֵתִים, מַה מֵּת מְטַמֵּא בְאֹהֶל אַף תִּקְרֹבֶת עֲבוֹדָה זָרָה מְטַמְּאָה בְאֹהֶל. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (ט) דְּאִלּוּ בְּמֵזִיד, לֹא יֵאָכֵל עוֹלָמִית. כִּדְאַשְׁכְּחָן בִּמְבַשֵּׁל בְּשַׁבָּת בְּמֵזִיד בְּמִשְׁנָה ג' פֶּרֶק ב' דִּתְרוּמוֹת, וּבְפֵרוּשׁ הָרַ"ב שָׁם, דְּלַאֲחֵרִים מֻתָּר לְמוֹצָאֵי שַׁבָּת. וְכֵן כָּתְבוּ הָרַמְבַּ"ם וְהַטּוּר. וְלִכְאוֹרָה בִּשְׁחִיטָה נַמִּי דִּינָא הָכִי. וּמִכָּל מָקוֹם מַתְנִיתִין לֹא מִתּוֹקְמָא בְּמֵזִיד, דִּשְׁחִיטָתוֹ כְשֵׁרָה קָתָנֵי לֹא שְׁנָא לוֹ וְלֹא שְׁנָא לַאֲחֵרִים כוּ'. וְכִי תֵימָא אַמַּאי כָּשֵׁר לַאֲחֵרִים הָא הֲוֵי מְחַלֵּל שַׁבָּת, דְּפֵרֵשׁ הָרַ"ב בַּמִּשְׁנָה דִלְקַמָּן דִּשְׁחִיטָתוֹ פְסוּלָה, יֵשׁ לוֹמַר דִּלְקַמָּן מַיְרֵי בְּפַרְהֶסְיָא וְהָכָא בְּצִנְעָא. וְעוֹד יֵשׁ לוֹמַר דְּמִשּׁוּם פַּעַם אַחַת לָא חֲשִׁיב מוּמָר. תּוֹסָפוֹת. אֲבָל בְּהָרַמְבַּ"ם וְהַטּוּר מַשְׁמַע דִּבְמֵזִיד אָסוּר אַף לַאֲחֵרִים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (י) גְּמָרָא, מַאי טַעֲמָא, כֵּיוָן דְּאִי אֶפְשָׁר לְכַזַּיִת בָּשָׂר בְּלֹא שְׁחִיטָה, אַדַּעְתָּא דְחוֹלֶה קָא שָׁחַט. פֵּרֵשׁ רַשִׁ"י, וְלֵיכָּא לְמִגְזַר שֶׁמָּא יַרְבֶּה בַּשְּׁחִיטָה בִּשְׁבִיל בָּרִיא, דְּהָא מִשּׁוּם הַהוּא זַיִת דְּחוֹלֶה בָּעֵי לְמִשְׁחָט כֻּלָּהּ: (יא) בְּנַפְשׁוֹ. בְּשַׁבָּת סְקִילָה. וּבְיוֹם הַכִּפּוּרִים כָּרֵת. הָרַ"ן:

(יב) מִשּׁוּם צוֹר אִצְטְרִיךְ לְאוֹקְמֵיהּ הָכִי. אֲבָל קָנֶה הֲוָה מָצֵי לְמֵימַר דְּמַיְרֵי בְּקָנֶה שֶׁאֵינוֹ גָדֵל בַּאֲגָם כְּדִלְקַמָּן. תּוֹסָפוֹת: (יג) בַּגְּמָרָא יָלְפִינַן מִקְּרָא דִּשְׁחִיטָה בְּתָלוּשׁ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (יד) שֶׁאָמְרוּ מוּמָר לַעֲבוֹדָה זָרָה מוּמָר לְכָל הַתּוֹרָה כֻלָּהּ. הָרַמְבַּ"ם: (טו) לְפִי שֶׁעִקָּר בְּיָדֵינוּ גַּם כֵּן הַמְחַלֵּל שַׁבָּת בְּפַרְהֶסְיָא הֲרֵי הוּא כְּעוֹבֵד עֲבוֹדָה זָרָה לְכָל דְּבָרָיו. הָרַמְבַּ"ם: (טז) דִּשְׁחִיטָתוֹ פְסוּלָה, כְּדִתְנַן פֶּרֶק ב' מִשְׁנָה ח': (יז) לְשׁוֹן רַשִׁ"י אַף עַל גַּב דִּתְנַן בְּפֶרֶק ב' מִשְׁנָה ט' אֵין שׁוֹחֲטִין לְתוֹךְ יַמִּים, בְּרֹאשׁ הַסְּפִינָה שָׁרֵי דְּמוֹכְחָא מִלְּתָא דִּלְנַקֵּר כוּ': (יח) דְּאִלּוּ אֶחָד לֹא מִקְרֵי מְחֻבָּר בְּמַה שֶּׁהוּא בַּלֶּחִי, דְּלֶחִי הֲוֵי כְּבֵית יָד. הָרַמְבַּ"ם: (יט) בִּמְחֻבֶּרֶת לַגּוּף הוּא. דְּאִלּוּ לַיָּד וְהַיָּד תְּלוּשָׁה מִן הַגּוּף, לֹא מִקְרֵי מְחֻבָּר. כְּמוֹ שֵׁן בַּלֶּחִי:

(כ) וְהָא דְקָרֵי לָהּ טַבַּעַת סְתָמָא, מִשּׁוּם דְּאֵין בְּכָל הַקָּנֶה טַבַּעַת גְּמוּרָה אֶלָּא הִיא. שֶׁאֵינָן שְׁלֵמוֹת, אֶלָּא רְצוּעַת בָּשָׂר מְהַלֶּכֶת לְאָרְכּוֹ שֶׁל קָנֶה וּמְחַבֶּרֶת רָאשֵׁי הַטַּבָּעוֹת. רַשִׁ"י: (כא) וְהַיְנוּ דִּבְסָמוּךְ פֵּרֵשׁ מְלֹא חוּט הַשַּׂעֲרָה. וּמַה שֶּׁחָסֵר בָּזֶה גִּלָּה בָזֶה:

הַכֻּלְיָא: (טז) וְהַצֶּלַע שֶׁכְּנֶגְדָּהּ מְחֻבֶּרֶת יָפֶה בַּחֲצִי חֻלְיָא הַקַּיֶּמֶת. רַשִׁ"י: (יז) אֶלָּא מָקוֹם שֶׁאֵין בּוֹ מִילַת. גְּמָרָא. וּפֵרֵשׁ רַשִׁ"י, מָקוֹם יֵשׁ בַּכֶּרֶס חָלָק חַיָּה אֱכֹל [זֹאת מִעוּט הוּא], שֶׁאֵינָהּ חַיָּה לֹא תֵיכוֹל. גְּמָרָא:

(כא) שֶׁעַל יְדֵי הַנֶּקֶב הוּא הוֹלֵךְ וּמוֹסִיף מַכְאוֹב וְהַחֹלִי רַב בְּגוּפָהּ. רַשִׁ"י: (כב) כִּי חָזִינַן שֶׁהַקְּרוּם שֶׁלָּהּ כָּוֻּץ. הָרַ"ן: (כג) מִשּׁוּם דְּבַגְּמָרָא מַיְתֵי עוּבְדָא שֶׁצָּמְקוּ הַרְבֵּה רֵאוֹת, תָּפַס הָרַ"ב גַּם כֵּן בִּלְשׁוֹן רַבִּים כִּלְשׁוֹן הַגְּמָרָא: (כד) דְּאִלּוּ עַל מִקְצָת הַשִּׁדְרָה וְכָל הָעוֹר קַיָּם מַתִּירִין. בֵּית יוֹסֵף. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

(כה) בָּעוֹף. אַף עַל גַּב דְּכָל טְרֵפוֹת דִּבְהֵמָה כְּנֶגְדָּן בָּעוֹף, הָנֵי אִצְטְרִיכֵי לֵיהּ לְמִתְנֵי. נְקוּבַת הַוֶּשֶׁט מִשּׁוּם פְּסוּקַת הַגַּרְגֶּרֶת. וְהַהִיא אִצְטְרִיךְ, דְּלֹא תֵימָא הוֹאִיל וְחַיּוּתוֹ שֶׁל עוֹף לֹא נָפֵישׁ, דְּהָא כָּשֵׁר בְּסִימָן אֶחָד, אֵימָא גַרְגֶּרֶת נַמִּי תִטָּרֵף בְּמַשֶּׁהוּ. וְהִכַּתָּהּ חֻלְדָּה, מִשּׁוּם דְּקָתָנֵי מְקוֹם שֶׁעוֹשֶׂה אוֹתָהּ טְרֵפָה דְּהַיְנוּ נְקִיבַת הַקְּרוּם, דְּלֹא תֵימָא תִטָּרֵף בְּנִשְׁבַּר הַעֶצֶם בְּלֹא נְקִיבַת קְרוּם. וְנִקְּבוּ הַדַּקִּין, מִשּׁוּם דְּקָבָעֵי לְמִתְנֵי סֵיפָא גַּבֵּי אֵלּוּ כְשֵׁרוֹת יָצְאוּ בְנֵי מֵעֶיהָ וְלֹא נִקְּבוּ, וְלֹא תֵימָא עוֹף שֶׁחַיּוּתוֹ מֻעָט תִּטָּרֵף בִּיצִיאָה לְחוּדָא. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (כו) וּמִיהוּ דַּוְקָא שֶׁנִּקַּב מֵעַצְמוֹ מֵחֲמַת חֹלִי. דְּאִי מֵחֲמַת קוֹץ וְכַיּוֹצֵא בוֹ וַדַּאי טְרֵפָה הִיא. דְּהַאי קוֹץ מֵהֵיכָא אָתָא, אִי דֶּרֶךְ חוּץ, הֲרֵי נִקְּבָה לֶחָלָל וּטְרֵפָה, דְּחָיְשִׁינַן שֶׁמָּא נִקַּב אֶחָד מִן הָאֵבָרִים הַפְּנִימִים שֶׁנְּקִיבָתָן בְּמַשֶּׁהוּ. וְאִי דֶּרֶךְ וֶשֶׁט עַל כָּרְחִין כֵּיוָן דְּקֻרְקְבָן מִבַּחוּץ אִינַקַּב, פְּנִימִי נַמִּי אִינַקַּב כוּ'. אֶלָּא וַדַּאי עַל יְדֵי חֹלִי קָאָמַר. הָרַ"ן:
