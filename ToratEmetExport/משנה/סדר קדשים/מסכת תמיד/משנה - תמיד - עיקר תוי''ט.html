(מְפָרֵשׁ הַמַּסֶּכְתָּא. וְאֵינוֹ פֵּרוּשׁ רַשִׁ"י): (ב) וְאַף עַל גַּב דִּקְרָא קָאָמַר וְהַחוֹנִים וְגוֹ' מֹשֶׁה וְאַהֲרֹן וּבָנָיו, הָכִי קָאָמַר וְהַחוֹנִים וְגוֹ' מֹשֶׁה, וַהֲדַר אַהֲרֹן וּבָנָיו. מִדִּכְתִיב וְהַחוֹנִים וּכְתִיב שׁוֹמְרֵי, חוֹנִים לְחוּד וְשׁוֹמְרִים לְחוּד. גְּמָרָא: (ג) הַנִּיצוֹץ. לֹא שָׁמַעְתִּי טַעַם לָמָּה נִקְרָא כֵן. מְפָרֵשׁ: (ד) וַאֲפִלּוּ הָיוּ אוֹתָן עֲלִיּוֹת מִקְצָתָן בָּעֲזָרָה, יְכוֹלִין כֹּהֲנִים לֵישֵׁב כְּשֶׁהָיוּ עֲיֵפִים, דְּגַגִּים וַעֲלִיּוֹת לֹא נִתְקַדְּשׁוּ. וּבָעֲזָרָה אֵין אָדָם יָכוֹל לֵישֵׁב, דְּאֵין יְשִׁיבָה בָעֲזָרָה אֶלָּא לְמַלְכֵי בֵּית דָּוִד. מְפָרֵשׁ: (ה) וְהֵם כֹּהֲנִים פְּחוּתִים מִשְּׁלֹשׁ עֶשְׂרֵה, וּלְפִי שֶׁעֲדַיִן לֹא הִגִּיעוּ לַעֲשׂוֹת עֲבוֹדָה עָבְדִינַן לְהוּ שׁוֹמְרִים, שֶׁאוֹתָן שֶׁיְּכוֹלִין לַעֲבֹד אֵינָן רוֹצִין לִשְׁמֹר. וְגַם אֵין מַנִּיחִין אוֹתָן לִשְׁמֹר. מְפָרֵשׁ: (ו) וּבִלְשׁוֹן אַשְׁכְּנַז גיוועל"ב: (ז) יְשֵׁנִים כוּ'. בְּחֶצְיָהּ שֶׁבָּחֹל: (ח) בְּיָדָם. כְּלוֹמַר, בִּידֵי אוֹתָם הַכֹּהֲנִים הַיְשֵׁנִים לְשָׁם הָיוּ מֻנָּחוֹת כָּל מַפְתְּחוֹת הָעֲזָרָה. וּמְפָרֵשׁ בְּמַסֶּכֶת מִדּוֹת סוֹף פֶּרֶק קַמָּא הֵיכָן הָיְתָה הַנָּחָתָן, מָקוֹם הָיָה שָׁם אַמָּה עַל אַמָּה. מְפָרֵשׁ: (ט) וּבַגְּמָרָא אָמְרִינַן, הָתָם קָרֵי לְהוּ רוֹבִים הָכָא קָרֵי לְהוּ פִרְחֵי כְהֻנָּה. אָמְרֵי, אִין, הָתָם דְּלֹא מְטוֹ לְמֶעֱבַד עֲבוֹדָה קָרֵי לְהוּ רוֹבִים, הָכָא דִמְטוֹ לְהוּ לְמֶעֱבַד עֲבוֹדָה קָרֵי לְהוּ פִּרְחֵי: (י) אֲבָל לֵהָנוֹת נִתְּנוּ אֲפִלּוּ שֶׁלֹּא בִּשְׁעַת עֲבוֹדָה. כְּמוֹ שֶׁכָּתַב הָרַ"ב בְּרֵישׁ פֶּרֶק ז' דְּיוֹמָא: (יא) וְטָבַל. וְהָא דִתְנַן בְּמַסֶּכֶת מְגִלָּה פֶּרֶק ב' מִשְׁנָה ד' לֹא מַזִּין וְלֹא טוֹבְלִין אֶלָּא בַיּוֹם, הַיְנוּ בְּחַיָּבֵי טְבִילָה שֶׁצְּרִיכִין סְפִירָה כְּמוֹ טְמֵא מֵת אוֹ זָב אוֹ מְצֹרָע כוּ'. וְנִדָּה וְיוֹלֶדֶת טְבִילָתָן בְּלֵיל שְׁמִינִי. מְפָרֵשׁ: (יב) עַיֵּן פֵּרוּשׁ הָרַ"ב בְּסוֹף פֶּרֶק קַמָּא דְּמִדּוֹת:

(יג) וְדוֹפֵק. מוֹרֶה עַל הַמַּקִּישׁ בַּפֶּתַח בְּנַחַת. כְּמוֹ קוֹל דּוֹדִי דוֹפֵק [שִׁיר הַשִּׁירִים ה]. הָרַמְבַּ"ם: (יד) וְיָפִיס. בְּכָאן בַּחֲצִי בֵּית הַמּוֹקֵד הַבָּנוּי בְּקֹדֶשׁ. דְּאֵין מַטִּילִין פַּיִס אֶלָּא בָעֲזָרָה בַּמָּקוֹם הַקָּדוֹשׁ, דִּכְתִיב בְּבֵית אֱלֹהִים נְהַלֵּךְ בְּרָגֶשׁ כוּ' [תְּהִלִּים נה]. וְאַף עַל גַּב דִּשְׁאָר פְּיָסוֹת הָיוּ בְּלִשְׁכַּת הַגָּזִית כוּ', הָתָם הַיְנוּ טַעֲמָא מִפְּנֵי שֶׁאוֹתָם הַכֹּהֲנִים שֶׁהָיוּ בָעֲזָרָה הָיוּ קְרוֹבִים לְלִשְׁכַּת הַגָּזִית וְהָתָם הָיוּ מְפַיְּסִים, אֲבָל הָכָא כֵּיוָן דְּהָיוּ בְּבֵית הַמּוֹקֵד שֶׁעֲדַיִן לֹא יָרְדוּ לָעֲזָרָה וְאִיכָּא נַמִּי בְּבֵית הַמּוֹקֵד מָקוֹם מְקֻדָּשׁ הָיוּ מְפַיְּסִים הָתָם, וּבָעֲזָרָה אִי אֶפְשָׁר לְפַיֵּס כוּ'. מְפָרֵשׁ. אֲבָל הָרַמְבַּ"ם כָּתַב, וְיָפִיסוּ בְּלִשְׁכַּת הַגָּזִית פַּיִס רִאשׁוֹן וְשֵׁנִי:

(טו) וְהַךְ סְתָמָא רַבִּי יְהוּדָה דְּמִדּוֹת פֶּרֶק קַמָּא מִשְׁנָה ז': (טז) בְּיָדָם. וּבְלֵילֵי שַׁבָּת וְיוֹם טוֹב כָּתַב הָרַמְבַּ"ם דְּאֵין בְּיָדָם אוּר, אֶלָּא בּוֹדְקִין בַּנֵּרוֹת הַדּוֹלְקִים שָׁם מֵעֶרֶב שַׁבָּת. וְאַף עַל גַּב דְּאֵין שְׁבוּת בַּמִּקְדָּשׁ, הָכָא גָזְרוּ, דְּלִשָּׂא אֲבוּקָה שֶׁל נֵר בְּיָדוֹ קָרוֹב לְוַדַּאי לְהַטּוֹת הַנֵּר שֶׁהוּא מַבְעִיר מַמָּשׁ: (יז) כְּלוֹמַר, רָחוֹק מִכָּתְלֵי הָעֲזָרָה. וְאָמְרִינַן בַּגְּמָרָא דְּשֶׁל בִּנְיָן הָיוּ, כְּלוֹמַר שֶׁל אֲבָנִים. דְּהָא כְתִיב לֹא תִטַּע לְךָ אֲשֵׁרָה כָּל עֵץ אֵצֶל מִזְבַּח ה' אֱלֹהֶיךָ [דְּבָרִים טז], וְהָכִי קָאָמַר, לֹא תִטַּע אֲשֵׁרָה וְכָל עֵץ אֵצֶל וְגוֹ': (יח) שֶׁבֵּית הַמּוֹקֵד הָיָה בַּצָּפוֹן, כְּדִתְנַן בְּפֶרֶק ג' מִשְׁנָה ג': (יט) שָׁלוֹם כוּ'. כֵּפֶל לָשׁוֹן. מְפָרֵשׁ: (כ) הֶעֱמִידוּ כוּ'. לָאו דַּחֲבִתִּין קְדִימֵי, שֶׁאֵין הַסֵּדֶר שָׁנוּי כָּךְ לְקַמָּן. אֶלָּא מַעֲמִיד אוֹתָם לְהָחֵם חַמִּין לִרְבוּכָה. גְּמָרָא:

(כא) הִזָּהֵר. אַף עַל פִּי שֶׁהוּא יוֹדֵעַ בַּדָּבָר צָרִיךְ לְהַזְהִירוֹ, שֶׁהוּא הוֹלֵךְ יְחִידִי וְשֶׁמָּא יִשְׁכַּח וְיִגַּע, וּמִתּוֹךְ שֶׁמַּזְהִירִין אוֹתוֹ נִזְהָר. מְפָרֵשׁ: (כב) עַד כוּ'. דְּלִינָה פּוֹסֶלֶת מִן קִדּוּשׁ יָדַיִם וְרַגְלַיִם, וְצָרִיךְ לַחֲזֹר וּלְקַדֵּשׁ. מְפָרֵשׁ: (כג) וַהֲרֵי כוּ'. כָּל זֶה הָיָה אוֹמֵר לוֹ, הֲרֵי הַמַּחְתָּה נְתוּנָה כוּ'. מְפָרֵשׁ: (כד) בְּמַעֲרָבוֹ. כְּלוֹמַר בַּאֲלַכְסוֹן [הִיא נְתוּנָה] מְמֻצַּעַת בֵּין כֶּבֶשׁ לַמִּזְבֵּחַ בְּזָוִית הַכֶּבֶשׁ הִיא נְתוּנָה אֵצֶל מָקוֹם שֶׁכָּלֶה הַכֶּבֶשׁ, דְּהַיְנוּ בֵּין הַכֶּבֶשׁ לַמִּזְבֵּחַ. מְפָרֵשׁ: (כה) אֵין כוּ'. שֶׁבֵּין אוּלָם לַמִּזְבֵּחַ אֵין אָדָם יָכוֹל לִכָּנֵס בָּהֶן אֶלָּא בִּשְׁעַת עֲבוֹדָה, וְכֶבֶשׁ הוּא כְּמִזְבֵּחַ לְכָל דָּבָר. מְפָרֵשׁ: (כו) וְלֹא נֵר כוּ'. לְפִי שֶׁהוֹלֵךְ יְחִידִי. מְפָרֵשׁ. וְנִרְאֶה לִי שֶׁרְצוֹנוֹ לוֹמַר דְּמִשּׁוּם שֶׁהוֹלֵךְ יְחִידִי הֲוָה אֲמֵינָא שֶׁיִּהְיֶה נֵר בְּיָדוֹ, קָא מַשְׁמַע לָן מַתְנִיתִין דְּלֹא: (כז) לֹא כוּ'. לְפִי שֶׁהָיוּ מִצַּד אַחֵר שֶׁל כֶּבֶשׁ בְּמִזְרָחוֹ, דְּהָא גֹּבַהּ הַכֶּבֶשׁ שֶׁהוּא גָבוֹהַּ הַרְבֵּה אַמּוֹת, הָיָה מַפְסִיק. מְפָרֵשׁ: (כח) וְלֹא כוּ'. מִפְּנֵי שֶׁרוֹצֶה לוֹמַר עַד שֶׁיִּהְיוּ שׁוֹמְעִין כוּ' תָּנֵי וְלֹא הָיוּ כוּ'. מְפָרֵשׁ: (כט) עֵת. לְקַדֵּשׁ. [עַיֵּן פ"ג מ"ב. בֵּית דָּוִד]. מְפָרֵשׁ: (ל) דַּאֲשֶׁר תֹּאכַל הָאֵשׁ כְּתִיב: (לא) לַצָּפוֹן. כְּלוֹמַר, לֹא הָיָה מְהַלֵּךְ כְּדַרְכּוֹ דֶּרֶךְ מִזְרָח וְיוֹצֵא חוּץ לָעֲזָרָה אֶלָּא הוֹפֵךְ פָּנָיו לִצְפוֹן הָעֲזָרָה דְּהַיְנוּ לִדְרוֹם הַמִּזְבֵּחַ: (לב) הָלַךְ כוּ'. וְאִם תֹּאמַר הָא בָעִינַן וְשָׂמוֹ אֵצֶל הַמִּזְבֵּחַ, וּנְהִי דְּכֶבֶשׁ כְּמִזְבֵּחַ לְכָל מִלֵּי, מִכָּל מָקוֹם הָיָה לָנוּ לְקָרְבוֹ אֵצֶל הַמִּזְבֵּחַ וַעֲדַיִן הוּא רָחוֹק מִן הַמִּזְבֵּחַ עֶשְׂרִים אַמָּה. וְאָמַר לִי רַבִּי, דְּמִוְּהִשְׁלִיךְ כוּ' דָּיְקִינַן דְּקִים לְהוּ לְרַבָּנָן דְּלֹא מִקְּרֵי הַשְׁלָכָה פָּחוֹת מֵעֶשְׂרִים אַמָּה, וְעוֹלַת הָעוֹף הִיא נַעֲשֵׂית עַל קֶרֶן דְּרוֹמִית מִזְרָחִית, הִלְכָּךְ מַנִּיחַ אֶת הַדֶּשֶׁן רָחוֹק מִן הַמִּזְבֵּחַ עֶשְׂרִים אַמָּה מְקוֹם שֶׁנּוֹתְנִים שָׁם מֻרְאָה וְנוֹצָה כוּ'. מְפָרֵשׁ: (לג) מֻרְאוֹת כוּ'. וְכָל אֵלּוּ נִבְלָעִים בִּמְקוֹמָן. אֲבָל תְּרוּמַת הַדֶּשֶׁן נִרְאֶה לִי דְּלֹא הָיְתָה נִבְלַעַת, כִּדְמוּכָח בְּיוֹמָא דַּף כ"א וּבִכְרֵתוֹת דַּף ו'. מְפָרֵשׁ. אֲבָל הַתּוֹסָפוֹת בְּיוֹמָא שָׁם וּבִזְבָחִים דַּף ס"ד כָּתְבוּ דְּאֵין רְאָיָה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:
