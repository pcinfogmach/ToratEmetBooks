פרק א - משנה א

(א) גֵּט. לְשׁוֹן אֲרַמִּי. סֵפֶר כְּרִיתוּת (דְּבָרִים כד) מְתֻרְגָּם גֵּט פִּטּוּרִין. וּלְכָךְ נָהֲגוּ לִכְתֹּב בּוֹ שְׁתֵּים עֶשְׂרֵה שׁוּרוֹת כְּמִנְיַן גֵּט. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (ב) וְכֵן פָּסַק רַשִׁ"י. וְהָרַ"ן פֵּרֵשׁ דְּחוּץ לָאָרֶץ שֶׁל צַד מַעֲרָבִי שֶׁל אֶרֶץ יִשְׂרָאֵל נָקַט, לְפִי שֶׁבְּאוֹתוֹ צַד לֵיכָּא עֲיָרוֹת מֻבְלָעוֹת וּסְמוּכוֹת שֶׁהֲרֵי הַיָּם מַפְסִיק, וּבִצְדָדִים אֲחֵרִים לֹא פְּסִיקָא לֵיהּ מִלְּתָא מִשּׁוּם דְּאִיכָּא עֲיָרוֹת סְמוּכוֹת וּמֻבְלָעוֹת (וּלְקַמָּן אָמְרִינַן דְּתַנָּא קַמָּא מַפִּיק אֵלּוּ מִכְּלָל). וּמִשּׁוּם הָכִי אָמַר מִמְּדִינַת הַיָּם שֶׁהִיא מַעֲרַב אֶרֶץ יִשְׂרָאֵל, וְהוּא הַדִּין לְכָל צְדָדִים אֲחֵרִים מֵאוֹתָן עֲיָרוֹת שֶׁהֵן דּוֹמוֹת לְאֵלּוּ שֶׁבִּמְדִינַת הַיָּם, שֶׁאֵינָן לֹא מֻבְלָעוֹת וְלֹא סְמוּכוֹת. וְאֶפְשָׁר שֶׁלָּזֶה כִּוֵּן רַשִׁ"י זִכְרוֹנוֹ לִבְרָכָה בְּמַה שֶּׁסִּיֵּם, כְּלוֹמַר כָּל חוּץ לָאָרֶץ שֶׁהִיא דוֹמָה לְחוּץ לָאָרֶץ שֶׁבִּמְדִינַת הַיָּם: (ג) וּקְרָא כְּתִיב וְכָתַב לָהּ, דְּהַיְנוּ לִשְׁמָהּ. וְאֵין בְּקִיאִין הַיְנוּ דְּאֵין חוֹשְׁשִׁין לְהַךְ דְּרָשָׁה, דְּאִי לָאו הָכִי מַאי שְׁנָא מִשְּׁאָר הִלְכוֹת גִּטִּין דְּלֹא חָיְשִׁינַן לְהוּ דְּלֹא בְּקִיאֵי. תּוֹסָפוֹת: (ד) וְרַשִׁ"י פֵּרֵשׁ דִּסְתָמָא כִּי מַסְהִיד בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם לִשְׁמָהּ קָמַסְהִיד. וְסִיֵּם הָרַ"ן דְּמֵימַר אָמַר הַאי שָׁלִיחַ לָמָּה לְהוּ לְרַבָּנָן לְמִשְׁאָל אִם נִכְתַּב וְנֶחְתַּם בְּפָנַי, שְׁמַע מִנַּהּ שֶׁיֵּשׁ כְּתִיבָה וַחֲתִימָה בְּגֵט שֶׁהִיא פְסוּלָה, וְאִלּוּ לֹא נִכְתַּב לִשְׁמָהּ הָיָה אוֹמֵר הַשָּׁלִיחַ כָּךְ אֵרַע מַעֲשֶׂה דִלְאַחֵר נִכְתַּב. וְהָא דְסַגִּי בְּעֵד אֶחָד, מִשּׁוּם דְּרֹב בְּקִיאִין הֵן. וַאֲפִלּוּ לְרַבִּי מֵאִיר דְּחָיֵישׁ לְמִעוּטָא, הָכָא הָוֵי מִעוּטָא דְמִעוּטָא, וְרַבָּנָן הוּא דְאַצְרוּךְ, וּמִשּׁוּם עִגּוּנָא הֵקֵלּוּ. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (ה) וּפָרֵיךְ בַּגְּמָרָא אִם כֵּן בְּפָנַי נִכְתַּב לָמָּה לִי, וּמְשָׁנֵי דִּלְמָא אָתֵי לְאַחְלוֹפֵי בְּקִיּוּם שְׁטָרוֹת דְּעָלְמָא בְּעֵד אֶחָד: (ו) דְּעֵדִים הַחֲתוּמִים עַל הַשְּׁטָר כְּמוֹ שֶׁנֶּחְקְרָה עֵדוּתָן בְּבֵית דִּין, וְרַבָּנָן הוּא דְאַצְרוּךְ, וּמִשּׁוּם עִגּוּנָא הֵקֵלּוּ. גְּמָרָא: (ז) הַחֶגֶר. בַּגְּמָרָא מְפָרֵשׁ דִּבְהָא פְלִיגֵי, דְתַנָּא קַמָּא סָבַר דְּמֵרֶקֶם וְחֶגֶר אַף עַל פִּי שֶׁהֵן מִחוּץ לָאָרֶץ אֵין צָרִיךְ שֶׁיֹּאמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם, לְפִי שֶׁהֵן עֲיָרוֹת הַסְּמוּכוֹת לְאֶרֶץ יִשְׂרָאֵל, וְכָל שֶׁכֵּן מִכְּפַר לוּדִים לְלוֹד שֶׁאֵין צָרִיךְ, לְפִי שֶׁהוּא מֻבְלָע בִּתְחוּם אֶרֶץ יִשְׂרָאֵל כוּ'. וְאָתֵי רַבָּן גַּמְלִיאֵל לָאַפְלוֹגֵי עֲלַהּ וּלְמֵימַר דְּדַוְקָא בְּמֻבְלָעוֹת כְּמוֹ מִכְּפַר לוּדִים גְּמִירֵי וּשְׁכִיחֵי, אֲבָל סְמוּכוֹת כְּגוֹן רֶקֶם לֹא גְמִירֵי וְלֹא שְׁכִיחֵי. וַאֲתָא רַבִּי אֱלִיעֶזֶר לְמֵימַר דַּאֲפִלּוּ מִכְּפַר לוּדִים שֶׁהוּא מֻבְלָע צָרִיךְ לוֹמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם, דִּנְהִי דְמוֹדֶה לְרַבָּנָן קַמָּאֵי דְמֻבְלָעִים גְּמִירֵי וּשְׁכִיחֵי, אֲפִלּוּ הָכִי מַצְרִיךְ בְּהוּ לוֹמַר בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתַּם, כְּדֵי שֶׁלֹּא תַחְלֹק בִּמְדִינַת הַיָּם. הָרַ"ן: (ח) מִכְּפַר כוּ'. מִקְרֵי כְּפַר לוּדִים, שֶׁבְּנֵי לוֹד מְצוּיִין בָּהּ תָּמִיד. וּמִשּׁוּם הָכִי נָקַט לְלוֹד לִרְבוּתָא. הָרַ"ן: (ט) וְהַמּוֹלִיךְ. וּבְהָא פְלִיגֵי אַתַּנָּא קַמָּא, דִּלְתַנָּא קַמָּא הַמּוֹלִיךְ מֵאֶרֶץ יִשְׂרָאֵל לְחוּץ לָאָרֶץ אֵין צָרִיךְ, לְפִי שֶׁבְּנֵי אֶרֶץ יִשְׂרָאֵל בְּקִיאִין לִשְׁמָהּ, וַחֲכָמִים גָּזְרוּ מוֹלִיךְ אַטּוּ מֵבִיא. וּלְמַאן דְּאָמַר טַעֲמָא מִשּׁוּם קִיּוּם, חֲכָמִים לִפְרוּשֵׁי מִלְּתָא דְּתַנָּא קַמָּא אָתוּ דְּמֵבִיא לָאו דַּוְקָא, אֶלָּא הוּא הַדִּין מוֹלִיךְ. גְּמָרָא: (י) בִּמְדִינַת הַיָּם. לְטַעֲמָא דְאֵין בְּקִיאִין הוּא הַדִּין בְּאוֹתָהּ מְדִינָה נַמִּי צָרִיךְ שֶׁיֹּאמַר, אֶלָּא דְּנָקַט מִמְּדִינָה לִמְדִינָה לְמֵידַק מִנַּהּ דְּהָא מִמְּדִינָה לִמְדִינָה בְּאֶרֶץ יִשְׂרָאֵל לֹא צָרִיךְ, וַאֲפִלּוּ לְכַתְּחִלָּה. וּלְאִידָךְ טַעֲמָא נַמִּי בְּאֶרֶץ יִשְׂרָאֵל דִּקְבִיעֵי בֵית דִּין שְׁכִיחֵי שַׁיָּרוֹת. גְּמָרָא: (יא) הֶגְמוֹנְיָא. פְּקִידוּת: (יב) גְּמָרָא. וּפֵרֵשׁ רַשִׁ"י שֶׁלֹּא יֵלֵךְ מִזּוֹ לְזוֹ. וּבְאֶרֶץ יִשְׂרָאֵל נַמִּי לְטַעֲמָא דְקִיּוּם, דְּמַקְפִּידִין. וּלְמַאן דְּאָמַר לִשְׁמָהּ נַמִּי, דְּאִיהוּ סְבִירָא לֵיהּ נַמִּי טַעֲמָא דְקִיּוּם, דְּאִלּוּ בְּחוּץ לָאָרֶץ אֲפִלּוּ בְּאוֹתָהּ הֶגְמוֹנְיָא כִדְאִיתָא בַּגְּמָרָא:

(יג) לַמִּזְרָח. אֲבָל מַעֲרַב אֶרֶץ יִשְׂרָאֵל אֵין צָרִיךְ גְּבוּל, שֶׁהַיָּם הַגָּדוֹל גְּבוּלָהּ, דִּכְתִיב וּגְבוּל יָם וְהָיָה לָכֶם הַיָּם הַגָּדוֹל [בְּמִדְבָּר לה]. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (יד) וְאַשְׁקְלוֹן. וּבְסֵפֶר יְהוֹשֻׁעַ מוּכָח דְּמֵאֶרֶץ יִשְׂרָאֵל הִיא. גַּם בַּסִּפְרִי חָשִׁיב פָּרָשַׁת אַשְׁקְלוֹן מִכִּבּוּשׁ עוֹלֵי בָּבֶל. וְהָרַ"ן תֵּרֵץ דְּפָרָשַׁת אַשְׁקְלוֹן אֵינוֹ אֶלָּא פָּרָשַׁת דְּרָכִים שֶׁלָּהּ, וְשֶׁכֵּן הוּא בַּיְרוּשַׁלְמִי:

(יד) וּלְשׁוֹן (טֹפֶס) [דְּפוּס] הוּא, שֶׁהוּא הֶעְתֵּק וְתֹאַר הַגֵּט. וְתוֹרֵף, הוּא גִלּוּי הַגֵּט, מִלְּשׁוֹן יַנִּיחֶנָּהּ בְּמָקוֹם הַתֻּרְפָּה: (טו) וְלֵיכָּא לְמֵימַר דְּדִיעֲבַד אֲפִלּוּ תוֹרֵף כָּשֵׁר, דְּהָא כֵּיוָן דִּמְחֻבָּר פְּסוּלָא דְאוֹרַיְתָא הוּא אֲפִלּוּ דִיעֲבַד אִית לָן לְמִפְסְלֵיהּ. הָרַ"ן: (טז) יֵשׁ לְעַיֵּן כֵּיוָן דִּכְתִיבַת זְמַן בַּגֵּט לֵיתָא אֶלָּא מִדְּרַבָּנָן, כָּל שֶׁשְּׁאָר הַתּוֹרֵף נִכְתַּב כְּהִלְכָתוֹ, זְמַן לֹא יִהְיֶה לוֹ אֶלָּא דִין טֹפֶס. וְאֶפְשָׁר דְּאַלְּמוּהוּ כְּאִלּוּ הָיָה שֶׁל תּוֹרָה. בֵּית יוֹסֵף: (יז) וְיַעֲבִיר תְּנַאי שֶׁהָיָה בוֹ. רַשִׁ"י: (יח) וְאַף עַל גַּב דְּאֵינוֹ דּוֹמֶה נִמְחַק פַּעַם אֶחָד לְנִמְחָק שְׁתֵּי פְעָמִים, וְיִהְיֶה נִכָּר אֵצֶל מַחַק דְּעֵדִים, סָבַר תַּנָּא קַמָּא דְגָזְרוּ דִּלְמָא דָמִי הַאי מַחְקָא לְהַאי מַחְקָא. וַחֲכָמִים מַכְשִׁירִין, דְּאִי אִיכָּא חֲשָׁשָׁא אָזְלִינַן לְעֵדֵי מְסִירָה. בֵּית יוֹסֵף, וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (יט) מַכְשִׁירִין. אֵין לִתְמֹהַּ אֵיךְ מַכְשִׁיר רַבִּי אֱלִיעֶזֶר בְּדָבָר שֶׁיָּכוֹל לְהִזְדַּיֵּף, וְהָא מוֹדֶה רַבִּי אֱלִיעֶזֶר בִּמְזֻיָּף מִתּוֹכוֹ שֶׁהוּא פָּסוּל. דְּהָא מוֹדֶה נַמִּי הֵיכָא דְּלֹא אָתֵי לְמִסְמָךְ עֲלַיְהוּ, כְּגוֹן בְּשֵׁמוֹת מֻבְהָקִים, דְּכָשֵׁר, הָכָא נַמִּי כָּשֵׁר. הַגָּהוֹת אָשְׁרִי: (כ) וְאַף עַל גַּב דְּבִשְׁאָר שְׁטָרוֹת נַמִּי קָיְמָא לָן דְּעַל יְדֵי עֵדֵי מְסִירָה גּוֹבֶה מִמְּשֻׁעְבָּדִים, מִכָּל מָקוֹם כֵּיוָן דִּלְגוֹבַיְנָא קָיְמָא דִלְמָא לֹא מַפִּיק לְהוּ עַד זְמַן אָרֹךְ וְשָׁכְחוּ עֵדֵי מְסִירָה אֶת הַדָּבָר וְסַמְכִינַן אַעֵדֵי חֲתִימָה, וְהוּא מְזַיֵּף כְּתָב הָעֶלְיוֹן וְחוֹתַם הָעֵדִים עוֹמֵד בִּמְקוֹמוֹ. וְכָתְבוּ הַתּוֹסָפוֹת דְּהַיְנוּ דַּוְקָא בִשְׁטָרוֹת הָעוֹמְדִים לִרְאָיָה, אֲבָל שְׁטַר מֶכֶר וּמַתָּנָה דָמִי לְגִטִּין. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

(א) בְּעִדִּית. אֶת כָּל נְאוֹת יַעֲקֹב [אֵיכָה ב] תִּרְגּוּם יַת כָּל עִדִּית בֵּית יַעֲקֹב. נְאוֹת אֱלֹהִים [תְּהִלִּים פג] תַּרְגּוּם עִדִּית. עָרוּךְ: (ב) כְּלוֹמַר דְּהָא לְהָכִי תְנַן לָהּ הָכָא בְגִטִּין וְאַף עַל פִּי שֶׁאֵין עִנְיָנוֹ מֵעִנְיָן גִּטִּין, אֶלָּא מִשּׁוּם דְּתָנֵי בְּפֶרֶק דִּלְעֵיל מִלֵּי דְתִקּוּן הָעוֹלָם: (ג) שֶׁנּוֹחַ לוֹ לְאָדָם לִגְבּוֹת מֻעָט מִקַּרְקַע עִדִּית וְלֹא הַרְבֵּה מִזִּבּוּרִית. רַשִׁ"י: (ד) גְּמָרָא. וּפֵרֵשׁ הָרַ"ב אַף עַל פִּי שֶׁהַזִּבּוּרִית שָׁוָה כְנֶגֶד הַחוֹב לֹא קָפִיץ עֲלֵיהּ זְבִינָא: (ה) בְּזִבּוּרִית. הַפָּחוּת וְהָרַע. הָרַמְבַּ"ם. וְלֹא נוֹדַע לִי מֵאֵיזֶה לָשׁוֹן הוּא (נִרְאֶה לִי מִלָּשׁוֹן זִבּוּרָא וְעַקְרַבָּא. בֵּית דָּוִד): (ו) וְקָשֶׁה דִלְמַאן דְּאָמַר כְּתֻבָּה דְאוֹרַיְתָא דְּיָלֵיף לָהּ מִכְּמֹהַר הַבְּתוּלוֹת כוּ', מִדְּאוֹרַיְתָא דִינָהּ בְּעִדִּית, וְאַמַּאי הִפְקִיעוּ חֲכָמִים דִּינָהּ מֵהַאי טַעֲמָא. וְרַבִּי מֵאִיר נַמִּי דְּאָמַר בְּבֵינוֹנִית קָסָבַר כְּתֻבָּה דְאוֹרַיְתָא בְּרֵישׁ פֶּרֶק אַף עַל פִּי. וְיֵשׁ לוֹמַר דְּטַעְמָא דְהִפְקִיעוּ חֲכָמִים דִּינָהּ כִּדְאָמְרִינַן בַּגְּמָרָא גַּבֵּי בַעַל חוֹב, שֶׁלֹּא יִרְאֶה אָדָם שָׂדֶה נָאֶה וַיֹּאמַר אֶקְפֹּץ וְאַלְוֶּנּוּ, הָכִי נַמִּי חָשׁוּ חֲכָמִים בְּאִשָּׁה שֶׁמָּא תַקְנִיטֶנּוּ כְּדֵי שֶׁיְּגָרְשֶׁנָּהּ וְתִגְבֶּה כְּתֻבָּה. וְטַעְמָא דְיוֹתֵר מִשֶּׁהָאִישׁ כוּ' לֹא הָוֵי אֶלָּא מִשּׁוּם דְּלֹא תֵימָא כְמוֹ שֶׁבַּעַל חוֹב בְּבֵינוֹנִית מִשּׁוּם נְעִילַת דֶּלֶת, אִשָּׁה נַמִּי מִשּׁוּם חִנָּא שֶׁיְּהוּ הָאֲנָשִׁים נוֹשְׂאִים חֵן בְּעֵינֵי הַנָּשִׁים וְיִנָּשְׂאוּ לָהֶן תְּהֵא בְבֵינוֹנִית. תּוֹסָפוֹת:

(ז) וְחָשִׁיב תִּקּוּן הָעוֹלָם, דְּאִי לָאו תִּקּוּן הָעוֹלָם הָיָה גוֹבֶה אֲפִלּוּ מִמְּשַׁעְבְּדֵי בְּבֵינוֹנִית מִשּׁוּם נְעִילַת דֶּלֶת. עַד כָּאן. וּמִשּׁוּם הָכִי נָקַט הָרַ"ב בַּעַל חוֹב. וְהוּא הַדִּין לְנִזָּקִין: (ח) יְתוֹמִים. מְפָרֵשׁ טַעְמָא בַגְּמָרָא מִשּׁוּם דְּלֹא מַסִּיק אַדַּעְתֵּיהּ דְּמַלְוֶה דְמִית לֹוֶה וְנָפְלֵי נִכְסֵי קָמֵי יַתְמֵי דַּהֲוָה לֵיהּ נוֹעֵל דֶּלֶת וְהִלְכָּךְ אֲפִלּוּ יְתוֹמִים גְּדוֹלִים. וּבְנִזָּקִין אַף עַל גַּב דְּמִדְּאוֹרַיְתָא בְּעִדִּית, הַיְנוּ דְנִזַּק, וְגַבֵּי יַתְמֵי אוֹקְמוּהָ אַדְּאוֹרַיְתָא. וְעַיֵּן תּוֹסָפוֹת:

(ט) וּלְשׁוֹן לַאֲכִילַת פֵּרוֹת לֹא דָיְקָא, דִלְפֵרוֹת בִּלְחוּד הֲוָה לֵיהּ לְמִתְנֵי. וְהָרַמְבַּ"ם מְפַרֵשׁ כְּשֶׁגָּזַל רְאוּבֵן קַרְקַע מִשִּׁמְעוֹן וְהָיְתָה מְלֵאָה פֵרוֹת וְאָכַל רְאוּבֵן הַפֵּרוֹת וְהִפְסִיד הַקַּרְקַע כוּ', כְּשֶׁיָּבֹא שִׁמְעוֹן לִגְבּוֹת מֵרְאוּבֵן הַפֵּרוֹת שֶׁאָכַל כוּ'. וּשְׁנֵי הַפֵּרוּשִׁים בַּגְּמָרָא בָּבָא מְצִיעָא דַף י"ד. וּפֵרוּשׁ הָרַ"ב נִדְחָה: (י) אֲבָל דְּמֵי פֵרוֹת הֲוָה כְמִלְוֶה עַל פֶּה. דְּאַף עַל פִּי שֶׁכָּתַב מוֹכֵר לַלּוֹקֵחַ אֲנָא אֵיקוּם וַאַשַׁפֵּי אִינוּן וְעַמְלֵיהוֹן וְשִׁבְחֵיהוֹן, אֲפִלּוּ הָכִי אֵין קוֹל יוֹצֵא אֶלָּא עַל הַמִּקָּח, לֹא עַל הַשְּׁאָר, כֵּיוָן שֶׁעֲדַיִן אֵינוֹ בָּעוֹלָם. רַשִׁ"י וְהָרַ"ן:

(יא) אֲבִיהֶן. וְכָל שֶׁכֵּן הַמְּמֻנֶּה מֵהַבֵּית דִּין שֶׁהוּא שְׁלִיחַ בֵּית דִּין: (יב) לְעַשֵּׂר. קוֹרְאִין כָּל הַמַּתָּנוֹת בְּשֵׁם מַעֲשֵׂר בִּלְבַד. וּבַגְּמָרָא רָמִינַן דִּתְנַן (בְּרֵישׁ תְּרוּמוֹת) וְהַתּוֹרֵם אֶת שֶׁאֵינוֹ שֶׁלּוֹ אֵין תְּרוּמָתוֹ תְּרוּמָה, וּמְשָׁנִינַן לֹא קָשְׁיָא כָּאן לְהַאֲכִיל כָּאן לְהַנִּיחַ. פֵּרֵשׁ רַשִׁ"י לְהַאֲכִיל יְתוֹמִים לְאַלְתַּר חַיָּב לְעַשֵּׂר, לְהַנִּיחַ הַתְּבוּאָה בָאוֹצָר עַד שֶׁיִּגְדְּלוּ, לָמָּה לִי עִשּׂוּרֵי, לִכְשֶׁיִּגְדְּלוּ יְעַשְּׂרוּ הֵם: (יג) שֶׁמִּנָּהוּ. וּמִי שֶׁסָּמְכוּ אֶצְלוֹ שַׁבְקֵיהּ מִלָּדוּן בּוֹ. וְיֵשׁ בָּזֶה מַחֲלֹקֶת הַפּוֹסְקִים לְמִי דָמִי יוֹתֵר: (יד) וְאַף עַל גַּב דְּהַיְנוּ מְדַמֵּעַ, שֶׁהוּא נַמִּי מְעָרֵב, אִצְטְרִיךְ לְמִתְנֵי בְּהֶדְיָא, מִשּׁוּם דִּקְנָסָא מִקְּנָסָא לֹא יָלְפִינַן, דְּהָא לָאו דִּינָא הוּא, וְהֵיכָא דְקָנוּס קָנוּס וְהֵיכָא דְלָא קָנוּס לֹא קָנוּס. גְּמָרָא: (טו) וְצָרִיךְ לוֹמַר דְּמַזִּיק שִׁעְבּוּדוֹ שֶׁל חֲבֵרוֹ דִתְנַן בַּפֶּרֶק דִּלְעֵיל מִשְׁנָה ד', דִּלְרַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל דְּקָיְמָא לָן כְּוָתֵיהּ מְחָיְבִינַן לֵיהּ, דְּחָשִׁיב הֶזֵּק נִכָּר לְגַבֵּי הָנָךְ דְּהָכָא. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (טז) וְהוּא הַדִּין לַעֲבוֹדַת הוֹלָכָה וְקַבָּלָה:

(יז) הַחֵרֶשֶׁת. מִשּׁוּם תַּרְתֵּי בַּתְרָיָתָא דְּמִשּׁוּם תַּקָּנָה נִינְהוּ, תָּנָא לֵיהּ הָכָא. רַשִׁ"י: (יח) קְטַנָּה כוּ'. אֲבָל חֵרֶשֶׁת דְּאַיְרֵי בַּהּ וְשָׁבְקָה, שְׁמַע מִנַּהּ דְּלֹא אָכְלָה. וְטַעְמָא בַגְּמָרָא, גְזֵרָה שֶׁמָּא חֵרֵשׁ שֶׁנָּשָׂא פִּקַּחַת יַאֲכִילֶנָּהּ, וּגְזֵרָה מִשּׁוּם תְּרוּמָה דְאוֹרַיְתָא: (יט) וְאִם תֹּאמַר וְלוֹקְמָא אֲפִלּוּ בִּדְאוֹרַיְתָא, דְקָטָן אוֹכֵל נְבֵלוֹת הוּא דְאֵין מְצֻוִּין לְהַפְרִישׁוֹ. וְיֵשׁ לוֹמַר דְּבַגְּמָרָא מַשְׁמַע דַּאֲפִלּוּ מַאֲכִילָהּ בְּיָדַיִם. וְעוֹד מַשְׁמַע דַּאֲפִלּוּ לְכִי גָּדְלָה אוֹכֶלֶת מִכֹּחַ זֶה אַף עַל פִּי שֶׁעֲדַיִן לֹא בָּעַל. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (כ) יוֹרְשָׁהּ. דְּהֶפְקֵר בֵּית דִּין הֶפְקֵר כִּדְיָלְפִינַן לֵיהּ בַּגְּמָרָא מִקְּרָאִי: (כא) שֶׁלֹּא נוֹדְעָה. דְּאִי נוֹדְעָה לָרַבִּים, מִלְּתָא דְלָא שְׁכִיחָא הוּא שֶׁיְבִיאֶנָּה הַגַּזְלָן, וְלֹא תִקְּנוּ בְּמִלְּתָא דְלָא שְׁכִיחַ: (כב) לָרַבִּים. אָמְרִינַן בַּיְרוּשַׁלְמִי דְרַבִּים הַיְנוּ שְׁלֹשָׁה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (כג) מְכַפֶּרֶת. וְאִי קָשְׁיָא הָאֵיךְ הִתְנוּ בֵּית דִּין לַעֲקֹר דָּבָר מֵהַתּוֹרָה וְלִפְטֹר אֶת זֶה מֵחַטָּאת שֶׁהוּא חַיָּב, הָא אָמְרִינַן בִּיבָמוֹת דַּף צ' שֵׁב וְאַל תַּעֲשֶׂה שָׁאנֵי דְלָאו מִעַקַּר הוּא. רַשִׁ"י. וְקָשֶׁה דַעֲדַיִן אֵינוֹ מְיֻשָּׁב הָאֵיךְ הַכֹּהֲנִים אוֹכְלִין חֻלִּין בָּעֲזָרָה. וּלְהַמַּסְקָנָא מְתָרֵץ דְּרַבָּנָן אוֹקְמֵיהּ בְּיַד הַגַּזְלָן שֶׁיְּהֵא שֶׁלּוֹ וְשֶׁיּוּכַל לְהַקְדִּישׁוֹ. וּפֵרֵשׁ רַשִׁ"י דְהֶפְקֵר בֵּית דִּין הֶפְקֵר, אִם כֵּן הָוֵי שֶׁלּוֹ נַמִּי לְעִנְיַן כַּפָּרָה, וּמְיֻשַּׁב הַכֹּל. תּוֹסָפוֹת חֲדָשִׁים:

(יב) כֵּן פֵּרֵשׁ רַשִׁ"י. וְהִקְשׁוּ בַּתּוֹסְפוֹת דְּלִישְׁנָא דְּמַה הִיא שֶׁהֻזְכַּר כוּ' מַשְׁמַע דְּקָאֵי אַמַּאי דִלְעֵיל מִנֵּיהּ. לְפִיכָךְ פֵּרְשׁוּ דְּקָאֵי אַמֵּהַיּוֹם אִם מַתִּי. וְרַבִּי יְהוּדָה סָבַר שֶׁדַּעְתּוֹ לְאַחֵר הַגֵּט כָּל מַה שֶּׁיּוּכַל רַק שֶׁיָּחוּל מֵחַיִּים שָׁעָה אַחַת סָמוּךְ לְמִיתָתוֹ, וּלְפִיכָךְ הִיא כְאֵשֶׁת אִישׁ לְכָל דְּבָרֶיהָ. וּלְדִידֵיהּ בָּרֵישָׁא דְלָא תִּתְיַחֵד כוּ' לֹא הָוֵי טַעֲמָא מִשּׁוּם חֲשָׁשָׁא דְקִדּוּשִׁין, דְּהָא עֲדַיִן אֵינָהּ מְגֹרֶשֶׁת, אֶלָּא טַעֲמָא מִשּׁוּם גֵּט יָשָׁן. וְרַבִּי יוֹסֵי מְסָפְקָא לֵיהּ אִי מֵהַיּוֹם מַמָּשׁ קָאָמַר וַהֲרֵי הִיא כִפְנוּיָה, אוֹ מֵעֵת שֶׁאֲנִי בָּעוֹלָם וְהַוְיָא אֵשֶׁת אִישׁ עַד שָׁעָה אֶחַת סָמוּךְ לְמִיתָתוֹ. וְלֹא הָוֵי טַעֲמָא מִשּׁוּם דְּאֵין בְּרֵרָה, דְּאִי מִשּׁוּם דְּאֵין בְּרֵרָה, לֹא הָיָה לָחוּל הַגֵּרוּשִׁין כְּלָל, אַף עַל גַּב דְּהֻבְרַר הַדָּבָר לְבַסּוֹף, כֵּיוָן דְּאֵין בְּרֵרָה: (יג) מְגֹרֶשֶׁת וְאֵינָהּ מְגֹרָשֶׁת. וְלֹא תְּנַן סְפֵק מְגֹרֶשֶׁת דִּכְשֶׁאָמְרוּ בְּלָשׁוֹן מְגֹרֶשֶׁת וְאֵינָהּ מְגֹרֶשֶׁת, בַּעְלָהּ חַיָּב בִּמְזוֹנוֹתֶיהָ, כִּדְאִיתָא בַגְּמָרָא:

(מג) כְּמוֹ אֵשׁ קָדְחָה בְאַפִּי [דְּבָרִים לב]: (מד) וְאִצְטְרִיךְ לִכְתֹּב עֶרְוָה, לוֹמַר שֶׁאֲפִלּוּ יָצְאָה מִשּׁוּם עֶרְוָה מֻתֶּרֶת לִנָּשֵׂא לְאַחֵר, כְּדִכְתִיב וְיָצְאָה וְהָיְתָה לְאִישׁ אַחֵר. וּלְבֵית שַׁמַּאי אִצְטְרִיךְ דָּבָר לִגְזֵרָה שָׁוָה, דִּכְתִיב הָכָא עֶרְוַת דָּבָר וּכְתִיב לְהַלָּן עַל פִּי שְׁנַיִם עֵדִים יָקוּם דָּבָר, מַה לְּהַלָּן בִּשְׁנַיִם אַף כָּאן בִּשְׁנַיִם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (מה) דְּאָמַר רֵישׁ לָקִישׁ כִּי מִשְׁמֵשׁ בְּאַרְבַּע לְשׁוֹנוֹת. אִי, אֶלָּא, דִּלְמָא, דְּהָא. בֵּית שַׁמַּאי סָבְרֵי כִי מָצָא, דְּהָא מָצָא. וְרַבִּי עֲקִיבָא סָבַר כִּי מָצָא, אִי נַמִּי מָצָא: (מו) הַיְנוּ דְּמַסִּיק הַתּוֹסָפוֹת דְּרַבִּי עֲקִיבָא כְבֵית הִלֵּל אָתֵי. דְּאִי לָאו דַּאֲיַתֵּר לֵיהּ דָּבָר לְהִקְדִּיחָה תַּבְשִׁילוֹ לֹא הֲוָה דָרֵישׁ כִּי מָצָא בָּהּ בִּלְשׁוֹן אִי נַמִּי:
