(א) לָאו מֵהַמְּקַנֵּא דִּיֵּק, שֶׁיָּכוֹל לְפָרֵשׁ הַבָּא לְקַנְאוֹת, אֶלָּא כִּדְפֵרֵשׁ רַשִׁ"י, מִדְּלֹא תָּנֵי רַבִּי אֱלִיעֶזֶר אוֹמֵר מְקַנֵּא אָדָם לְאִשְׁתּוֹ. תּוֹסָפוֹת. וְלִישְׁנָא דִמְקַנֵּא לְשׁוֹן כַּעַס הוּא, שֶׁמַּכְעִיסָהּ וּמַקְנִיטָהּ. וּפֵרֵשׁ רַשִׁ"י כְּמוֹ הֵם קִנְאוּנִי [דְּבָרִים לב]. כִּי קִנְאָה חֲמַת גָּבֶר [מִשְׁלֵי ו]. וְלַהֲלָכָה שֶׁכָּתַב הָרַ"ב דְּחַיָּב לְקַנְאוֹת לִישְׁנָא דְּהַתְרָאָה כְּמוֹ וַיְקַנֵּא ה' לְאַרְצוֹ [יוֹאֵל ב]. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (ב) עַל פִּי כוּ'. דִּסְתִירָה אִתַּקַּשׁ לְטֻמְאָה, דִּכְתִיב וְנִסְתְּרָה וְהִיא נִטְמָאָה. וּבְטֻמְאָה כֻּלֵּי עָלְמָא מוֹדוּ דְּעֵד אֶחָד נֶאֱמָן לְאַחַר קִנּוּי כוּ'. גְּמָרָא: (ג) הָכִי יָלֵיף לָהּ בַּגְּמָרָא דַּף כ"ח: (ד) פֵּרוּשׁ, אֲפִלּוּ עַל פִּי שְׁנַיִם כוּ'. אֲבָל בִּזְמַן שֶׁיֵּשׁ מֵי הַמָּרִים, אַף עַל פִּי שֶׁאִם תִּסָּתֵר עַל פִּי עַצְמוֹ אוֹ עַל פִּי עֵד אֶחָד תֵּאָסֵר עָלָיו עוֹלָמִית, הוֹאִיל וְאֶפְשָׁר שֶׁיָּכוֹל לִהְיוֹת שֶׁתִּסָּתֵר עַל פִּי שְׁנַיִם וְאָז יַשְׁקֶנָּה, לְפִיכָךְ רְשׁוּת אוֹ חוֹבָה לְקַנְאוֹת. מַה שֶּׁאֵין כֵּן בַּזְּמַן הַזֶּה שֶׁאִי אֶפְשָׁר שֶׁתָּשׁוּב לְהֵתֵּרָהּ, אָסְרוּ לְקַנְאוֹת כְּלָל:

(ה) וְאָמַר לָהּ לָאו פֵּרוּשָׁא דְכֵיצַד מְקַנֵּא לָהּ הוּא, אֶלָּא תַּרְתֵּי קָתָנֵי, וְהָכִי קָאָמַר, לֹא תְדַבְּרִי כוּ'. רַשִׁ"י: (ו) לְבֵיתָהּ. כְּלוֹמַר לְבַעְלָהּ, שֶׁלֹּא אֲסָרָתָהּ תּוֹרָה מִסָּפֵק אֶלָּא עַל יְדֵי קִנּוּי: (ז) בְּאֵשׁ בֵּינוֹנִית. הָרַ"מ: (ח) בַּתְּרוּמָה. כִּדְיָלְפִינַן לְקַמָּן שְׁלֹשָׁה וְנִטְמְאָה אֲמוּרִים בַּפָּרָשָׁה, אֶחָד לַבַּעַל וְאֶחָד לַבּוֹעֵל וְאֶחָד לַתְּרוּמָה. רַשִׁ"י:

(ט) עֵדִים. הוּא הַדִּין לְעֵד אֶחָד, שֶׁכָּל מָקוֹם שֶׁהֶאֱמִינָה תּוֹרָה לְעֵד אֶחָד הֲרֵי כָּאן שְׁנַיִם. תּוֹסָפוֹת. וְעַיֵּן מַה שֶּׁכָּתַבְנוּ מִשְׁנָה ב' פֶּרֶק ט"ז דִּיבָמוֹת: (י) אַף עַל גַּב דְּדָרְשִׁינַן לֵיהּ לְעֵד אֶחָד נֶאֱמָן בְּסוֹטָה, אֵין מִקְרָא יוֹצֵא מִיְדֵי פְשׁוּטוֹ וְדָרְשִׁינַן לֵיהּ נַמִּי הָכִי. רַשִׁ"י: (יא) וַהֲדַר אָמַר לְהוּ רַבִּי יְהוּדָה מִקְּרָא דִּכְתִיב וְהֵבִיא הָאִישׁ אֶת אִשְׁתּוֹ אֶל הַכֹּהֵן, מִן הַתּוֹרָה הָאִישׁ מֵבִיא אֶת אִשְׁתּוֹ. גְּמָרָא:

(יב) כְּדֶרֶךְ. כְּלוֹמַר כְּמוֹ שֶׁמְּאַיְּמִין יוֹתֵר עַל עֵדֵי נְפָשׁוֹת מֵעַל עֵדֵי מָמוֹנוֹת, כִּדְאִיתָא בְּסַנְהֶדְרִין, כְּמוֹ כֵן מְאַיְּמִין יוֹתֵר עַל הָאִשָּׁה. אֲבָל לֹא שֶׁאוֹתוֹ אִיּוּם עַצְמוֹ אוֹמְרִים בָּאִשָּׁה, שֶׁאֵין עִנְיָנוֹ לְעִנְיָנָהּ כְּלָל: (יג) כְּשֶׁאָמַר יְהוּדָה צָדְקָה מִמֶּנִּי, עָמַד רְאוּבֵן וְאָמַר בִּלְבַּלְתִּי יְצוּעֵי אָבִי. תַּנְחוּמָא: (יד) וְכָל מִשְׁפַּחַת כוּ'. שֶׁאִי אֶפְשָׁר שֶׁלֹּא יִהְיוּ נִמְצָאִין אַנְשֵׁי בֵּיתָהּ וּמִשְׁפַּחְתָּהּ בְּזִכְרוֹן אֵלּוּ הַדְּבָרִים. הָרַמְבַּ"ם:

(טו) וְיוֹצֵאת. וְאֵינָהּ נֶהֱרֶגֶת, שֶׁלֹּא הִתְרוּ בָּהּ עֵדִים בִּשְׁעַת מַעֲשֶׂה: (טז) הַמִּזְרָח. שַׁעַר הַחִיצוֹן שֶׁבּוֹ נִכְנָסִין לְהַר הַבַּיִת. לְשַׁעַר נִקָּנוֹר, וּמִשָּׁם לְשַׁעַר נִקָּנוֹר הוּא שַׁעַר הָעֶלְיוֹן שֶׁבֵּין עֶזְרַת יִשְׂרָאֵל לְעֶזְרַת נָשִׁים. רַשִׁ"י: (יז) דְּאִי לָאו הָכִי מַאי מַעֲלִין, הָא הָתָם קָיְמָא לִפְנֵי הַסַּנְהֶדְרִין שֶׁבְּלִשְׁכַּת הַגָּזִית שֶׁבָּעֲזָרָה. גְּמָרָא. וּרְצוֹנוֹ לוֹמַר מוֹרִידִין וּמַעֲלִין וּמוֹרִידִין: (יח) שֶׁשָּׁם כוּ'. וּבְמִשְׁנָה ג' פֶּרֶק ו' דִשְׁקָלִים פֵּרֵשׁ הָרַ"ב שַׁעַר הַנָּשִׁים לַעֲמֹד עַל קָרְבָּנָן, הַיְנוּ בְּנִדָּר וְנִדָּב אוֹ שְׁאָר חוֹבוֹת שֶׁאֵינָן מְחֻסְּרֵי כַפָּרָה: (יט) מִפְּנֵי שֶׁלֹּא נִתְקַדֵּשׁ בִּקְדֻשַּׁת עֲזָרָה עֹבִי חָלָל אוֹתוֹ שַׁעַר, מִפְּנֵי הַמְּצֹרָעִין שֶׁמַּכְנִיסִין יְדֵיהֶן לַבְּהוֹנוֹת לִתֵּן מִדַּם הָאָשָׁם עַל בֹּהֶן יָדָן, וּמְחֻסָּר כִּפּוּרִים שֶׁנִּכְנַס לָעֲזָרָה בְּכָרֵת, לְפִיכָךְ לֹא קִדְּשׁוּם, שֶׁיּוּכַל לַעֲמֹד בַּחֲלַל הַשַּׁעַר כוּ'. שֶׁאִם הָיָה כֹּהֵן מוֹצִיא אֶת דַּם הָאָשָׁם חוּץ לָעֲזָרָה הֲרֵי הוּא נִפְסַל בְּיוֹצֵא., לְפִיכָךְ צָרִיךְ לְהַכְנִיס יָדוֹ לְתוֹךְ חֲלַל הָעֲזָרָה, וּבִיאַת מִקְצָת שָׁרֵי רַחֲמָנָא. רַשִׁ"י: (כ) לֹא הָיָה כוּ'. אַף עַל גַּב דִּגְזֵרַת הַכָּתוּב הוּא, שֵׁב וְאַל תַּעֲשֶׂה לָאו מִעֲקַר הוּא. תּוֹסָפוֹת: (כא) עַל שֵׁם שֶׁהָיוּ מְצוּיִין שָׁם בָּעֲזָרָה יוֹתֵר מִשְּׁאָר הָעָם. רַשִׁ"י. וְאִם הָיוּ פּוֹרְסִין סָדִין בֵּינָהּ לְבֵין הָאֲנָשִׁים לֹא הָיָה נִוּוּל כָּל כָּךְ בִּפְנֵי הַנָּשִׁים כְּמוֹ בָּאֲנָשִׁים, שֶׁאֵין אִשָּׁה בּוֹשָׁה מִפְּנֵי חַבְרוֹתֶיהָ, וְלֹא יִוָּסְרוּ. תּוֹסָפוֹת:

(כב) נְזָמִים. הֵם תַּכְשִׁיטֵי הָאַף. הָרַמְבַּ"ם. וּמִלְּשׁוֹן הָרַ"ב פֶּרֶק ו' דְּשַׁבָּת גַּם דְּאֹזֶן בִּכְלַל: (כג) לְפִי שֶׁנֶּאֱמַר בִּתְחִלַּת פָּרָשַׁת עֲרָיוֹת כְּמַעֲשֵׂה אֶרֶץ מִצְרַיִם וְגוֹ'. וְאַף עַל פִּי שֶׁנֶּאֱמַר כְּמוֹ כֵן וּכְמַעֲשֵׂה אֶרֶץ כְּנַעַן וְגוֹ', וְיִקְּחוּ חֶבֶל שֶׁבַּמְּדִינָה, כָּל הַשִּׁנּוּיִים שֶׁנּוּכַל לַעֲשׂוֹת עוֹשִׂין כְּדֵי לְנַוְּלָהּ, וּלְפִיכָךְ לוֹקְחִין חֶבֶל מִצְרִי שֶׁלֹּא מֵהַמְּדִינָה. תּוֹסָפוֹת. וְכֵן יֵשׁ לְפָרֵשׁ כְּפִיפָה מִצְרִית דְּרֵישׁ פֶּרֶק ב'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (כד) גַּס. כְּשֶׁהָאָדָם רוֹאֶה בְּנֵי בֵיתוֹ דַּעְתּוֹ מִתְגַּבֶּרֶת עָלָיו וְלֹא תִירָא וְלֹא תוֹדֶה. וְאָנוּ מְבַקְּשִׁין שֶׁתּוֹדֶה וְלֹא יִמָּחֶה שֵׁם הַקָּדוֹשׁ עַל הַמַּיִם. רַשִׁ"י:

(כה) תְּחִלָּה. לְקַלֵּל אוֹתָהּ, דִּכְתִיב בְּתֵת ה' אֶת יְרֵכֵךְ נוֹפֶלֶת וְאֶת בִּטְנֵךְ צָבָה. רַשִׁ"י. וּבַגְּמָרָא הָא כְּתִיב וְצָבְתָה בִּטְנָהּ וְנָפְלָה יְרֵכָהּ. אָמַר אַבַּיֵי כִּי לָיֵיט לָיֵיט תְּחִלָּה יָרֵךְ וַהֲדַר בֶּטֶן, וּמַיָּא כִּי אוֹרְחַיְהוּ בַּדְקֵי בֶּטֶן בָּרֵישָׁא וַהֲדַר יָרֵךְ. בִּקְלָלָה נַמִּי הָכְתִּיב לַצְבּוֹת בֶּטֶן וְלַנְפִּל יָרֵךְ, הַהִיא דְמוֹדַע לָהּ כֹּהֵן דְּבֶטֶן בָּרֵישָׁא וַהֲדַר יָרֵךְ, שֶׁלֹּא לְהוֹצִיא לַעַז עַל מַיִם הַמָּרִים:

(א) דְּאִי לָאו קְרָא הֲוָה אֲמִינָא נֵילַף בְּקַל וָחֹמֶר מִיְבָמָה דְּבִלְשׁוֹן הַקֹּדֶשׁ. רַשִׁ"י: (ב) וְרַשִׁ"י כָּתַב דְּהָכִי גָּרְסִינַן, וְאָמַר אֶל הָאִשָּׁה בְּכָל לָשׁוֹן שֶׁהִיא שׁוֹמַעַת, אֶל הָאִשָּׁה מַשְׁמַע דְּבָרִים הַנִּכְנָסִין בְּלִבָּהּ, שֶׁתְּהֵא מַכֶּרֶת בַּלָּשׁוֹן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (ג) וְיָלֵיף אֲמִירָה מִסּוֹטָה. גְּמָרָא. וְאָתֵי שַׁפִּיר לְגִרְסַת רַשִׁ"י, דְּאִם לֹא כֵן מַאי אוּלְמֵיהּ דְּהַאי מֵהַאי: (ד) וְאִצְטְרִיךְ לַאֲפוֹקֵי מֵרַבִּי דְּאָמַר בַּגְּמָרָא וְהָיוּ כִּכְתָבָם כוּ'. גְּמָרָא: (ה) וּתְפִלָּה. גְּמָרָא, רַחֲמֵי הוּא, כָּל דְּבָעֵי מְצַלֵּי. פֵּרֵשׁ רַשִׁ"י דְּהָא רַחֲמֵי נִינְהוּ, וְהַהוּא לִישָׁנָא דְּיָדַע לְכַוֵּן לְצַלֵּי: (ו) דְּאָמַר רַבִּי יוֹחָנָן כָּל הַשּׁוֹאֵל צְרָכָיו בִּלְשׁוֹן אֲרַמִּי אֵין מַלְאֲכֵי הַשָּׁרֵת נִזְקָקִין לוֹ. פֵּרֵשׁ רַשִׁ"י יָחִיד צָרִיךְ שֶׁיְּסַיְּעוּהוּ מַלְאֲכֵי הַשָּׁרֵת, צִבּוּר לֹא צְרִיכִי לְהוּ, דִּכְתִיב הֵן אֵל כַּבִּיר לֹא יִמְאַס [אִיּוֹב לו], אֵינוֹ מוֹאֵס בִּתְפִלָּתָן שֶׁל רַבִּים. וּמַשְׁמַע לְהָרַ"ב דְּכָל שֶׁכֵּן שְׁאָר לְשׁוֹנוֹת דְּלֹא, דְּהָא אֲרַמִּי קָרוֹב לִלְשׁוֹן הַקֹּדֶשׁ הוּא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב דִּלְגִרְסָא שֶׁלָּנוּ צָרִיךְ לוֹמַר דְּתָנֵי וְשִׁיֵּר (אַף עַל פִּי שֶׁתָּנֵי אֵלּוּ) הַלֵּל וְקִדּוּשׁ וּבִרְכַּת הַפֵּרוֹת וְהַמִּצְוֹת: (ז) וְלֹא יָלְפִינַן מִבִּרְכוֹת הַר גְּרִזִּים כִּדְיָלְפִינַן מִנֵּיהּ בִּרְכַּת כֹּהֲנִים, לְהָכִי כְּתִיב וּבֵרַכְתָּ אֶת ה', כְּלוֹמַר מֵאַחַר שֶׁהַבְּרָכָה וְהַשֶּׁבַח כְּלַפֵּי הַשְּׁכִינָה תּוּכַל לְבָרֵךְ בְּכָל לָשׁוֹן שֶׁתִּרְצֶה, וּמִסְתָמָא בְּלָשׁוֹן שֶׁאַתָּה מַכִּיר שֶׁתִּתֵּן שֶׁבַח לְהַקָּדוֹשׁ בָּרוּךְ הוּא בְּלֵב שָׁלֵם לַהֲנָאָתְךָ: (ח) מִדְּלֹא כְּתִיב וְהִשְׁבִּיעוֹ בְּאָלָה. וְאִי לָאו קְרָא הֲוָה אֲמִינָא נֵילַף קוֹל קוֹל מִמֹּשֶׁה כְּדִלְקַמָּן מִשְׁנָה ג'. תּוֹסָפוֹת: (ט) אַגַּב דְּיָלְפִינַן בִּשְׁבוּעוֹת גְּזֵרָה שָׁוָה זוֹ לְעִנְיָן אַחֵר נָקַט לָהּ נַמִּי הָכָא. דִּמְהֵיכִי תֵּיתֵי דְּנִבְעֵי בִּלְשׁוֹן הַקֹּדֶשׁ. תּוֹסָפוֹת:

(י) וְאַיְדֵי דְּמִשְׁנָה ה' וּלְהַלָּן, תָּנֵי נַמִּי הָכָא כֵּיצַד: (יא) בְּמַתַּן תּוֹרָה. וּבִלְשׁוֹן הַקֹּדֶשׁ נִתְּנָה. רַשִׁ"י. וְלֹא יָלֵיף קוֹל קוֹל מִשְּׁבוּעַת הָעֵדוּת, דַּעֲנִיָּה וְקוֹל יָלֵיף. תּוֹסָפוֹת:

פרק ז - משנה ד

(יב) עַד כוּ'. וְהָא דִּבְמִשְׁנָה ג' פֶּרֶק י"ב דִּיבָמוֹת קְרִיאָה אֵינָהּ מְעַכֶּבֶת בֵּין לְרַבִּי אֱלִיעֶזֶר בֵּין לְרַבִּי עֲקִיבָא, וְאִם קָרָא בִּלְשׁוֹן חֹל מְעַכֵּב, כָּתְבוּ הַתּוֹסָפוֹת דְּיֵשׁ לוֹמַר דִּסְבִירָא לְהוּ דִקְרִיאָה מְעַכֶּבֶת, וְקָסָבַר לֹא נֶחְלְקוּ רַבִּי עֲקִיבָא וְרַבִּי אֱלִיעֶזֶר בְּדָבָר זֶה. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

(יג) וְגוֹ'. וְסֵיפָא דִּקְרָא, נֶגֶד הַכֹּהֲנִים וְהַלְוִיִּם: (יד) בַּבְּרָכָה. כָּל הָאֲרוּרִים הָיוּ אוֹמְרִים תְּחִלָּה בִּלְשׁוֹן בְּרָכָה. רַשִׁ"י. וּבַגְּמָרָא מַפִּיק לָהּ מִקְּרָא: (טו) אֵלּוּ וָאֵלּוּ. שֶׁבְּהַר גְּרִזִּים וּבְהַר עֵיבָל. וְהָא דִּכְתִיב וְאֵלֶּה יַעַמְדוּ לְבָרֵךְ, לֹא שֶׁהָיוּ הַשְּׁבָטִים מְבָרְכִין וּמְקַלְּלִין, אֶלָּא הַלְוִיִּם הָעוֹמְדִים בֵּין שְׁנֵי הֶהָרִים הוֹפְכִין פְּנֵיהֶם לְהַר גְּרִזִּים בַּבְּרָכָה וּלְהַר עֵיבָל בַּקְּלָלָה. רַשִׁ"י. שֶׁנֶּאֱמַר וְעָנוּ הַלְוִיִּם וְאָמְרוּ: (טז) וּפָתְחוּ כוּ'. בַּגְּמָרָא שָׁם, יָכוֹל יִהְיוּ כָּל הַבְּרָכוֹת קוֹדְמוֹת לַקְּלָלוֹת. תַּלְמוּד לוֹמַר בְּרָכָה וּקְלָלָה, בְּרָכָה אַחַת קוֹדֶמֶת לַקְּלָלָה וְאֵין כָּל הַבְּרָכוֹת קוֹדְמוֹת לַקְּלָלוֹת. וּלְהָקִישׁ בְּרָכָה לַקְּלָלָה, שֶׁשְּׁנֵיהֶם בַּלְוִיִּם, וּבְקוֹל רָם, וּבִלְשׁוֹן הַקֹּדֶשׁ, וּבִכְלָל וּפְרָט, וְאֵלּוּ וָאֵלּוּ עוֹנִין אָמֵן: (יז) בְּשִׁבְעִים לָשׁוֹן. כָּתַב רַבֵּנוּ בְּחַיֵּי וְכָתַבְתָּ עֲלֵיהֶן אֶת כָּל דִּבְרֵי הַתּוֹרָה הַזֹּאת, זֶה יוֹרֶה שֶׁכָּל הַתּוֹרָה כֻּלָּהּ כְּתוּבָה עֲלֵיהֶן. וְכֵן כָּתַב הָרַמְבַּ"ם שֶׁכָּל הַתּוֹרָה הָיְתָה כְּתוּבָה מִבְּרֵאשִׁית עַד לְעֵינֵי כָּל יִשְׂרָאֵל בְּתָגֶיהָ וְזִיּוּנֶיהָ, וּמִשָּׁם נֶעְתְּקוּ. וְדֶרֶךְ נֵס הָיָה בְּלֹא סָפֵק לְמַלְאוֹת שִׁבְעִים לָשׁוֹן בְּכָל הַתּוֹרָה עַל שְׁתֵּים עֶשְׂרֵה אֲבָנִים וּכְתִיבָתָם בְּיוֹם אֶחָד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (יח) כְּלוֹמַר בִּכְתָב וְלָשׁוֹן: (יט) הֵיטֵב. דְּאִם לֹא כֵן בַּאֵר הֵיטֵב לָמָּה לִי. וְיֵשׁ מְפָרְשִׁים תֵּבַת הֵיטֵב עוֹלָה שִׁבְעִים כָּזֶה ה' ה"י הי"ט הֵיטֵ"ב (וְכֵן הִיא גִּרְסָא הַנְּכוֹנָה): (כ) וְנָטְלוּ כוּ'. וּבַיְרוּשַׁלְמִי, עוֹד הוּא מַעֲשֵׂה נִסִּים, נָתַן הַקָּדוֹשׁ בָּרוּךְ הוּא בִּינָה בְּלֵב כָּל אֻמָּה וְאֻמָּה וְהִשִּׂיאוּ אֶת הַתּוֹרָה שֶׁהָיְתָה כְּתוּבָה בְּשִׁבְעִים לָשׁוֹן: (כא) כְּדִכְתִיב וְהֶעֱלֵיתָ עָלָיו עוֹלוֹת. רַשִׁ"י. וּכְתִיב וְזָבַחְתָּ שְׁלָמִים. וְקָשֶׁה לִי, דַּעֲדִיפָא הֲוָה לֵיהּ לְפָרֵשׁ דְּאַף הַכְּתִיבָה הָיְתָה אַחַר הַקָּרְבָּנוֹת, שֶׁכֵּן סִדְרָן הַכָּתוּב:
