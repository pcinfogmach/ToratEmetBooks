(א) שֶׁל גּוֹיִם. עוֹבְדֵי עֲבוֹדָה זָרָה. וְעַיֵּן בְּפֵרוּשׁ הָרַ"ב מִשְׁנָה ב': (ב) שְׁלֹשָׁה יָמִים. דִּלְהָכִין צָרְכֵי הַקְרָבָה (עַיֵּן פֶּרֶק ה' דְּחֻלִּין מִשְׁנָה ג') צָרִיךְ שְׁלֹשָׁה יָמִים. הִלְכָּךְ אָסוּר לָשֵׂאת וְלָתֵת כוּ', מִשּׁוּם דְּשֵׁם עֲבוֹדָה זָרָה שָׁגוּר בְּפִיו וְאָזֵיל וּמוֹדֶה. גְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (ג) וְעָבַר יִשְׂרָאֵל מִשּׁוּם לֹא יִשָּׁמַע עַל פִּיךָ. רַשִׁ"י. כְּלוֹמַר לֹא יִשָּׁמַע בִּגְרָם שֶׁלְּךָ. הָרַ"ן: (ד) וְלִשְׁאֹל. שֶׁהַנָּכְרִי יִשְׂמַח כְּשֶׁיִּשְׂרָאֵל צָרִיךְ לוֹ. וּבַגְּמָרָא עֲבִיד צְרִיכוּתָא: (ה) לְהַלְוֹתָן. פֵּרֵשׁ רַבֵּנוּ תָם דַּוְקָא בְחִנָּם, אֲבָל בְּרִבִּית שָׁרֵי. מִשּׁוּם דְּמִצְטַעֵר טוּבָא, כִּדְאָמְרִינַן בְּקִדּוּשִׁין דַּף כ', לְזַבֵּין אֱינִישׁ בְּרַתֵּיהּ וְלֹא לוֹזִיף בְּרִבִּית:

(ו) וַחֲכָמִים אוֹמְרִים כוּ'. חֲכָמִים הַיְנוּ תַנָּא קַמָּא. הֵן בְּלֹא אֵידֵיהֶן אִיכָּא בֵינַיְהוּ. תַּנָּא קַמָּא סָבַר הֵן בְּלֹא אֵידֵיהֶן, וְרַבָּנָן בָּתְרָאֵי סָבְרֵי הֵן וְאֵידֵיהֶן. וְאִי בָעֵית אֵימָא כוּ'. גְּמָרָא: (ז) לְשׁוֹן הָרַמְבַּ"ם בְּמִשְׁנָה ד', אֲבָל אֲנַחְנוּ אֲנוּסִים, וְנִתְקַיֵּם בָּנוּ מַה שֶּׁנֶּאֱמַר (דְּבָרִים ד) וַעֲבַדְתֶּם שָׁם אֱלֹהִים עֵץ וָאֶבֶן. עַד כָּאן: (ח) לְאַחַר אֵידֵיהֶן מֻתָּר. וְהַאִידְנָא מֻתָּר לָשֵׂאת וְלָתֵת עִם הָעַמִּים אֲשֶׁר אֲנַחְנוּ יוֹשְׁבִים בֵּינֵיהֶם, דְּכָל הָנֵי דְאִתַּסְּרוּ בְזוֹ הַמַּסֶּכֶת מַיְרֵי בְעוֹבְדֵי אֱלִילִים וַעֲבוֹדָה זָרָה מַמָּשׁ. וְאֵלּוּ הַגּוֹיִים לֹא עוֹבְדֵי עֲבוֹדָה זָרָה מַמָּשׁ הֵם, כְּלוֹמַר שֶׁאֵינָם יוֹדְעִים בְּטִיב עֲבוֹדָה זָרָה כָּל כָּךְ, וְכִדְאָמְרִינַן בְּפֶרֶק קַמָּא דְחֻלִּין דַּף י"ג ב', גּוֹיִים שֶׁבְּחוּץ לָאָרֶץ לָאו עוֹבְדֵי עֲבוֹדָה זָרָה הֵם, אֶלָּא מִנְהַג אֲבוֹתֵיהֶם בִּידֵיהֶם. בֵּית יוֹסֵף סוֹף סִימָן קמ"ח:

(ט) גְּמָרָא. וְנִרְאֶה דְּמִשּׁוּם שֶׁאֵלּוּ הַשְּׁמֹנָה יָמִים שֶׁאַחַר הַתְּקוּפָה עָשָׂה אוֹתָן יָמִים טוֹבִים קֹדֶם שֶׁעָשָׂה לְאוֹתָן שֶׁקֹּדֶם הַתְּקוּפָה, שֶׁלֹּא עֲשָׂאָן אֶלָּא לְשָׁנָה הַבָּאָה, לְפִיכָךְ הִקְדִימָן לִשְׁנוֹת, קָלֶנְדָּא קֹדֶם סְטַרְנוּרָא. וּבַיְרוּשַׁלְמִי כֵּיוָן דַּחֲזָא אַיְּמָמָא אָמַר קָלֶנְדָּא. קְלָן דִיא"וֹ. וְנִרְאֶה לִי כִּי קְלָן הוּא לְשׁוֹן שְׂרָרָה וּגְדֻלָּה. כְּמוֹ שֶׁכָּתַב הָרַ"ב פֶּרֶק ד' דְּסֻכָּה מִשְׁנָה ה', קָלֶנְיָא חָפְשִׁי מִמַּס, וְאֵין לְךָ חָפְשִׁי כְּמֶלֶךְ מַלְכֵי הַמְּלָכִים הַקָּדוֹשׁ בָּרוּךְ הוּא, וְהוּא דִיא"וֹ, שֶׁבִּלְשׁוֹן יָוָן קוֹרִין אֱלוֹהַּ דִיא"וֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (י) בִּלְשׁוֹן יָוָן הוּא קָרַט, לְשׁוֹן תְּפִיסָה. תּוֹסָפוֹת: (יא) בַּגְּמָרָא, גְּנִיסֵי מַלְכוּת, פֵּרוּשׁ תַּכְסִיסֵי מַלְכוּת. עָרוּךְ: (יב) וּבְלוֹרִיתוֹ. פֵּרוּשׁ צִיצִית שֶׁל שֵׂעָר. עָרוּךְ:

(יג) לְשָׁם. לְאוֹתָהּ הָעִיר בְּיוֹם עֲבוֹדָה זָרָה שֶׁלָּהֶם לְסַפֵּר עִם אֶחָד מִבְּנֵי הָעִיר. רַשִׁ"י: (יד) אֲסוּרוֹת. מִשּׁוּם דְּנֶהֱנֶה מִשֶּׁל עֲבוֹדָה זָרָה אָסוּר, כְּדִכְתִיב [דְּבָרִים יג] וְלֹא יִדְבַּק בְּיָדְךָ מְאוּמָה מִן הַחֵרֶם, כָּל שֶׁכֵּן לִמְהַנֶּה לַעֲבוֹדָה זָרָה דְּאָסוּר. הָרַ"ן: (טו) וְאִי מִשּׁוּם דְּנוֹשֵׂא וְנוֹתֵן בְּיוֹם אֵידָם, הָנֵי מִלֵּי לְזַבּוֹנֵי לְהוּ, דִּלְמָא מְזַבֵּן לֵיהּ בְּהֵמָה וְאָזֵיל וּמַקְרִיב לַהּ לַעֲבוֹדָה זָרָה. אִי נַמִּי דֶּרֶךְ לוֹקֵחַ לִהְיוֹת שָׂמֵחַ וְאָזֵיל וּמוֹדֶה. אֲבָל לְמִזְבַּן מִנֵּיהּ דָּבָר הַמִּתְקַיֵּם שָׁרֵי, דִּסְתָם מוֹכֵר עָצֵב הוּא. רַשִׁ"י. וְאַף עַל פִּי שֶׁהָרַ"ב פֵּרֵשׁ בְּרֵישׁ פִּרְקִין לִמְכֹּר וְלִקְנוֹת, לֹא קַשְׁיָא כִּדְפֵרֵשׁ הָרֹא"שׁ דִּלְרַשִׁ"י יֵשׁ בִּמְכִירָה צַד אִסּוּר וְצַד הֶתֵּר וְכֵן בִּלְקִיחָה. כְּלוֹמַר צַד אִסּוּר דִּמְכִירָה דָּבָר הַמִּתְקַיֵּם, וְצַד הֶתֵּר שֶׁאֵינוֹ מִתְקַיֵּם. וְדִלְקִיחָה הוּא בְהֵפֶךְ:

(טז) הָאֲסוּרִים לִמְכֹּר. לְעוֹלָם. הָרַמְבַּ"ם. וְטַעֲמָא, דְּכֵיוָן דִּלְהַקְרָבָה צָרִיךְ לָהֶם קָא עָבַר עַל לִפְנֵי עִוֵּר לֹא תִתֵּן מִכְשׁוֹל, דְּבֶן נֹחַ מֻזְהָר עַל עֲבוֹדָה זָרָה. נִרְאֶה לִי. וְהַבֵּית יוֹסֵף כָּתַב דְּאֶפְשָׁר שֶׁכָּתַב לְעוֹלָם, שֶׁאֲפִלּוּ שֶׁלֹּא בְיוֹם חַגָּם אָסוּר. אִי נַמִּי לוֹמַר שֶׁלְּעִנְיָן זֶה אֵין חִלּוּק [בֵּין בִּימֵיהֶם לְהָאִידְנָא כְדִלְעֵיל בְּמִשְׁנָה ב']. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (יז) קָשׁוּר. רַשִׁ"י: (יח) רַשִׁ"י וְתוֹסָפוֹת. עוֹד פֵּרְשׁוּ, שֶׁהוּא מִין עֵשֶׂב שֶׁבּוֹ תִחֵם יְהוֹשֻׁעַ אֶת הָאָרֶץ:

(לב) בְּכָל שֶׁהֵן. הַאי בְּכָל שֶׁהֵן אֵין פֵּרוּשׁוֹ שֶׁאֲפִלּוּ בְמַשֶּׁהוּ מֵאִסּוּרִין הַלָּלוּ יֶאֱסֹר תַּעֲרָבְתּוֹ בַהֲנָאָה, דְּהָא דַוְקָא דָבָר שֶׁבְּמִנְיָן קָא חָשֵׁיב, הִלְכָּךְ וַדַּאי הַאי כָּל שֶׁהֵן הָכִי קָאָמַר, שֶׁכָּל אֶחָד מֵאִסּוּרִים הַלָּלוּ אִם הוּא דָבָר שֶׁבְּמִנְיָן אוֹסֵר תַּעֲרָבְתּוֹ אֲפִלּוּ רִבָּה עָלָיו אֶלֶף כַּיּוֹצֵא בוֹ. הָרַ"ן: (לג) וְהָכָא כְּשֶׁאִסּוּר וְהֶתֵּר כָּל אֶחָד בְּעֵין. וְלֹא דָמֵי לְתַעֲרֹבֶת דִּלְעֵיל וּמַתְנִיתִין דִּלְקַמָּן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (לד) וְהוּא הַדִּין לִשְׁאָר הָאוֹסְרִין בְּכָל שֶׁהֵן, שֶׁיִּמָּכְרוּ חוּץ מִדְּמֵי אִסּוּר. הָרַמְבַּ"ם. וְהָרַאֲבַ"ד: (לה) דְּהַשְׁתָּא הָוֵי דָבָר הַנִּמְכָּר בְּמִנְיָן, כַּמָּה גִזּוֹת בְּסֶלַע, וְעוֹשִׂין מֵהֶן שַׂקִּין. רַשִׁ"י: (לו) וְאִם כֵּן מִשְׁנָה ה' פֶּרֶק ז' דְּחֻלִּין דְּאוֹסֶרֶת גִּיד הַנָּשֶׁה מִשּׁוּם דִּבְרִיָּה הִיא, וַחֲתִיכַת נְבֵלָה דִרְאוּיָה לְהִתְכַּבֵּד הִיא, פְּלִיגָא אַמַּתְנִיתִין, וְכֵן מַתְנִיתִין דְּרֵישׁ פֶּרֶק ח' דִּזְבָחִים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

(לז) שֶׁנָּפַל. כְּשֶׁנָּפַל בְּחָבִית גְּדוֹלָה וְנִשְׁתַּבְּרָה וְנָפַל כֻּלּוֹ כְאֶחָד לְגוֹ הֶתֵּרָא, וְלֹא אָמְרִינַן קַמָּא קַמָּא בָטִיל לֵיהּ בְּרֹב. רַשִׁ"י: (לח) וְקָא מַשְׁמַע לָן דַּאֲפִלּוּ נִתְעָרֵב הַיַּיִן מַמָּשׁ. אֲבָל כִּי נִתְעָרֵב חָבִית בְּחָבִיּוֹת אֲפִלּוּ יֵין נֶסֶךְ גָּמוּר נַמִּי שָׁרֵי בְּהוֹלָכַת הֲנָאָה לְיָם הַמֶּלַח. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

פרק ה - משנה יא

(לט) יִקְלֹף. וְאַחַר כָּךְ יְנַגְּבוֹ. הָרַ"ן וְהָרֹא"שׁ: (מ) אֲסוּרָה. וְאֵינָהּ נִתֶּרֶת בְּנִגּוּב אֶלָּא בְעֵרוּי. דְּנִגּוּב אֵינוֹ מַפְלִיט בִּכְלִי חֶרֶס. וְכֵיוָן דְּצָרִיךְ עֵרוּי, אֵין צָרִיךְ קְלִיפָה, דְּעֵרוּי מַפְלִיט אַף בְּלֹא קְלִיפָה כָּל זְמַן שֶׁלֹּא דָרַךְ בָּהּ, דְּלֹא גָרַע מִקַּנְקַנִּים שֶׁמַּכְנִיסִין לְקִיּוּם. הָרֹא"שׁ: (מא) כְּלוֹמַר, שֶׁאַף עַל פִּי שֶׁהַזֶּפֶת נִקְלַף, לֹא הוּסַר בָּזֶה טַעַם הַיַּיִן, לְפִי שֶׁהַחֶרֶס עַצְמוֹ מִלְּבַד הַזֶּפֶת בָּלַע כְּשֶׁנָּתַן הַיַּיִן בַּזֶּפֶת לְזַפְּתָהּ כוּ'. וּלְשׁוֹן הָרַ"ן, מִשּׁוּם דְּחֶרֶס בָּלַע טוּבָא:
