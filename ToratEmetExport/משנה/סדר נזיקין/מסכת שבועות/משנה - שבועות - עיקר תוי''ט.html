(ו) בַּתְּחִלָּה כוּ'. בַּגְּמָרָא יָלְפִינַן לְהוּ וּמַסִּיק דִּידִיעַת בֵּית רַבּוֹ (שֶׁלָּמַד בְּבֵית רַבּוֹ שֶׁהַנּוֹגֵעַ בְּטֻמְאָה טָמֵא) שְׁמָהּ יְדִיעָה. פְּרָט לְנִשְׁבָּה בֵּין הַנָּכְרִים: (ז) בִּפְנִים. שֶׁהֲרֵי הוּא אוֹמֵר בְּשָׂעִיר זֶה וְכִפֶּר עַל הַקֹּדֶשׁ, מִטֻּמְאָה שֶׁבַּקֹּדֶשׁ. יָכוֹל עַל כָּל טֻמְאָה שֶׁבַּקֹּדֶשׁ יְהֵא שָׂעִיר זֶה מְכַפֵּר, תַּלְמוּד לוֹמַר וּמִפִּשְׁעֵיהֶם לְכָל חַטֹּאתָם, חֲטָאִים דּוּמְיָא דִפְשָׁעִים, מַה פְּשָׁעִים שֶׁאֵינָן בְּנֵי קָרְבָּן, אַף חֲטָאִים שֶׁאֵינָן בְּנֵי קָרְבָּן. וּמִנַּיִן לְיֵשׁ בָּהּ יְדִיעָה בַתְּחִלָּה, פֵּרֵשׁ הָרַ"ב מִשְׁנָה דִלְקַמָּן: (ח) שְׁלֹשָׁה שְׂעִירִים נַעֲשִׂין לַצִּבּוּר בְּיוֹם הַכִּפּוּרִים, אֶחָד מַתַּן דָּמוֹ לִפְנִים, וַחֲבֵרוֹ מִשְׁתַּלֵּחַ לַעֲזָאזֵל, וְאֶחָד קָרֵב בַּמִּזְבֵּחַ הַחִיצוֹן בְּמוּסָפִין. וְהוּא כָתוּב בְּפָרָשַׁת פִּינְחָס עִם שְׁאָר כָּל מוּסְפֵי הַמּוֹעֲדוֹת. רַשִׁ"י: (ט) וְיוֹם הַכִּפּוּרִים. הוּא עַצְמוֹ מוֹעִיל עִם הַקָּרְבָּנוֹת. שֶׁנֶּאֱמַר [וַיִּקְרָא כג] כִּי יוֹם כִּפֻּרִים הוּא. רַשִׁ"י: (י) שֶׁאַף עַל פִּי שֶׁאֵינוֹ יוֹדֵעַ שֶׁחָטָא, יֵשׁ לוֹ לִדְאֹג. שֶׁכָּל הַשּׁוֹגְגִים צְרִיכִין כַּפָּרָה לִכְשֶׁיֵּדְעוּ, אַלְמָא קֹדֶם יְדִיעָה עֲנוּשִׁים הֵן. רַשִׁ"י: (יא) וַחֲדָא מִתַּרְתֵּי נָקַט. וְהוּא הַדִּין שֶׁנִּכְנַס לַמִּקְדָּשׁ:

(יב) לְמַסְקָנָא דִגְמָרָא שֶׁכָּתַבְתִּי בְרֵישׁ מַתְנִיתִין דִּלְעֵיל, לֹא מַשְׁכַּחַת אֶלָּא בְתִינוֹק שֶׁנִּשְׁבָּה בֵּין הַנָּכְרִים: (יג) וְאִם תֹּאמַר כֵּיוָן שֶׁהֻקְּשׁוּ זֶה לָזֶה יְכַפֵּר כָּל אֶחָד אַדִּידֵיהּ וְאַדְּחַבְרֵיהּ. וְנָפְקָא מִנַּהּ לְטֻמְאָה שֶׁאֵרַע בֵּין זֶה לָזֶה, וְאִי נַמִּי אִי לֹא עֲבַד חִיצוֹן. מְשָׁנֵי בַגְּמָרָא אָמַר קְרָא אַחַת, כַּפָּרָה אַחַת מְכַפֵּר, וְאֵינוֹ מְכַפֵּר שְׁתֵּי כַפָּרוֹת. אָמַר קְרָא אַחַת בַּשָּׁנָה, כַּפָּרָה זוֹ לֹא תְהֵא אֶלָּא אַחַת בַּשָּׁנָה:

(יד) גְּמָרָא, וּנְכַפֵּר נַמִּי אֲפִלּוּ בִשְׁאָר עֲבֵרוֹת (כָּל חֵטְא שֶׁלֹּא נוֹדַע). תָּנָא דְבֵי רַבִּי יִשְׁמָעֵאל הוֹאִיל וְזֶה בָא בִזְמַן קָבוּעַ (שְׂעִיר יוֹם הַכִּפּוּרִים דְּחִיצוֹן) וְזֶה בָא בִזְמַן קָבוּעַ, מַה זֶּה אֵינוֹ מְכַפֵּר אֶלָּא עַל טֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו אַף כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

(טו) בַּגְּמָרָא, עֲצֶרֶת [וְיוֹם הַכִּפּוּרִים] דְּלֹא כְתִיב בְּהוּ וּשְׂעִיר, מְנָלָן. אֶלָּא אָמַר רַבִּי יוֹנָה אָמַר קְרָא אֵלֶּה תַּעֲשׂוּ לַה' בְּמוֹעֲדֵיכֶם, הֻקְּשׁוּ כָל הַמּוֹעֲדוֹת כֻּלָּן זֶה לָזֶה. וְהָא רֹאשׁ חֹדֶשׁ לָאו מוֹעֵד הוּא, אִבְּרָא כוּ' רֹאשׁ חֹדֶשׁ נַמִּי אִקְּרֵי מוֹעֵד, דִּכְתִיב [אֵיכָה א] קָרָא עָלַי מוֹעֵד כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (טז) הָרְגָלִים. דְּאִית לֵיהּ נַמִּי הֶקֵּשָׁא דִבְמוֹעֲדֵיכֶם דְּהֻקְּשׁוּ כֻלָּן לִדְרֹאשׁ חֹדֶשׁ, מַה דְּרֹאשׁ חֹדֶשׁ בְּמִלְּתָא דְקֹדֶשׁ מְכַפֵּר, אַף דִּרְגָלִים כוּ'. וְכִי תֵימָא נִיכַפְּרוּ אַדְּרֹאשׁ חֹדֶשׁ, וְאֹתָהּ נָתַן לָכֶם לָשֵׂאת אֶת עֲוֹן הָעֵדָה וְגוֹ', אֹתָהּ נוֹשֵׂא עָוֹן, וְאֵין אַחֵר נוֹשֵׂא עָוֹן כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (יז) עַל הַטָּהוֹר כוּ'. דְּאָמַר קְרָא וְאֹתָהּ נָתַן לָכֶם לָשֵׂאת אֶת עֲוֹן הָעֵדָה, וְהַאי קְרָא בְּשָׂעִיר דְּרֹאשׁ חֹדֶשׁ כְּתִיב, וְיָלֵיף עָוֹן עָוֹן מִצִּיץ, מַה לְּהַלָּן טֻמְאַת בָּשָׂר, שֶׁאֵין הַצִּיץ מְרַצֶּה עַל טֻמְאַת הַגּוּף אֶלָּא עַל טֻמְאַת חֵלֶב דָּם וּבָשָׂר שֶׁקָּרַב בְּטֻמְאָה, אַף כָּאן טֻמְאַת בָּשָׂר כוּ'. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (יח) שָׁוָה. דְּיָלֵיף הֶקֵּשָׁא דְמוֹעֲדֵיכֶם אַף לַחִיצוֹן דְּיוֹם הַכִּפּוּרִים. דַּעֲדִיפָא לֵיהּ הַךְ הֶקֵּשָׁא מֵהֶקֵּשָׁא דְּלַפְּנִימִי. תּוֹסָפוֹת:

(יט) שֶׁהוּא אֵינוֹ מְכַפֵּר כַּפָּרָתָן, כַּפָּרָה אַחַת מְכַפֵּר וְאֵינוֹ מְכַפֵּר שְׁתֵּי כַפָּרוֹת. וְהֵן אֵינָן מְכַפְּרִין כַּפָּרָתוֹ, אַחַת בַּשָּׁנָה, כַּפָּרָה זוֹ לֹא תְהֵא אֶלָּא אַחַת בַּשָּׁנָה. גְּמָרָא:

(כ) מִשּׁוּם דִּלְרַבִּי שִׁמְעוֹן אַלִּימָא קֻשְׁיָא, דַּחֲלוּקִים הֵם שְׂעִירֵי רָאשֵׁי חֳדָשִׁים גַּם מִשְּׂעִירֵי הָרְגָלִים, לְהָכִי פָּרֵיךְ לֵיהּ. וְהוּא הַדִּין נַמִּי דַּהֲוָה מָצֵי לְמִפְרַךְ לְתַנָּא קַמָּא מִשְּׂעִירֵי יוֹם הַכִּפּוּרִים עִם שְׁאָר הַשְּׂעִירִים. וּמִסְתַּבֵּר נַמִּי דְּמַה שֶּׁכָּתַב הָרַ"ב וְכֵן רַשִׁ"י, רַבִּי מֵאִיר הוּא דְּקָאָמַר לֵיהּ, לָאו דַּוְקָא כוּ', אֶלָּא חֲכָמִים הֵשִׁיבוּ לוֹ לְרַבִּי שִׁמְעוֹן אַלִּיבָּא דְרַבִּי מֵאִיר:

(כא) מוֹסִיף כוּ'. דְּלֹא דָרֵישׁ אֹתָהּ נוֹשֵׂא עָוֹן וְאֵין אַחֵר כוּ'. וּסְבִירָא לֵיהּ נַמִּי דְּאִתַּקּוּשׁ רְגָלִים לַשָּׂעִיר הַחִיצוֹן כוּ'. תּוֹסָפוֹת: (כב) מוֹסִיף כוּ'. וְדִרְגָלִים לֹא מְכַפְּרֵי אַדְּיוֹם הַכִּפּוּרִים, דִּסְבִירָא לֵיהּ דְּרָשָׁא דְאַחַת בַּשָּׁנָה. וּדְיוֹם הַכִּפּוּרִים מוֹסִיף, דְּאָמַר קְרָא וְכִפֶּר אַהֲרֹן עַל קַרְנֹתָיו אַחַת בַּשָּׁנָה, קַרְנוֹתָיו דְּמִזְבֵּחַ הַפְּנִימִי הוּא דְאֵינוֹ מְכַפֵּר שְׁתֵּי כַפָּרוֹת, הָא דְחִיצוֹן אֲפִלּוּ שְׁתֵּי כַפָּרוֹת. גְּמָרָא:

(א) שְׁבוּעַת כוּ'. שְׁבוּעָה לֹא כְתִיב בְּהֶדְיָא בַקְּרָא, אֶלָּא יָלְפִינַן מִקּוֹל, כְּמוֹ שֶׁכָּתוּב בְּרֵישׁ מַתְנִיתִין דְּסוֹף פִּרְקִין: (ב) כְּלוֹמַר, אַף עַל גַּב דְּאֵינוֹ בַר מַלְקוֹת, שֶׁנִּתָּק לַעֲשֵׂה דְהִשָּׁבוֹן: (ג) מִכְּלָל דְּמֶלֶךְ מִדְּאוֹרַיְתָא אֵינוֹ כָשֵׁר לְהָעִיד, וְהָכִי אִיתָא בְהֶדְיָא בַגְּמָרָא דִּמְמַעֲטִינַן מְשַׂחֵק בְּקֻבְיָא וְכָל שֶׁכֵּן מֶלֶךְ. וּכְלוֹמַר שֶׁהוּא מִדְּאוֹרַיְתָא. וּפֵרֵשׁ רַשִׁ"י, דִּכְתִיב (דְּבָרִים יז) שׂוֹם תָּשִׂים עָלֶיךָ מֶלֶךְ, שֶׁתְּהֵא אֵימָתוֹ עָלֶיךָ, וְהַבָּא לְהָעִיד צָרִיךְ לַעֲמֹד לִפְנֵי הַדַּיָּן וְדֶרֶךְ בִּזָּיוֹן הוּא לוֹ. וּלְפִי זֶה אַף בְּמַלְכֵי בֵית דָּוִד אָיְרִינַן, דְּאֵינָן רְאוּיִין לְהָעִיד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (ד) מִפִּי עַצְמוֹ. דְּיָלֵיף לַהּ מִגְּזֵרָה שָׁוָה דְתֶחֱטָא תֶּחֱטָא מִפִּקָּדוֹן. וְסָבַר דּוּן מִנַּהּ. וּמִנַּהּ כְּלוֹמַר, דְּדָיְנִינַן מִנַּהּ לְחַיְּבוֹ אַף מִפִּי עַצְמוֹ, וּמִנַּהּ כִּי הָתָם דְּמִחַיְּבֵי בֵּין בְּבֵית דִּין בֵּין שֶׁלֹּא בְבֵית דִּין. דְּנִשְׁבַּע עַל שָׁקֶר כְּתִיב, מַשְׁמַע בֵּין בְּבֵית דִּין בֵּין שֶׁלֹּא בְבֵית דִּין. רַשִׁ"י: (ה) וְכֵן אִם עָנָה אָמֵן אַחַר הַשְּׁבוּעָה. רַשִׁ"י: (ו) אַף עַל גַּב דְּלֹא עָנָה אָמֵן. שֶׁכֵּן הוּא אוֹמֵר וְשָׁמְעָה קוֹל אָלָה אִם לוֹא יַגִּיד וְנָשָׂא עֲוֹנוֹ: (ז) שֶׁנֶּאֱמַר וְשָׁמְעָה קוֹל אָלָה: (ח) בֵּין כוּ'. דִּסְבִירָא לְהוּ דּוּן מִנַּהּ וְאוֹקֵי בְאַתְרַהּ. כְּלוֹמַר, לְאַחַר שֶׁהֵבֵאתָ מִפִּי עַצְמוֹ לִשְׁבוּעַת הָעֵדוּת מִפִּקָּדוֹן, אוֹקֵי בְאַתְרַהּ בְּדִין הַכָּתוּב בִּשְׁבוּעַת הָעֵדוּת, דְּמִפִּי אֲחֵרִים שֶׁמִּשְׁפָּטוֹ דַוְקָא בְבֵית דִּין, הָכִי נַמִּי מִפִּי עַצְמוֹ שֶׁהֵבֵאתָ אֵינוֹ אֶלָּא בְבֵית דִּין:

(ט) וּמִמֵּילָא שֶׁיּוֹדֵעַ לוֹ עֵדוּת. דְּאִי לָאו הָכִי אֵינָהּ אֲסוּרָה. וְהַיְנוּ זְדוֹן הָעֵדוּת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (י) רַשִׁ"י. וְתָמְהוּ הַתּוֹסָפוֹת, דְּתֵיפּוּק לֵיהּ דִּבְקוּשְׁטָא קָא מִשְׁתַּבְעֵי. וְתֵרֵץ הָרַ"ן, דְּרַשִׁ"י הָכִי קָאָמַר, שֶׁבְּאוֹתָהּ שָׁעָה שֶׁנִּשְׁבְּעוּ אֵין אָנוּ יוֹדְעִין, יוֹדְעִין הָיוּ אֶלָּא שֶׁלֹּא הָיוּ מִתְכַּוְּנִים לָשׂוּם לֵב, אֲבָל לֹא הָיוּ צְרִיכִין יְדִיעָה אַחֶרֶת, וְנִמְצָא מַה שֶּׁנִּשְׁבְּעוּ אֵין אָנוּ יוֹדְעִים לֹא נִשְׁבְּעוּ עַל הָאֱמֶת, שֶׁאִלּוּ הָיוּ שָׂמִים לֵב בְּאוֹתָהּ שָׁעָה יוֹדְעִים הָיוּ, שֶׁהֲרֵי מִיָּד כְּשֶׁנִּתְיַשְּׁבוּ נִזְכְּרוּ בְּלֹא שׁוּם חִדּוּשׁ יְדִיעָה, וַאֲפִלּוּ הָכִי כֵּיוָן שֶׁהָיוּ סְבוּרִים לִשָּׁבַע עַל הָאֱמֶת, פְּטוּרִים, מִשּׁוּם דַּאֲנוּסִים הָיוּ כוּ'. אֲבָל וַדַּאי דְּהוּא הַדִּין וְכָל שֶׁכֵּן כְּשֶׁסְּבוּרִים שֶׁמֻּתָּרִים לְגַמְרֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

(יא) שֶׁאָמְרוּ. בְּלֹא שְׁבוּעָה, אֵין אָנוּ כוּ', וְאָמַר לָהֶם מַשְׁבִּיעַ אֲנִי עֲלֵיכֶם שֶׁיְּהֵא אֱמֶת מַה שֶּׁאֲמַרְתֶּם, וְאָמְרוּ אָמֵן, חַיָּבִים. אֲבָל אִם שָׁתְקוּ אַחַר הַשְּׁבוּעָה, אֵין כָּאן קַבָּלַת שְׁבוּעָה. רַשִׁ"י. וְנִרְאֶה מִלְּשׁוֹנוֹ דְּדַוְקָא שָׁתְקוּ, אֲבָל כָּפְרוּ וְאָמְרוּ אֵין אָנוּ יוֹדְעִים כוּ' חַיָּבִים. וּמַיְרֵי בְבֵית דִּין. וּמַתְנִיתִין אֲפִלּוּ כְרַבָּנָן. וְאָמֵן לָאו דַּוְקָא, וְלֹא נָקְטוּ אֶלָּא כְדֵי לְקַצֵּר. וְכֵן כָּתְבוּ הַתּוֹסָפוֹת, וְעַיֵּן בְּסָמוּךְ (אוֹת יג): (יב) חוּץ לְבֵית דִּין. אָתְיָא נַמִּי כְרַבָּנָן, דִּלְרַבָּנָן נַמִּי לֹא בָעִינַן שְׁבוּעָה בְבֵית דִּין, אֶלָּא הַכְּפִירָה. דְּאַהַגָּדָה דַוְקָא קָפֵיד קְרָא דְּלֶהֱוֵי בְמָקוֹם הָרָאוּי לְהַגָּדָה. תּוֹסָפוֹת:

(יג) וְרֵישָׁא דְתָנֵי הֲרֵי אֵלּוּ חַיָּבִין, וְלֹא אָמְרִינַן דַּהֲוֵי כְפִירַת דְּבָרִים בְּעָלְמָא כֵּיוָן דִּכְבָר אָמְרוּ אֵין אָנוּ יוֹדְעִים כוּ' וְשׁוּב אֵין יְכוֹלִין לַחֲזֹר וּלְהוֹדוֹת, נִרְאֶה לִי מִשּׁוּם דְּלֹא קַפְּדָה הַתּוֹרָה אַחוֹזֵר וּמַגִּיד אֶלָּא בָתַר שְׁבוּעָה. דְּאִם לֹא יַגִּיד, בָּתַר וְשָׁמְעָה קוֹל אָלָה כְּתִיב. וְהַטַּעַם מִפְּנֵי שֶׁדֶּרֶךְ הָעֵדִים לְהִשָּׁמֵט שֶׁלֹּא לְהָעִיד, וּכְשֶׁאוֹמְרִים בָּתַר שְׁבוּעָה יוֹדְעִים אָנוּ, אֵין זֶה חוֹזֵר כוּ'. עוֹד נִרְאֶה לִי, דְּאֵין אָנוּ יוֹדְעִים לֹא מִקְּרֵי מַגִּיד כְּלָל, דְּהָא לֹא הִגִּידוּ כְלוּם, וְהָכָא לָאו מִשּׁוּם חוֹזֵר וּמַגִּיד אַיְתִּינַן עֲלַהּ, אֶלָּא מִשּׁוּם דְּכָפְרוּ בָתַר שְׁבוּעָה שׁוּב לֹא מָצוּ מְשַׁוּוּ נַפְשַׁיְהוּ רְשִׁיעֵי לוֹמַר יוֹדְעִים אָנוּ. וּבְהָכִי מִתָּרְצָא אַמַּאי נָקַט רֵישָׁא בְּמֻשְׁבָּע מִפִּי עַצְמוֹ, דִּלְאַשְׁמוֹעִינַן הַךְ רְבוּתָא נָקַט בְּמִפִּי עַצְמוֹ. הָרַ"ן. כְּלוֹמַר דְּאִלּוּ הֲוָה נָקַט בְּמֻשְׁבָּע מִפִּי אֲחֵרִים, לֹא הָיָה יָכוֹל לְהַתְחִיל אָמְרוּ אֵין אָנוּ יוֹדְעִים כוּ', לְפִי שֶׁשּׁוּב אַחַר כָּךְ כְּשֶׁמַּשְׁבִּיעָן אֵין רְגִילוּת לְהָשִׁיב אֶלָּא אָמֵן, וְאִם כֵּן לֹא הֲוָה יָדְעִינַן רְבוּתָא זוֹ:

(יב) וְלֹא קַשְׁיָא דְּאִם כֵּן יִתְבָּעֵהוּ לְאוֹתוֹ אַחֵר. דְּהָכִי קָאָמַר, שֶׁמָּא אַחֵר עָשָׂה עַל פִּיו דְּנֶחְבָּל, שֶׁאָמַר לוֹ לַחֲבֹל בּוֹ כְּדֵי שֶׁיִּתְבַּע לָזֶה, מִידֵי דַהֲוָה דְחָיְשִׁינַן שֶׁמָּא חָבַל בְּעַצְמוֹ. וְהָכִי אִיתָא בַגְּמָרָא:

(יג) הָעֵדוּת. הִקְשָׁה הָרַשְׁבָּ"א זַ"ל, מַאן מוֹכָח דְּעָבַר עֲלֵיהּ בְּמֵזִיד, דִּלְמָא אִשְׁתְּלֵי וְלֹא מֵזִיד הוּא. וּלְדִידִי לֹא קַשְׁיָא, דְּכָל שֶׁהָיָה עֵדוּת קָרוֹב כָּל כָּךְ דְּאִי אֶפְשָׁר לִתְלוֹת בּוֹ שִׁכְחָה, לֹא אָמְרִינַן דִּלְמָא אִשְׁתְּלֵי. וּבְמוֹדֶה שֶׁהֵזִיד נַמִּי מַשְׁכַּחַת לַהּ, וְלֹא אָמְרִינַן בָּזֶה אֵין אָדָם מֵשִׂים עַצְמוֹ רָשָׁע, דְּהָא הוֹדָאַת מָמוֹן הוּא, וְהוֹדָאַת בַּעַל דִּין כְּמֵאָה עֵדִים דָּמֵי. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (יד) לְהוֹצִיא שֶׁקֶר מִפִּיו. רַשִׁ"י: (טו) שֶׁכְּנֶגְדּוֹ כוּ'. וְלֹא אָמְרִינַן מִתּוֹךְ שֶׁאֵינוֹ יָכוֹל לִשָּׁבַע מְשַׁלֵּם, כְּמוֹ שֶׁכָּתַב הָרַ"ב פֶּרֶק ח' דְּבָבָא מְצִיעָא מִשְׁנָה ב', כְּדֵי שֶׁלֹּא יַפְקִיעוּ מִמֶּנּוּ מָמוֹן, שֶׁמִּי שֶׁהָיָה רוֹצֶה הָיָה מַלְוֶה לוֹ פְרוּטָה וְיִתְבַּע כָּל אֲשֶׁר לוֹ, וִיהֵא מוֹדֶה מִקְצָת כַּמָּה שֶׁחַיָּב, וְיִתְחַיֵּב שְׁבוּעָה וּמִתּוֹךְ שֶׁאֵינוֹ יָכוֹל לִשָּׁבַע יִקַּח כָּל אֲשֶׁר לוֹ בְּלֹא שְׁבוּעָה, לְכָךְ תִּקְּנוּ שֶׁיִּשָּׁבַע כְּנֶגְדּוֹ. מַה שֶּׁאֵין כֵּן הָתָם שֶׁזֶּה שֶׁאֵינוֹ יָכוֹל לִשָּׁבַע לֹא מֵחֲמַת שֶׁהוּא חָשׁוּד אֶלָּא עַתָּה בִתְבִיעָה זוֹ אֵינוֹ יָכוֹל לִשָּׁבַע מֵחֲמַת הַמְאֹרָע, לֹא הִצְרִיכוּ לִשָּׁבַע שֶׁכְּנֶגְדּוֹ. תּוֹסָפוֹת: (טז) וְכֵן פֵּרֵשׁ הָרַמְבַּ"ם. וּמְסַיֵּם, וְאֵין לוֹ עַל אוֹתוֹ שֶׁתְּבָעוֹ זוּלַת חֵרֶם סְתָם. וַהֲלָכָה כְּרַבִּי מֵאִיר וְלֹא כְרַבִּי יוֹסֵי דְּאָמַר יַחֲלֹקוּ. עַד כָּאן. הָכִי מַסְּקִינַן בַּגְּמָרָא, דְּרַבִּי יוֹסֵי הוּא דְּאָמַר יַחֲלֹקוּ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

פרק ז - משנה ה

(יז) לֹא כוּ'. אֶלָּא בִזְמַן שֶׁיֵּשׁ רַגְלַיִם לַדָּבָר, שֶׁבַּעַל הַבַּיִת מוֹדֶה שֶׁאָמַר לוֹ תֵּן כוּ'. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (יח) לִבְנִי כוּ'. וְהָא דְקָתָנֵי פוֹעֲלִים וּבָנִים, רְבוּתָא אַשְׁמְעִינַן, דְּלֹא תֵימָא הָא דְחֶנְוָנִי מְחֻיָּב שְׁבוּעָה הַיְנוּ כְשֶׁהִמְחָה פוֹעֲלִים אֶצְלוֹ שֶׁהוּא צָרִיךְ לְשַׁלֵּם נַמִּי לַפּוֹעֲלִים, אֲבָל הֵיכָא דְאֵינוֹ חַיָּב לְשַׁלֵּם אֶלָּא לַחֶנְוָנִי בִלְבַד, סָלְקָא דַעְתָּךְ שֶׁיִּטֹּל חֶנְוָנִי בְּלֹא שְׁבוּעָה, קָא מַשְׁמַע לָן. הָרַ"ן: (יט) בְּסֶלַע כוּ'. כְּלוֹמַר, בְּסֶלַע שֶׁאֶתֵּן לְךָ תֵּן לָהֶם מָעוֹת שֶׁיּוּכַל הַפּוֹעֵל לְהוֹצִיא אוֹתָם בִּפְרוֹטְרוֹט: (כ) וְהֵן כוּ'. וְלֹא הָוֵי כְאוֹמֵר הִלְוִיתַנִי וְאֵינִי יוֹדֵעַ אִם פְּרַעְתִּיךָ דְּחַיָּב לְשַׁלֵּם, דְּהָכָא כֵּיוָן דְּמַפְסִיד בַּעַל הַבַּיִת, סְבִירָא לֵיהּ לְתַנָּא קַמָּא דְּתִקְּנוּ רַבָּנָן שְׁבוּעָה שֶׁיִּשָּׁבְעוּ פוֹעֲלִים. הָרַ"ן: (כא) אֵלּוּ וָאֵלּוּ. בָּאִים לְבֵית דִּין לִשָּׁבַע שְׁבוּעַת שָׁוְא מִבֵּינֵיהֶם, דְּלֵית בְּהוּ דְּאֵינוֹ מְשַׁקֵּר. רַשִׁ"י: (כב) שֶׁלֹּא כוּ'. הָרַ"ן הֶעְתִּיק, אֶלָּא הוּא נוֹטֵל בִּשְׁבוּעָה, לְפִי שֶׁאֵין חִיּוּבוֹ בָרוּר. וְהֵן נוֹטְלִין בְּלֹא שְׁבוּעָה, דַּהֲוֵי לֵיהּ כְּמָנֶה הִלְוִיתַנִי וְאֵינִי יוֹדֵעַ אִם פְּרַעְתִּיךָ:

(כג) וְנָתַן לוֹ. פֵּרוּשׁ שֶׁהִגְבִּיהָן לוֹקֵחַ וְהִנִּיחָן בִּרְשׁוּת הָרַבִּים. הַמַּגִּיד. אֲבָל לְהָרַ"ן דִּלְקַמָּן אֵין נִרְאֶה שֶׁהִגְבִּיהָן הַלּוֹקֵחַ. וְכֵן נִרְאֶה דַּעַת הָרַ"ב:

(כד) וְאֵינוֹ תוֹבֵעַ מִמֶּנּוּ אֶלָּא דְמֵיהֶן. הָרַ"ן: (כה) וְזֶה לְשׁוֹן הָרַ"ן, מִשּׁוּם דְּטַעַנְתֵּיהּ דְּחֶנְוָנִי מְעַלְּיָא בְהוּ טְפֵי, לְפִי שֶׁלִּדְבָרָיו אֵין לְבַעַל הַבַּיִת שׁוּם דְּרָרָא בַּפֵּרוֹת הַלָּלוּ, וּבַעַל הַבַּיִת מוֹדֶה לַחֶנְוָנִי שֶׁהֵן שֶׁלּוֹ אֶלָּא שֶׁנָּתַן לוֹ הַדִּינָר בִּשְׁבִילָן. הָרַ"ן:

(כו) אָמַר לוֹ כוּ'. פֵּרוּשׁ חֶנְוָנִי טוֹעֵן שֶׁמַּעֲשֶׂה שֶׁהָיָה כָּךְ הָיָה, שֶׁבָּא אֵלָיו בַּעַל הַבַּיִת וְאֵין בְּיָדוֹ דִינָר וְאוֹמֵר לוֹ תֵן לִי בְדִינָר פֵּרוֹת וְעַכְשָׁיו אָבִיא לְךָ אֶת הַדִּינָר וְנָתַן לוֹ אֶת הַפֵּרוֹת וְהוֹלִיכָם לְתוֹךְ בֵיתוֹ וְאַחַר כָּךְ חָזַר אֵלָיו וּפָרַע לוֹ אֶת הַדִּינָר, וּבַעַל הַבַּיִת טוֹעֵן לֹא כִי, אֶלָּא קֹדֶם שֶׁנָּתַתִּי הַדִּינָר לֹא לָקַחְתִּי פֵרוֹת מֵעוֹלָם, וְלֹא נָתַתִּי לְךָ הַדִּינָר אֶלָּא בִשְׁבִיל אֵלּוּ הַפֵּרוֹת שֶׁהֵן בִּרְשׁוּת הָרַבִּים. הָרַ"ן: (כז) לַשֻּׁלְחָנִי. הָכָא נַמִּי שֶׁהַמָּעוֹת צְבוּרִין וּמֻנָּחִין בִּרְשׁוּת הָרַבִּים: (כח) מִשּׁוּם דְּפֵירֵי עֲבִידֵי דְּמַרְקְבֵי, לֹא מַשְׁהֵי לְהוּ. גְּמָרָא. וְהַיְנוּ טַעֲמָא דְּרַבִּי יְהוּדָה לֹא אָמַר בַּחֶנְוָנִי בְּהַהוּא לִישְׁנָא דְּאָמַר בַּשֻּׁלְחָנִי, לְפִי שֶׁדֶּרֶךְ הַחֶנְוָנִי לְהַקִּיף אֶלָּא דְהָכָא מוֹדֶה הַחֶנְוָנִי שֶׁלֹּא מָכַר בְּהַקָּפָה. הָרַ"ן וְתוֹסָפוֹת. וּבַסֵּיפָא לֹא אָמַר כֹּל שֶׁהָאִסָּר כוּ', מִשּׁוּם דְּלִפְעָמִים בַּעַל הַבַּיִת נוֹתֵן אִסָּר בִּפְרוּטוֹת, וְלֹא שַׁיָּךְ לְמֵימַר כֹּל שֶׁהָאִסָּר כוּ'. תּוֹסָפוֹת:
