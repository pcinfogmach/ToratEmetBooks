פרק א - משנה א

שְׁבוּעוֹת שְׁתַּיִם. כְּתִיבֵי בִקְרָא בְהֶדְיָא (וַיִּקְרָא ה) אוֹ נֶפֶשׁ כִּי תִשָּׁבַע לְבַטֵּא בִשְׂפָתַיִם לְהָרַע אוֹ לְהֵיטִיב. אֹכַל, לְהֵיטִיב. לֹא אֹכַל, לְהָרַע, שֶׁמְּעַנֶּה אֶת עַצְמוֹ. וְהוּא הַדִּין לְכָל דָּבָר שֶׁהוּא הֵן וְלָאו לְהַבָּא, דִּלְהָרַע אוֹ לְהֵיטִיב לְהַבָּא מַשְׁמַע: שֶׁהֵן אַרְבַּע. יֵשׁ לְרַבּוֹת עוֹד שְׁתַּיִם אֲחֵרִים דְּלֹא כְתִיבֵי. וְהֵן, הֵן וְלָאו דִּלְשֶׁעָבַר (א). כְּמוֹ אָכַלְתִּי וְלֹא אָכַל, אוֹ לֹא אָכַלְתִּי וְאָכַל: יְדִיעוֹת הַטֻּמְאָה. אוֹ נֶפֶשׁ אֲשֶׁר תִּגַּע בְּכָל דָּבָר טָמֵא (שָׁם), הָאָמוּר בְּוַיִּקְרָא אֵצֶל טֻמְאַת מִקְדָּשׁ וְקָדָשָׁיו (ב): שְׁתַּיִם. כְּתִיבֵי בִקְרָא. דִּכְתִיב (שָׁם) וְנֶעְלַם מִמֶּנּוּ וְהוּא טָמֵא, מַשְׁמַע שֶׁנֶּעֶלְמָה מִמֶּנּוּ טֻמְאָה (ג) וְעַל יְדֵי אוֹתוֹ הֶעְלֵם אָכַל בְּשַׂר קֹדֶשׁ, הֲרֵי אַחַת. אוֹ נִכְנַס לַמִּקְדָּשׁ, הֲרֵי שְׁתַּיִם: שֶׁהֵן אַרְבַּע. יֵשׁ לְרַבּוֹת עוֹד שְׁנַיִם חִיּוּבִים אֲחֵרִים (ד). הֶעְלֵם קֹדֶשׁ, וְהֶעְלֵם מִקְדָּשׁ, וְזָכוּר הוּא אֶת הַטֻּמְאָה: יְצִיאוֹת הַשַּׁבָּת. הוֹצָאָה מֵרְשׁוּת הַיָּחִיד לִרְשׁוּת הָרַבִּים, יָלְפִינַן מִקְּרָא מִן וַיְצַו מֹשֶׁה וַיַּעֲבִירוּ קוֹל בַּמַּחֲנֶה וְגוֹ' (שְׁמוֹת לו), לֹא תַּפִּיקוּ מֵרְשׁוּת הַיָּחִיד לִרְשׁוּת הָרַבִּים: שְׁתַּיִם. אַחַת לָעוֹמֵד בַּחוּץ וְהוֹשִׁיט יָדוֹ לִפְנִים וְנָטַל הַחֵפֶץ וְהוֹצִיאוֹ, וְאַחַת לָעוֹמֵד בִּפְנִים וְנָטַל הַחֵפֶץ מִמְּקוֹמוֹ וְהוֹצִיאוֹ לַחוּץ: שֶׁהֵן אַרְבַּע. יֵשׁ לְהוֹסִיף עוֹד שְׁתַּיִם דְּהַכְנָסָה (ה), אַף עַל גַּב דְּלֹא כְתִיבֵי. שֶׁכְּשֵׁם שֶׁהִקְפִּידָה תוֹרָה מֵרְשׁוּת לִרְשׁוּת בְּהוֹצָאָה, כָּךְ הִקְפִּידָה עַל הַהַכְנָסָה. אַחַת לָעוֹמֵד בִּפְנִים וְהוֹצִיא יָדוֹ לַחוּץ וְנָטַל חֵפֶץ וְהִכְנִיס, וְאַחַת לָעוֹמֵד בַּחוּץ וְנָטַל חֵפֶץ מִמְּקוֹמוֹ וְהוֹשִׁיטוֹ לִפְנִים וְהִנִּיחוֹ: מַרְאוֹת נְגָעִים שְׁנַיִם. כְּתִיבֵי בַקְּרָא (וַיִּקְרָא יג), שְׂאֵת וּבַהֶרֶת: שֶׁהֵן אַרְבָּעָה. תּוֹלָדָה דִשְׂאֵת וְתוֹלָדָה דְבַהֶרֶת, דְּלֹא כְתִיבֵי בַקְּרָא בְהֶדְיָא:

פרק א - משנה ב

שֶׁיֵּשׁ בָּהּ יְדִיעָה בַתְּחִלָּה. אַיְּדִיעוֹת הַטֻּמְאָה קָאֵי. וְאַיְדֵי דְזוּטְרֵי מִלַּיְהוּ מְפָרֵשׁ לְהוּ בָרֵישָׁא וַהֲדַר מְפָרֵשׁ שְׁבוּעוֹת דִּנְפִישֵׁי מִלַּיְהוּ. וִיצִיאוֹת וּמַרְאוֹת, כָּל חֲדָא וַחֲדָא מִפָּרְשָׁא בְמַסֶּכֶת דִּידֵיהּ, לְהָכִי לֹא פָרְשִׁינְהוּ בְּהַאי מַסֶּכְתָּא. וְלֹא תְנִינְהוּ הָכָא אֶלָּא מִשּׁוּם דְּדָמוּ לַהֲדָדֵי בִּשְׁתַּיִם שֶׁהֵן אַרְבַּע: יְדִיעָה בַתְּחִלָּה. שֶׁיָּדַע שֶׁנִּטְמָא: וִידִיעָה בַסּוֹף. מִשֶּׁאָכַל אֶת הַקֹּדֶשׁ בְּהֶעְלֵם אוֹ נִכְנַס לַמִּקְדָּשׁ בְּהֶעְלֵם וְיָצָא, נוֹדַע לוֹ שֶׁבְּטֻמְאָה אָכַל אוֹ בְּטֻמְאָה נִכְנַס: וְהֶעְלֵם בֵּינְתַּיִם. כְּשֶׁאָכַל אֶת הַקֹּדֶשׁ נֶעֶלְמָה מִמֶּנּוּ טֻמְאָה אוֹ קֹדֶשׁ, אוֹ נִכְנַס לַמִּקְדָּשׁ וְנֶעֶלְמָה מִמֶּנּוּ טֻמְאָה אוֹ מִקְדָּשׁ: הֲרֵי זֶה בְּעוֹלֶה וְיוֹרֵד. עוֹלֶה לֶעָשִׁיר, וְיוֹרֵד לֶעָנִי. עָשִׁיר מֵבִיא חַטַּאת בְּהֵמָה, וְעָנִי חַטַּאת הָעוֹף, וְדַל שֶׁבַּדַּלִּים עֲשִׂירִית הָאֵיפָה, וְהִיא מִנְחַת חוֹטֵא הָאֲמוּרָה בְּכָל מָקוֹם: שָׂעִיר הַנַּעֲשֶׂה בִפְנִים. שָׂעִיר שֶׁל יוֹם הַכִּפּוּרִים שֶׁמֵּבִיא מִדָּמוֹ לִפְנִים (ח): תּוֹלֶה. לְהָגִין עָלָיו מִן הַיִּסּוּרִין (י): עַד שֶׁיִּוָּדַע לוֹ. שֶׁאָכַל קֹדֶשׁ בְּטֻמְאָה (יא) וְיָבִיא קָרְבָּן עוֹלֶה וְיוֹרֵד:

פרק א - משנה ג

אֵין בָּהּ יְדִיעָה בַתְּחִלָּה. שֶׁלֹּא יָדַע בְּטֻמְאָה זוֹ מִיָּמָיו (יב), זֶה לֹא בָא לִכְלַל קָרְבָּן עוֹד לְעוֹלָם. שֶׁהֲרֵי אֵין קָרְבָּן בָּא אֶלָּא עַל שֶׁיֵּשׁ בּוֹ יְדִיעָה בַתְּחִלָּה: שָׂעִיר הַנַּעֲשֶׂה בַחוּץ. הַקָּרֵב בַּמִּזְבֵּחַ הַחִיצוֹן בְּמוּסָפִין שֶׁל יוֹם הַכִּפּוּרִים: וְיוֹם הַכִּפּוּרִים. עַצְמוֹ מְכַפֵּר עִם הַשָּׂעִיר: שֶׁנֶּאֱמַר. בְּשָׂעִיר הַנַּעֲשֶׂה בַחוּץ (בְּמִדְבָּר כט) שְׂעִיר עִזִּים אֶחָד חַטָּאת מִלְּבַד חַטַּאת הַכִּפֻּרִים, וְחַטַּאת הַכִּפֻּרִים הוּא הַשָּׂעִיר הַנַּעֲשֶׂה בִּפְנִים, וְהִקִּישָׁן הַכָּתוּב, לְלַמֶּדְךָ שֶׁעַל מַה שֶּׁזֶּה מְכַפֵּר זֶה מְכַפֵּר. מַה פְּנִימִי אֵינוֹ בָא אֶלָּא עַל דָּבָר שֶׁיֵּשׁ בּוֹ יְדִיעָה, וּמַאי נִיהוּ, יְדִיעָה בַתְּחִלָּה, כִּדְיָלְפִינַן מִקְּרָא דִּכְתִיב לְכָל חַטֹּאתָם, דְּמַשְׁמַע לְאוֹתָם שֶׁיָּבֹאוּ לִכְלַל חַטָּאת כְּשֶׁתִּהְיֶה לָהֶם יְדִיעָה בַסּוֹף, וְאִי אֶפְשָׁר לָבֹא לִכְלַל חַטָּאת אֶלָּא אִם כֵּן הָיְתָה לוֹ יְדִיעָה בַתְּחִלָּה, אַף שָׂעִיר הַנַּעֲשֶׂה בַחוּץ אֵינוֹ מְכַפֵּר אֶלָּא עַל שֶׁיֵּשׁ בּוֹ יְדִיעָה, וּמַאי נִיהוּ, יְדִיעָה בַסּוֹף וְלֹא בַתְּחִלָּה (יג) דְּלֹא אָתְיָא לִכְלַל קָרְבָּן, וּמְכַפֵּר זֶה כַּפָּרָה גְמוּרָה בְּלֹא תְלִיָּה:

שְׂעִירֵי הָרְגָלִים. בְּכָל הָרְגָלִים נֶאֶמְרוּ שְׂעִירֵי חַטָּאת בְּמוּסָפִין: וּשְׂעִירֵי רָאשֵׁי חֳדָשִׁים מְכַפְּרִים. דִּכְתִיב בִּשְׂעִיר רֹאשׁ חֹדֶשׁ (שָׁם כה) לְחַטָּאת לַה', חֵטְא שֶׁאֵין מַכִּיר בּוֹ אֶלָּא ה' שָׂעִיר זֶה יְהֵא מְכַפֵּר, וְהַיְנוּ שֶׁאֵין בּוֹ יְדִיעָה לֹא בַתְּחִלָּה וְלֹא בַסּוֹף (יד). וּשְׂעִירֵי רְגָלִים יָלְפִינַן מִשְּׂעִיר רֹאשׁ חֹדֶשׁ בְּהֶקֵּשׁ, דִּשְׂעִיר הֲוָה מָצֵי לְמִכְתַּב בְּכֻלְּהוּ רְגָלִים, וְכָתַב וּשְׂעִיר (טו), וָא"ו מוֹסִיף עַל עִנְיָן רִאשׁוֹן, לְהַקִּישׁ שֶׁעַל מַה שֶּׁזֶּה מְכַפֵּר אֵלּוּ מְכַפְּרִים: כָּל הַשְּׂעִירִים כַּפָּרָתָן שָׁוָה. כָּל שְׂעִירֵי הַמּוּסָפִין, בֵּין שְׂעִירֵי רְגָלִים בֵּין שְׂעִיר רֹאשׁ חֹדֶשׁ בֵּין שָׂעִיר הַנַּעֲשֶׂה בַחוּץ בְּיוֹם הַכִּפּוּרִים, כַּפָּרָתָן שָׁוָה, שֶׁהֵם מְכַפְּרִין בֵּין עַל שֶׁאֵין בָּהּ יְדִיעָה בַתְּחִלָּה וְיֵשׁ בָּהּ יְדִיעָה בַסּוֹף בֵּין עַל שֶׁאֵין בָּהּ יְדִיעָה לֹא בַתְּחִלָּה וְלֹא בַסּוֹף, בֵּין עַל טָהוֹר שֶׁאָכַל אֶת הַטָּמֵא, וְנָפְקָא מִנַּהּ לְטֻמְאָה שֶׁאֵרְעָה בֵּין זוֹ לְזוֹ. אֲבָל שָׂעִיר הַנַּעֲשֶׂה בִּפְנִים בְּיוֹם הַכִּפּוּרִים לֹא נֶחְלְקוּ עָלָיו (יט) שֶׁהוּא תוֹלֶה עַל שֶׁיֵּשׁ בָּהּ יְדִיעָה בַתְּחִלָּה וְאֵין בָּהּ יְדִיעָה בַסּוֹף, כִּדְאָמְרִינַן: הָיָה רַבִּי שִׁמְעוֹן אוֹמֵר. הַאי דַּהֲדַר נָקַט לַהּ, מִשּׁוּם דְּבָעֵי לְמֵימַר אָמְרוּ לוֹ מַהוּ שֶׁיִּקְרְבוּ זֶה בָזֶה: מַהוּ שֶׁיִּקְרְבוּ. אִם אָבַד שָׂעִיר שֶׁהֻפְרַשׁ שְׁבוּעַת הָעֵדוּת. נוֹהֶגֶת בַּאֲנָשִׁים וְלֹא בְנָשִׁים. דִּכְתִיב (וַיִּקְרָא ה) וְהוּא עֵד, בְּרָאוּי לְהָעִיד הַכָּתוּב מְדַבֵּר, וְאִשָּׁה אֵינָהּ רְאוּיָה לְהָעִיד, דִּכְתִיב (דְּבָרִים יט) וְעָמְדוּ שְׁנֵי הָאֲנָשִׁים, אֲנָשִׁים וְלֹא נָשִׁים. וְהַהוּא קְרָא בְּעֵדִים הוּא דִכְתִיב, נֶאֱמַר כָּאן שְׁנַיִם וְנֶאֱמַר לְהַלָּן שְׁנַיִם, עַל פִּי שְׁנַיִם עֵדִים (שָׁם יז): וְלֹא בִקְרוֹבִים. דִּפְסוּלֵי עֵדוּת נִינְהוּ, דִּכְתִיב (שָׁם כד) לֹא יוּמְתוּ אָבוֹת עַל בָּנִים, בְּעֵדוּת בָּנִים. וְהוּא הַדִּין לִשְׁאָר קְרוֹבִים: וְלֹא בִפְסוּלִים. כְּגוֹן חַיָּבֵי מִיתוֹת וְחַיָּבֵי מַלְקֻיּוֹת וְגַזְלָן (ב), דְּאִקְּרוּ רְשָׁעִים, וְהַתּוֹרָה אָמְרָה אַל תָּשֶׁת רָשָׁע עֵד: וְאֵינָהּ נוֹהֶגֶת אֶלָּא בָרְאוּיִין לְהָעִיד. לַאֲפוֹקֵי מֶלֶךְ שֶׁאֵינוֹ מֵעִיד, וְהַפְּסוּלִים לְעֵדוּת מִדְּרַבָּנָן (ג) כְּגוֹן מְשַׂחֲקֵי בְקֻבְיָא וּמַפְרִיחֵי יוֹנָה: בִּפְנֵי בֵית דִּין וְשֶׁלֹּא בִפְנֵי בֵית דִּין מִפִּי עַצְמוֹ. אִם מִפִּי עַצְמוֹ נִשְׁבַּע שְׁבוּעָה שֶׁאֵינִי יוֹדֵעַ לְךָ עֵדוּת (ה), חַיָּב, בֵּין נִשְׁבַּע בִּפְנֵי בֵית דִּין בֵּין שֶׁלֹּא בִּפְנֵי בֵית דִּין: וּמִפִּי אֲחֵרִים. כְּגוֹן מַשְׁבִּיעַ אֲנִי עֲלֵיכֶם שֶׁתָּבֹאוּ וּתְעִידוּנִי, וְאָמְרוּ לוֹ אֵין אָנוּ יוֹדְעִים לְךָ עֵדוּת (ו): אֵינָם חַיָּבִין עַד שֶׁיִּכְפְּרוּ בְּבֵית דִּין. דִּכְתִיב (וַיִּקְרָא ה) אִם לוֹא יַגִּיד וְנָשָׂא עֲוֹנוֹ, בְּמָקוֹם שֶׁאִלּוּ הָיָה מַגִּיד הָיָה מוֹעִיל. וּקְרָא בְּמֻשְׁבָּע מִפִּי אֲחֵרִים כְּתִיב (ז). וְאֵין הֲלָכָה כְּרַבִּי מֵאִיר:

פרק ד - משנה ב

וְחַיָּבִין. קָרְבָּן עוֹלֶה וְיוֹרֵד עַל זְדוֹן הַשְּׁבוּעָה. שֶׁהֲרֵי לֹא נֶאֱמַר בָּהּ וְנֶעְלַם: וְעַל שִׁגְגָתָהּ עִם זְדוֹן הָעֵדוּת. שֶׁיּוֹדְעִים שֶׁשְּׁבוּעָה זוֹ אֲסוּרָה (ט) אֲבָל אֵין יוֹדְעִים שֶׁחַיָּבִים עָלֶיהָ קָרְבָּן: וְאֵין חַיָּבִים עַל שִׁגְגָתָהּ. אִם שׁוֹגְגִים הֵם לְגַמְרֵי, כִּסְבוּרִים שֶׁאֵין יוֹדְעִין לוֹ עֵדוּת וְאַחַר כָּךְ נִזְכְּרוּ (י). שֶׁהֲרֵי הֵן אֲנוּסִים וְלֹא קָרִינַן בְּהוּ וְנִשְׁבַּע עַל שָׁקֶר:

פרק ד - משנה ג

וּבָאוּ לְבֵית דִּין וְהוֹדוּ פְּטוּרִים. וַאֲפִלּוּ כָפְרוּ חוּץ לְבֵית דִּין עַל כָּל שְׁבוּעָה. שֶׁאֵין כְּפִירָה חוּץ לְבֵית דִּין חֲשׁוּבָה כְפִירָה: כָּפְרוּ בְּבֵית דִּין חַיָּבִין עַל כָּל אַחַת וְאֶחָת. דְּאָמַר קְרָא (וַיִּקְרָא ה) וְהָיָה כִי יֶאְשַׁם לְאַחַת מֵאֵלֶּה, לְחַיֵּב עַל כָּל אַחַת וְאַחַת: מַה טַּעַם. אֵין חַיָּבִים אֶלָּא אַחַת אֲפִלּוּ שָׁתְקוּ וְכָפְרוּ לְבַסּוֹף, וְלֹא אָמְרִינַן כְּפִירָה קַיְמָא אַכֻּלְּהוּ לְחַיְּבָם עַל כָּל אַחַת וְאַחַת: הוֹאִיל. וְאִלּוּ כָפְרוּ בְבֵית דִּין בְּפַעַם רִאשׁוֹנָה שׁוּב לֹא הָיוּ יְכוֹלִין לַחֲזֹר וּלְהוֹדוֹת, שֶׁכְּבָר הִגִּידוּ שֶׁאֵינָם יוֹדְעִים לוֹ שׁוּם עֵדוּת וְשׁוּב אֵין חוֹזְרִים וּמַגִּידִים (יג). הִלְכָּךְ אַף עַל גַּב דְּלֹא כָפְרוּ אֶלָּא לְבַסּוֹף, כָּל הַשְּׁבוּעוֹת הָיוּ לְבַטָּלָה חוּץ מִן הָרִאשׁוֹנָה. דְּאִם מַה שֶּׁשָּׁתְקוּ בָרִאשׁוֹנָה כְּפִירָה הִיא, שׁוּב אֵינָם רְאוּיִים לְהַשְׁבִּיעָן, וְאִם אֵינָהּ כְּפִירָה הֲרֵי מֻשְׁבָּעִין וְעוֹמְדִין, וּמַה זּוֹ שְׁבוּעָה עַל שְׁבוּעָה. עַל כָּרְחֲךָ כִּי כְתִיב לְאַחַת לְחַיֵּב עַל כָּל אַחַת וְאַחַת, אַשְּׁבוּעַת חוּץ לְבֵית דִּין וְכָפְרוּ בְבֵית דִּין כְּתִיב, דִּשְׁבוּעוֹת חוּץ לְבֵית דִּין שֶׁהָיָה מַשְׁבִּיעָן שֶׁיָּבֹאוּ לְבֵית דִּין וְיָעִידוּ לֹא יָצְאוּ לְבַטָּלָה, וּרְאוּיוֹת לֵחָלֵק, שֶׁאִם כָּפְרוּ שָׁם בָּרִאשׁוֹנָה הָיָה רָאוּי לַחֲזֹר וּלְהַשְׁבִּיעָן:

פרק ד - משנה ד

כָּפְרוּ שְׁנֵיהֶם כְּאֶחָד. בְּתוֹךְ כְּדֵי דִבּוּר (יד): בָּזֶה אַחַר זֶה. שֶׁהָיָה בֵין זֶה לָזֶה יוֹתֵר מִכְּדֵי דִבּוּר: הָרִאשׁוֹן חַיָּב. אֲבָל הַשֵּׁנִי פָּטוּר, שֶׁכֵּיוָן שֶׁכָּפַר הָרִאשׁוֹן שׁוּב אֵין הַשֵּׁנִי רָאוּי לְהָעִיד, שֶׁהוּא יְחִידִי (טו): שְׁתֵּיהֶן חַיָּבוֹת. בַּגְּמָרָא פָרֵיךְ, רִאשׁוֹנָה אַמַּאי חַיֶּבֶת, הָא קַיְמָא כַּת שְׁנִיָּה, וּמַה הִפְסִידוּהוּ הָרִאשׁוֹנִים בִּכְפִירָתָם (יז). וּמְשָׁנֵי, דְּמַתְנִיתִין אַיְרֵי כְּגוֹן שֶׁהָיוּ הָעֵדִים שֶׁל כַּת שְׁנִיָּה קְרוֹבִים בִּנְשׁוֹתֵיהֶן (יח), שֶׁלֹּא הָיוּ כְשֵׁרִים לְהָעִיד כְּשֶׁכָּפְרָה כַת רִאשׁוֹנָה, וּנְשׁוֹתֵיהֶן גּוֹסְסוֹת. מַהוּ דְּתֵימָא כֵּיוָן דְּקַיְמָא לָן רֹב גּוֹסְסִים לְמִיתָה הָווּ לְהוּ כְּאִלּוּ מֵתוּ וְתִפָּטֵר כַּת רִאשׁוֹנָה דְּהָא קַיְמָא שְׁנִיָּה, קָא מַשְׁמַע לָן הַשְׁתָּא מִיהָא לֹא שָׁכוּב (יט), וְנִמְצָא שֶׁלֹּא הָיָה שָׁם בִּשְׁעַת כְּפִירָה אֶלָּא כַת רִאשׁוֹנָה בִלְבַד, וּלְפִיכָךְ חַיָּבִין:

פרק ד - משנה ה

תְּשׂוּמֶת יָד. הַלְוָאָה. שֶׁשָּׂם בְּיָדוֹ מָמוֹן: וַאֲבֵדָה. שֶׁמָּצָא אֲבֵדָתִי: חִטִּין וּשְׂעֹרִין וְכֻסְּמִין. רֵישָׁא אַשְׁמוֹעִינַן מִין אֶחָד וּתְבִיעוֹת חֲלוּקוֹת. וְסֵיפָא אַשְׁמוֹעִינַן תְּבִיעָה אַחַת כֻּלָּם פִּקָּדוֹן, וּמִינִין חֲלוּקִים:

פרק ד - משנה ו

נֶזֶק וַחֲצִי נֶזֶק. בַּחֲצִי נֶזֶק צְרוֹרוֹת דְּמָמוֹנָא הוּא וְלֹא קְנָסָא (כב). דְּאָמְרִינַן בְּעֵדוּת שֶׁל קְנָס אֵין חַיָּבִין (כג): תַּשְׁלוּמֵי כֶפֶל וְתַשְׁלוּמֵי אַרְבָּעָה וַחֲמִשָּׁה. מִשּׁוּם קַרְנָא: וְשֶׁאָנַס אִישׁ פְּלוֹנִי וּפִתָּה אֶת בִּתִּי. מִשּׁוּם בֹּשֶׁת וּפְגָם דְּמָמוֹנָא הוּא (כד): וְשֶׁהִכַּנִי בְנִי. הַכָּאָה שֶׁאֵין בָּהּ חַבּוּרָה, שֶׁאֵינוֹ חַיָּב עָלֶיהָ מִיתָה. וְכֵיוָן דְּלֵיכָּא מִיתָה אִיכָּא מָמוֹן: וְשֶׁחָבַל בִּי חֲבֵרִי בְּיוֹם הַכִּפּוּרִים. וְנָקַט לַהּ, דְּאַף עַל גַּב דְּעָנוּשׁ כָּרֵת חַיָּב מָמוֹן:

פרק ד - משנה ז

שֶׁאֲנִי כֹהֵן. פְּטוּרִין, שֶׁאֵין הָעֵדִים חַיָּבִין אֶלָּא כְּשֶׁכָּפְרוּ בְדָבָר שֶׁיֵּשׁ בּוֹ תְּבִיעוֹת מָמוֹן: שֶׁאָנַס אִישׁ פְּלוֹנִי אוֹ פִתָּה בִתּוֹ שֶׁל פְּלוֹנִי. וּבִתּוֹ דְקָתָנֵי, אַאִישׁ פְּלוֹנִי כֹהֵן דִּלְעֵיל קָאֵי, [דְּתָנָא לְעֵיל מִנֵּיהּ שֶׁאִישׁ פְּלוֹנִי כֹהֵן אוֹ לֵוִי וְאֵינוֹ בֶן גְּרוּשָׁה, וַעֲלֵיהּ קָאֵי], אוֹ שֶׁהִשְׁבִּיעָן שֶׁאָנַס אִישׁ פְּלוֹנִי אֶת בִּתּוֹ שֶׁל אוֹתוֹ פְלוֹנִי דְאַיְרֵי בֵיהּ עַד הַשְׁתָּא. פְּטוּרִים, דְּבָעִינַן שֶׁיִּשְׁמְעוּ מִפִּי הַתּוֹבֵעַ. וּבַגְּמָרָא מוֹקֵי לַהּ בְּבָא בְהַרְשָׁאָה, וְאִי תְבִיעַת מָמוֹן אַחֵר הָוֵי, הָיוּ חַיָּבִים. וְאַשְׁמוֹעִינַן דְּאֵין נֶחְשָׁב הַמֻּרְשֶׁה בַּעַל דִּין בָּזֶה כִּשְׁאָר מָמוֹן, דְּכֵיוָן דְּמָמוֹן שֶׁהוּא שׁוֹאֵל לֹא מְטָא לִידֵיהּ מֵעוֹלָם, לֹא מָצֵי לְמִכְתַּב הַרְשָׁאָה עֲלֵיהּ: וְשֶׁחָבַל בִּי בְנִי. חֲבָלָה הִיא חַבּוּרָה. פְּטוּרִין, שֶׁאִם הֱעִידוּהוּ הָיָה מִתְחַיֵּב מִיתָה וְלֹא מָמוֹן: וְשֶׁחָבַל בִּי חֲבֵרִי וְשֶׁהִדְלִיק אֶת גְּדִישִׁי בְשַׁבָּת. דְּתַרְוַיְהוּ חִיּוּב מִיתָה נִינְהוּ וּפְטוּרִים מִן הַמָּמוֹן:

פרק ד - משנה ח

פרק ד - משנה יא

פרק ד - משנה יב

עַד שֶׁיִּשְׁמְעוּ מִפִּי הַתּוֹבֵעַ. דִּכְתִיב (שָׁם) אִם לוֹא יַגִּיד, מָלֵא בְּוָי"ו וְאָלֶ"ף. לְמִדְרַשׁ הָכִי, אִם לוֹ לֹא יַגִּיד, כְּלוֹמַר לַתּוֹבֵעַ, וְנָשָׂא עֲוֹנוֹ, הָא אִם לְאַחֵר לֹא יַגִּיד פָּטוּר:

פרק ד - משנה יג

מְצַוֶּה אֲנִי עֲלֵיכֶם וְכוּ'. וְהוּא שֶׁיֹּאמַר מְצַוֶּה אֲנִי עֲלֵיכֶם בִּשְׁבוּעָה (לא), אוֹסֶרְכֶם אֲנִי בִּשְׁבוּעָה: הַמְקַלֵּל בְּכֻלָּם. מְבָרֵךְ אֶת הַשֵּׁם בְּאֶחָד מֵהֶם: רַבִּי מֵאִיר מְחַיֵּב. סְקִילָה: וַחֲכָמִים פּוֹטְרִין. דְּעַל שֵׁם הַמְיֻחָד דַּוְקָא בִּסְקִילָה, דִּכְתִיב (וַיִּקְרָא כד) בְּנָקְבוֹ שֵׁם יוּמָת. וְעַל הַכִּנּוּיִים בְּאַזְהָרָה דֵּאלֹהִים לֹא תְקַלֵּל (שְׁמוֹת כב). וּבִשְׁבוּעַת הָעֵדוּת מוֹדִים חֲכָמִים לְרַבִּי מֵאִיר דְּחַיָּב עַל הַכִּנּוּיִים כְּמוֹ שֶׁחַיָּב עַל הַשֵּׁם הַמְיֻחָד. דְּבִשְׁבוּעַת [עֵדוּת] כְּתִיב (וַיִּקְרָא ה) וְשָׁמְעָה קוֹל אָלָה: הַמְקַלֵּל עַצְמוֹ. דִּכְתִיב (דְּבָרִים ד) הִשָּׁמֶר לְךָ וּשְׁמֹר נַפְשְׁךָ מְאֹד, וְכָל הִשָּׁמֶר פֶּן וְאַל אֵינוֹ אֶלָּא לֹא תַעֲשֶׂה: וְהַמְקַלֵּל חֲבֵרוֹ. דִּכְתִיב (וַיִּקְרָא יט) לֹא תְקַלֵּל חֵרֵשׁ, אֲפִלּוּ חֵרֵשׁ שֶׁאֵינוֹ שׁוֹמֵעַ וְלֹא קָפֵיד אַקִּלְלָתוֹ לֹא תְקַלֵּל, וְכָל שֶׁכֵּן שְׁאָר כָּל אָדָם דְּשָׁמַע וְקָפֵיד: יַכְּכָה אֱלֹהִים וְכֵן יַכְּכָה אֱלֹהִים. אִם אָמַר לָעֵדִים יַכְּכָה אֱלֹהִים אִם לֹא תְעִידוּנִי. אוֹ שֶׁשָּׁמַע אֶחָד שֶׁקּוֹרֵא בַקְּלָלוֹת שֶׁבַּתּוֹרָה יַכְּכָה אֱלֹהִים, וְאָמַר הוּא לָעֵדִים וְכֵן יַכְּכָה אֱלֹהִים אִם לֹא תְעִידוּנִי. אוֹ שֶׁאָמַר אַל יַכְּכָה אֱלֹהִים אִם תְּעִידוּנִי. אוֹ יְבָרֶכְךָ ה' אִם תְּעִידוּנִי. וְיֵיטִיב לְךָ אֱלֹהִים אִם תְּעִידוּנִי. בְּכָל אֵלּוּ רַבִּי מֵאִיר מְחַיֵּב, דְּמִכְּלַל לָאו אַתָּה שׁוֹמֵעַ הֵן, אַל יַכְּכָה אֱלֹהִים אִם תְּעִידוּנִי, הָא אִם לֹא תְעִידוּנִי יַכְּכָה. וְכֵן יְבָרֶכְךָ אֱלֹהִים אִם תְּעִידוּנִי, הָא אִם לֹא תְעִידוּנִי יְקַלֶּלְךָ. וּבְכָל אֵלּוּ אֵין הֲלָכָה כְּרַבִּי מֵאִיר:

פרק ה - משנה א

שְׁבוּעַת הַפִּקָּדוֹן. בַּאֲנָשִׁים וּבְנָשִׁים. אַיְדֵי דְתָנָא בִשְׁבוּעַת הָעֵדוּת בַּאֲנָשִׁים וְלֹא בְנָשִׁים, תָּנָא נַמִּי הָכָא לְכֻלְּהוּ: בִּקְרוֹבִים. שֶׁבַּעַל הַפִּקָּדוֹן קָרוֹב לָזֶה שֶׁהַפִּקָּדוֹן אֶצְלוֹ: בִּפְנֵי בֵית דִּין וְשֶׁלֹּא בִפְנֵי בֵית דִּין. אִם מִפִּי עַצְמוֹ נִשְׁבַּע, שֶׁהוֹצִיא שְׁבוּעָה מִפִּיו אוֹ שֶׁעָנָה אָמֵן אַחַר הַשְּׁבוּעָה וְאַחַר כָּךְ הוֹדָה (א), חַיָּב קָרְבָּן. דִּכְתִיב (וַיִּקְרָא ה) וְכִחֵשׁ בַּעֲמִיתוֹ, כָּל דְּהוּ (ב): וּמִפִּי אֲחֵרִים. וְלֹא עָנָה אָמֵן. כְּגוֹן מַשְׁבִּיעַ אֲנִי עָלֶיךָ שֶׁתַּחֲזִיר לִי פִקְדוֹנִי, וְאָמַר אֵין לְךָ בְיָדִי כְלוּם, אֵינוֹ חַיָּב עַד שֶׁיִּכְפֹּר בְּבֵית דִּין: וַחֲכָמִים אוֹמְרִים וְכוּ'. וַהֲלָכָה כַּחֲכָמִים: וְחַיָּב עַל זְדוֹן הַשְּׁבוּעָה. דְּלֹא כְתִיב בָּהּ וְנֶעְלַם. זְדוֹן הַשְּׁבוּעָה, שֶׁזָּכוּר עַל הַפִּקָּדוֹן וְיוֹדֵעַ שֶׁמִּתְחַיֵּב עַל כְּפִירָתוֹ קָרְבָּן: וְעַל שִׁגְגָתָהּ עִם זְדוֹן הַפִּקָּדוֹן. שֶׁאֵינוֹ יוֹדֵעַ שֶׁחַיָּב בִּשְׁבוּעָה זוֹ קָרְבָּן (ה), אֲבָל זָכוּר הוּא שֶׁהַפִּקָּדוֹן אֶצְלוֹ: בְּכֶסֶף שְׁקָלִים. הַקָּנוּי בִּשְׁנֵי סְלָעִים. דְּהָכִי כְתִיב גַּבֵּי אֵיל אָשָׁם (וַיִּקְרָא ה), בְּעֶרְכְּךָ כֶּסֶף שְׁקָלִים (ו):

פרק ה - משנה ב

מִפְּנֵי שֶׁיָּכוֹל לַחֲזֹר וּלְהוֹדוֹת. אַחַר כְּפִירָה. נִמְצָא שֶׁבְּכָל שְׁבוּעָה הוּא כוֹפֵר מָמוֹן:

פרק ה - משנה ג

רַבִּי מֵאִיר אוֹמֵר יֵשׁ דְּבָרִים שֶׁהֵם כַּקַּרְקַע. עֲנָבִים הָעוֹמְדוֹת לִבָּצֵר הוּא דְאִיכָּא בֵּין חֲכָמִים לְרַבִּי מֵאִיר, דְּרַבִּי מֵאִיר סָבַר עֲנָבִים הָעוֹמְדוֹת לִבָּצֵר כִּבְצוּרוֹת דָּמְיָן, וַחֲכָמִים סָבְרֵי לָאו כִּבְצוּרוֹת דָּמְיָן. וַהֲלָכָה כַּחֲכָמִים. וְדַוְקָא בְּדִין שׁוֹמְרִים (כט), אֲבָל גַּבֵּי מִקָּח וּמִמְכָּר וְאוֹנָאָה וּמוֹדֶה בְמִקְצָת הַטַּעֲנָה, בְּכָל הָנֵי קַיְמָא לָן דְּדָבָר הָעוֹמֵד לִבָּצֵר כְּבָצוּר דָּמִי (ל). וְכֵן הֲלָכָה: עַד הַזִּיז. קוֹרָה שֶׁל עֲלִיָּה הַבּוֹלֶטֶת מִתּוֹךְ הַבַּיִת. וּכְלָלָא דְמִלְּתָא, לְעוֹלָם אֵינוֹ חַיָּב שְׁבוּעָה דְאוֹרַיְתָא עַד שֶׁיִּטְעֲנֶנּוּ דָבָר שֶׁבַּמִּדָּה אוֹ שֶׁבַּמִּשְׁקָל אוֹ שֶׁבַּמִּנְיָן, וְיוֹדֶה לוֹ בְמִקְצָת (לג) הַמִּדָּה אוֹ בְמִקְצָת הַמִּשְׁקָל אוֹ מִקְצָת מִנְיָן:

כָּל הַנִּשְׁבָּעִין. נִשְׁבָּעִין וְלֹא מְשַׁלְּמִין. לֹא תִקְּנָה תוֹרָה לַתּוֹבֵעַ לִשָּׁבַע וְלִטֹּל, אֶלָּא לַנִּתְבָּע שֶׁיִּשָּׁבַע וְלֹא יְשַׁלֵּם, דִּכְתִיב (שְׁמוֹת כב) וְלָקַח בְּעָלָיו וְלֹא יְשַׁלֵּם (א), מִי שֶׁעָלָיו לְשַׁלֵּם הוּא נִשְׁבָּע: וְאֵלּוּ נִשְׁבָּעִין וְנוֹטְלִין. שֶׁתִּקְּנוּ לָהֶם חֲכָמִים לִשָּׁבַע וְלִטֹּל. וְכֻלְּהוּ מְפָרֵשׁ לְהוּ לְקַמָּן בְּמַתְנִיתִין: הַשָּׂכִיר. תִּקְּנוּ לוֹ חֲכָמִים לִשָּׁבַע וְלִטֹּל, מִפְּנֵי שֶׁבַּעַל הַבַּיִת טָרוּד בְּפוֹעֲלָיו וְאֵינוֹ נִזְכָּר. וְהָנֵי מִלֵּי כְּשֶׁתּוֹבְעוֹ בִזְמַנּוֹ, שְׂכִיר יוֹם זְמַנּוֹ כָל הַלַּיְלָה שֶׁאַחֲרָיו, וּשְׂכִיר לַיְלָה זְמַנּוֹ כָל הַיּוֹם שֶׁלְּאַחֲרָיו. אֲבָל תְּבָעוֹ לְאַחַר זְמַנּוֹ, בַּעַל הַבַּיִת נִשְׁבָּע הֶסֵּת שֶׁפְּרָעוֹ וְנִפְטָר. וְאִם שְׂכָרוֹ שֶׁלֹּא בְעֵדִים (ג), נַמִּי אֵין הַשָּׂכִיר נִשְׁבָּע וְנוֹטֵל, דְּמִתּוֹךְ שֶׁיָּכוֹל לוֹמַר לוֹ לֹא שְׂכַרְתִּיךָ מֵעוֹלָם, יָכוֹל לוֹמַר לוֹ נַמִּי שְׂכַרְתִּיךָ וְנָתַתִּי (ד) לְךָ שְׂכָרְךָ: הוּא אוֹמֵר נָתַתִּי. אֲבָל אָמַר לוֹ שְׁנַיִם קָצַצְתָּ לִי וְהוּא אוֹמֵר לֹא קָצַצְתִּי לְךָ אֶלָּא אֶחָד, נִשְׁבָּע בַּעַל הַבַּיִת שְׁבוּעַת הַתּוֹרָה שֶׁהוּא כִדְבָרָיו וְאֵינוֹ נוֹתֵן לוֹ אֶלָּא אֶחָד (ה): רַבִּי יְהוּדָה אוֹמֵר. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה לֹא בְשָׂכִיר וְלֹא בְנִגְזָל וְלֹא בְנֶחְבָּל:

פרק ז - משנה ב

לְמַשְׁכְּנוֹ שֶׁלֹּא בִרְשׁוּת. וּכְגוֹן שֶׁרָאוּהוּ עֵדִים שֶׁנִּכְנַס לְבֵית חֲבֵרוֹ וְאֵין בְּיָדוֹ כְּלוּם וְיָצָא מִשָּׁם וְכֵלִים תְּחוּבִים לוֹ תַּחַת כְּנָפָיו (ח): הֲרֵי זֶה נִשְׁבָּע וְנוֹטֵל. שֶׁהֲרֵי הַדְּבָרִים מוֹכִיחִים, שֶׁהָעֵדִים מְעִידִים שֶׁמִּשְׁכְּנוֹ שֶׁלֹּא בִרְשׁוּת. וְהוּא דְטָעֵין לֵיהּ הַאִיךְ בְּמִידֵי דְאָמֵיד בֵּיהּ, אֲבָל אִם טָעֵין לֵיהּ בְּכָסָא דְכַסְפָּא וְכִי הַאי גַּוְנָא בְּמִידֵי דְלָא אָמֵיד בֵּיהּ, לָאו כָּל כְּמִינֵיהּ לִשָּׁבַע וְלִטֹּל, אֶלָּא יִשָּׁבַע הַנִּתְבָּע וְיִפָּטֵר:

פרק ז - משנה ג

הֲרֵי זֶה נִשְׁבָּע וְנוֹטֵל. וְדַוְקָא שֶׁהַחֲבָלָה בְּמָקוֹם שֶׁאֶפְשָׁר שֶׁחָבַל בְּעַצְמוֹ, מִשּׁוּם הָכִי צָרִיךְ שְׁבוּעָה. אֲבָל בְּמָקוֹם שֶׁאֵינוֹ יָכוֹל לַחֲבֹל בְּעַצְמוֹ וְנִכָּר שֶׁאֲחֵרִים עָשׂוּ לוֹ, כְּגוֹן שֶׁעָלְתָה לוֹ נְשִׁיכָה בְגַבּוֹ (יא), וְאֵין שָׁם אַחֵר, שֶׁיּוּכַל לוֹמַר שֶׁמָּא אַחֵר עָשָׂה (יב), נוֹטֵל בְּלֹא שְׁבוּעָה:

פרק ז - משנה ד

וַאֲפִלּוּ שְׁבוּעַת שָׁוְא. לֹא מִבַּעְיָא שְׁבוּעַת עֵדוּת וּשְׁבוּעַת הַפִּקָּדוֹן דְּאִית בְּהוּ כְּפִירַת מָמוֹן, דַּהֲוֵי רַע לַשָּׁמַיִם וְלַבְּרִיּוֹת, אֶלָּא אֲפִלּוּ שְׁבוּעַת שָׁוְא, דְּאֵינוֹ אֶלָּא רַע לַשָּׁמַיִם, שֶׁכְּנֶגְדּוֹ נִשְׁבָּע וְנוֹטֵל. וּשְׁבוּעַת בִּטּוּי לֹא קָתָנֵי, דִּשְׁבוּעַת בִּטּוּי דִּלְהַבָּא כְּגוֹן אֹכַל וְלֹא אֹכַל, אִיכָּא לְמֵימַר כִּי אִשְׁתָּבַע בְּקוּשְׁטָא אִשְׁתָּבַע, שֶׁבְּדַעְתּוֹ הָיָה לְקַיְּמוֹ, וְאַף עַל פִּי שֶׁכְּפָאוֹ יִצְרוֹ וְעָבַר עָלֶיהָ אֵינוֹ חָשׁוּד בְּכָךְ עַל הַשְּׁבוּעָה (יד). אֲבָל בִּטּוּי דִּלְשֶׁעָבַר, אָכַלְתִּי וְלֹא אָכַלְתִּי, דָּמֵי לִשְׁבוּעַת שָׁוְא, שֶׁהֲרֵי הוֹצִיא שְׁבוּעַת שֶׁקֶר מִפִּיו: הָיָה אֶחָד מֵהֶן מְשַׂחֵק בְּקֻבְיָא. תָּנָא פְסוּלָא דְאוֹרַיְתָא וְקָתָנֵי פְּסוּלָא דְרַבָּנָן: מַפְרִיחֵי יוֹנִים. אִית דִּמְפָרְשֵׁי אִם תַּקְדִּים יוֹנָתְךָ לְיוֹנָתִי אֶתֵּן לְךָ כָּךְ וְכָךְ, וְהַיְנוּ שְׂחוֹק. וְאִית דִּמְפָרְשֵׁי שֶׁמְּגַדֵּל יוֹנָה מְלֻמֶּדֶת לְהָבִיא יוֹנִים לְבֵית בְּעָלֶיהָ, וְיֵשׁ בָּהֶן גָּזֵל מִפְּנֵי דַּרְכֵי שָׁלוֹם: וְסוֹחֲרֵי שְׁבִיעִית. עוֹשִׂים סְחוֹרָה מִפֵּרוֹת שְׁבִיעִית, וְהַתּוֹרָה אָמְרָה לְאָכְלָה וְלֹא לִסְחוֹרָה: חָזְרָה שְׁבוּעָה לִמְקוֹמָהּ. אִית דִּמְפָרְשֵׁי בַגְּמָרָא, חָזְרָה לְסִינַי, לִשְׁבוּעַת הַר סִינַי שֶׁהִשְׁבִּיעַ הַקָּדוֹשׁ בָּרוּךְ הוּא אֶת יִשְׂרָאֵל לֹא תִגְזֹל, וְהוּא יִפָּרַע מִן הַכּוֹפֵר מָמוֹן לַחֲבֵרוֹ, אֲבָל בֵּית דִּין אֵין נִזְקָקִין לֹא לִשְׁבוּעָה וְלֹא לְפֵרָעוֹן. וְאִית דִּמְפָרְשֵׁי חָזְרָה שְׁבוּעָה לַמְחֻיָּב לָהּ, עַל זֶה שֶׁהוֹדָה בְמִקְצָת, וְכֵיוָן שֶׁאֵינוֹ יָכוֹל לִשָּׁבַע שֶׁהֲרֵי חָשׁוּד הוּא, יְשַׁלֵּם (טז):

פרק ז - משנה ה

[שְׁנֵיהֶן נִשְׁבָּעִין וְנוֹטְלִין. מִבַּעַל הַבַּיִת. דְּאָמַר לֵיהּ חֶנְוָנִי, פּוֹעֲלִים לֹא מְהֵימְנוּ לִי בִּשְׁבוּעָה], אַתְּ הֶאֱמַנְתִּינְהוּ, דְּלֹא אָמַרְתְּ לִי בְּסַהֲדֵי הַב לְהוּ. וְכֵן פּוֹעֲלִים אָמְרֵי לֵיהּ, חֶנְוָנִי לֹא מְהֵימָן לָן בִּשְׁבוּעָה. וּכְשֶׁשְּׁנֵיהֶם נִשְׁבָּעִין וְנוֹטְלִים מִבַּעַל הַבַּיִת, נִשְׁבָּעִים זֶה בִּפְנֵי זֶה, כִּי הֵיכִי דְּלִכְסוּף חֶנְוָנִי מִפּוֹעֲלִים אוֹ פּוֹעֲלִים מֵחֶנְוָנִי: אֵלּוּ וְאֵלּוּ בָאִין לִידֵי שְׁבוּעַת שָׁוְא. שֶׁהֲרֵי עַל כָּרְחֲךָ אֶחָד מֵהֶן נִשְׁבַּע לַשָּׁוְא וְנִמְצָא שֵׁם שָׁמַיִם מִתְחַלֵּל. אֶלָּא אֵלּוּ וָאֵלּוּ נוֹטְלִין בְּלֹא שְׁבוּעָה. וְאֵין הֲלָכָה כְּבֶן נַנָּס:

וְנָתַן לוֹ. וְהַפֵּרוֹת צְבוּרִין וּמֻנָּחִים בִּרְשׁוּת הָרַבִּים וְאֵין מֻחְזָק בָּהֶן לֹא חֶנְוָנִי וְלֹא בַּעַל הַבַּיִת: וּנְתַתּוֹ בָאֹנְפָּלִי. תִּיק הֶעָשׂוּי לְמָעוֹת: יִשָּׁבַע בַּעַל הַבָּיִת. שְׁבוּעָה כְעֵין שֶׁל תּוֹרָה, וְיִטֹּל. דְּכֵיוָן דְּחֶנְוָנִי מוֹדֶה שֶׁמָּכַר (כד), וְהַפֵּרוֹת הֵם חוּץ לַחֲנוּתוֹ, בַּעַל הַבַּיִת נִשְׁבָּע וְנוֹטֵל: אָמַר לוֹ נְתַתִּים לְךָ. וְהוֹלַכְתָּן לְתוֹךְ בֵּיתֶךָ. וְאִלּוּ הַפֵּרוֹת הַצְּבוּרִין וּמֻנָּחִים שֶׁלִּי הֵם שֶׁנָּתַתִּי כָאן לְמָכְרָן, וְהַלָּה טוֹעֵן אֵלּוּ הֵם הַפֵּרוֹת שֶׁמָּכַרְתָּ לִּי בְדִינָר. הוֹאִיל וְהַלּוֹקֵחַ מוֹדֶה בַמִּקָּח (כה) וְהַחֶנְוָנִי כוֹפֵר שֶׁלֹּא מָכַר אֶת אֵלּוּ, יִשָּׁבַע הַחֶנְוָנִי שְׁבוּעָה כְעֵין שֶׁל תּוֹרָה וְיִטֹּל: רַבִּי יְהוּדָה אוֹמֵר וְכוּ'. רַבִּי יְהוּדָה אַסֵּיפָא פָלֵיג, וְאָמַר דְּבֵין בְּזוֹ בֵּין בְּזוֹ בַּעַל הַבַּיִת נִשְׁבָּע וְנוֹטֵל. דְּכֵיוָן שֶׁהַפֵּרוֹת הֵם חוּץ לַחֲנוּת הָוֵי כְאִלּוּ הַפֵּרוֹת הֵם בְּיַד בַּעַל הַבַּיִת, וְכֹל שֶׁהַפֵּרוֹת בְּיָדוֹ, יָדוֹ עַל הָעֶלְיוֹנָה וְהוּא נִשְׁבָּע וְנוֹטֵל: אָמַר לַשֻּׁלְחָנִי וְכוּ'. אַשְׁמְעִינַן תַּנָּא פְּלֻגְתָּא דְרַבִּי יְהוּדָה וְרַבָּנָן בֵּין בְּמָעוֹת שֶׁל שֻׁלְחָנִי בֵּין בְּפֵרוֹת שֶׁל חֶנְוָנִי. דְּאִי אַשְׁמְעִינַן פְּלוּגְתַּיְהוּ בְחֶנְוָנִי, הֲוָה אֲמֵינָא בְּפֵרוֹת אָמְרוּ רַבָּנָן דְּכִי אָמַר לוֹ נְתַתִּים לְךָ וְהוֹלַכְתָּן לְתוֹךְ בֵּיתֶךָ יִשָּׁבַע הַחֶנְוָנִי וְיִטֹּל, לְפִי שֶׁהַחֶנְוָנִי עָשׂוּי לִתֵּן פֵּרוֹת קֹדֶם שֶׁיִּקַּח (כח) הַדִּינָר, אֲבָל בְּשֻׁלְחָנִי שֶׁאֵינוֹ עָשׂוּי לִתֵּן הָאִסָּרִין קֹדֶם שֶׁיִּקַּח הַדִּינָר, אֵימָא מוֹדוּ לֵיהּ לְרַבִּי יְהוּדָה דִּלְעוֹלָם בַּעַל הַבַּיִת נִשְׁבָּע וְנוֹטֵל. וְאִי אִתְּמַר בְּהָא, בְּהָא קָאָמַר רַבִּי יְהוּדָה דִּלְעוֹלָם בַּעַל הַבַּיִת נִשְׁבָּע וְנוֹטֵל, מִפְּנֵי שֶׁאֵין הַשֻּׁלְחָנִי עָשׂוּי לִתֵּן הָאִסָּרִין עַד שֶׁיִּקַּח הַדִּינָר, אֲבָל בְּחֶנְוָנִי שֶׁהוּא עָשׂוּי לִתֵּן הַפֵּרוֹת קֹדֶם שֶׁיִּקַּח הַמָּעוֹת אֵימָא מוֹדֵי לְהוּ לְרַבָּנָן, צְרִיכָא. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה:

הַפּוֹגֶמֶת כְּתֻבָּתָהּ. שֶׁמּוֹצִיאָה שְׁטַר כְּתֻבָּתָהּ וּמוֹדָה שֶׁנִּתְקַבְּלָה מִקְצָתָהּ: לֹא תִפָּרַע אֶלָּא בִשְׁבוּעָה. אִם טוֹעֵן הַבַּעַל נִתְקַבְּלָה כֻלָּהּ: עֵד אֶחָד אוֹמֵר וְכוּ'. וּמִנְּכָסִים מְשֻׁעְבָּדִים וּמִנִּכְסֵי יְתוֹמִים כוּ'. וְנִפְרַעַת שֶׁלֹּא בְפָנָיו. כֻּלְּהוּ אַכְּשֵׁם קַיְמֵי, כְּשֵׁם שֶׁאֵין אֶחָד מֵאֵלּוּ נִפְרָע אֶלָּא בִשְׁבוּעָה, כָּךְ הַיְתוֹמִים אֵינָם נִפְרָעִים אֶלָּא בִשְׁבוּעָה. יְתוֹמִים הַנִּפְרָעִים מִן הַיְתוֹמִים לֹא יִפָּרְעוּ אֶלָּא בִשְׁבוּעָה. וּמַיְרֵי מַתְנִיתִין כְּגוֹן שֶׁאֵלּוּ הַיְתוֹמִים שֶׁבָּאִים לְהִפָּרַע מֵהֶן אוֹמְרִים אֵין אָנוּ יוֹדְעִים אִם פָּרַע אָבִינוּ זֶה הַחוֹב אִם לָאו, אֲבָל אִם טָעֲנוּ אָמַר לָנוּ אָבִינוּ שֶׁלֹּא לָוָה מָעוֹת הַלָּלוּ וְלֹא נִתְחַיֵּב בְּחוֹב זֶה מֵעוֹלָם, הֲרֵי הַיְתוֹמִים שֶׁמּוֹצִיאִים שְׁטַר חוֹב עַל הַיְתוֹמִים הַלָּלוּ נִפְרָעִים מֵהֶם שֶׁלֹּא בִשְׁבוּעָה, שֶׁכָּל הָאוֹמֵר לֹא לָוִיתִי כְּאוֹמֵר לֹא פָרַעְתִּי דָמֵי, וְהֵם אֵינָם יְכוֹלִים לְהַכְחִישׁ הָעֵדִים שֶׁמְּעִידִים שֶׁאֲבִיהֶם לָוָה הַמָּעוֹת הַלָּלוּ. וְלֹא שָׁנוּ דְּבִשְׁבוּעָה מִיהַת שָׁקְלֵי יְתוֹמִים מִן הַיְתוֹמִים, אֶלָּא שֶׁמֵּת מַלְוֶה בְּחַיֵּי לֹוֶה, אֲבָל מֵת לֹוֶה בְּחַיֵּי מַלְוֶה, כְּבָר נִתְחַיֵּב מַלְוֶה שְׁבוּעָה לִבְנֵי לֹוֶה שֶׁלֹּא נִתְקַבֵּל כְּלוּם, שֶׁהַנִּפְרָע מִן הַיְתוֹמִים אֲפִלּוּ בִשְׁטָר צָרִיךְ שְׁבוּעָה, וְאֵין אָדָם מוֹרִישׁ מָמוֹן שֶׁהוּא חַיָּב עָלָיו שְׁבוּעָה לְבָנָיו, לְפִי שֶׁאֵין הַבָּנִים יְכוֹלִים לִשָּׁבַע (ל) שְׁבוּעָה שֶׁנִּתְחַיֵּב אֲבִיהֶם. וְאַף עַל פִּי שֶׁכֵּן הַדִּין, אִם דַּיָּן אֶחָד דָּן שֶׁיִּשָּׁבְעוּ הַיְתוֹמִים שְׁבוּעָה שֶׁלֹּא פְקָדָנוּ אַבָּא וְיִפָּרְעוּ מִמָּמוֹן הַיְתוֹמִים הָאֲחֵרִים בִּשְׁבוּעָה זוֹ, מַה שֶּׁעָשׂוּי עָשׂוּי (לא): שֶׁלֹּא פְקָדָנוּ אַבָּא. בִּשְׁעַת מִיתָה: וְשֶׁלֹּא אָמַר לָנוּ. קֹדֶם לָכֵן, שֶׁשְּׁטָר זֶה פָרוּעַ: הֲרֵי זֶה נִשְׁבָּע. שֶׁלֹּא מָצָא בֵין שְׁטָרוֹתָיו שֶׁל אָבִיו שֶׁשְּׁטָר זֶה פָּרוּעַ. וְכֵן הֲלָכָה: רַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל אוֹמֵר. וַהֲלָכָה כְּרַבָּן שִׁמְעוֹן בֶּן גַּמְלִיאֵל:

שֶׁלֹּא בְטַעֲנָה. שֶׁלֹּא בְּטַעֲנַת בָּרִי אֶלָּא בְטַעֲנַת שֶׁמָּא, שֶׁטּוֹעֲנוֹ שֶׁמָּא עִכַּבְתָּ מִשֶּׁלִּי. וּמִשּׁוּם דְּכָל הָנֵי מוֹרוּ הֶתֵּר לְעַצְמָם לְפִי שֶׁטָּרְחוּ בַנְּכָסִים, מִשּׁוּם הָכִי רָמוּ רַבָּנָן שְׁבוּעָה עֲלַיְהוּ: וְאַפּוֹטְרוֹפִּים. שֶׁנִּתְעַסְּקוּ בְמָמוֹן אָדָם לְהַכְנִיס וּלְהוֹצִיא וְלִשָּׂא וְלִתֵּן. אֲבָל בְּאַפּוֹטְרוֹפּוֹס שֶׁל יְתוֹמִים, אִם מִנָּהוּ בֵית דִּין יִשָּׁבַע, וְאִם אֲבִי יְתוֹמִים מִנָּהוּ לֹא יִשָּׁבַע: וְהָאִשָּׁה הַנּוֹשֵׂאת וְנוֹתֶנֶת בְּתוֹךְ הַבַּיִת. שֶׁהוֹשִׁיבָהּ בַּעְלָהּ חֶנְוָנִית אוֹ אַפּוֹטְרוֹפָּא עַל נְכָסָיו: וּבֶן הַבָּיִת. אֶחָד מִן הָאַחִין שֶׁנִּתְעַסֵּק בַּנְּכָסִים מִשֶּׁמֵּת אֲבִיהֶן: חָלְקוּ. וְלֹא הִשְׁבִּיעוּהוּ בִשְׁעַת הַחֲלוּקָה: נִתְגַּלְגְּלָה לוֹ שְׁבוּעָה. אֶצְלָם אַחֲרֵי כֵן, מְגַלְגְּלִין גַּם אֶת זוֹ עָלָיו. וּכְדֶרֶךְ שֶׁמְּגַלְגְּלִין בִּשְׁבוּעַת הַתּוֹרָה וּבִשְׁבוּעָה שֶׁהִיא כְעֵין שֶׁל תּוֹרָה, כָּךְ מְגַלְגְּלִין בִּשְׁבוּעַת הֶסֵּת: וְהַשְּׁבִיעִית מְשַׁמֶּטֶת אֶת הַשְּׁבוּעָה. לָאו אַהָנָךְ שְׁבוּעוֹת דְּשֻׁתָּפִין קָאֵי, דְּהָא אֵין שְׁבִיעִית מְשַׁמֶּטֶת שֻׁתָּפוּת וְלֹא שְׁבוּעָתָהּ. אֶלָּא מִלְוָה וּשְׁבוּעָתָהּ הִיא מְשַׁמֶּטֶת (לו):

אַרְבָּעָה שׁוֹמְרִים הֵם. וְדִינֵיהֶן שְׁלֹשָׁה: נוֹשֵׂא שָׂכָר. שׁוֹמֵר בְּשָׂכָר: נִשְׁבָּע עַל הַכֹּל. שֶׁלֹּא פָשַׁע: מְשַׁלֵּם אֶת הַכֹּל. כָּל הָנָךְ דִּתְנִינַן בְּמַתְנִיתִין, שְׁבוּיָה וּשְׁבוּרָה וּמֵתָה וּגְנֵבָה וַאֲבֵדָה. אֲבָל מֵתָה מֵחֲמַת מְלָאכָה, אֵינוֹ מְשַׁלֵּם, דְּלָאו לְאוֹקְמַהּ בְּכִלְּתָא שְׁאָלָהּ. וְהַמִּשְׁנָה הַזֹּאת מְפֹרֶשֶׁת בְּהַשּׂוֹכֵר אֶת הַפּוֹעֲלִים (ב):

פרק ח - משנה ב

וְאָמַר אָמֵן פָּטוּר. שֶׁאִלּוּ הוֹדָה לֹא הָיָה מְשַׁלֵּם כְּלוּם, נִמְצָא שֶׁאֵין כָּאן כְּפִירַת מָמוֹן. וְאַף עַל פִּי שֶׁנִּשְׁבַּע לַשֶּׁקֶר, לֹא חִיְּבָה תוֹרָה קָרְבַּן שְׁבוּעָה (ג) אֶלָּא הֵיכָא שֶׁכְּפָרוֹ מָמוֹן. וְכֵן הַדַּיָּן אֵינוֹ יָכוֹל לְהַשְׁבִּיעַ אֶלָּא עַל דָּבָר שֶׁאִם הוֹדָה הָיָה חַיָּב לְשַׁלֵּם מָמוֹן:

פרק ח - משנה ג

הֵיכָן שׁוֹרִי אָמַר לוֹ אָבָד. אָמַר לְשׁוֹמֵר חִנָּם הֵיכָן שׁוֹרִי, אָמַר לֵיהּ אָבַד: מַשְׁבִּיעֲךָ אָנִי וְאָמַר אָמֵן וְהָעֵדִים מְעִידִים אוֹתוֹ שֶׁאֲכָלוֹ מְשַׁלֵּם אֶת הַקֶּרֶן. וְלֹא כֶפֶל. וְאִם הוֹדָה לוֹ מֵעַצְמוֹ עַד שֶׁלֹּא בָאוּ עֵדִים, מְשַׁלֵּם קֶרֶן וְחֹמֶשׁ וְאָשָׁם, כְּדִין שְׁבוּעַת הַפִּקָּדוֹן דְּאֵינָהּ בָּאָה עַד שֶׁמּוֹדֶה וְשָׁב מֵרִשְׁעָתוֹ וּבָא לְהִתְכַּפֵּר: אָמַר לוֹ נִגְנָב מַשְׁבִּיעֲךָ אָנִי וְכוּ'. דְּשׁוֹמֵר חִנָּם הַפּוֹטֵר עַצְמוֹ בְּטַעֲנַת גַּנָּב וְנִשְׁבַּע וּבָאוּ עֵדִים, מְשַׁלֵּם כֶּפֶל, דִּכְתִיב (שְׁמוֹת כב) אִם לֹא יִמָּצֵא הַגַּנָּב, אִם לֹא יִמָּצֵא כְּמוֹ שֶׁאָמַר אֶלָּא שֶׁהוּא עַצְמוֹ גְנָבוֹ, יְשַׁלֵּם שְׁנַיִם. אֲבָל אִם בָּא לִפְטֹר עַצְמוֹ בְּטַעֲנַת אָבַד, אֵינוֹ מְשַׁלֵּם כֶּפֶל: הוֹדָה מֵעַצְמוֹ מְשַׁלֵּם קֶרֶן וְחֹמֶשׁ וְאָשָׁם. אֲבָל כֶּפֶל לֹא, דְּמוֹדֶה בִקְנָס פָּטוּר:

פרק ח - משנה ד

אָמַר לְאֶחָד מִן הַשּׁוּק וְכוּ'. הָכָא לֹא גָרְסִינַן מַשְׁבִּיעֲךָ אָנִי, דְּהָא בְלֹא שְׁבוּעָה מִחַיַּב כֶּפֶל. שֶׁזֶּה גַנָּב מַמָּשׁ הוּא, וּבְגַנָּב כְּתִיב יְשַׁלֵּם שְׁנַיִם, וְאֵין צָרִיךְ שְׁבוּעָה: עֵדִים שֶׁמְּמַשְׁמְשִׁין וּבָאִין. רְבוּתָא אַשְׁמְעִינַן, דְּאַף עַל גַּב דְּמֵחֲמַת בִּעֲתוּתָא דְעֵדִים מוֹדֶה, אֲפִלּוּ הָכִי מַהֲנְיָא הוֹדָאָה וְנִפְטַר מִן הַכֶּפֶל. וְכֵיוָן דְּאֵין כֶּפֶל, פָּטוּר אַף מִן הַטְּבִיחָה שֶׁכָּפַר בָּהּ וּבָאוּ עֵדִים שֶׁגָּנַב (ד) וְטָבַח וּמָכַר, דְּכָל הֵיכָא דְּלֵיכָּא כֶפֶל לֵיכָּא חִיּוּב טְבִיחָה, דְּתַשְׁלוּמֵי אַרְבָּעָה וַחֲמִשָּׁה אָמַר רַחֲמָנָא, וְלֹא תַשְׁלוּמֵי שְׁלֹשָׁה וְאַרְבָּעָה, וְכֵיוָן דְּכֶפֶל לֵיכָּא, בְּצִיר לֵיהּ חַד:

פרק ח - משנה ה

אָמַר לַשּׁוֹאֵל וְכוּ' פָּטוּר. שֶׁהֲרֵי אַף כְּשֶׁנִּשְׁבַּע לַשֶּׁקֶר חִיֵּב אֶת עַצְמוֹ בְתַשְׁלוּמִין וְאֵין בִּשְׁבוּעָה זוֹ כְּפִירַת מָמוֹן:

פרק ח - משנה ו

חַיָּב. שֶׁהֲרֵי כָפַר מָמוֹן, שֶׁאִלּוּ הוֹדָה הָיָה מִתְחַיֵּב: מֵת וְהוּא נִשְׁבַּר וְכוּ' פָּטוּר. שֶׁאִם הוֹדָה לֹא הָיָה מְשַׁלֵּם. וְכֵן נִגְנַב וְהוּא שֶׁאָבַד, אוֹ אָבַד וְהוּא שֶׁנִּגְנַב, פָּטוּר, שֶׁהֲרֵי שִׁנָּה מֵחוֹבָה לְחוֹבָה וְלֹא כָפַר מָמוֹן בִּשְׁבוּעָה זוֹ: אָמַר לוֹ אָבַד אוֹ נִגְנַב וְהוּא שֶׁמֵּת אוֹ נִשְׁבָּה וְכוּ' פָּטוּר. שֶׁהֲרֵי שִׁנָּה מִפְּטוֹר לְחוֹבָה וְהִפְסִיד בִּשְׁבוּעָתוֹ:
