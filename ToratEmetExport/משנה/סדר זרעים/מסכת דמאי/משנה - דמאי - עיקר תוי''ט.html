(א) וְאֵין לְהַקְשׁוֹת דִּבְלֹא הִדִּירוֹ אֵיךְ אוֹכֵל אֶצְלוֹ וְהָתְנַן בְּפֶרֶק ג' שֶׁאֵינוֹ מִתְאָרֵחַ אֵצֶל עַם הָאָרֶץ, דְּיֵשׁ לוֹמַר דְּהָכָא בָּעֵי תַּקָּנָה לְמִי שֶׁרוֹצֶה לֶאֱכֹל מְתֻקָּן וְאַף עַל פִּי שֶׁלֹּא קִבֵּל עָלָיו לִהְיוֹת נֶאֱמָן. אִי נַמִּי מַיְרֵי בְּיִחֵד לוֹ עֵדִים. תּוֹסְפוֹת יוֹם טוֹב: (ב) אוֹמֵר מֵעֶרֶב שַׁבָּת. כִּי בִדְמַאי אָדָם מַתְנֶה עַל הַדָּבָר שֶׁאֵינוֹ בִרְשׁוּתוֹ הָרַ"מ: (ג) מְשַׁעֵר בְּאֹמֶד כַּמָּה שִׁעוּר אֲכִילָתוֹ וּמְחַשֵּׁב בְּלִבּוֹ לְהַתְנוֹת עַכְשָׁיו וְשֶׁיַּפְרִישׁ כָּזֶה לְמָחָר:

(ד) וְהוּא הַדִּין לְמַה שֶּׁיֹּאכַל שֶׁצָּרִיךְ שֶׁיַּחֲזֹר וְיִרְחֹשׁ בִּשְׂפָתָיו בְּשַׁבָּת בְּשָׁעָה שֶׁמַּפְרִישׁ זוֹ אַחַר זוֹ. יְרוּשַׁלְמִי: (ה) וּמַעֲשֵׂר שֵׁנִי בְּפִיו. בְּכוֹס לֹא שַׁיָּךְ כֻּלֵּי הַאי בִּצְפוֹנוֹ כוּ' וְהַוְיָא יוֹתֵר קְבִיעַת מָקוֹם בְּפִיו:

(ו) וְחוֹשֵׂךְ. מִלְּשׁוֹן וְלֹא חָשַׂכְתָּ אֶת בִּנְךָ: (ז) וּלְכָךְ רַשַּׁאי לֶאֱכֹל, וְאֵין רְצוֹנוֹ לוֹמַר שֶׁנְּחַיְּבֵהוּ לְבַעַל הַבַּיִת שֶׁיִּתֵּן לוֹ כוּ' דְּמִכֵּיוָן שֶׁהוּא עַם הָאָרֶץ לֹא יִשְׁמַע לָנוּ לְעַשֵּׂר פֵּרוֹתָיו דְּמַאי:

(ח) מִבֵּין הַכּוּתִים. דְּאִלּוּ יִשְׂרָאֵל אֵינוֹ רַשַּׁאי לִמְכֹּר טֶבֶל אֶלָּא לְצֹרֶךְ, וְכֵיוָן דְּלֹא שְׁכִיחַ לֹא בָעֵי לְמַתְנֵיהּ הָכָא:

(ט) לְהַפְרִישׁ. לֹא גָרְסִינַן לְמָחָר לְפִי מַה שֶּׁכָּתַב הָרַ"ב לְעֵיל לְחַלֵּק בֵּין שֶׁאֵינוֹ מְתַקֵּן אֶלָּא מַה שֶּׁאוֹכֵל לַמְתַקֵּן אֶת כָּל הַנֹּאד. דְּאִי גָרַס אֵין צָרִיךְ לְחַלֵּק בְּהָכִי אֶלָּא הֲוָה לֵיהּ לְחַלֵּק בֵּין וַדַּאי לִדְמַאי. וְעַיֵּן בְּתוֹסְפוֹת יוֹם טוֹב:

(י) מַעַשְׂרוֹת. הֻשְׁאַל שֵׁם מַעַשְׂרוֹת לִכְלַל הַמַּתָּנוֹת וּבִכְלָלָן גַּם הַתְּרוּמָה: (יא) מְעֻשֶּׂרֶת. בְּהַךְ קָמַיְתָא לֵיכָּא חִדּוּשׁ שֶׁהָרִאשׁוֹנָה מְעֻשֶּׂרֶת, וּמִשּׁוּם סֵיפָא נָקְטָהּ. הָרַ"שׁ: (יב) דְּרַבִּי מֵאִיר הִיא דְּאָמַר אֵינוֹ תּוֹפֵס אֶלָּא לָשׁוֹן רִאשׁוֹן כִּדְאִיתָא בְּמַסֶּכֶת תְּמוּרָה: (יג) וְאֵין לְהַקְשׁוֹת דְּהָא בְּמַפְרִישׁ מִפָּטוּר עַל הַחִיּוּב תְּרוּמָתוֹ תְּרוּמָה אֶלָּא דְּיַחֲזֹר וְיִתְרֹם, וְאִם כֵּן אֵיךְ קָאָמַר וְלֹא אָמַר כְּלוּם. דְּשָׁאנֵי הָתָם שֶׁהִפְרִישׁ וְהַתְּרוּמָה נִכֶּרֶת לְכָךְ תְּרוּמָתוֹ תְּרוּמָה כְּדֵי שֶׁלֹּא לְזַלְזוֹלֵי בִּתְרוּמָה, אֲבָל הָכָא שֶׁלֹּא נִתְרְמָה וְלֹא נִכֶּרֶת אֵין כָּאן בֵּית מֵחוּשׁ כְּלָל:

פרק ז - משנה ז

(יד) וְאַמַּתְנִיתִין דִּשְׁתֵּי כַלְכָּלוֹת נַמִּי קָאֵי מִדְּלֹא תָנֵי בְּהָא אֶלָּא מַעַשְׂרוֹת: (טו) חֻלִּין מְתֻקָּנִים. הַטֶּבֶל כְּשֶׁיּוֹצִיא מִמֶּנּוּ תְּרוּמָה גְדוֹלָה וּתְרוּמַת מַעֲשֵׂר וְלֹא שְׁאָר מַעַשְׂרוֹת יִקָּרֵא חֻלִּין מִפְּנֵי שֶׁהַקְּדֻשָּׁה נִטְּלָה מִמֶּנּוּ, וְחֻלִּין מְתֻקָּנִים הַיְנוּ שֶׁהוֹצִיא מִמֶּנּוּ כָּל הַתְּרוּמוֹת וּמַעַשְׂרוֹת. הָרַ"מ: (טז) וְכֻלָּהּ מַתְנִיתִין כְּשֶׁאֵין לוֹ פַּרְנָסָה מִמָּקוֹם אַחֵר. וְעַיֵּן בְּפֶרֶק ג' דְּחַלָּה מִשְׁנָה ט': (יז) וּלְהָכִי בָּעִינַן עֲשָׂרָה. דְּהָא מִשּׁוּם דְּדַיְנִינַן לֵיהּ כְּאִלּוּ יֵשׁ לוֹ פַּרְנָסָה מִמָּקוֹם אַחֵר הוּא, וְכָל שֶׁיֵּשׁ לוֹ עֲשָׂרָה חֲלָקִים יְתֵרִים. שֶׁיֵּשׁ לָנוּ לְהַפְרִישׁ מֵהֶן אֶחָת לִתְרוּמַת מַעֲשֵׂר הָרְאוּיָה לֵחַלֵּק עֲשִׂירִית מֵעֲשָׂרָה חֶלְקֵי הַטֶּבֶל אוֹ הַמַּעֲשֵׁר, הִלְכָּךְ דַּיְנִינַן לֵיהּ כְּפַרְנָסָה מִמָּקוֹם אַחֵר, וּבְפָחוֹת מִזֶּה לֹא נוּכַל לְדַיְנוֹ כָּךְ. תּוֹסְפוֹת יוֹם טוֹב:

פרק ז - משנה ח

(יח) שְׁתֵּי חָבִיּוֹת. פָּתַח בְּכַד וְסִיֵּם בְּחָבִית, לוֹמַר לְךָ הַיְנוּ כַּד הַיְנוּ חָבִית, כִּדְאִיתָא בְּבָבָא קַמָּא (כז.): (יט) וְכָל הַחֲרָדָה הַזֹּאת בִּשְׁבִיל תְּרוּמַת מַעֲשֵׂר שֶׁיֵּשׁ בּוֹ. תּוֹסְפוֹת יוֹם טוֹב: (כ) חֲצִי שׁוּרָה אַחַת מַעֲשֵׂר. לְשׁוֹן הָרַ"מ וְאִם אָמַר חָבִית אַחַת מֵחֲצִי שׁוּרָה אַחַת מַעֲשֵׂר וְלֹא יִחֵד אֵיזוֹ חֲצִי שׁוּרָה יִקַּח עֶשְׂרִים חָבִיּוֹת עֶשֶׂר עַל אֲלַכְסוֹן אֶחָד וְעֶשֶׂר עַל אֲלַכְסוֹן אַחֵר הָעוֹבֵר עָלָיו וְנִמְצָא לוֹקֵחַ חָבִית מִכָּל חֲצִי שׁוּרָה מִן הַשּׁוּרוֹת וְאַחַר כָּךְ יוֹצִיא מִכֻּלָּם חָבִית אַחַת:
