פרק א - משנה א

הַנּוֹטֵעַ לִסְיָג. שֶׁיִּהְיֶה הָאִילָן גָּדֵר לַכֶּרֶם: וּלְקוֹרוֹת. לְגַדֵּל עֵצִים לַעֲשׂוֹת מֵהֶן קוֹרוֹת (א) לְבִנְיָן, וְאֵין עִקַּר נְטִיעָתוֹ לֶאֱכֹל הַפֵּרוֹת: פָּטוּר מִן הָעָרְלָה. דִּכְתִיב (וַיִּקְרָא יט) וּנְטַעְתֶּם כָּל עֵץ מַאֲכָל, אֶת שֶׁהוּא לְמַאֲכָל חַיָּב, לִסְיָג וּלְקוֹרוֹת וּלְעֵצִים פָּטוּר: הַפְּנִימִי לְמַאֲכָל. נִתְכַּוֵּן שֶׁיִּחוּר פְּנִימִי יִהְיֶה לְמַאֲכָל וְיִחוּר הַחִיצוֹן לִסְיָג, אַף עַל פִּי שֶׁהַכֹּל אִילָן אֶחָד, הַחִיצוֹן פָּטוּר. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי:

פרק א - משנה ב

וּמָצְאוּ נָטוּעַ. פָּטוּר. דִּכְתִיב (וַיִּקְרָא יט) כִּי תָבֹאוּ אֶל הָאָרֶץ וּנְטַעְתֶּם, פְּרָט לְנָטְעוּ עוֹבְדֵי כוֹכָבִים עַד שֶׁלֹּא בָאוּ לָאָרֶץ: נָטְעוּ. בֵּין יִשְׂרָאֵל בֵּין עוֹבְדֵי כוֹכָבִים לְאַחַר שֶׁבָּאוּ לָאָרֶץ, אַף עַל פִּי שֶׁלֹּא כָבְשׁוּ יִשְׂרָאֵל אֶת הָאָרֶץ, חַיָּב, דִּכְתִיב (שָׁם) כָּל עֵץ בֵּין נְטָעוֹ יִשְׂרָאֵל בֵּין נְטָעוֹ עוֹבֵד כּוֹכָבִים: הַנּוֹטֵעַ לָרַבִּים. שֶׁנָּטַע בִּרְשׁוּתוֹ לְצֹרֶךְ רַבִּים: חַיָּב. דִּכְתִיב (שָׁם) יִהְיֶה לָכֶם עֲרֵלִים, לְהָבִיא הַנָּטוּעַ לָרַבִּים: וְרַבִּי יְהוּדָה פּוֹטֵר. סָבַר וּנְטַעְתֶּם נַמִּי מַשְׁמַע לָרַבִּים, וְאֵין רִבּוּי אַחַר רִבּוּי אֶלָּא לְמַעֵט. וְרַבָּנָן סָבְרֵי וּנְטַעְתֶּם כָּל אֶחָד לְעַצְמוֹ מַשְׁמַע. וְאֵין הֲלָכָה כְּרַבִּי יְהוּדָה: וְהַנּוֹטֵעַ בִּרְשׁוּת הָרַבִּים. לְצֹרֶךְ עַצְמוֹ: וְהָעוֹבֵד כּוֹכָבִים שֶׁנָּטַע. בְּשָׂדֶה שֶׁל יִשְׂרָאֵל (ג): וְהַגַּזְלָן. שֶׁגָּזַל קַרְקַע וּנְטָעָהּ, וְנִתְיָאֲשׁוּ הַבְּעָלִים מִמֶּנָּה: וְהַנּוֹטֵעַ בִּסְפִינָה. אַף עַל פִּי שֶׁאֵינָהּ נְקוּבָה, אִם הִיא שֶׁל חֶרֶס, לְפִי שֶׁאֵין כְּלִי חֶרֶס עוֹמֵד בִּפְנֵי הַשָּׁרָשִׁים (ד) וְהֵן מְפַעְפְּעִין דֶּרֶךְ הַחֶרֶס וְיוֹנְקִים מִן הַקַּרְקַע. אֲבָל שֶׁל עֵץ בָּעֵי נְקִיבָה: וְהָעוֹלֶה מֵאֵלָיו. וְדַוְקָא בִּמְקוֹם יִשּׁוּב, אֲבָל בִּמְקוֹם יְעָרִים וּמִדְבָּרוֹת לֹא:

פרק א - משנה ג

אִילָן שֶׁנֶּעֱקַר. שֶׁעֲקַרְתּוֹ הָרוּחַ אוֹ שְׁטָפוֹ נָהָר וְהוֹלִיכוֹ לְמָקוֹם אַחֵר: וְהַסֶּלַע עִמּוֹ. הֶעָפָר שֶׁסְּבִיב הַשָּׁרָשִׁים קָרוּי סֶלַע, לְפִי שֶׁהוּא נַעֲשֶׂה שָׁם קָשֶׁה כְּסֶלַע. וְקָאָמַר שֶׁאִם הָיָה אוֹתוֹ עָפָר עִמּוֹ וְהוֹסִיף עָלָיו עָפָר וְנִשְׁרַשׁ שָׁם בָּאָרֶץ, רוֹאִים, אִם יָכוֹל לִחְיוֹת מִן הֶעָפָר שֶׁבָּא עִמּוֹ בְּלֹא תוֹסֶפֶת עָפָר אַחֵר, [הֲרֵי] הוּא כְּנָטוּעַ בִּמְקוֹמוֹ הָרִאשׁוֹן וּפָטוּר מִן הָעָרְלָה, וְאִם לָאו, הֲרֵי הוּא כְּאִילָן הָעוֹלֶה מֵאֵלָיו וְחַיָּב: זִעְזַעְתּוּ הַמַּחֲרֵשָׁה. וְנִתְפַּזֵּר מִמֶּנּוּ הֶעָפָר וְנִתְגַּלּוּ הַשָּׁרָשִׁים:

פרק א - משנה ד

וְנִשְׁתַּיֵּר בּוֹ שֹׁרֶשׁ. מְחֻבָּר לַקַּרְקַע שֶׁלֹּא נֶעֱקַר: כְּמַחַט שֶׁל מִתּוּחַ. גָּרְסִינַן, כְּלוֹמַר שִׁעוּר עֳבִי הַשֹּׁרֶשׁ כְּמַחַט שֶׁהָאוֹרְגִים מוֹתְחִים בּוֹ (ח) הַבֶּגֶד אַחַר שֶׁאֲרָגוּהוּ, לְהַפְשִׁיטוֹ וּלְהַרְחִיבוֹ:

פרק א - משנה ה

וּבוֹ בְרֵכָה. מִנְהַג עוֹבְדֵי אֲדָמָה שֶׁחוֹפְרִים גֻּמָּא וְלוֹקְחִים יִחוּר אֶחָד מִן הָאִילָן וְטוֹמְנִין אוֹתוֹ בְּאוֹתוֹ גֻּמָּא וְיוֹצֵא שֹׁרֶשׁ (ט) הַיִּחוּר מִצַּד אַחֵר וְנַעֲשֶׂה שָׁם אִילָן, וְרֹאשׁ הַיִּחוּר נִשְׁאָר מְחֻבָּר בָּאִילָן, וְאֵין דִּין עָרְלָה נוֹהֵג בּוֹ בְּעוֹדוֹ מְחֻבָּר לָאִילָן וְיוֹנֵק מִמֶּנּוּ, אַף עַל פִּי שֶׁהוּא יוֹנֵק גַּם מִן הַקַּרְקַע. וְאִם נֶעֱקַר אִילָן הַזָּקֵן מִן הַקַּרְקַע וְנִמְצָא שֶׁכָּל חַיּוּתוֹ וִינִיקָתוֹ עַתָּה מִן הַבְּרֵכָה, חָזְרָה הַזְּקֵנָה לִהְיוֹת כַּבְּרֵכָה. וּשְׁנֵיהֶם, הַזְּקֵנָה וְהַבְּרֵכָה הֲווֹ כְּאִלּוּ נִטְּעוּ עַתָּה, וּמוֹנִים לָהֶם שָׁלֹשׁ שָׁנִים מִשָּׁעָה שֶׁנֶּעֱקַר הָאִילָן: הִבְרִיכָהּ שָׁנָה אַחַר שָׁנָה. לְאַחַר שֶׁהִגְדִּילָה בְּרֵכָה רִאשׁוֹנָה לָקַח מִמֶּנּוּ יִחוּר וְהִבְרִיכוֹ בָאָרֶץ, וְכֵן עָשָׂה כַּמָּה פְעָמִים זֶה אַחַר זֶה: מוֹנֶה. לְכֻלָּן שְׁנֵי עָרְלָה מִשֶּׁנִּפְסְקָה הַבְּרֵכָה הָרִאשׁוֹנָה מִן הָאִילָן: סִפּוּק גְּפָנִים. דֶּרֶךְ עוֹבְדֵי אֲדָמָה שֶׁלּוֹקְחִים זְמוֹרָה אֲרֻכָּהּ מִגֶּפֶן זוֹ וּמוֹשְׁכִים אוֹתָהּ לַגֶּפֶן שֶׁבְּצִדָּהּ וּמַרְכִּיבִין הַזְּמוֹרָה בַּגֶּפֶן, וְזֶהוּ הַנִּקְרָא סִפּוּק (יא), וּפָטוּר מִן הָעָרְלָה. וְהָא דִתְנַן אֶחָד הַנּוֹטֵעַ וְאֶחָד הַמַּבְרִיךְ וְאֶחָד הַמַּרְכִּיב חַיָּב בָּעָרְלָה, הָתָם מַיְרֵי כְּשֶׁחָתַךְ הַזְּמוֹרָה מִן הַגֶּפֶן אוֹ הַיִּחוּר מִן הָאִילָן, וְאַחַר כָּךְ הִבְרִיכָן אוֹ הִרְכִּיבָן: מָקוֹם שֶׁכֹּחָהּ יָפֶה מֻתָּר. אִם הַזְּמוֹרָה שֶׁהִבְרִיכוּהָ בַּגֶּפֶן כֹּחָהּ יָפֶה וְיוֹנְקָה מִן הַגֶּפֶן אֲשֶׁר מִמֶּנָּה לֻקָּחָה (יג), מֻתָּר וּפְטוּרָה מִן הָעָרְלָה. וְאִם כֹּחָהּ רָע חַיֶּבֶת בָּעָרְלָה. וְאֵין הֲלָכָה כְּרַבִּי מֵאִיר: וְהִיא מְלֵאָה פֵרוֹת. אִם תְּלָשָׁן מִיָּד, הֵן מֻתָּרִין מִפְּנֵי שֶׁגָּדְלוּ בְּהֶתֵּר. וְאִם הִנִּיחָן בָּאִילָן וְגָדְלוּ לְאַחַר שֶׁנִּפְסְקָה וְאֵין בַּפֵּרוֹת שֶׁגָּדְלוּ כְּבָר מָאתַיִם לְבַטֵּל הַתּוֹסֶפֶת, אָסוּר, דְּעָרְלָה אוֹסֶרֶת עַד מָאתַיִם. וְהוּא הַדִּין בְּאִילָן זָקֵן שֶׁנֶּעֱקַר וּבוֹ פֵּרוֹת, וְחָזַר וּשְׁתָלוֹ:

פרק א - משנה ו

שֶׁנִּתְעָרְבָה בִנְטִיעוֹת. שֶׁל הֶתֵּר, וְאֵינוֹ מַכִּיר הַנְּטִיעָה שֶׁל עָרְלָה: וְשֶׁל כִּלְאֵי הַכֶּרֶם. כְּגוֹן שֶׁהֶעֱבִיר עָצִיץ נָקוּב תַּחַת גֶּפֶן אַחַת וְנֶאֶסְרָה (יד), וְאֵינוֹ מַכִּירָהּ בֵּין שְׁאָר גְּפָנִים: הֲרֵי זֶה לֹא יִלְקֹט. דְּכֵיוָן שֶׁאֵינוֹ מַכִּיר אוֹתָהּ שֶׁל אִסּוּר כֻּלָּן אֲסוּרוֹת: אַף יִתְכַּוֵּן וְיִלְקֹט. דַּחֲזָקָה אֵין אָדָם אוֹסֵר כַּרְמוֹ בִּנְטִיעָה אַחַת (טו), וְלֹא חַיְשִׁינַן דִּלְמָא אָתֵי לְעָרֵב הַנְּטִיעָה לְכַתְּחִלָּה. וְאַף עַל גַּב דִּבְעַלְמָא אֵין מְבַטְּלִין אִסּוּר לְכַתְּחִלָּה (טז), הָכָא מִלְּתָא דְּלֹא שְׁכִיחָא הִיא וְלֹא גָזוּר. וְאֵין הֲלָכָה כְּרַבִּי יוֹסֵי:

הַתְּרוּמָה מַעֲלָה אֶת הָעָרְלָה. מִצְטָרֶפֶת עִם הַחֻלִּין לְבַטֵּל אֶת הָעָרְלָה. וְכֵן הָעָרְלָה מִצְטָרֶפֶת עִם הַחֻלִּין לְבַטֵּל אֶת הַתְּרוּמָה: שֶׁנָּפְלָה לְמֵאָה. לָאו דַּוְקָא מֵאָה אֶלָּא פָּחוֹת מִמֵּאָה, דְּאִלּוּ מֵאָה מַמָּשׁ עוֹלֶה וְלֹא בָעֵי צֵרוּף, אֶלָּא הָכִי קָאָמַר הַתְּרוּמָה שֶׁנָּפְלָה לְחֻלִּין וְנַעֲשָׂה הַכֹּל מֵאָה: וְאַחַר כָּךְ נָפְלוּ שְׁלֹשָׁה קַבִּין שֶׁל עָרְלָה. סְתָם מַתְנִיתִין כְּרַבִּי יְהוֹשֻׁעַ דְּאָמַר בְּפֶרֶק קַמָּא דִתְרוּמוֹת דִּתְרוּמָה עוֹלָה בְּמֵאָה וְעוֹד, הִלְכָּךְ כְּשֶׁנָּפְלָה סְאָה שֶׁל תְּרוּמָה לְתִשְׁעִים וָתֵשַׁע שֶׁל חֻלִּין, וְחָזְרוּ וְנָפְלוּ שָׁם שְׁלֹשָׁה קַבִּין שֶׁל עָרְלָה וּמִצְטָרְפִין עִם הַחֻלִּין, שֶׁנִּמְצְאוּ שָׁם תִּשְׁעִים וָתֵשַׁע וְעוֹד שְׁלֹשָׁה קַבִּין חֻלִּין, הֲרֵי סְאָה שֶׁל תְּרוּמָה בְּטֵלָה בְּמֵאָה וְעוֹד. אִי נַמִּי מָצִינַן לְאוֹקְמָהּ כְּרַבִּי אֱלִיעֶזֶר דְּהִלְכְתָא כְּוָתֵיהּ דְּאָמַר תְּרוּמָה עוֹלָה בְּאֶחָד וּמֵאָה, וּכְגוֹן שֶׁנָּפְלָה סְאָה שֶׁל תְּרוּמָה לְתוֹךְ תִּשְׁעִים וְתֵשַׁע סְאִין וָחֵצִי (ד) סְאָה שֶׁל חֻלִּין, וּשְׁלֹשָׁה קַבִּין הַלָּלוּ שֶׁל עָרְלָה שֶׁהוּא חֲצִי סְאָה מִצְטָרְפִין עִם הַחֻלִּין לְהַשְׁלִים לְשִׁעוּר מֵאָה שֶׁל חֻלִּין כְּדֵי שֶׁתַּעֲלֶה בְּאֶחָד וּמֵאָה: זוֹ הִיא שֶׁהַתְּרוּמָה מַעֲלָה אֶת הָעָרְלָה. שֶׁאוֹתָהּ סְאָה שֶׁל תְּרוּמָה נִצְטָרְפָה עִם הַחֻלִּין לְהַשְׁלִים מָאתַיִם חֲצָאֵי סְאִין, לְהַעֲלוֹת שְׁלֹשָׁה קַבִּין שֶׁל עָרְלָה בְּאֶחָד וּמָאתַיִם: וְהָעָרְלָה אֶת הַתְּרוּמָה. דִּשְׁלֹשָׁה קַבִּין שֶׁל עָרְלָה שֶׁהֵן חֲצִי סְאָה, נִצְטָרְפוּ עִם הַחֻלִּין לְבַטֵּל סְאָה שֶׁל תְּרוּמָה בְּאֶחָד וּמֵאָה. דִּכְשֶׁנָּפְלָה סְאָה לְתוֹךְ תִּשְׁעִים וְתֵשַׁע וָחֵצִי נֶאֶסְרוּ, וּכְשֶׁחָזַר וְנָפַל שָׁם שְׁלֹשָׁה קַבִּין עָרְלָה אוֹ כִלְאַיִם, הֻתְּרוּ:

פרק ב - משנה ג

וְהָעָרְלָה אֶת הָעָרְלָה. צָרִיךְ לוֹמַר דְּאֶחָד מֵהֶם נֶטַע רְבָעִי, שֶׁהֵם שְׁנֵי שֵׁמוֹת, דְּאִי אֶפְשָׁר לְאִסּוּר שֶׁל שֵׁם אֶחָד שֶׁיִּהְיֶה מְבַטֵּל קְצָתוֹ לִקְצָתוֹ. וְקָרֵי תַּנָא לְנֶטַע רְבָעִי עָרְלָה, שֶׁמִּן הָעָרְלָה הוּא בָא. וּלְהָכִי קָתָנֵי הָעָרְלָה אֶת הָעָרְלָה וְלֹא קָתָנֵי הַכִּלְאַיִם אֶת הַכִּלְאַיִם, שֶׁאֵין בַּכִּלְאַיִם שְׁנֵי שֵׁמוֹת כְּמוֹ שֶׁיֵּשׁ בָּעָרְלָה: שֶׁנָּפְלָה לְמָאתַיִם. לָאו לְמָאתַיִם דַּוְקָא דְּאִלּוּ מָאתַיִם מַמָּשׁ לֹא בָעֵי צֵרוּף. וּכְשֶׁנָּפְלָה סְאָה עָרְלָה לְתוֹךְ מֵאָה תִּשְׁעִים וָתֵשַׁע שֶׁל הֶתֵּר נֶאֶסְרוּ, וּכְשֶׁחָזַר וְנָפַל שָׁם סְאָה וָעוֹד שֶׁל כִּלְאַיִם מִצְטָרְפִים עִם הַמֵּאָה תִּשְׁעִים וָתֵשַׁע לְבַטֵּל הָעָרְלָה. וְכֵן הָעָרְלָה מִצְטָרֶפֶת עִם הַמֵּאָה תִּשְׁעִים וָתֵשַׁע לְבַטֵּל סְאָה וְעוֹד שֶׁל כִּלְאֵי הַכֶּרֶם. וְהַיְנוּ כְּרַבִּי יְהוֹשֻׁעַ דְּאִית לֵיהּ דְּעָרְלָה בְּטֵלָה בְּמָאתַיִם וְעוֹד, כְּמוֹ תְּרוּמָה בְּמֵאָה וְעוֹד, הִלְכָּךְ סְאָה שֶׁל כִּלְאַיִם בְּטֵלָה בְּמֵאָה תִּשְׁעִים וָתֵשַׁע וּמַשֶּׁהוּ, וְהָעוֹד בָּטֵל בִּסְאָה חָסֵר מַשֶּׁהוּ (ה). וְכֵן נַמִּי אִם נָפַל שָׁם סְאָה וְעוֹד שֶׁל נֶטַע רְבָעִי, מִצְטָרְפִין עִם הַמֵּאָה תִּשְׁעִים וָתֵשַׁע לְבַטֵּל אֶת הַסְּאָה שֶׁל עָרְלָה שֶׁנָּפְלָה, וְאָסְרָה הַכֹּל, וְכֵן סְאָה שֶׁל עָרְלָה מִצְטָרֶפֶת עִם הַמֵּאָה תִּשְׁעִים וָתֵשַׁע שֶׁל הֶתֵּר לְבַטֵּל סְאָה וְעוֹד שֶׁל נֶטַע רְבָעִי שֶׁנָּפְלוּ בָּאַחֲרוֹנָה:

כָּל הַמְחַמֵּץ וְהַמְתַבֵּל וְכוּ'. הָכִי קָאָמַר, כָּל הַמְחַמֵּץ וְהַמְתַבֵּל בָּעָרְלָה וּבְכִלְאֵי הַכֶּרֶם אָסוּר: וְהַמְדַמֵּעַ בַּתְּרוּמָה (ו). הַמְחַמֵּץ אֶת הָעִסָּה בְּתַפּוּחִים (ז) אוֹ בְשִׁמְרֵי יַיִן שֶׁל עָרְלָה וְשֶׁל כִּלְאֵי הַכֶּרֶם, וְהַמְתַבֵּל אֶת הַקְּדֵרָה בִּתְבָלִין שֶׁל כִּלְאֵי הַכֶּרֶם אוֹ שֶׁל עָרְלָה, אָסוּר, וְאֵין עוֹלִין בְּאֶחָד וּמָאתַיִם. וְכֵן בַּתְּרוּמָה, כֵּיוָן שֶׁהֶחֱמִיצָה עִסָּה שֶׁל חֻלִּין בִּשְׂאֹר שֶׁל תְּרוּמָה וְנִתְבְּלָה קְדֵרָה שֶׁל חֻלִּין בִּתְבָלִין שֶׁל תְּרוּמָה, נַעֲשֶׂה הַכֹּל מְדֻמָּע, וְאָסוּר לְזָרִים, וְאֵינוֹ עוֹלֶה בְּאֶחָד וּמֵאָה: בֵּית שַׁמַּאי אוֹמְרִים אַף מְטַמֵּא. שְׂאֹר טָמֵא שֶׁהֶחֱמִיץ עִסָּה טְהוֹרָה אוֹ תְּבָלִין טְמֵאִין שֶׁתִּבְּלוּ קְדֵרָה טְהוֹרָה, אַף עַל פִּי שֶׁאֵין בַּשְּׂאֹר אוֹ בַתְּבָלִין כַּבֵּיצָה שֶׁהוּא שִׁעוּר טֻמְאַת אֳכָלִין, נִטְמֵאת הָעִסָּה וְנִטְמֵאת הַקְּדֵרָה: וּבֵית הִלֵּל אוֹמְרִים עַד שֶׁיְּהֵא בּוֹ כַּבֵּיצָה. דְּאֵין אֹכֶל מְטַמֵּא פָּחוֹת מִכַּבֵּיצָה, לֹא שְׁנָא שְׂאֹר וּתְבָלִין וְלֹא שְׁנָא שְׁאָר אֳכָלִין:

וּלְמָה אָמְרוּ. כְּמוֹ וּבַמֶּה אָמְרוּ, כְּלוֹמַר בְּאֵיזֶה עִנְיָן אָמְרוּ שֶׁהַמְחַמֵּץ חֻלִּין בִּשְׂאֹר שֶׁל תְּרוּמָה, וְהַמְתַבֵּל קְדֵרַת חֻלִּין בִּתְבָלִין שֶׁל תְּרוּמָה, וְהַמְדַמֵּעַ שֶׁמְּעָרֵב [וּמְבַשֵּׁל] חֻלִּין וּתְרוּמָה יַחַד, שֶׁאָנוּ הוֹלְכִים הַכֹּל לְהַחֲמִיר, וּמְשָׁנֵי מִין בְּמִינוֹ. וּבַמֶּה אָמְרוּ פְּעָמִים לְהָקֵל ופְעָמִים לְהַחֲמִיר, בְּמִין שֶׁנִּתְעָרֵב עִם שֶׁאֵינוֹ מִינוֹ. וּמְפָרֵשׁ, לְהַחֲמִיר כֵּיצַד, שְׂאֹר שֶׁל תְּרוּמַת חִטִּין שֶׁנָּפַל לְתוֹךְ עִסַּת חִטִּין שֶׁל חֻלִּין, דְּהַיְנוּ מִין בְּמִינוֹ, אוֹסֵר בְּנוֹתֵן טַעַם אֲפִלּוּ יוֹתֵר מִמֵּאָה, וְעַד מֵאָה אוֹסֵר אֲפִלּוּ בְּלֹא נְתִינַת טַעַם:

גְּרִיסִין. שֶׁל פּוֹל שֶׁל תְּרוּמָה, שֶׁנִּתְבַּשְּׁלוּ עִם עֲדָשִׁים שֶׁל חֻלִּין וְנָתְנוּ בָהֶם טַעַם, טַעֲמָא לֹא בָטִיל אֲפִלּוּ בְּיוֹתֵר מִמֵּאָה. וּבְלֹא טַעַם אֲפִלּוּ בְּפָחוֹת מִמֵּאָה שָׁרֵי, דְּהָא דְבָעִינַן מֵאָה הַיְנוּ דַּוְקָא חִטִּים בְּחִטִּים דַּהֲוֵי מִין בְּמִינוֹ, דְּאַסְמְכוּהָ אַקְּרָא אֶת מִקְדְּשׁוֹ מִמֶּנּוּ, מַה שֶּׁאַתָּה מֵרִים מִמֶּנּוּ אִם נָפַל לְתוֹכוֹ מְקַדְּשׁוֹ, וְזֶהוּ מִין בְּמִינוֹ:

וְיֵשׁ בּוֹ כְדֵי לְחַמֵּץ אָסוּר. הָא קָא מַשְׁמַע לָן דְּלֹא אָמְרִינַן הוֹאִיל וּבְלֹא שְׂאֹר זֶה שֶׁל אִסּוּר הָיְתָה הָעִסָּה מִתְחַמֶּצֶת עַל יְדֵי שְׂאֹר שֶׁל הֶתֵּר שֶׁנָּפַל בּוֹ תְּחִלָּה, לֹא נָחוּשׁ לִשְׂאֹר שֶׁל אִסּוּר:

חֲבִילֵי תִלְתָּן. אֵין חֲבִילָה פְּחוּתָה מֵעֶשְׂרִים וַחֲמִשָּׁה זִירִים: נִתְעָרְבוּ בַאֲחֵרִים. גָּרְסִינַן, וְלֹא גָרְסִינַן וַאֲחֵרִים בַּאֲחֵרִים. וְאִי גָרְסִינַן לַהּ צָרִיךְ לוֹמַר שֶׁנִּתְעָרְבוּ כָּל אוֹתָן אֲחֵרִים בַּאֲחֵרִים, דְּלֹא הֲוֵי אֶלָּא חַד סְפֵקָא, וְאַשְׁמְעִינַן אַף עַל גַּב דְּאִיכָּא תְּרֵי רֻבֵּי, אֲסִירֵי: כֻּלָּם יִדָּלֵקוּ. וְהָא דְלֹא אָמְרִינַן יִמָּכְרוּ כֻּלָּן חוּץ מִדְּמֵי אִסּוּר שֶׁבָּהֶן, כִּדְאָמַר רַבִּי שִׁמְעוֹן בֶּן גַּמְלִיאֵל בְּפֶרֶק [בַּתְרָא] דַּעֲבוֹדָה זָרָה גַּבֵּי יֵין נֶסֶךְ, מִשּׁוּם דְּהָכָא חַיְשִׁינַן בְּכוּלְהוּ שֶׁמָּא יַחֲזֹר הָעוֹבֵד כּוֹכָבִים וְיִמְכְּרֶנּוּ לְיִשְׂרָאֵל, מַה שֶּׁאֵין כֵּן בְּיֵין נֶסֶךְ דְּלֵיכָּא לְמֵיחַשׁ לְהָכִי (ט):

פרק ג - משנה ז

שֶׁדַּרְכּוֹ לִמָּנוֹת. שֶׁאֵין אָדָם מוֹכְרוֹ אֶלָּא בְּמִנְיָן מִפְּנֵי חֲשִׁיבוּתוֹ: פֶּרֶךְ וּבָדָן. שְׁמוֹת מְקוֹמוֹת הֵן. וַאֲנִי שָׁמַעְתִּי אֱגוֹזֵי פֶרֶךְ, אֱגוֹזִים שֶׁקְּלִפָּתָן רַכָּה וְנִפְרָכִין: חֻלְפֵי תְּרָדִין. עָלִין מִצַּלְעוֹת הַתְּרָדִין (יא): קֻלְסֵי כְרוּב. קִלְחֵי כְרוּב שֶׁבְּאֶרֶץ יִשְׂרָאֵל, שֶׁהֵם גְּדוֹלִים וְטוֹבִים: שֶׁל בַּעַל הַבָּיִת. שֶׁהֵן גְּדוֹלִים, אֲבָל שֶׁל נַחְתּוֹם קְטַנִּים הֵן: הָרָאוּי לְעָרְלָה. כְּגוֹן אֱגוֹזִים וְרִמּוֹנִים וְחָבִיּוֹת סְתוּמוֹת: וְהָרָאוּי לְכִלְאֵי הַכֶּרֶם. כְּגוֹן חֻלְפֵי תְּרָדִין וְקֻלְסֵי כְרוּב וּדְלַעַת יְוָנִית. וְאֵין הֲלָכָה כְּרַבִּי מֵאִיר וְלֹא כְרַבִּי עֲקִיבָא:

פרק ג - משנה ח

נִתְפַּצְּעוּ הָאֱגוֹזִים. גָּרְסִינַן, וְלֹא גָרְסִינַן כֵּיצַד, וְקָא מַשְׁמַע לָן מַתְנִיתִין דְּדַוְקָא שְׁלֵמִים לֹא בְּטִילֵי, אֲבָל לֹא שְׁבוּרִים וּפְצוּעִים: נִתְפָּרְסוּ הַכִּכָּרוֹת. הַאי סְתָמָא כְּרַבִּי עֲקִיבָא, דְּהִלְכְתָא כְּוָתֵיהּ (טו):

פרק ג - משנה ט

סְפֵק עָרְלָה. כְּגוֹן עוֹבֵד כּוֹכָבִים שֶׁיֵּשׁ לוֹ נְטִיעוֹת שֶׁל עָרְלָה בְּגִנָּתוֹ, וּבְיָדוֹ פֵּרוֹת, וְאֵין יָדוּעַ אִם מִשֶּׁל עָרְלָה אִם מִשֶּׁל זְקֵנָה: בְּאֶרֶץ יִשְׂרָאֵל. דַּהֲוֵי סְפֵקָא דְאוֹרַיְתָא לְחֻמְרָא: וּבְסוּרְיָא מֻתָּר. דְּכִבּוּשׁ יָחִיד הוּא וְלָאו שְׁמֵיהּ כִּבּוּשׁ, וְהֶחֱמִירוּ בוֹ קְצָת: וּבְחוּצָה לָאָרֶץ יוֹרֵד וְלוֹקֵחַ. בְּתוֹךְ הַגִּנָּה וְקוֹנֶה מִן הָעוֹבֵד כּוֹכָבִים מֵאוֹתָם שֶׁלִּקֵּט כְּבָר, וּבִלְבַד שֶׁלֹּא יִרְאֶנּוּ לוֹקֵט: וּבְחוּצָה לָאָרֶץ יוֹרֵד וְלוֹקֵט. הָעוֹבֵד כּוֹכָבִים יוֹרֵד וְלוֹקֵט בִּפְנֵי הַיִּשְׂרָאֵל וְיִשְׂרָאֵל קוֹנֶה מִמֶּנּוּ, וּבִלְבַד שֶׁלֹּא יִלְקֹט יִשְׂרָאֵל בַּיָּד: בְּכָל מָקוֹם. וַאֲפִלּוּ בְּחוּצָה לָאָרֶץ, דִּכְתִיב (וַיִּקְרָא כד) וְלֶחֶם וְקָלִי וְכַרְמֶל לֹא תֹאכְלוּ וְגוֹ' בְּכָל מוֹשְׁבוֹתֵיכֶם: וְהָעָרְלָה הֲלָכָה. לְמֹשֶׁה מִסִּינַי שֶׁאָסוּר בְּחוּצָה לָאָרֶץ, וְאַף עַל פִּי כֵן סְפֵקָהּ מֻתָּר, שֶׁכָּךְ נֶאֶמְרָה הַהֲלָכָה לְמֹשֶׁה בְּסִינַי, שֶׁיִּהְיֶה סְפֵקָהּ מֻתָּר בְּחוּצָה לָאָרֶץ: וְהַכִּלְאַיִם מִדִּבְרֵי סוֹפְרִים. דַּוְקָא כִּלְאֵי הַכֶּרֶם, כֵּיוָן דְּבָאָרֶץ חֲמִירֵי וַאֲסוּרִים בַּהֲנָאָה מִן הַתּוֹרָה, בְּחוּצָה לָאָרֶץ גָּזְרוּ בָהּ רַבָּנָן, אֲבָל כִּלְאֵי זְרָעִים דְּבָאָרֶץ שָׁרוּ בַּהֲנָאָה, בְּחוּצָה לָאָרֶץ לֹא גָזְרוּ בְּהוּ רַבָּנָן. וְהַרְכָּבַת הָאִילָן אֲסוּרָה בְּחוּצָה לָאָרֶץ מִן הַתּוֹרָה, דִּכְתִיב (שָׁם יט) בְּהֶמְתְּךָ לֹא תַרְבִּיעַ כִּלְאַיִם שָׂדְךָ לֹא תִזְרַע כִּלְאָיִם, מַה בְּהֶמְתְּךָ בְּהַרְבָּעָה אַף שָׂדְךָ בְּהַרְכָּבָה, וּמַה בְּהֶמְתְּךָ בֵּין בָּאָרֶץ בֵּין בְּחוּצָה לָאָרֶץ, אַף שָׂדְךָ בְּהַרְכָּבָה בֵּין בָּאָרֶץ בֵּין בְּחוּצָה לָאָרֶץ (יח):
