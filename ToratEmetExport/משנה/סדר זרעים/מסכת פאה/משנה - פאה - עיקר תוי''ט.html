(א) אֵלּוּ דְבָרִים. הָא דְלֹא תְנַן תְּרוּמָה עִמָּהֶן כֵּיוָן דְּאִיכָּא תְּרוּמַת מַעֲשֵׂר שֶׁיֵּשׁ לָהּ שִׁעוּר לֹא נָקַט אַף סְתָם תְּרוּמָה. יְרוּשַׁלְמִי. וְעוֹד יֵשׁ לוֹמַר שֶׁהִיא חֲלוּקָה מִן הַשְּׁאָר כְּדִמְפָרֵשׁ אַחַר כָּךְ בְּפֵאָה בִּקֵּשׁ לַעֲשׂוֹת כָּל שָׂדֵהוּ פֵּאָה עוֹשֶׂה, בִּקֵּשׁ לַעֲשׂוֹת כָּל כִּרְיוֹ תְּרוּמָה אֵינוֹ עוֹשֶׂה. (וְעוֹד יֵשׁ לוֹמַר דְּלֹא תָנֵי רַק מַה שֶּׁיֵּשׁ בְּהוֹסָפָתָן מִצְוָה כְּדִמְשָׁנֵי בַּיְרוּשַׁלְמִי אַאֵינָךְ שֶׁלֹּא נִשְׁנוּ עַיֵּן שָׁם, וּתְרוּמָה גְדוֹלָה אַדְּרַבָּא מַאֲכִיל טֶבֶל לַכֹּהֵן שֶׁמּוֹסִיף אַחַר שֶׁנָּתַן כַּשִּׁעוּר. הָרַ"שׁ. וְצָרִיךְ עִיּוּן הָא פֵּאָה נַמִּי דִּכְוָתָהּ, עַיֵּן לְהַלָּן מִשְׁנָה ג' וְצָרִיךְ עִיּוּן). תּוֹסְפוֹת יוֹם טוֹב: (ב) הַפֵּאָה. וְעוֹשֶׂה מִצְוָה בְּהוֹסָפָתוֹ. מִיהוּ הֵיכָא דְנָתַן פֵּאָה כְּדִינוֹ בְּסוֹף שָׂדֵהוּ חַיָּב בְּמַעַשְׂרוֹת מַה שֶּׁהוֹסִיף: (ג) וְאַף מִדְּרַבָּנָן אֵין לָהֶן שִׁעוּר: (ד) תַּקָּנַת אוּשָׁא הוּא, בְּמַסֶּכֶת כְּתֻבּוֹת דַּף נ'. וְהַשְׁתָּא מִדְּאוֹרַיְתָא אֵין שִׁעוּר לִגְמִילוּת חֲסָדִים בְּמָמוֹנוֹ: (ה) שֶׁאָדָם אוֹכֵל. כָּתַב הָרַ"מ כִּי הַמִּצְוֹת כֻּלָּם נֶחֱלָקִים בִּתְחִלָּה עַל שְׁנֵי חֲלָקִים. הָאֶחָד, בְּמַה שֶּׁיֵּשׁ בֵּינוֹ וּבֵין הַקָּדוֹשׁ בָּרוּךְ הוּא כְּגוֹן צִיצִית וּתְפִילִין כוּ'. וְהַשֵּׁנִי, מִצְוֹת הַתְּלוּיוֹת בְּתוֹעֶלֶת בְּנֵי אָדָם קְצָתָם עִם קְצָתָם כְּגוֹן הָאַזְהָרָה עַל הַגְּנֵבָה וּכְגוֹן הַצִּוּוּי בְּאַהֲבַת אִישׁ אֶת רֵעֵהוּ. וּכְשֶׁיַּעֲשֶׂה הַמִּצְוֹת שֶׁבַּחֵלֶק הָרִאשׁוֹן יִגְמְלֵהוּ הַקָּדוֹשׁ בָּרוּךְ הוּא עֲלֵיהֶם לָעוֹלָם הַבָּא. וּכְשֶׁיַּעֲשֶׂה הַמִּצְוֹת שֶׁבַּחֵלֶק הַשֵּׁנִי תֵּחָשֵׁב לוֹ לִצְדָקָה לָעוֹלָם הַבָּא וְיִמְצָא טוֹבָה בָּעוֹלָם הַזֶּה בַּעֲבוּר שֶׁנָּהַג מִנְהַג הַטּוֹב בֵּין בְּנֵי אָדָם כוּ', וְכָל הַמִּצְוֹת שֶׁבֵּין אָדָם לַחֲבֵרוֹ נִכְנָסוֹת בִּגְמִילוּת חֲסָדִים: (ו) וְהַקֶּרֶן. וְהֵשִׁיב אֶת אֲשָׁמוֹ בְרֹאשׁוֹ תַּרְגּוּם יְרוּשַׁלְמִי אָשָׁם קֶרֶן. עָרוּךְ. וְהָרַמְבַּ"ן כָּתַב שֶׁבַּעֲבוּר שֶׁהַכֹּחַ בַּקַּרְנַיִם יִקָּרֵא גוּף הַדָּבָר קַרְנוֹ: (ז) כִּבּוּד אָב וָאֵם. כְּתִיב לְמַעַן יַאֲרִיכוּן יָמֶיךָ וּלְמַעַן יִיטַב לָךְ, חַד בָּעוֹלָם הַזֶּה וְחַד בָּעוֹלָם הַבָּא. בִּגְמִילוּת חֲסָדִים כְּתִיב רוֹדֵף צְדָקָה וָחֶסֶד יִמְצָא חַיִּים צְדָקָה וְכָבוֹד. בַּהֲבָאַת שָׁלוֹם כְּתִיב בַּקֵּשׁ שָׁלוֹם וְרָדְפֵהוּ כוּ' אַתְיָא רְדִיפָה מִגְּמִילוּת חֲסָדִים. בְּתַלְמוּד תּוֹרָה כְּתִיב כִּי הוּא חַיֶּיךָ וְאֹרֶךְ יָמֶיךָ: (ח) וְתַלְמוּד תּוֹרָה כוּ'. כִּי בְּתַלְמוּד תּוֹרָה יִזְכֶּה הָאָדָם בְּכָל זֶה, שֶׁהַתַּלְמוּד מֵבִיא לִידֵי מַעֲשֶׂה. הָרַ"מ:

(י) נוֹתְנִין כוּ'. וְלֹא אָמְרִינַן דִּפְאַת שָׂדְךָ מַשְׁמַע דַּוְקָא בַּסּוֹף כִּדְמַשְׁמַע לְשׁוֹן פֵּאָה, וְאִם נָתַן מִתְּחִלָּה וּמֵאֶמְצַע לֹא תִקְדַּשׁ בְּדִין פֵּאָה וּלְחַיֵּב בְּמַעֲשֵׂר. הָרַ"שׁ: (יא) וְהַתּוֹסְפוֹת יוֹם טוֹב הִקְשָׁה עַל זֶה, וּמְחֻוָּר פֵּרוּשׁ הָרַ"שׁ אִם שִׁיֵּר קֶלַח אֶחָד כוּ', כְּשֶׁנָּתַן שִׁעוּר פֵּאָה וּמְבַקֵּשׁ לְהוֹסִיף אַיְרֵי דְּיָכוֹל לְהוֹסִיף עַל הַמְחֻבָּר וְיֵשׁ לוֹ תּוֹרַת פֵּאָה לִפְטֹר מִן הַמַּעֲשֵׂר, וְאִם לָאו, כְּלוֹמַר שֶׁכְּבָר תְּלָשׁוֹ, אֵינוֹ נוֹתֵן אֶלָּא מִשּׁוּם הֶפְקֵר, דְּכָל מַה שֶּׁמּוֹסִיף מִן הַתָּלוּשׁ אֵין לוֹ תּוֹרַת פֵּאָה וְאֵינוֹ נִפְטָר מִן הַמַּעֲשֵׂר עַד שֶׁיַּפְקִיר. וְעַיֵּן עוֹד: (יב) וְנַפְקָא מִנַּהּ דְהֶפְקֵר בֵּין לַעֲנִיִּים בֵּין לַעֲשִׁירִים, מַה שֶּׁאֵין כֵּן פֵּאָה דְּלַעֲנִיִּים כוּ'. אֲבָל לְעִנְיַן מַעֲשֵׂר פֵּאָה נַמִּי פָּטוּר כְּדִתְנַן בְּסוֹף פִּרְקִין. תּוֹסְפוֹת יוֹם טוֹב:

(יג) וְנָקַט סְפִיחֵי וְלֹא סִטִּיס עַצְמוֹ דִּפְשִׁיטָא דְּלֹא חָזֵי כְּלָל. הָרַ"שׁ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (יד) פֵּרוּשׁ בְּאִילָן אֶחָד בְּעַצְמוֹ, וְזֶה לְשׁוֹן הָרַ"מ, יֵשׁ בְּאִילָן זֶה מַה שֶּׁנִּגְמַר בּוֹ הַיּוֹם וְיֵשׁ בּוֹ מַה שֶּׁנִּגְמַר לְאַחַר כַּמָּה יָמִים: (טו) לְשׁוֹן הָרַ"שׁ, דִּסְתָם קָצִיר מִתְבַּשֵּׁל בְּיַחַד: (טז) רֹב יָרָק קָאָמַר, דְּיֵשׁ יָרָק חַיָּב בְּפֵאָה כְּגוֹן שׁוּמִים וּבְצָלִים כְּדִלְקַמָּן פֶּרֶק ג' מִשְׁנָה ג' הַנְהוּ מַכְנִיסָן לְקִיּוּם. הָרַ"שׁ: (יז) וּפֵרוּשׁוֹ דָּחוּק. וְהָרַ"שׁ כָּתַב עַל פִּי תּוֹרַת כֹּהֲנִים, וּסְתָם קָצִיר מַכְנִיסָן לְקִיּוּם, יָצְאוּ יְרָקוֹת שֶׁאַף עַל פִּי שֶׁלְּקִיטָתוֹ כְּאֶחָד אֵינוֹ מַכְנִיסוֹ לְקִיּוּם: (יח) וְהַתְּבוּאָה. בַּיְרוּשַׁלְמִי דָּרְשִׁינַן וּבְקֻצְרְכֶם אֵין לִי אֶלָּא תְּבוּאָה קִטְנִיּוֹת מִנַּיִן תַּלְמוּד לוֹמַר אַרְצְכֶם: (יט) וְדִבְרֵי תֵימַהּ הֵם דַּהֲרֵי וַדַּאי שֶׁנִּשְׁלְמוּ בָהֶם כוּ'. וּלְשׁוֹן הָרַ"מ בְּחִבּוּרוֹ כָּל הַדּוֹמֶה לְקָצִיר בְּחָמֵשׁ דְּרָכִים אֵלּוּ הוּא שֶׁחַיָּב בַּפֵּאָה כְּגוֹן הַתְּבוּאָה כוּ', וְאוּלַי מִשּׁוּם נִשְׁמָר כָּתַב כֵּן. תּוֹסְפוֹת יוֹם טוֹב:
