(א) שֶׁהֲרֵי מָנָה שָׁנָה זוֹ שְׁנַת אַרְבַּע מֵאוֹת וּשְׁמוֹנִים לִיצִיאַת מִצְרַיִם, וְשָׁנָה רְבִיעִית לִמְלֹךְ שְׁלֹמֹה. רַשִׁ"י: (ב) כִּדְּיָלֵיף לַהּ מִקְּרָאֵי בַּגְּמָרָא: (ג) לְמַלְכֵי יִשְׂרָאֵל אַסְבְּרַא לֵיהּ, דִּבְהוּ קַיְמִינַן הַשְׁתָּא: (ד) דְּאִלּוּ בְּנִיסָן שֶׁלְּאַחַר אֲדָר וְכָתַב שֶׁל שָׁנָה רִאשׁוֹנָה, אַף עַל פִּי שֶׁלְּפִי הָאֱמֶת הִיא שָׁנָה הַשְּׁנִיָּה, לֵיכָּא לְמֵיחַשׁ כוּ', שֶׁכֵּיוָן שֶׁזֶּה הַמֶּלֶךְ לֹא מָלַךְ שׁוּם נִיסָן, וְשֵׁם הַמֶּלֶךְ הֲרֵי כָּתוּב בַּשְּׁטָר, וְנִמְצָא שֶׁלֹּא יוּכַל לִטְרֹף לָקוֹחוֹת אֶלָּא שֶׁלָּקְחוּ אַחַר הַנִּיסָן שֶׁמָּלַךְ זֶה הַמֶּלֶךְ, וּלְאוֹתוֹ כְּדִין הוּא טוֹרֵף: (ה) עַיֵּן בְּהָרַ"ב מִשְׁנָה ה' פֶּרֶק י' דִּשְׁבִיעִית: (ו) כְּלוֹמַר וְרַבִּי שִׁמְעוֹן הִיא, גְּמָרָא. וְטַעֲמַיְיהוּ דְּרַבִּי שִׁמְעוֹן וְרַבָּנָן מִקְּרָאֵי יָלְפִינַן לְהוּ בַּגְּמָרָא: (ז) בְּתִשְׁרֵי. וְאַרְבָּעָה רָאשֵׁי שָׁנִים דְּחָשְׁבֵי לַחֲמִשָּׁה עָשָׂר בְּנִיסָן שֶׁהוּא רֹאשׁ הַשָּׁנָה לָרְגָלִים, הָכִי קָאָמַר, אַרְבָּעָה רָאשֵׁי שָׁנִים לְדִבְרֵי הַכֹּל. לְרַבִּי שִׁמְעוֹן דְּבָעֵי שְׁלֹשָׁה רְגָלִים כְּסִדְרָן דַּל מַעֲשַׂר בְּהֵמָה, וּלְדִפְלִיגֵי עֲלֵיהּ דַּל רְגָלִים. גְּמָרָא: (ח) וְהַאי גָּמְרוּ הַיְנוּ שֶׁנִּגְמַר לְהִתְחַיֵּב בְּמַעֲשֵׂר, שֶׁאֵין אוֹכְלִים אֲפִלּוּ עֲרַאי, וְהוּא מִשֶּׁיִרְאֶה פְּנֵי הַבַּיִת: (ט) גְּמָרָא. וּזְמַן לֵידַת בְּהֵמָה גַּסָּה אַף עַל גַּב דְּיוֹלֶדֶת לְתִשְׁעָה חֳדָשִׁים, נַמִּי בְּאָב הוּא, אֶלָּא שֶׁזְּמַן עִבּוּרָן קָדִים אַרְבָּעָה חֳדָשִׁים. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (י) לַשָּׁנִים כוּ'. מִקְּרָאֵי יָלְפִינַן לְהוּ בַּגְּמָרָא: (יא) שֶׁיּוֹם הַנְּטִיעָה אֵינוֹ מִכְּלָל יְמֵי הַקְּלִיטָה: (יב) בַּגְּמָרָא יָלֵיף לַהּ: (יג) הָכִי יָלְפִינַן לְהוּ בַּגְּמָרָא: (יד) כִּדְמְפָרֵשׁ בַּגְּמָרָא דְתַנָּא תָּנֵי יְרָקוֹת שֶׁמַּעַשְׂרוֹתֵיהֶן בִּשְׁבָט. הוֹאִיל וְיָצְאוּ רֹב גִּשְׁמֵי שָׁנָה וְעָלָה הַשָּׂרָף בָּאִילָנוֹת, וְנִמְצָא הַפֵּרוֹת חוֹנְטִין מֵעַתָּה. גְּמָרָא: (יז) דִּלְעִנְיַן עָרְלָה וּרְבָעִי כְּבָר נִתְבָּאֵר לְעֵיל:

(יח) זֶה לְשׁוֹן הַבְּרַיְתָא, מִפְּנֵי מָה אָמְרָה הַתּוֹרָה הָבִיאוּ עֹמֶר בְּפֶסַח, מִפְּנֵי שֶׁהַפֶּסַח זְמַן תְּבוּאָה הוּא, אָמַר הַקָּדוֹשׁ בָּרוּךְ הוּא הָבִיאוּ כוּ' כְּדֵי כוּ', וְכֵן בְּכֻלָּם: (יט) בְּרֹאשׁ הַשָּׁנָה. בַּגְּמָרָא (דַּף ח) יָלְפִינַן לַהּ מִקְּרָאֵי. וְאַף עַל גַּב דְּוַדַּאי עַל כָּל הַמְּאֹרָעוֹת נִדּוֹנִין וּבִכְלָלָם אִינָךְ שְׁלֹשָׁה דִּינִים, מִכָּל מָקוֹם בִּשְׁלֹשָׁה פְּרָקִים הַלָּלוּ הוּא גְּזַר דִּין שֶׁל כָּל הָעוֹלָם, וּבְרֹאשׁ הַשָּׁנָה כָּל בָּאֵי כוּ' וְגוֹזְרִין לְכָל אֶחָד חֶלְקוֹ מִדְּבָרִים אֵלּוּ. הָרַ"ן. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (כ) גְּמָרָא. וּפֵרֵשׁ רַשִׁ"י וְאַקְּרָא דִּלְעֵיל מִנֵּיהּ קָמְהַדַּר, הִשְׁגִּיחַ אֶל כָּל כוּ', הַיּוֹצֵר אוֹתָם הִשְׁגִּיחַ יַחַד אֶת לִבָּם. וּלְהָכִי נָסֵיב הַאי תַּנָּא לְהַךְ קְרָא, לוֹמַר שֶׁאַף עַל פִּי שֶׁעוֹבְרִין כִּבְנֵי מָרוֹן בְּדִקְדּוּק גָּדוֹל, מִכָּל מָקוֹם כֻּלָּן נִסְקָרִין בִּסְקִירָה אַחַת. וְזֶהוּ שֶׁכָּתַב הָרַמְבַּ"ם, וְהַנִּגְלֶה מִזֶּה הַמַּאֲמָר מְבֹאָר כַּאֲשֶׁר תִּרְאֶה, אֲבָל הַנִּסְתָּר עִנְיָנוֹ קָשֶׁה בְּלִי סָפֵק, עַד כָּאן. כִּי הָאָדָם אִי אֶפְשָׁר לְצַיֵּר שְׁנֵי הֲפָכִים בְּנוֹשֵׂא אֶחָד בִּזְמַן אֶחָד שֶׁיַּעַבְרוּ אֶחָד אֶחָד וְיִהְיוּ נִסְקָרִים בִּסְקִירָה אַחַת, אֲבָל כְּמוֹ שֶׁאִי אֶפְשָׁר לְהַשִּׂיג מַהוּתוֹ יִתְבָּרֵךְ וְיִתְעַלֶּה זִכְרוֹ לָנֵצַח, כֵּן אִי אֶפְשָׁר לְהַשִּׂיג הַשְׁגָּחָתוֹ וּסְקִירָתוֹ. תּוֹסְפוֹת יוֹם טוֹב: (כא) וּבֶחָג. וְאַף עַל פִּי שֶׁשְּׁלָשְׁתָּן בְּשֵׁם חַגִּים נִקְרְאוּ בַּכָּתוּב, מִכָּל מָקוֹם מִלַּת חָג הֲנָחָתוֹ עַל הַמָּחוֹל וְהָרִקּוּד שֶׁהוּא בְּחַג הַסֻּכּוֹת, וְהָיְתָה שִׂמְחָה יְתֵירָה שֶׁל בֵּית הַשּׁוֹאֵבָה: (כב) כְּדֵי שֶׁיִּתְבָּרְכוּ לָכֶם גִּשְׁמֵי הַשָּׁנָה. בְּרַיְתָא:

(כג) נִיסָן. אֲבָל סִיוָן מִפְּנֵי הָעֲצֶרֶת לֹא צָרִיךְ, דְּבִסְפִירַת עֹמֶר הַדָּבָר תָּלוּי. תּוֹסָפוֹת: (כד) וְאַף עַל גַּב דִּבְשִׁבְעָה עָשָׂר בְּתַמּוּז נַמִּי אֵרְעוּ חֲמִשָּׁה דְּבָרִים חֻרְבַּן בֵּית הַמִּקְדָּשׁ תַּקִיפָא טוּבָא. וְעוֹד דִּבְתִשְׁעָה בְּאָב צָרָה אַחַת הֻכְפְּלָה בּוֹ. תּוֹסָפוֹת: (כה) הַפּוּרִים. וְאִלּוּ עַל אֲדָר הַשֵּׁנִי מִפְּנֵי פּוּרִים לֹא קָתָנֵי, דִּסְבִירָא לֵיהּ דִּלְעוֹלָם אֲדָר הָרִאשׁוֹן מְעֻבָּר וְיָדְעוּ בְּנֵי הַגּוֹלָה דִּבְיוֹם שְׁלֹשִׁים וְאֶחָד נִתְקַדֵּשׁ. גְּמָרָא: (כו) פֶּסַח קָטָן. לְפִיכָךְ קָרֵי לֵיהּ פֶּסַח קָטָן, דְּלֹא הָיָה רַק יוֹם אֶחָד. מְפָרֵשׁ הָרַמְבַּ"ם בְּהִלְכוֹת קִדּוּשׁ הַחֹדֶשׁ:

(כז) הַיְנוּ לְאַחַר שֶׁיִּשְׁמְעוּ מִפִּי בֵּית דִּין מְקֻדָּשׁ, דְּאַף שֶׁנִּרְאֶה בַּעֲלִיל [בְּיוֹם כ"ט] אוֹ בְּלֵיל שְׁלֹשִׁים צְרִיכִין לְהַמְתִּין עַד שֶׁיִּשְׁמְעוּ כוּ', דְּדִלְמָא מִמְלְכֵי בֵית דִּין וּמְעַבְּרֵי לֵיהּ וְאָתֵי לְקִלְקוּלֵי מוֹעֲדוֹת, וְאִם כֵּן אִם לֹא יָבֹאוּ עֵדִים וְיִתְעַבֵּר הַחֹדֶשׁ לֹא יֵצְאוּ הַשְּׁלוּחִין עַד לְמָחָר, שֶׁאֵין מְקַדְּשִׁין בַּלַּיְלָה (וְאַף דְּאָז אִי אֶפְשָׁר לִדְחוֹתוֹ יוֹתֵר צְרִיכִין לְהַמְתִּין, דְּגָזְרֵי הַאי אַטּוּ הַאי), נִמְצְאוּ מְמַעֲטִין הִלּוּכָן מַהֲלַךְ לַיְלָה וְיוֹתֵר. אֲבָל בִּשְׁאָר חֳדָשִׁים אֵינָם צְרִיכִין לְהַמְתִּין עַד שֶׁיִּשְׁמְעוּ מְקֻדָּשׁ, דְּאֵין הַמּוֹעֲדוֹת תְּלוּיִן בָּהֶם, וְאִם כֵּן אֵין חִלּוּל שַׁבָּת [שֶׁל עֵדִים] מְמַהֵר יְצִיאַת הַשְּׁלוּחִים. גְּמָרָא וּפֵרוּשׁ רַשִׁ"י וְתוֹסָפוֹת: (כח) וּבָהֶן כוּ'. אֲבָל אֱלוּל אַף עַל גַּב דְּמִנֵּיהּ חַשְׁבִינַן לַעֲשׂוֹת בְּיוֹם שְׁלֹשִׁים רֹאשׁ הַשָּׁנָה, לֹא נָפְקָא מִינָהּ מִידֵי, לְפִי שֶׁעֲדַיִן בְּיוֹם שְׁלֹשִׁים יוֹשְׁבִים וּמְצַפִּים כוּ', וְאִם לֹא נוֹדַע לָהֶם נוֹהֲגִין הַיּוֹם וְיוֹם שְׁלֹשִׁים וְאֶחָד קֹדֶשׁ עַד שֶׁיָּבֹאוּ לָהֶם שְׁלוּחֵי תִּשְׁרֵי. הָרַמְבַּ"ם:

(כב) דְּסֵיפֵיהּ דִּקְרָא דִּוְהַעֲבַרְתָּ שׁוֹפַר תְּרוּעָה אִיהוּ: (כג) מִשּׁוּם דִּמְרָמֵי לַהּ רָמֵי מַתְנִיתִין וּבְרַיְתָא אַהֲדָדֵי, מְשָׁנֵי לַהּ בַּגְּמָרָא דְּתַנָּא דִּידָן קָא חָשִׁיב תְּקִיעוֹת דְּכֻלְּהוּ בָּבֵי. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (כד) כְּשָׁלֹשׁ. רַ"ח גּוֹרֵס שָׁלֹשׁ יְבָבוֹת, וְלֹא כְּשָׁלֹשׁ, דְּהָא תְּרוּעָה עַצְמָה שָׁלֹשׁ יְבָבוֹת, וְהֵיכִי שַׁיָּךְ לְמִתְנֵי כְּשָׁלֹשׁ. אֶלָּא שֶׁאֶפְשָׁר שֶׁלְּפִי שֶׁעִקַּר יְבָבוֹת בְּפִיו, קָאָמַר דְּשִׁעוּר תְּרוּעָה כְּשָׁלֹשׁ יְבָבוֹת שֶׁאָדָם מְיַלֵּל בְּפִיו: (כה) יְבָבוֹת. בַּגְּמָרָא, דִּתְרוּעָה מְתַרְגְמִינַן יְבָבָא, וּכְתִיב בְּאִמֵּיהּ דְּסִיסְרָא [שֹׁפְטִים ה] בְּעַד הַחַלּוֹן נִשְׁקְפָה וַתְּיַבֵּב. עַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (כו) בָּרִאשׁוֹנָה כוּ'. שֶׁאֵין לַתְּקִיעוֹת שִׁעוּר לְמַעְלָה, שֶׁאִם רָצָה לְהַאֲרִיךְ מַאֲרִיךְ, וְאֵין הַדָּבָר נִכָּר כְּשֶׁהוּא מַאֲרִיךְ שֶׁיִּתְכַּוֵּן לִשְׁתֵּי תְּקִיעוֹת, אֶלָּא מֵחֲמַת כְּשֶׁהִיא כִּפְלַיִם כְּשִׁעוּר תְּקִיעָה רִאשׁוֹנָה, מִשּׁוּם הָכִי תָּנֵי תָּקַע בָּרִאשׁוֹנָה. הָרֹא"שׁ וְהָרַ"ן: (כז) וְאַף עַל פִּי שֶׁיִּמְשֹׁךְ עַד שֶׁיֹּאמַר עֲשָׂרָה זִכְרוֹנוֹת יֵשׁ לַעֲשׂוֹת כָּךְ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (כח) אֶלָּא אֶחָת. הַיְנוּ תְּקִיעָה שֶׁלַּאֲחָרֶיהָ, וְלֹא כַּיֵּשׁ מְפָרְשִׁים דְּהַיְנוּ הָרִאשׁוֹנָה וְשֶׁמִּשּׁוּם הָכִי תָּנֵי תָּקַע בָּרִאשׁוֹנָה: (כט) שָׁלֹשׁ פְּעָמִים. בִּשְׁבִיל מַלְכוּיוֹת, וְכֵן בִּשְׁבִיל זִכְרוֹנוֹת, וְכֵן בִּשְׁבִיל שׁוֹפָרוֹת. רַשִׁ"י: (ל) דְּאַף עַל גַּב דְּכָל יִשְׂרָאֵל עֲרֵבִים הֵם זֶה לָזֶה, וּמֵהַאי טַעְמָא מוֹצִיא זֶה אֶת זֶה בְּבִרְכוֹת הַמִּצְוֹת, הָא תַּנְיָא בַּגְּמָרָא [אַמַּתְנִיתִין דְּהָכָא] יָחִיד שֶׁלֹּא בֵּרַךְ אֵין חֲבֵרוֹ מְבָרֵךְ עָלָיו, וְטַעְמָא כָּתַב הָרַ"ן בְּשֵׁם יְרוּשַׁלְמִי מִשּׁוּם דִּתְפִלָּה מִסְתַּבֵּר שֶׁיִּהְיֶה כָּל אֶחָד מְבַקֵּשׁ רַחֲמִים עַל עַצְמוֹ:
