(ח) וּדְקָתָנֵי בֵּין שְׁנַיִם הַחִיצוֹנִים הָכִי קָאָמַר, אִם יֵשׁ אֲוִיר בֵּין שְׁנֵי הַחִיצוֹנִים כְּדֵי לְהִתְמַלְּאוֹת אֶמְצָעִי וְלַעֲמֹד אֲוִיר הַנִּשְׁאָר בֵּין אֶמְצָעִי לַחִיצוֹנִים עַל מֵאָה אַרְבָּעִים וְאַחַת וּשְׁלִישׁ לְכָאן וּמֵאָה אַרְבָּעִים וְאַחַת וּשְׁלִישׁ לְכָאן, דְּהַיְנוּ שְׁתֵּי קַרְפֵּפוֹת לִשְׁנֵיהֶן, רוֹאִין כוּ', אֲבָל אִם רָבָה אֲוִיר יוֹתֵר מִכָּאן לֹא אָמְרִינַן רוֹאִין. רַשִׁ"י:

(ט) אֲבָל בְּיָדַיִם לֹא. גְּמָרָא. וּפֵרֵשׁ רַשִׁ"י דְּחַיְשִׁינַן דְּשָׁקֵיל טוּבָא וּמְמַעֵט, אֲבָל בְּהֵמָה כוּ'. וְקָשֶׁה דִּלְקַמָּן פֶּרֶק ט' מִשְׁנָה ג' מְפָרֵשׁ דִּמְחִצָּה שֶׁנִּפְרְצָה בְּשַׁבָּת לֹא בָּטֵיל עֵרוּב. וְהָרֹא"שׁ מְפָרֵשׁ דִּבְחֹל קָאָמַר, וְאָסוּר בַּיָּדַיִם דִּלְמָא מְמַעֵט לְהוּ בְּעֶרֶב שַׁבָּת סָמוּךְ לַחֲשֵׁכָה וְלָאו אַדַּעְתֵּיהּ, וּמְטַלְטֵל בְּשַׁבָּת בֶּחָצֵר הָאֲסוּרָה, וְכֵן מוּכָח לִישְׁנָא דְּמַתְנִיתִין דְּנִתְמַעֵט [דְּמַיְרֵי בְּחֹל] כִּדְפֵרֵשׁ הָרַ"ב. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

(י) הַגְּדוֹלִים. וַאֲפִלּוּ סְמוּכִים עַל שֻׁלְחָנוֹ, דְּכֵיוָן דְּרוֹצֶה לְזַכּוֹת לַאֲחֵרִים זוֹכִים הֵם לַאֲחֵרִים כֶּאֱינַשׁ דְּעָלְמָא, וְהוּא הַדִּין בְּקָטָן דְּלֹא, אַף עַל פִּי שֶׁאֵינוֹ סָמוּךְ: (יא) הָעִבְרִים. בְּפֶרֶק קַמָּא דְּקִדּוּשִׁין מִשְׁנָה ב' תְּנַן דְּאָמָה הָעִבְרִיָּה קוֹנָה אֶת עַצְמָהּ בְּסִימָנִין, וְהַשְׁתָּא לֹא מַשְׁכַּחַת לַהּ שִׁפְחָה עִבְרִיָּה אֶלָּא קְטַנָּה, וְהָכִי אִיתָא בַּגְּמָרָא, וּבְשִׁתּוּפֵי מְבוֹאוֹת דְּרַבָּנָן אָמְרֵי שֶׁתְּזַכֶּה אֲפִלּוּ לַאֲחֵרִים, וְשָׁאנֵי שִׁפְחָה מִבִּתּוֹ הַקְּטַנָּה, דְּיָדָהּ כְּיַד אָבִיהָ אֲבָל לֹא כְּיַד רַבָּהּ, שֶׁאֵין קִנְיָנוֹ בָּהּ קִנְיָן גָּמוּר. וְהָכִי נַמִּי מַסִּיק בַּגְּמָרָא בָּבָא מְצִיעָא דַּף י"ב: (יב) אִשְׁתּוֹ. בְּפֶרֶק י"א דִּנְדָרִים מִשְׁנָה ח' תְּנַן סְתָמָא כְּרַבִּי מֵאִיר דְּיַד הָאִשָּׁה כְּיַד בַּעֲלָהּ. וּמִשּׁוּם הָכִי מַסִּיק הָרֹא"שׁ דְּהָכָא כְּשֶׁאֵינוֹ זָנָהּ, וּלְפִיכָךְ מַעֲשֵׂה יָדֶיהָ שֶׁלָּהּ וְאֵין יָדָהּ כְּיָדוֹ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (יג) שֶׁיָּדָן כוּ'. וּבְרֵישׁ פֶּרֶק ח' אִיתָא בַּגְּמָרָא מְעָרֵב עַל יְדֵי עַבְדּוֹ כוּ', פֵּרוּשׁ בִּשְׁבִיל, וְעַל יְדֵי בְּנוֹ וּבִתּוֹ מִפְּנֵי שֶׁיָּדָן כוּ', וְכָתְבוּ הַתּוֹסָפוֹת אַעַבְדּוֹ וְשִׁפְחָתוֹ קָאֵי, כִּדְאָמְרִינַן בְּגִטִּין דְּיַד עֶבֶד קָנֵי לֵיהּ רַבֵּיהּ. וְנִרְאֶה דְּהָכִי נַמִּי לֹא קָאֵי אֶלָּא אַעַבְדּוֹ כוּ', וְטַעֲמָא דִּבְנוֹ וּבִתּוֹ הַקְּטַנִּים אַף עַל גַּב דְּיֵשׁ לָהֶן יַד לִזְכּוֹת כְּשֶׁהִגִּיעַ לִכְלַל אֱגוֹז וְנוֹטְלוֹ כוּ', הָכָא לְעִנְיַן עֵרוּב כֵּיוָן שֶׁיָּכוֹל לְעָרֵב בִּשְׁבִילָם שֶׁהֵם סְמוּכִים עָלָיו כִּדְאָמְרִינַן בַּגְּמָרָא, הִלְכָּךְ אָמְרוּ שֶׁאֵין יָכוֹל לְזַכּוֹת עַל יְדֵיהֶם. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

(מז) כּוֹרֵךְ. וְאִם תֹּאמַר הָא הַוְיָא חֲצִיצָה, מוֹקֵי בַּגְּמָרָא דְּהָכָא בִּשְׂמֹאל, דַּעֲבוֹדַת כֹּהֵן בְּיָמִין. וְאִי נַמִּי בְּיָמִין, שֶׁלֹּא בִּמְקוֹם עֲבוֹדָה, וּפֵרֵשׁ רַשִׁ"י כְּגוֹן עַל גַּב הָאֶצְבַּע: (מח) קָשֶׁה, דְּתֵפּוֹק לֵיהּ דְּאֵין זֶה צֹרֶךְ עֲבוֹדָה. וְזֶה לְשׁוֹן רַשִׁ"י, אָסוּר שֶׁאֵין זֶה צֹרֶךְ עֲבוֹדָה, וְעוֹד דַּהֲוָה לֵיהּ חוֹבֵל כוּ' (עַד כָּאן). וּבְהָכִי נִיחָא נַמִּי מֵהַאי דְּפֶרֶק י"ט דְּשַׁבָּת: (מט) פֵּרוּשׁ וְחַיָּב מִשּׁוּם זוֹרֵעַ, שֶׁכְּמוֹ שֶׁהַזּוֹרֵעַ כַּוָּנָתוֹ לְהַצְמִיחַ הַפְּרִי, כֵּן כַּוָּנַת הַמַּשְׁקֶה צְמָחִים וְאִילָנוֹת: (נ) לְהָרַמְבַּ"ם אֲפִלּוּ בְּחָצֵר, כָּל שֶׁאֵין שָׁם גִּנָּה. וּלְרַבֵּנוּ תָּם דַּוְקָא בְּגַלְגַּל קָטָן כְּאוֹתָן שֶׁבַּבָּתִּים. טוּר אֹרַח חַיִּים סִימָן של"ח: (נא) וְכֻלְּהוּ בּוֹרוֹת הַנּוֹבְעִים מַיִם חַיִּים נַמִּי נִקְרָאִים כָּךְ. וְלֹא זֶהוּ הֵתֵּרוֹ מִשּׁוּם שֶׁהָיָה קַר, אֶלָּא הֵתֵּרוֹ מִפְּנֵי הַצֹּרֶךְ הָיָה:

פרק י - משנה טו

(נב) שֶׁרֶץ שֶׁנִּמְצָא. כְּלוֹמַר שֶׁמְּצָאוּהוּ מֵת. הָרִי"כ: (נג) תַּרְגּוּם אַבְנֵט הֶמְיָנָא. הָרַמְבַּ"ם: (נד) הָאוּלָם וְלַמִּזְבֵּחַ. דְּמִשָּׁם וּלְהַלָּן קְדֻשָּׁה יְתֵרָה כְּדִתְנַן לַהּ סוֹף פֶּרֶק קַמָּא דְּכֵלִים, עַל שְׁבוּת כָּזֶה גָּזְרוּ אַף בַּמִּקְדָּשׁ: (נה) וְהָא דְּנָטַר עַד הָכָא, מִשּׁוּם דְּבָעֵי לְאַסּוֹקֵי מִלֵּי דְּמִקְדָּשׁ אֲבָל לֹא בַּמְּדִינָה. תּוֹסָפוֹת: (נו) כְּפֵרוּשׁוֹ דִּלְעֵיל דִּלְתַּנָּא קַמָּא עַצְמוֹ לֹא הִתִּירוּ אֶלָּא שֶׁאֵינוֹ שֶׁל קְיָמָא: (נז) דִּתְחוּמִין אִיכָּא מַאן דְּאָמַר דְּאִית בֵּיהּ דְּרָרָא דְּאִסּוּר דְּאוֹרַיְתָא, וְכָאן שֶׁאָנוּ מַחְמִירִין דַּאֲפִלּוּ שְׁבוּת הַקַּל אֵין מַתִּירִין בַּמְּדִינָה, רַבִּי שִׁמְעוֹן מַחְמִיר יוֹתֵר מֵהֶם, שֶׁאֲפִלּוּ בַּמִּקְדָּשׁ אֵינוֹ מַתִּיר שְׁבוּת שֶׁאֶפְשָׁר לָבֹא לִידֵי חִיּוּב. רִי"כ:
