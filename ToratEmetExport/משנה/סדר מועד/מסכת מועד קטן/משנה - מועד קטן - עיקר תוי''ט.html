פרק א - משנה א

(א) וְהֵ"א מִתְחַלֵּף בְּחֵי"ת. רַשִׁ"י. שֶׁשְּׁנֵיהֶם מִמּוֹצָא הַגָּרוֹן. וְעָיֵף הַיְנוּ לְמַיִם, כְּדִכְתִיב [מִשְׁלֵי כה] מַיִם קָרִים עַל נֶפֶשׁ עֲיֵפָה: (ב) בַּמּוֹעֵד. הַמִּשְׁנָה תִּקְרָא [מוֹעֵד, מַה שֶּׁאָנוּ קוֹרִים] חֻלּוֹ שֶׁל מוֹעֵד, [וְהֵם] הַיָּמִים שֶׁבֵּין מוֹעֵד לְמוֹעֵד. הָרַמְבַּ"ם: (ג) רַשִׁ"י. וְהָרִי"ף וְהָרַמְבַּ"ם וְהָרֹא"שׁ וְהַטּוּר פֶּה אֶחָד הִסְכִּימוּ דְּאֵין לְחֻלּוֹ שֶׁל מוֹעֵד עִקָּר מִן הַתּוֹרָה. וּבְרַיְתָא דַּחֲגִיגָה דְּלַמְּדוּהָ מִקְּרָא אֵינוֹ אֶלָּא אַסְמַכְתָּא. וְהַבֵּית יוֹסֵף כָּתַב שֶׁנִּרְאֶה לוֹ מִזֹּאת הַבְּרַיְתָא שֶׁיֵּשׁ לוֹ עִקָּר מִן הַתּוֹרָה, וְהַכָּתוּב מְסָרָהּ לַחֲכָמִים לֶאֱסֹר וּלְהַתִּיר כְּפִי שֶׁנִּרְאֶה לָהֶם: (ד) אֲבָל הָרַמְבַּ"ם נִרְאֶה שֶׁאֵין סוֹבֵר כָּךְ, אֶלָּא דְּבֵית הַבַּעַל נַמִּי אָסוּר בִּשְׁבִיעִית. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (ה) כְּלוֹמַר שֶׁאֲפִלּוּ דָּבָר הָאָבֵד לֹא הִתִּירוּ כְּשֶׁיֵּשׁ טִרְחָא: (ו) שֶׁלֹּא. וְאִצְטְרִיךְ לְמִתְנֵי לְמִעוּטֵי שְׂדֵה הַבַּעַל, שֶׁאֲפִלּוּ מִמַּעְיָן שֶׁלֹּא יָצָא בִּתְחִלָּה אָסוּר. גְּמָרָא: (ז) אֵין מַשְׁקִין. אַמּוֹעֵד קָאֵי, דְּגַבֵּי שְׁבִיעִית לֹא חָיְישִׁינַן לְטִרְחָא יְתֵרָא, וְטַעְמָא מְפָרֵשׁ בַּיְרוּשַׁלְמִי דִּשְׁבִיעִית עַל יְדֵי שֶׁהוּא מֻתָּר בִּמְלָאכָה שֶׁאֵינָהּ דְּקַרְקַע הִתִּירוּ אֲפִלּוּ בְּדָבָר שֶׁל טֹרַח, אִי נַמִּי שְׁבִיעִית עַל יְדֵי שֶׁזְּמַנָּהּ מְרֻבֶּה כוּ' וְלֹא מִסְתַּבֵּר לַעֲשׂוֹת שִׁבְעָה יָמִים הָאַחֲרוֹנִים כְּשִׁבְעָה דְּרֶגֶל: (ח) וּמִלְּשׁוֹן קוּלְתָּא שֶׁהוּא תִּרְגּוּם כַּדָּהּ עַל שִׁכְמָה, נִקְרֵאת קִילוֹן. הָרַ"ן: (ט) הַקִּילוֹן. כֵּיוָן דְּמֵי הַגְּשָׁמִים לֹא נֶאֶסְרוּ אֶלָּא מִשּׁוּם גְּזֵרָה דְּמֵי הַקִּילוֹן, הִלְכָּךְ שַׁפִּיר נַקְטֵיהּ אַחַר מֵי גְּשָׁמִים. הָרַ"ן: (י) וְאֵין עוֹשִׂין כוּ'. מִדְּרֵישָׁא דְּהַךְ אֲבָל אֵין מַשְׁקִין לֹא קָאֵי אַשְּׁבִיעִית, הַךְ סֵיפָא נַמִּי לֹא קָאֵי אַשְּׁבִיעִית. הָרַמְבַּ"ם: (יא) כְּדֵי שֶׁיִּתְעַכְּבוּ שָׁם הַמַּיִם, מִלְּשׁוֹן [רוּת א] הֲלָהֶן תֵּעָגֵנָה, שֶׁהוּא לְשׁוֹן עִכּוּב. הָרַ"ן: (יב) לִישְׁנָא דִּבְרַיְתָא, וּבְאַתְרֵיהּ דְּתַנָּא דְּמַתְנִיתִין שְׁכִיחֵי גְּפָנִים וְלֹא זֵיתִים:

פרק א - משנה ב

(יג) וַחֲכָמִים אוֹמְרִים כוּ'. דְּכָל שֶׁנִּרְאֶה כְּחוֹפֵר לֹא הֲוֵי עֲבוֹדָה, דְּעוֹדֵר לֹא הֲוֵי כֵּיוָן דְּמַשְׁלִיךְ אֶת הֶעָפָר לְמֵרָחוֹק, וְהָעוֹדֵר מֵנִיחַ בִּמְקוֹמוֹ. הָרַ"ן. וּבָרַמְבַּ"ם מַשְׁמָע דְּמַתִּיר אֲפִלּוּ אֵינוֹ מַשְׁלִיךְ לְמֵרָחוֹק, וְעַיֵּן שָׁם הַטַּעַם: (יד) הַמְקֻלְקָלוֹת. סִיּוּמָא דְּמִלְּתָא דְּרַבָּנָן הוּא: (טו) וְלֹא דָּמִי לְמַעְיָן שֶׁיָּצָא בַּתְּחִלָּה כְּדִלְעֵיל דְּחַיְּשִׁינַן שֶׁיִּפְּלוּ גְּדוֹתָיו וְאָתֵי לְתַקֵּן, דְּתִקּוּנָא דְּהָתָם עֲשִׂיָּה בַּתְּחִלָּה הִיא כִּדְקָתָנֵי שֶׁיָּצָא בַּתְּחִלָּה: (טז) שֶׁבִּרְשׁוּת הָרַבִּים. אֲפִלּוּ אֵין רַבִּים צְרִיכִים לָהֶם. וְטַעְמָא כָּתַב הָרַ"ן בְּשֵׁם הָרַאבַ"ד לְפִי שֶׁצָּרְכֵי רַבִּים אֵינָם נִגְמָרִים אֶלָּא בִּשְׁעַת הַבַּטָּלָה, מִשּׁוּם דְּדָמִי לִקְדֵרָה דְּבֵי שֻׁתְּפֵי כוּ', וּמַסִּיק דְּכִי הִתִּירוּ בִּמְלֶאכֶת הֶדְיוֹט [וּכְדִלְקָמָּן]. וְנִרְאֶה לִי דְּהַיְנוּ דְּתָנֵי וְחוֹטְטִין [דְּהַיְנוּ שֶׁהוּא מְלֶאכֶת הֶדְיוֹט] וּפֵרוּשֵׁי קָמְפָרֵשׁ דְּאִי לֹא תֵּימָא הָכִי לִישָׁנָא יְתֵרָא הוּא דְּהַיְנוּ מְתַקְּנִין: (יז) וּמְתַקְּנִין כוּ'. בְּרֵישׁ שְׁקָלִים תְּנַן דִּבְט"ו בַּאֲדָר מְתַקְּנִין. וּלְפִי שֶׁהַקִּלְקוּלִים מְצוּיִין בַּדְּרָכִים לְכָךְ הֻצְרַךְ תּוּ לְתַקֵּן בְּחֹל הַמּוֹעֵד: (יח) וְעוֹשִׂין כוּ'. לַאֲתוֹיֵי בּוֹרוֹת שִׁיחִין וּמְעָרוֹת כְּשֶׁרַבִּים צְרִיכִין לָהֶם עַתָּה. אֲבָל שֶׁלֹּא לְצֹרֶךְ הַמּוֹעֵד לֹא הִתִּירוּ, מִפְּנֵי שֶׁהֵן מְלֶאכֶת אֻמָּן, וְאֵינוֹ יָכֹל לְשַׁנּוֹת בָּהֶן, וְאִם יְשַׁנֶּה בָּהֶן יַפְסִיד. הָרֹא"שׁ. וְעַיֵּן פֵּרוּשׁ הָרַ"ב בְּרֵישׁ שְׁקָלִים, וְהוּא מֵהַיְרוּשַׁלְמִי. וּגְמָרָא דִּידָן לֹא סְבִירָא לֵיהּ הַאי דִּיּוּקָא, מִשּׁוּם דְּלֹא דָּמִי לְהָנָךְ פְּרָטֵי דְּמַתְנִיתִין: (יט) וְיוֹצְאִין כוּ'. בַּגְּמָרָא רָמֵי וְהָתְנַן בְּרֵישׁ שְׁקָלִים בְּא' בַּאֲדָר כוּ', וּמְשָׁנֵי כָּאן בְּבָכִיר כָּאן בְּאָפִיל: (כ) וְלֹא הֲוֵי מְעִילָה, דְּלֵב בֵּית דִּין מַתְנֶה עֲלֵיהֶן:

פרק א - משנה ג

(כא) דְּאִי בְּבֵית הַשְּׁלָחִין הָתְנַן בְּרֵישׁ פִּרְקִין דְּמַשְׁקִין, וְהַיְינוּ כָּל הַשָּׂדֶה. אֲבָל תֵּימַהּ מַה דּוֹחְקֵיהּ לְאוֹקְמֵיהּ בִּשְׂדֵה הַבַּעַל, לוֹקְמֵיהּ בִּשְׁלָחִין, וְאִלּוּ בֵּית הַבַּעַל מַאן דְּכַר שְׁמֵיהּ כוּ', וְאִי קַשְׁיָא מַשְׁקִין דִּלְעֵיל הֲרֵי כָּתַב בְּעַצְמוֹ לְעֵיל דְּשֶׁל אִילָן לֹא שָׁרֵי בַּמּוֹעֵד דְּלֹא פָּסְדֵי, וּמַתְנִיתִין הָכִי דַּיְיקָא דִּמְשִׁיכָה מֵאִילָן לְאִילָן הוּא דְּמֻתָּר הָא הַשְׁקָאָה לֹא. וְכֵן דִּבְרֵי הָרַ"ן לְחַלֵּק בְּכָךְ. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (כב) סְבָרָתוֹ בָּזֶה כְּפֵרוּשׁ רַשִׁ"י דִּמְפָרֵשׁ זְרָעִים שֶׁלֹּא שָׁתוּ לִפְנֵי הַמּוֹעֵד שֶׁלֹּא הָיוּ מְלֻמָּדִים לִשְׁתּוֹת תָּמִיד לִפְנֵי הַמּוֹעֵד לֹא יַשְׁקֶה בַּמּוֹעֵד דְּהוֹאִיל וְלֹא מַשְׁקֶה לְהוּ תָּדִיר לִפְנֵי הַמּוֹעֵד לֹא הֲוֵי לְהוּ פְּסִידָא אִי לֹא מַשְׁקֶה לְהוּ בַּמּוֹעֵד. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

פרק א - משנה ד

(כג) וְהַגֻּמָּא אֵינוֹ אֶלָּא כְּדֵי שֶׁיִּתְקַבְּצוּ שָׁמָּה אִישׁוּת הַרְבֵּה, וְאֶפְשָׁר שֶׁנּוֹתֵן שָׁם אֳכָלִין כְּדֵי שֶׁיָּבֹאוּ וְיֵאָחְזוּ בַּמְּצוּדָה. הָרַ"ן: (כד) וּבַשְּׁבִיעִית. לְשׁוֹן רַשִׁ"י אַף עַל גַּב דִּמְתַקֵּן אֶת הַשָּׂדֶה. וּלְשׁוֹן הָרַ"ן דְּלֹא גָּזְרִינַן דִּלְמָא מֵחֲזֵי כַּחֲרִישָׁה כְּשֶׁאָדָם חוֹפֵר וְסוֹתֵם אֶת חֹרֵיהֶם: (כה) וּמְקָרִין. נִרְאֶה לִי שֶׁהוּא נִגְזַר מִשֵּׁם קִיר. וְלֹא תָּנֵי מְקִירִין, שֶׁלֹּא יִתְעָרֵב בְּמִלַּת מֵקִיר שֶׁעִנְיָנוֹ קְרִירוּת:

פרק א - משנה ה

(כו) בַּתְּחִלָּה. אֲפִלּוּ בַּתְּחִלָּה שֶׁעֲדַיִן לֹא נִזְקָק לְהֶסְגֵּר, דְּמִכָּל מָקוֹם שִׂמְחָה הִיא לוֹ כְּשֶׁאוֹמֵר לוֹ אֵינְךָ צָרִיךְ לַהֶסְגֵּר. וּמִלְּשׁוֹן רַשִׁ"י וְהָרַ"ב נִרְאֶה דְּלֹא גָּרְסִינַן בַּתְּחִלָּה. וְכֵן מַשְׁמָע בַּגְּמָרָא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (כז) דְּאַף הַמֻּסְגָּר מִשְׁתַּלֵּחַ, וְהָא וַדַּאי דְּכֵיוָן דְּמִטַּמֵּא נַמִּי דְּגַם הוּא צַעַר לוֹ, מִכָּל מָקוֹם צַוְתָּא דְּאִינְשֵׁי עִקָּר צַעֲרָא הוּא: (כח) לֹא לְהָקֵל כוּ'. סְבִירָא לְהוּ נַמִּי דִּבְכֹהֵן תַּלְיָא מִלְּתָא, אֶלָּא מִדְּרָשָׁא דִּלְטַהֲרוֹ אוֹ לְטַמְּאוֹ נָפֵיק לְהוּ דְּמִכֵּיוָן שֶׁנִּזְקַק לִרְאוֹתוֹ אֵינוֹ רַשַּׁאי שֶׁיִּשְׁתֹּק: (כט) וְעוֹד. כְּלוֹמַר וְעוֹד מֵקֵל: (ל) וּרְצוֹנוֹ לוֹמַר בֵּין לְעַרְעֵר בֵּין לִסְפֹּד. וְלָשׁוֹן רַשִׁ"י אַתַּרְוַיְיהוּ קָאֵי. וְיוֹתֵר נִרְאֶה לִי שֶׁתֵּבַת תָּמִיד צָרִיךְ לוֹמַר אִית:

(כא) הַקּוֹבֵר. לְהָכִי נָקַט קוֹבֵר, דַּאֲבֵלוּת מַתְחֶלֶת בִּשְׁעַת סְתִימַת הַגּוֹלֵל כִּדְאִיתָא בַּגְּמָרָא. תּוֹסָפוֹת: (כב) שְׁלֹשָׁה יָמִים. כָּךְ הוּא עִקַּר אֲבֵלוּת לְכַמָּה מִלֵּי, דְּאָמְרִינַן בַּגְּמָרָא שְׁלֹשָׁה יָמִים לִבְכִי. תּוֹסָפוֹת: (כג) שִׁבְעָה. אַסְמָכוּהוּ אַקְּרָא דִּכְתִיב [עָמוֹס ח] וְהָפַכְתִּי חַגֵּיכֶם לְאֵבֶל, מַה חַג שִׁבְעָה אַף אֲבֵלוּת שִׁבְעָה. גְּמָרָא: (כד) וְטַעֲמָא דְּסַגִּי בְּחַד יוֹמָא גַּבֵּי שְׁלֹשִׁים, וְגַבֵּי שִׁבְעָה בָּעִינַן שְׁלֹשָׁה, מִשּׁוּם דְּדִין כָּל שְׁלֹשִׁים שָׁוֶה, מַה שֶּׁאֵין כֵּן בְּשִׁבְעָה. הָרַ"ן: (כה) גְּזֵרַת שְׁלֹשִׁים. יָלֵיף פֶּרַע פֶּרַע מִנָּזִיר. כְּתִיב הָכָא [וַיִּקְרָא י] רָאשֵׁיכֶם אַל תִּפְרָעוּ, דִּשְׁמַע מִינַהּ שֶׁשְּׁאָר אָבֵל חַיָּב לִפְרֹעַ, וּכְתִיב הָתָם גָּדֵל פֶּרַע שְׂעַר רֹאשׁוֹ, מַה לְּהַלָּן שְׁלֹשִׁים אַף כָּאן שְׁלֹשִׁים. גְּמָרָא. וְהָרַמְבַּ"ם כָּתַב מִדִּכְתִיב [דְּבָרִים כא] וּבָכְתָה אֶת אָבִיהָ וְאֶת אִמָּהּ יֶרַח יָמִים, מִכְּלָל שֶׁהָאָבֵל מִצְטַעֵר כָּל שְׁלֹשִׁים: (כו) מִפְּנֵי כוּ'. כְּלוֹמַר וְאִם תֹּאמַר מַה טַּעַם לֹא אָמְרוּ חֲכָמִים הַקּוֹבֵר אֶת מֵתוֹ קֹדֶם שַׁבָּת בָּטְלָה מִמֶּנּוּ גְּזֵרַת שִׁבְעָה, מִפְּנֵי שֶׁאָמְרוּ שַׁבָּת עוֹלָה מִשּׁוּם דְּלֹא כְּתִיב בֵּיהּ שִׂמְחָה אֶלָּא עֹנֶג וּמִקְצָת אֲבֵלוּת נוֹהֵג בּוֹ וְכוּ'. הָרַ"ן. וְהַטּוּר כָּתַב דְּאִי אָמַרְתְּ דְּשַׁבָּת מַפְסִיק לֹא יִהְיֶה לְעוֹלָם שִׁבְעַת יָמִים. וְכָתַב בֵּית יוֹסֵף שֶׁהוּא מֵהַיְרוּשַׁלְמִי: (כז) וְזֶה תָּמוּהַּ, דְּהָא קַיְמָא לָן אַף בָּרֶגֶל דְּבָרִים שֶׁבְּצִנְעָה נוֹהֵג כְּמוֹ בְּשַׁבָּת. וְהַטַּעַם מִשּׁוּם דְּשַׁבָּת לֹא כְּתִיב בֵּיהּ שִׂמְחָה, כָּךְ נִרְאֶה לִי. וְכֵן כָּתַב בְּסֵפֶר מִשְׁנַת חֲכָמִים:

פרק ג - משנה ו

(כח) עֲצֶרֶת. וְכָל שֶׁכֵּן רֹאשׁ הַשָּׁנָה וְיוֹם הַכִּפּוּרִים. תּוֹסָפוֹת. וְכָתַב הָרַ"ן דְּטַעְמָא דְּרַבִּי אֱלִיעֶזֶר דִּסְבִירָא לֵיהּ דְּטַעְמָא [דִּרְגָלִים] מַפְסִיקִין, שֶׁאִם אַתָּה אוֹמֵר יַעֲלוּ לְמִנְיַן שִׁבְעָה הֲרֵי הִפְסִיקוּ הַכֹּל, וְכֵן עֲצֶרֶת בִּזְמַן הַמִּקְדָּשׁ, וְכֵיוָן שֶׁאֵין עוֹלִין אָמְרוּ מַפְסִיקִין, וְעָשׂוּ דִּין שְׁלֹשִׁים כְּדִין שִׁבְעָה, אֲבָל מִשֶּׁחָרַב בֵּית הַמִּקְדָּשׁ כוּ': (כט) כָּרְגָלִים. מִפְּנֵי שֶׁהֻקְשׁוּ כָּל הַמּוֹעֲדוֹת זֶה לָזֶה, דִּכְתִיב [וְיִּקְרָא כג] אֵלֶּה מוֹעֲדֵי ה'. הָרַמְבַּ"ן: (ל) כְּאַבָּא שָׁאוּל דְּפָלֵיג אַמַּתְנִיתִין דִּלְעֵיל דְּתָנֵי שְׁמֹנָה יָמִים. וְטַעֲמוֹ דְּסוֹבֵר מִקְצַת הַיּוֹם כְּכֻלּוֹ וּשְׁבִיעִי עוֹלֶה לְכָאן וּלְכָאן: (לא) וְכֵן יוֹם אֶחָד לִפְנֵי שָׁבוּעוֹת וְלִפְנֵי רֹאשׁ הַשָּׁנָה וְלִפְנֵי יוֹם הַכִּפּוּרִים: (לב) בַּגְּמָרָא דַּף ט"ו יָלֵיף לְהוּ מִקְּרָאֵי. וּנְטִילַת צִפָּרְנַיִם בִּכְלָל גִּלּוּחַ הוּא. נִרְאֶה לִי: (לג) פֵּרֵשׁ בָּעָרוּךְ שֶׁהוּא כְּמוֹ מַכְלִיב דְּמִשְׁנָה ח' פֶּרֶק קַמָּא. וְנִרְאֶה לִי שֶׁהוּא מִלְּשׁוֹן שָׁלָל וּבִיזָה, שֶׁכֵּן הַשָּׁלָל אַחַת הֵנָּה וְאַחַת הֵנָּה וְאֵינוֹ מֻנָּח וְצָבוּר:

פרק ג - משנה ז

(לד) אֵין כוּ'. דִּין הַקְּרִיעָה יָלֵיף בַּגְּמָרָא שָׁם. וְדִמְעֻמָּד בַּגְּמָרָא דַּף כ"א, וּדְבַעֲיָא טֶפַח דַּף כ"ב. וּפֵרוּשׁ מְאַחֶה תּוֹפֵר מַמָּשׁ, שֶׁכֵּן וְעֵת לִתְפֹּר מְתַרְגֵּם וְעִדַּן בְּחִיר לַאֲחָאָה: (לה) שֶׁהַכֹּהֵן מִטַּמֵּא לָהֶם: (לו) גְּמָרָא. וַאֲפִלּוּ בְּחֹל קָאָמַר, וְאַשְׁמְעִינַן הָא בְּחֹל הַמּוֹעֵד, דְּמַהוּ דְּתֵימָא כָּל הֵיכָא דְּלֹא יַרְאֶה דֶּרֶךְ אֲבֵלוּת אֶלָּא דּוֹרוֹן וַחֲשִׁיבוּת עָדִיף טְפֵי, קָא מַשְׁמַע לָן. הָרַ"ן:

פרק ג - משנה ח

(לז) שֶׁלֹּא כוּ'. שֶׁהַמּוֹעֵד אָסוּר בְּהֶסְפֵּד וּבְתַעֲנִית. הָרַמְבַּ"ם: (לח) מִפְּנֵי הַכָּבוֹד. שֶׁשּׁוֹפַעַת זִיבָה. רַשִׁ"י:

פרק ג - משנה ט

(לט) נִקְבַּר הַמֵּת. אֲפִלּוּ בְּרֹאשׁ חֹדֶשׁ חֲנֻכָּה וּפוּרִים, וְכָל שֶׁכֵּן בְּחֹל הַמּוֹעֵד דְּחָמִיר טְפֵי:
