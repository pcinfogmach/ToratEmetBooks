פרק א - משנה א

מַשְׁקִין בֵּית הַשְּׁלָחִין. אֶרֶץ עֲיֵפָה וּצְמֵאָה לְמַיִם. תַּרְגּוּם עָיֵף וְיָגֵעַ, מְשַׁלְהֵי וְלָאֵי (א): מַשְׁקִין בַּמּוֹעֵד. דְּדָבָר הָאָבֵד מֻתָּר בַּמּוֹעֵד (ג), וּשְׂדֵה בֵּית הַשְּׁלָחִין מִשָּׁעָה שֶׁמַּתְחִיל לְהַשְׁקוֹתוֹ אִם אֵינוֹ מַשְׁקֶה אוֹתוֹ תָּדִיר מִיָּד נֶאֱבָד, כְּדִתְנַן בְּסוֹף מַתְנִיתִין, זְרָעִים שֶׁלֹּא שָׁתוּ לִפְנֵי הַמּוֹעֵד לֹא יַשְׁקֵם בַּמּוֹעֵד דְּלֹא פָּסְדֵי, אֲבָל שָׁתוּ לִפְנֵי הַמּוֹעֵד יַשְׁקֵם בַּמּוֹעֵד. וְדַוְקָא שְׂדֵה לָבָן שֶׁל תְּבוּאָה שֶׁהוּא שְׁלָחִין מַשְׁקִין בַּמּוֹעֵד, אֲבָל שְׂדֵה שְׁלָחִין שֶׁל אִילָן לֹא שָׁרֵי בַּמּוֹעֵד, דְּלֹא פָּסְדֵי. וּבַשְּׁבִיעִית שָׁרֵי לְהַשְׁקוֹת בֵּין שְׂדֵה בֵּית הַשְּׁלָחִין בֵּין שְׂדֵה בֵּית הַבַּעַל. וּבֵית הַשְּׁלָחִין, מִשּׁוּם מוֹעֵד בִּלְבָד נָקַט לַהּ (ד): שֶׁיָּצָא בַתְּחִלָּה. שֶׁיּוֹצֵא עַכְשָׁיו מִתְּחִלָּה. וְאֵין חוֹשְׁשִׁין הוֹאִיל וְחָדָשׁ הוּא שֶׁמָּא יִפְּלוּ גְּדוֹתָיו וְאָתֵי לְתַקֵּן בַּמּוֹעֵד וְאִיכָּא טִרְחָא (ה): שֶׁלֹּא יָצָא בַתְּחִלָּה. שֶׁהוּא יָשָׁן: אֲבָל לֹא מִמֵּי גְּשָׁמִים. גְּזֵרָה מִשּׁוּם מֵי קִילוֹן. קִילוֹן הֵם מַיִם שֶׁל בּוֹר עָמֹק וּבוֹ מֵי גְּשָׁמִים מְכֻנָּסִים, דְּבֵיהּ אִיכָּא טִרְחָא יְתֵרָה לִדְלוֹת (ח): עוּגִיּוֹת. חֲרִיצִים (יא) שֶׁעוֹשִׂים בְּעִקְּרֵי הַזֵּיתִים וּבְעִקְּרֵי הַגְּפָנִים (יב) כְּדֵי שֶׁיִּתְמַלְּאוּ מַיִם:

פרק א - משנה ב

הָאַמָּה. חֲרִיצִים שֶׁעוֹשִׂים בַּקַּרְקַע שֶׁבָּהֶן הַמַּיִם הוֹלְכִים סְבִיב הַשָּׂדֶה וּמִשָּׂדֶה לְשָׂדֶה, קְרוּיִין אַמָּה, לְפִי שֶׁהֵן רֹחַב אַמָּה בְּרוּם אַמָּה: בַּתְּחִלָּה. אִם לֹא הָיְתָה מֵעוֹלָם אֵין עוֹשִׂין אוֹתָהּ בַּתְּחִלָּה בַּמּוֹעֵד, מִשּׁוּם טִרְחָא: וּבַשְּׁבִיעִית. מִפְּנֵי שֶׁנִּרְאֶה כְּחוֹפֵר בַּקַּרְקַע בַּשְּׁבִיעִית: וּמְתַקְּנִים אֶת הַמְּקֻלְקֶלֶת. אִם נָפַל לְתוֹכָהּ עָפָר וְאֵינָהּ מְקַלַּחַת, מְתַקְּנִים אוֹתָהּ (טו), אֲבָל אֵין עוֹשִׂים אוֹתָהּ בַּתְּחִלָּה בַּמּוֹעֵד. וַהֲלָכָה כַּחֲכָמִים: וּמְתַקְּנִים קִלְקוּלֵי הַמַּיִם. מַיִם הַמְּכֻנָּסִים בַּגֻּמּוֹת לִשְׁתּוֹת מֵהֶן: וְחוֹטְטִין. מְנַקִּין וּמְנַקְּרִין וּמוֹצִיאִין מִתּוֹכָן צְרוֹרוֹת וְקִסְמִין וְעַפְרוּרִית שֶׁנָּפַל בָּהֶן: וּמְצַיְּנִין. עוֹשִׂין צִיּוּנִין עַל הַקְּבָרוֹת. שֶׁהָיוּ מְמַחִין סִיד וְשׁוֹפְכִין עַל הַקֶּבֶר, וְהוּא סִימָן לְהוֹלְכֵי דְּרָכִים שֶׁלֹּא יַעַבְרוּ בִּמְקוֹם טֻמְאָה: וְיוֹצְאִין. שְׁלוּחֵי בֵּית דִּין בַּמּוֹעֵד לְעַיֵּן הַשָּׂדוֹת שֶׁנִּזְרְעוּ כִּלְאַיִם, לַעֲקֹר אוֹתָם. לְפִי שֶׁנּוֹטְלִין שָׂכָר עַל כָּךְ מִתְּרוּמַת הַלִּשְׁכָּה (כ), וּבִימֵי הַמּוֹעֵד נִשְׂכָּרִים בְּזוֹל, מִפְּנֵי שֶׁהֵן בְּטֵלִים:

פרק א - משנה ג

מוֹשְׁכִים אֶת הַמַּיִם. שֶׁתַּחַת אִילָן זֶה, לְאִילָן אַחֵר, דְּלֵיכָּא טִרְחָא. אֲבָל לֹא יַשְׁקֶה כָּל הַשָּׂדֶה. וּבִשְׂדֵה בֵּית הַבַּעַל אָיְרֵי (כא), דְּלֹא פָּסְדֵי (כב), שֶׁמֵּי גְּשָׁמִים מַסְפִּיקִין לוֹ, אֶלָּא שֶׁכְּשֶׁמַּשְׁקִין אוֹתוֹ הֲוֵי הַרְוָחָה טְפֵי: שֶׁלֹּא שָׁתוּ. דְּלֹא פָּסְדֵי: בָּזֶה וּבָזֶה. אֲפִלּוּ לֹא שָׁתוּ, וַאֲפִלּוּ שְׂדֵה הַבַּעַל, דְּמִשּׁוּם הַרְוָחָה שָׁרוּ חֲכָמִים. וַהֲלָכָה כְּרַבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב, דִּסְתָם מַתְנִיתִין מַשְׁקִין בֵּית הַשְּׁלָחִין דִּלְעֵיל כְּוָתֵיהּ. וּמִיהוּ שָׂדֶה לַחָה שֶׁהַקַּרְקַע שֶׁלָּהּ כְּמוֹ טִיט מֻתָּר לְהַשְׁקוֹתָהּ בַּמּוֹעֵד אֲפִלּוּ לְרַבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב, דְּכִזְרָעִים שֶׁשָּׁתוּ לִפְנֵי הַמּוֹעֵד דָּמוּ:

פרק א - משנה ד

הָאִישׁוּת. תִּנְשֶׁמֶת. וְהִיא בְּרִיָּה שֶׁאֵין לָהּ עֵינַיִם, וּמַפְסֶדֶת אֶת הַשָּׂדוֹת: וּבִשְׂדֵה הַלָּבָן. שֶׁחוֹרְשִׁין וְזוֹרְעִין אוֹתוֹ לִתְבוּאָה וְאֵין בּוֹ אִילָנוֹת: בִּשְׂדֵה הָאִילָן כְּדַרְכּוֹ. דְּמַפְסְדֵי טְפֵי, וּלְהֶפְסֵד מְרֻבֶּה חָשְׁשׁוּ. לְפִיכָךְ חוֹפֵר גֻּמָּא כְּדַרְכּוֹ וְתוֹלֶה בָּהּ מְצוּדָה (כג): שֶׁלֹּא כְדַרְכּוֹ. נוֹעֵץ שִׁפּוּד בַּקַּרְקַע בְּחֹזֶק וּמְנַעֲנֵעַ אֵילָךְ וָאֵילָךְ וּמוֹצִיא, וְנִמְצֵאת גֻּמָּא נַעֲשֵׂית מֵאֵלֶיהָ. אֲבָל לֹא עַל יְדֵי חֲפִירָה. וַהֲלָכָה כַּחֲכָמִים. וּבִשְׂדֵה לָבָן הַסְּמוּכָה לִשְׂדֵה אִילָן, חוֹפֵר כְּדַרְכּוֹ, שֶׁמָּא יֵצְאוּ מִשְּׂדֵה לָבָן וְיַחְרִיבוּ הָאִילָנוֹת: וּמְקָרִין אֶת הַפִּרְצָה. מְסַדְּרִין אֶת הָאֲבָנִים זוֹ עַל זוֹ כְּדֶרֶךְ בִּנְיָן, וְאֵינוֹ טָח בְּטִיט. וּבְפִרְצָה שֶׁל גִּנָּה מַיְרֵי, אֲבָל כֹּתֶל חָצֵר שֶׁנָּפַל, בּוֹנֶה כְּדַרְכּוֹ, מִפְּנֵי הַגַּנָּבִים:

פרק א - משנה ה

רוֹאִין אֶת הַנְּגָעִים בַּמּוֹעֵד לְהָקֵל. שֶׁאִם טָהוֹר הוּא אוֹמֵר לוֹ הַכֹּהֵן טָהוֹר אַתָּה, שֶׁשִּׂמְחָה הִיא לוֹ: אֲבָל לֹא לְהַחְמִיר. שֶׁאִם טָמֵא הוּא, יִשְׁתֹּק, וְלֹא יְטַמְּאֶנּוּ וְיַצְרִיכֶנּוּ לָצֵאת חוּץ לַמַּחֲנֶה (כז): לֹא לְהָקֵל וְלֹא לְהַחְמִיר. מִתּוֹךְ שֶׁנִּזְקַק לִרְאוֹתוֹ אִם הוּא טָהוֹר כְּדֵי לְהָקֵל עָלָיו, הוּא נִזְקָק לְהַחְמִיר, שֶׁאִם הוּא טָמֵא צָרִיךְ לוֹמַר לוֹ טָמֵא אַתָּה, כְּדִכְתִיב (וַיִּקְרָא יג) לְטַהֲרוֹ אוֹ לְטַמְּאוֹ, שֶׁאֵין כֹּהֵן רַשַּׁאי לִשְׁתֹּק, וּמוּטָב לוֹ שֶׁלֹּא יִרְאֵהוּ הַכֹּהֵן כְּלָל: מְלַקֵּט אָדָם עַצְמוֹת אָבִיו וְאִמּוֹ. בַּמּוֹעֵד, כְּדֵי לְקָבְרָם בְּמָקוֹם הָגוּן: שֶׁשִּׂמְחָה הִיא לוֹ. כְּשֶׁרוֹאֶה אוֹתָם שֶׁקּוֹבְרָן בְּקִבְרֵי אֲבוֹתָיו. וְאֵין הֲלָכָה כְּרַבִּי מֵאִיר בִּשְׁתֵּיהֶן: לֹא יְעַרְעֵר עַל מֵתוֹ. לֹא יָבִיא סַפְדָּן לְעַרְעֵר עַל מֵתוֹ שֶׁמֵּת זֶה יָמִים רַבִּים, לַחְזֹר עַל קְרוֹבֵי הַמֵּת כְּדֶרֶךְ שֶׁחוֹזֵר הַסַּפְדָּן שֶׁצּוֹעֵק וְאוֹמֵר בֹּאוּ בְּכוּ עִמִּי כָּל מָרֵי לֵב, וּמִי שֶׁלִּבּוֹ דָּוֶה עָלָיו הוֹלֵךְ וּמַסְפִּיד עַל קְרוֹבוֹ: לֹא יַסְפִּידֶנּוּ. לֹא יִשְׂכֹּר סַפְדָּן עַל מֵתוֹ שֶׁמֵּת לוֹ בְּאוֹתָהּ שָׁעָה: קֹדֶם לָרֶגֶל שְׁלֹשִׁים יוֹם. תָּמִיד (ל). דְּאָמְרֵי טַעְמָא בַּגְּמָרָא שֶׁדֶּרֶךְ בְּנֵי אָדָם לִכְנֹס מָעוֹת לְצֹרֶךְ הָרֶגֶל קֹדֶם לָרֶגֶל שְׁלֹשִׁים יוֹם מִכִּי שָׁמַע שֶׁדּוֹרְשִׁים בְּהִלְכוֹת הָרֶגֶל, וְשֶׁמָּא יִתֵּן הַמָּעוֹת שֶׁכִּנֵּס לְצֹרֶךְ הָרֶגֶל לַסַּפְדָּן וְיִמָּנַע מִשִּׂמְחַת הָרֶגֶל. וְאִית דְּאָמְרֵי שֶׁאֵין הַמֵּת מִשְׁתַּכֵּחַ מִן הַלֵּב אֶלָּא לְאַחַר שְׁלֹשִׁים יוֹם אַחַר הֶסְפֵּדוֹ:

(ח): וּפְרוֹזְבּוּלִין. שֶׁלֹּא תִּשְׁמֹט שְׁבִיעִית וְיַפְסִיד אֶת מְעוֹתָיו (ט). פְּרוֹזְבּוּל, פְּרוֹז בּוּלֵי וּפְרוֹז בּוּטֵי, כְּלוֹמַר תַּקָּנָה לָעֲשִׁירִים שֶׁלֹּא יַעַבְרוּ עַל מַה שֶּׁכָּתוּב בַּתּוֹרָה (דְּבָרִים טו) הִשָּׁמֵר לְךָ פֶּן יִהְיֶה דָּבָר עִם לְבָבְךָ בְּלִיַּעַל, וְתַקָּנָה לָעֲנִיִּים שֶׁיִּמְצְאוּ לִלְווֹת. בּוּלֵי עֲשִׁירִים. בּוּטֵי עֲנִיִּים. וְהִלֵּל שְׁבִיעִית מְשַׁמֶּטֶת הַלְוָאָתוֹ, דַּהֲוֵי כְּאִלּוּ גָּבוּ בֵּית דִּין אֶת חוֹבוֹ, וְשׁוּב לֹא קָרִינַן בֵּיהּ לֹא יִגֹּשׂ אֶת רֵעֵהוּ: אִגְּרוֹת שׁוּם. שֶׁשָּׁמוּ בֵּית דִּין נִכְסֵי לֹוֶה וְנָתְנוּ לַמַּלְוֶה (י): אִגְּרוֹת מָזוֹן. שֶׁמָּכְרוּ בֵּית דִּין אֶת הַקַּרְקַע לִמְזוֹן הָאִשָּׁה וְהַבָּנוֹת, וְכוֹתְבִין מַעֲשֵׂה בֵּית דִּין עַל זֶה. אִי נַמִּי מִי שֶׁקִּבֵּל עָלָיו לָזוּן אֶת בַּת אִשְׁתּוֹ: מֵאוּנִים. קְטַנָּה שֶׁהִשִּׂיאוּהָ אִמָּהּ וְאַחֶיהָ, יְכוֹלָה לְמָאֵן בְּבַעֲלָהּ וְלוֹמַר אִי אֶפְשִׁי בִּפְלוֹנִי בַּעֲלִי, וְיֹצְאָה בְּלֹא גֵּט. וּשְׁטָר שֶׁכּוֹתְבִין לְעֵדוּת דָּבָר זֶה, נִקְרָא שְׁטַר מֵאוּנִים (יא): שִׁטְרֵי בֵרוּרִין. שֶׁבָּרְרוּ לָהֶם דַּיָּנִין, זֶה בּוֹרֵר לוֹ אֶחָד וְזֶה בּוֹרֵר לוֹ אֶחָד, וְכוֹתְבִין שְׁטָר, שֶׁלֹּא יַחְזְרוּ בָּהֶן הַבַּעֲלֵי דִּינִים (יב): וּגְזֵרוֹת בֵּית דִּין. פְּסַק דִּין שֶׁפָּסְקוּ הַדַּיָּנִים (יג): אִגְּרוֹת שֶׁל רָשׁוּת. צַוֵּי הַשִּׁלְטוֹן (יד). כְּמוֹ וְאַל תִּתְוַדַּע לָרָשׁוּת, הֱוּוּ זְהִירִים בָּרָשׁוּת. וְיֵשׁ מְפָרְשִׁין אִגְּרוֹת שֶׁל שְׁאִילַת שָׁלוֹם שֶׁכּוֹתְבִים בֵּין אָדָם לַחֲבֵרוֹ (טו):

פרק ג - משנה ד

אֵין כּוֹתְבִין שִׁטְרֵי חוֹב בַּמּוֹעֵד. הוֹאִיל וְיָכֹל לְכָתְבוֹ לְאַחַר הַמּוֹעֵד (טז): וְאִם אֵינוֹ מַאֲמִינוֹ. וְאֵינוֹ רוֹצֶה לְהַלְווֹתוֹ בְּלֹא שְׁטָר, וְלֹוֶה צָרִיךְ לְמָעוֹת, כּוֹתְבִין: אוֹ שֶׁאֵין לוֹ. לַסּוֹפֵר מַה יֹּאכַל, הֲרֵי זֶה יִכְתֹּב וְיִטֹּל שְׂכָרוֹ, דִּשְׂכַר פְּעֻלָּה לְמִי שֶׁאֵין לוֹ מַה יֹּאכַל שָׁרֵי: וַאֲפִלּוּ בְּסֵפֶר הָעֲזָרָה. סֵפֶר תּוֹרָה שֶׁכֹּהֵן גָּדוֹל קוֹרֵא בּוֹ בְּיוֹם הַכִּפּוּרִים, וְאַף עַל גַּב דְּצָרִיךְ לָרַבִּים (יח): לְעַצְמוֹ. לְקַיֵּם הַמִּצְוָה. אֲבָל לֹא לִמְכֹּר אוֹ לְהַשְׂכִּיר: וְטוֹוֶה עַל יְרֵכוֹ תְּכֵלֶת. שֶׁנּוֹתֵן הַחוּט עַל יְרֵכוֹ וּמְשַׁפְשֵׁף בְּיָדוֹ וְהוּא נִטְוֶה מֵאֵלָיו. אֲבָל לֹא בְּיָדוֹ בֵּין אֶצְבְּעוֹתָיו, וְלֹא בַּפֶּלֶךְ כְּדֶרֶךְ שֶׁהוּא עוֹשֶׂה בְּחֹל. וַהֲלָכָה, טוֹוֶה אָדָם תְּכֵלֶת לְבִגְדּוֹ בֵּין בְּפֶלֶךְ בֵּין בְּאֶבֶן (כ):

פרק ג - משנה ה

הַקּוֹבֵר אֶת מֵתוֹ. שֶׁמֵּת לוֹ מֵת קֹדֶם הָרֶגֶל, וְנָהַג אֲבֵלוּת קֹדֶם הָרֶגֶל שְׁלֹשָׁה יָמִים: בָּטְלָה מִמֶּנּוּ גְּזֵרַת שִׁבְעָה. אֲבָל לֹא בָּטְלָה הֵימֶנּוּ גְּזֵרַת שְׁלֹשִׁים, דְּהַיְנוּ אִסּוּר תִּסְפֹּרֶת, וּמוֹנֶה אַחַר הָרֶגֶל שְׁלֹשִׁים יוֹם עִם הַשְּׁלֹשָׁה שֶׁעָבְרוּ כְּבָר מִן הַשְּׁלֹשִׁים. וּפְסַק הֲלָכָה, שֶׁהַקּוֹבֵר אֶת מֵתוֹ אֲפִלּוּ שָׁעָה אַחַת קֹדֶם הָרֶגֶל בָּטְלָה מִמֶּנּוּ גְּזֵרַת שִׁבְעָה: שְׁמֹנָה. יָמִים קֹדֶם הָרֶגֶל, הוֹאִיל וְנִכְנַס אֶחָד מִימֵי הַגִּלּוּחַ קֹדֶם לָרֶגֶל, אָתֵי רֶגֶל וּמְבַטֵּל אֶת הַשְּׁאָר לְגַמְרֵי (כד): שַׁבָּת עוֹלָה. שַׁבָּת שֶׁבִּימֵי אֶבְלוֹ עוֹלָה לְמִנְיַן שִׁבְעָה: וְאֵינָהּ מַפְסֶקֶת. שֶׁאֵינָהּ מְבַטֶּלֶת אֶת הַשְּׁאָר, אֶלָּא יוֹשֵׁב וּמִתְאַבֵּל אַחַר שַׁבָּת: הָרְגָלִים מַפְסִיקִין וְאֵינָם עוֹלִים. אִם נָהַג אֲבֵלוּת קֹדֶם הָרֶגֶל, הָרֶגֶל מַפְסִיק וּמְבַטֵּל גְּזֵרַת שְׁלֹשִׁים. וְאִם לֹא נָהַג אֲבֵלוּת קֹדֶם הָרֶגֶל אֶלָּא הִתְחִיל אֲבֵלוּתוֹ בָּרֶגֶל, אֵין יְמֵי הָרֶגֶל עוֹלִים לְמִנְיַן שִׁבְעָה, אֲבָל עוֹלִין לְמִנְיַן שְׁלֹשִׁים. וְטַעְמָא דְּשַׁבָּת עוֹלָה לְחֶשְׁבּוֹן שִׁבְעַת יְמֵי אֲבֵלוּת, מִשּׁוּם דִּדְבָרִים שֶׁבְּצִנְעָה נוֹהֵג בָּהּ, כְּגוֹן תַּשְׁמִישׁ הַמִּטָּה, וּפְרִיעַת הָרֹאשׁ, שֶׁאָסוּר בְּגִלּוּי הָרֹאשׁ, וּרְחִיצָה בְּחַמִּין, שֶׁהָאָבֵל אָסוּר בָּהֵן בְּשַׁבָּת. אֲבָל הָרֶגֶל אֵין שׁוּם דִּין מִדִּינֵי אֲבֵלוּת נוֹהֵג בּוֹ (כז), לְפִיכָךְ מַפְסִיק לְגַמְרֵי. וְאִם לֹא חָל אֲבֵלוּת כְּלָל קֹדֶם הָרֶגֶל, כְּגוֹן שֶׁנִּקְבַּר הַמֵּת בְּתוֹךְ הָרֶגֶל, אֵינוֹ עוֹלֶה לְמִנְיַן שִׁבְעָה, וּמַתְחִיל לִמְנוֹת שִׁבְעָה אַחַר הָרֶגֶל:

הַהֲלָכָה, דַּעֲצֶרֶת בַּזְּמַן הַזֶּה כָּרְגָלִים, וְרֹאשׁ הַשָּׁנָה וְיוֹם הַכִּפּוּרִים כָּרְגָלִים. וּמִי שֶׁמֵּת לוֹ מֵת קֹדֶם אֶחָד מֵאֵלּוּ אֲפִלּוּ שָׁעָה אֶחָת, בָּטְלָה מִמֶּנּוּ גְּזֵרַת שִׁבְעָה. וְאִם מֵת שִׁבְעָה יָמִים קֹדֶם אֶחָד מִיָּמִים טוֹבִים אֵלּוּ (ל), בָּטְלָה מִמֶּנּוּ גְּזֵרַת שְׁלֹשִׁים. וּמִי שֶׁמֵּת לוֹ מֵת שָׁעָה אַחַת קֹדֶם חַג הַסֻּכּוֹת בָּטְלָה מִמֶּנּוּ
