(ג) קִדּוּשׁ כוּ'. מִן הַכִּיּוֹר. רַשִׁ"י. וּבַגְּמָרָא בִּשְׁלָמָא רַגְלַיִם מִשּׁוּם נִיצוֹצוֹת אֶלָּא יָדַיִם מַאי טַעֲמָא, זֹאת אוֹמֶרֶת מִצְוָה לְשַׁפְשֵׁף בְּיָדוֹ הַנִּיצוֹצוֹת הַנִּתָּזִין עַל רַגְלָיו שֶׁלֹּא יֵצֵא בָּהֶן חוּץ, דְּנִרְאֶה כִּכְרוּת שָׁפְכָה וְיוֹצִיא לַעַז עַל בָּנָיו שֶׁאֵינָן שֶׁלּוֹ:

(יא) בֵּין דָּם לָאֵבָרִים. מִשּׁוּם דְּבִקְטֹרֶת כְּתִיב בַּבֹּקֶר בַּבֹּקֶר, וּבְתָמִיד חַד בַּבֹּקֶר. גְּמָרָא: (יב) בֵּין אֵבָרִים לַנְּסָכִים. דִּכְתִיב (בְּמִדְבָּר כח) כְּמִנְחַת הַבֹּקֶר וּכְנִסְכּוֹ תַּעֲשֶׂה, מַה מִּנְחַת הַבֹּקֶר קְטֹרֶת קוֹדֶמֶת לִנְסָכִים אַף כָּאן כוּ', וּמִדְּלֹא כְּתִיב כְּאֵבְרֵי הַבֹּקֶר אֶלָּא כְּמִנְחַת, הִלְכָּךְ לַמִּנְחָה קָדִים וְלֹא לָאֵבָרִים. גְּמָרָא:

פרק ג - משנה ו

(יג) פָּשַׁט קִדֵּשׁ. דִּסְבִירָא לֵיהּ דִּתְרֵי קִדּוּשֵׁי שֶׁהִצְרִיךְ הַכָּתוּב בֵּין בֶּגֶד לְבֶגֶד תַּרְוַיְהוּ מִשּׁוּם לְבִישַׁת בְּגָדִים הַמִּתְחַלְּפִים תַּחַת הָרִאשׁוֹנִים הוּא. וּלְדִידֵיהּ בְּמִשְׁנָה ד' פָּשַׁט וְקִדֵּשׁ וְיָרַד וְטָבַל כוּ' וְלָבַשׁ וְקִדֵּשׁ אֶלָּא הַהִיא מַתְנִיתִין כְּרַבָּנָן דְּהָכָא. גְּמָרָא: (יד) לָבַשׁ וְקִדֵּשׁ. הַכֹּל מוֹדִים בְּקִדּוּשׁ שֵׁנִי שֶׁלּוֹבֵשׁ וְאַחַר כָּךְ מְקַדֵּשׁ. מַאי טַעֲמָא, דְּאָמַר קְרָא [שְׁמוֹת כה] אוֹ בְּגִשְׁתָּם אֶל הַמִּזְבֵּחַ, מִי שֶׁאֵינוֹ מְחֻסָּר אֶלָּא גִּישָׁה, יָצָא זֶה שֶׁמְּחֻסָּר לְבִישָׁה וְגִישָׁה. גְּמָרָא:

פרק ג - משנה ז

(טו) בַּשַּׁחַר כוּ'. דְּכֻלֵּי עָלְמָא מִיהַת דְּשַׁחַר עֲדִיפֵי, דְּאָמַר קְרָא אַרְבָּעָה זִמְנֵי בַּד בִּלְבִישַׁת בִּגְדֵי שַׁחַר. חַד לְגוּפֵיהּ, וְחַד לוֹמַר לְבַד שֶׁאֵין לוֹבֵשׁ עִמָּהֶן בִּגְדֵי זָהָב כְּמוֹ בִּשְׁאָר יְמוֹת הַשָּׁנָה, וְחַד לוֹמַר שֶׁהוּא מֻבְחָר מִבַּד שֶׁל שְׁאָר הַיָּמִים, וְחַד לוֹמַר שֶׁטּוֹב מִשֶּׁל עַרְבִית. גְּמָרָא:

פרק ג - משנה ח

(טז) פָּרוֹ. שֶׁנֶּאֱמַר בְּזֹאת יָבֹא אַהֲרֹן אַל הַקֹּדֶשׁ בְּפַר בֶּן בָּקָר לְחַטָּאת. וּמִשֶּׁלּוֹ הָיָה, כְּדִכְתִיב אֲשֶׁר לוֹ: (יז) וּמִתְוַדֶּה. דִּכְתִיב וְכִפֵּר בַּעֲדוֹ. וְאִי אֶפְשָׁר לוֹמַר כַּפָּרַת דָּמִים, שֶׁעֲדַיִן לֹא נִשְׁחַט הַפָּר, אֶלָּא כַּפָּרַת דְּבָרִים דּוּמְיָא דְּשָׂעִיר הַמִּשְׁתַּלֵּחַ דִּכְתִיב גַּבֵּיהּ לְכַפֵּר עָלָיו. גְּמָרָא: (יח) אָנָּא. כְּתִיב הָכָא כַּפָּרָה וּכְתִיב בְּחוֹרֵב [שְׁמוֹת לב] אוּלַי אֲכַפְּרָה. וְנֶאֱמַר שָׁם אָנָּא חָטָא הָעָם הַזֶּה. אַף כָּאן בְּאָנָּא. גְּמָרָא: (יט) הַשֵּׁם. דְּיָלְפִינַן מִכַּפָּרָה דְּעֶגְלָה עֲרוּפָה דִּכְתִיב בֵּיהּ [דְּבָרִים כא] כַּפֵּר לְעַמְּךָ יִשְׂרָאֵל אֲשֶׁר פָּדִיתָ ה', אַף כָּאן בַּה'. וּלְפִי שֶׁהוּא אָמַר הַשֵּׁם כִּכְתָבוֹ לְפִיכָךְ לֹא הִזְכִּירוֹ הַתַּנָּא אֶלָּא תָּנֵי הַשֵּׁם. תּוֹסְפוֹת יוֹם טוֹב: (כ) בָּרוּךְ כוּ'. תַּנְיָא רַבִּי אוֹמֵר כִּי שֵׁם ה' אֶקְרָא הָבוּ גֹּדֶל לֵאלֹהֵינוּ, אָמַר לְהוּ מֹשֶׁה לְיִשְׂרָאֵל, בְּשָׁעָה שֶׁאֲנִי מַזְכִּיר שְׁמוֹ שֶׁל הַקָּדוֹשׁ בָּרוּךְ הוּא, אַתֶּם הָבוּ גֹּדֶל. גְּמָרָא:

פרק ג - משנה ט

(כא) לִצְפוֹן. מַקְשִׁינַן בַּגְּמָרָא מִקָּדַאֲמַר לִצְפוֹן כוּ' דְּמַשְׁמַע שֶׁהֻצְרַךְ לְמָשְׁכוֹ מִכְּנֶגֶד הַמִּזְבֵּחַ וְלַצָּפוֹן, מִכְּלַל דְּמִזְבֵּחַ לָאו בַּצָּפוֹן קָאֵי, מַנִּי רַבִּי אֱלִיעֶזֶר בֶּן יַעֲקֹב הִיא דִּסְבִירָא לֵיהּ הָכִי. וְהָא מַתְנִיתִין דִּלְעֵיל דִּתְנַן בֵּין הָאוּלָם וְלַמִּזְבֵּחַ, מִדְּמַכְשַׁר בֵּין הָאוּלָם וְלַמִּזְבֵּחַ שְׁמַע מִינַּהּ דִּסְבִירָא לֵיהּ דְּמִקְצָת מִזְבֵּחַ בַּצָּפוֹן קָאֵי, וְרַבִּי אֶלְעָזָר בֶּן שִׁמְעוֹן הוּא. וּמְשַׁנִּינַן תָּנֵי כְּבֵין הָאוּלָם וְלַמִּזְבֵּחַ, כְּלוֹמַר סָמוּךְ לְבֵין הָאוּלָם וְלַמִּזְבֵּחַ אֵצֶל מִקְצוֹעַ צְפוֹנִית שֶׁל הַמִּזְבֵּחַ, וְלֹא כְּנֶגֶד הַמִּזְבֵּחַ וּלְבֵין הָאוּלָם, דְּלָאו בַּצָּפוֹן הוּא. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב:

פרק ג - משנה י

(א) לִקְרוֹת. פֵּרֵשׁ רַשִׁ"י דְּיָלְפִינַן מִמִּלּוּאִים. וְאָמְרִינַן בְּפֶרֶק קַמָּא דַּף ה' מִנַּיִן שֶׁאַף מִקְרָא פָּרָשָׁה שֶׁנִּצְטַוָּה בִּוְאַתָּה תְּצַוֶּה מְעַכֵּב, תַּלְמוּד לוֹמַר זֶה הַדָּבָר אֲשֶׁר צִוָּה ה' לַעֲשׂוֹת, בַּעֲשִׂיַּת הַמִּלּוּאִים בְּצַו אֶת אַהֲרֹן, אֲפִלּוּ דִּבּוּר מְעַכֵּב: (ב) הַכְּנֶסֶת. בֵּית הַכְּנֶסֶת הָיָה סָמוּךְ לָעֲזָרָה, בְּהַר הַבַּיִת. רַשִׁ"י: (ג) וְנִקְרָא חַזָּן שֶׁהוּא לְשׁוֹן חוֹזֶה שֶׁצָּרִיךְ לִרְאוֹת מַה הֵם צָרְכֵי צִבּוּר: (ד) דְּלֹא תֵּימָא שֶׁזֶּה שֶׁהַסְּגָן אֵינוֹ מְקַבְּלוֹ מֵהַחַזָּן לְפִי שֶׁהַחַזָּן חוֹלֵק כָּבוֹד לַסְּגָן שֶׁיְּקַבְּלוֹ מֵרֹאשׁ הַכְּנֶסֶת, וְקַיְמָא לָן דְּאֵין חוֹלְקִים כָּבוֹד לַתַּלְמִיד בִּמְקוֹם רַב אֶלָּא הֵיכָא דְּרַבֵּיהּ פָּלֵיג לֵיהּ יְקָרָא, וַאֲנַן סְתָמָא תְּנַן, דִּלְעוֹלָם הַמִּנְהָג כֵּן. תּוֹסָפוֹת: (ה) גְּמָרָא. וְנִרְאֶה טַעֲמָא, אִי מִשּׁוּם יְקָרָא דְּכֹהֵן גָּדוֹל, שֶׁלֹּא יִהְיֶה עוֹמֵד כָּל כָּךְ זְמַן רַב בְּלֹא עֲבוֹדָה מִתְּחִלַּת שִׁלּוּחַ הַשָּׂעִיר עַד הַגִּיעוֹ לִמְקוֹמוֹ, וְאִי נַמִּי מִשּׁוּם חוּלְשָׁא דְּכֹהֵן גָּדוֹל. תּוֹסְפוֹת יוֹם טוֹב: (ו) שֶׁבְּחֻמַּשׁ הַפְּקוּדִים. סֵפֶר וַיְדַבֵּר, שֶׁמַּתְחִיל בְּמִנְיָנָן שֶׁל יִשְׂרָאֵל. רַשִׁ"י: (ז) וְהָרַמְבַּ"ם כָּתַב דְּהָנֵי שְׁמֹנֶה בְּרָכוֹת כֻּלָּן לְאַחֲרֵיהֶם הֵן, שֶׁכָּתַב וּמוֹסִיף לְאַחֲרֶיהָ שֶׁבַע בְּרָכוֹת: (ח) וְעַל הָעֲבוֹדָה כוּ'. מְבָרֵךְ בִּשְׁבִיל הָעֲבוֹדָה שֶׁעָבַד. וְהַהוֹדָאָה סְמוּכָה לָעֲבוֹדָה, דִּכְתִיב (תְּהִלִּים נ) זוֹבֵחַ תּוֹדָה יְכַבְּדָנְנִי, אַחַר הָעֲבוֹדָה תּוֹדָה, וְהִיא הוֹדָאָה. רַשִׁ"י: (ט) כְּמָה דְאַת אָמַר [מְלָכִים א ח] אֶל הָרִנָּה וְאֶל הַתְּפִלָּה, וְתִרְגֵּם יוֹנָתָן צְלוֹתָא וּבָעוּתָא. וּבַגְּמָרָא דִּבְרָכוֹת (לא.) אֶל הָרִנָּה וְאֶל הַתְּפִלָּה, רִנָּה זוֹ תְּפִלָּה, פֵּרוּשׁ שֶׁבַח. וּבְיִרְמְיָה ז' מְתַרְגֵּם רִנָּה בָּעוּתָא. וּבְמִדְרָשׁ רַבָּה [רִנָּה] זוֹ קִלּוּסוֹ שֶׁל הַקָּדוֹשׁ בָּרוּךְ הוּא:

פרק ז - משנה ב

(י) כְּאֶחָת. וְאַף עַל גַּב דִּכְתִיב יוֹצִיא אֶל מִחוּץ לַמַּחֲנֶה וּשְׂרָפוֹ בָּתַר אֵילוֹ וְאֵיל הָעָם וְאֵמוּרֵי חַטָּאת, יָלֵיף בַּגְּמָרָא דִּכְתִיב נַמִּי וְהַמְשַׁלֵּחַ בָּתַר הָכִי, וּמָה הַמְשַׁלֵּחַ דְּמֵעִקָּרָא דְּאֵין עֲמִידָה שֶׁלּוֹ אֶלָּא עַד מַתַּן דָּמִים דַּחֲבֵרוֹ, כְּמוֹ שֶׁכָּתַב [הָרַ"ב] בְּרֵישׁ פֶּרֶק ו', הָכִי נַמִּי וְהַשּׂוֹרֵף יְכַבֵּס דְּמֵעִקָּרָא וּקְרָאֵי דְּלֹא כְּסִדְרָן כְּדִלְקַמָּן. תּוֹסְפוֹת יוֹם טוֹב:

פרק ז - משנה ג

(יא) קִדֵּשׁ כוּ'. דְּאִלּוּ קָרָא בְּאִצְטְלִית כְּבָר קִדֵּשׁ קֹדֶם שֶׁפָּשַׁט בִּגְדֵי בּוּץ, וְאֵינוֹ צָרִיךְ עַתָּה אֶלָּא טְבִילָה וְקִדּוּשׁ. הָרֹא"שׁ: (יב) דְּאָמְרִינַן בַּגְּמָרָא דְּמַתְנִיתִין לָאו מִתַּקַּנְתָּא הוּא, דְּהָא לֹא קָאָמַר פַּר וְשָׂעִיר וְאֵמוּרֵי חַטָּאת אֵימָתַי עֲבֵיד לְהוּ, אֶלָּא כִּדְקָתָנֵי בַּבְּרַיְתָא כְּמוֹ שֶׁכָּתַב הָרַ"ב. וּפֵרְשׁוּ הַתּוֹסָפוֹת דְּמִכָּל מָקוֹם לָאו מְשַׁבַּשְׁתָּא הוּא, וְשִׁיֵּר. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (יג) עִם תָּמִיד כוּ'. קָאֵי אַשִּׁבְעָה כְּבָשִׂים, וּפַר הָעוֹלָה נַמִּי עִם שֶׁל שַׁחַר, וְכִדְּכָתַב הָרַ"ב מִטַּעַם הַגְּמָרָא: (יד) לְפִי שֶׁהוּא מִן הַמּוּסָפִין הָיָה רָאוּי לְהַקְדִּימוֹ, אֶלָּא דִּקְרָא קָאָמַר מִלְּבַד כוּ'. וּמִיהוּ בַּתְרֵיהּ לֹא מְאַחֲרִינַן לֵיהּ: (טו) וְהָא דְּלֹא קָתָנֵי לְהוּ מִשּׁוּם דְּרַבִּי אֱלִיעֶזֶר מוֹדֶה לוֹ דְּבִטְבִילָה שְׁלִישִׁית הָווּ: (טז) וְנִרְאֶה לִי דְּטַעֲמוֹ מִשּׁוּם דִּוְאֶת חֵלֶב הַחַטָּאת כְּתִיב בָּתַר וְיָצָא וְעָשָׂה אֶת עוֹלָתוֹ: (יז) וּרְצוֹנוֹ לוֹמַר בִּטְבִילָה רְבִיעִית וְטַעֲמָא מְפֹרָשׁ בַּגְּמָרָא דִּכְתִיב וּבָא אַהֲרֹן כוּ' וּפָשַׁט אֶת בִּגְדֵי הַבַּד אֲשֶׁר לָבַשׁ. כְּלוּם אָדָם פּוֹשֵׁט אֶלָּא מַה שֶּׁלָּבַשׁ, וּמַה תַּלְמוּד לוֹמַר אֲשֶׁר לָבַשׁ, שֶׁלָּבַשׁ כְּבָר. אַלְמָא זוֹ הִיא פְּשִׁיטָה שְׁנִיָּה. שְׁמַע מִינַהּ דְּיֵשׁ הַפְסָקַת חִלּוּף בְּגָדִים בֵּין עֲבוֹדַת הַיּוֹם לְהַךְ בִּיאָה כוּ', וּבְהוֹצָאַת כַּף וּמַחְתָּה אַיְרֵי. וְכֵיוָן דִּשְׁמַע מִינַהּ דְּאִיכָּא הֶפְסֵק, שְׁמַע מִינַהּ דְּהַךְ קְרָא שֶׁלֹּא כְּסִדְרוֹ, וְדִכְתִיב בַּתְרֵיהּ וְיָצָא וְעָשָׂה אֶת עוֹלָתוֹ כוּ' מֻקְדָּם לְשֶׁל זֶה: (יח) בִּטְבִילָה חֲמִשִּׁית. רַשִׁ"י. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב בַּאֲרִיכוּת:

פרק ז - משנה ד

(יט) דְּהָא בַּשַּׁחַר מֵטִיב אֶת הַנֵּרוֹת, דְּהַיְנוּ דִּשּׁוּנָן וְתִקּוּן הַפְּתִילוֹת הַחֲדָשׁוֹת, אֶלָּא הֲטָבָה זוֹ הַדְלָקָה הִיא. אֲבָל הָרַמְבַּ"ם סְבִירָא לֵיהּ דְּבַשַׁחַר נַמִּי מַדְלִיקָן אַחַר שֶׁמֵּטִיבָן, וְאִם כֵּן בָּעֶרֶב הָיָה צָרִיךְ לַחֲזֹר וּלְתַקְּנָן וּלְהַדְלִיקָן, וּלְדִידֵיהּ שַׁפִּיר תְּנַן לְהֵטִיב:

פרק ז - משנה ה

(כ) וּמִצְנֶפֶת. בַּתּוֹרָה נִקְרָא דְּשֶׁל הֶדְיוֹטוֹת מִגְבַּעַת, וּלְשׁוֹן תּוֹרָה לְחוּד וּלְשׁוֹן חֲכָמִים לְחוּד. תּוֹסָפוֹת. וְעַיֵּן תּוֹסְפוֹת יוֹם טוֹב: (כא) וְאַף עַל גַּב דְּלִישְׁנָא דִּבְרַיְתָא פָּנָיו כְּלַפֵּי הַכֹּהֵן, הָכִי פֵּרוּשׁוֹ, פָּנָיו פּוֹנוֹת לַמָּקוֹם שֶׁפּוֹנִים פְּנֵי הַנִּשְׁאָל: (כב) כְּמִי שֶׁמִּתְפַּלֵּל בֵּינוֹ לְבֵין עַצְמוֹ. הָרַמְבַּ"ם. וְהַיְנוּ דְּמִתְּפִלַּת חַנָּה יָלְפִינַן:

(א) דִּכְתִיב וְהַאֲבַדְתִּי אֶת הַנֶּפֶשׁ הַהִיא, עִנּוּי שֶׁהוּא אֲבֵדַת נֶפֶשׁ, וְאֵיזֶה זֶה אֲכִילָה וּשְׁתִיָּה. גְּמָרָא דַּף ע"ד: (ב) דְּכִי כָּל הַנֶּפֶשׁ אֲשֶׁר לֹא תְּעֻנֶּה לֹא קָחָשֵׁיב, דְּלֹא חָשֵׁיב אֶלָּא צִוּוּיִין, אֲבָל הַאי דְּלֹא צִוּוּי לֹא קָחָשֵׁיב. תּוֹסָפוֹת: (ג) בַּאֲכִילָה וּשְׁתִיָּה. אֲכִילָה וּשְׁתִיָּה דְּאִקְּרוּ עִנּוּי מְנָלָן, כְּתִיב הָכָא תְּעַנּוּ, וּכְתִיב [דְּבָרִים ח] וַיְעַנְּךָ וַיַּרְעִיבֶךָ, שְׁמַע מִינַהּ דִּרְעָבוֹן אִקְּרֵי עִנּוּי. רְחִיצָה, דִּכְתִיב [מְלָכִים א ב] כִּי נָשָׂאתָ כוּ' וְכִי הִתְעַנֵּיתָ בְּכָל אֲשֶׁר הִתְעַנָּה אָבִי, וּכְתִיב הָתָם (שְׁמוּאֵל ב יז) כִּי אָמְרוּ הָעָם רָעֵב וְעָיֵף וְצָמֵא, וּכְתִיב (מִשְׁלֵי כה) מַיִם קָרִים עַל נֶפֶשׁ עֲיֵפָה, וְלֹא כְתִיב בְּנֶפֶשׁ, שְׁמַע מִינַהּ עָיֵף מֵרְחִיצָה הוּא. סִיכָה, דִּכְתִיב (דָּנִיֵּאל י) לֶחֶם וְגוֹ' וְסוֹך לֹא סָכְתִּי, וּכְתִיב (שָׁם) כִּי מִן הַיּוֹם וְגוֹ' וּלְהִתְעַנּוֹת. תַּשְׁמִישׁ הַמִּטָּה, דִּכְתִיב (בְּרֵאשִׁית לא) אִם תְּעַנֶּה אֶת בְּנוֹתַי וְאִם תִּקַּח נָשִׁים, תְּעַנֶּה מִתַּשְׁמִישׁ, תִּקַּח צָרוֹת, גְּמָרָא: (ד) וְהַמֶּלֶךְ כוּ'. דְּאִסּוּרָא דִּידְהוּ לָאו דְּאוֹרָיְתָא, וּקְרָאֵי אַסְמַכְתָּא בְּעָלְמָא, כְּמוֹ שֶׁכָּתְבוּ הָרַמְבַּ"ם וְהָרֹא"שׁ, וְהִלְכָּךְ הֵקֵלּוּ בּוֹ, דְּהֵם אָמְרוּ וְהֵם אָמְרוּ. וְאִי נַמִּי דְּאוֹרַיְתָא, כֵּיוָן דְּלֹא פֵּרְשׁוּ בְּהֶדְיָא בִּקְרָא קִלֵּי טְפֵי, דִּמְסָרָן לַחֲכָמִים וְהֵם יְכוֹלִין לְהָקֵל וּלְהַתִּיר כְּפִי שֶׁנִּרְאֶה לָהֶם. הָרַ"ן:

(ה) כָּמוֹהָ וּכְגַרְעִינָתָהּ. פֵּרוּשׁ הַכּוֹתֶבֶת אֵינָהּ דְּבוּקָה לְגַרְעִינָתָהּ וְיֵשׁ בֵּינֵיהֶם חָלָל, וְקָא מַשְׁמַע לָן דְּצָרִיךְ לִמְעֹךְ אוֹתוֹ חָלָל לְשַׁעֵר בְּכָמוֹהָ וּכְגַרְעִינָתָהּ בִּלְבַד. תּוֹסָפוֹת וְהָרַ"ן: (ו) יֵשׁ לְכַוֵּן שֶׁאֵינָהּ פְּחוּתָה מִמֶּנָּה אֶלָּא מְעָט, דְּכֵיוָן דְּחַד אָמוֹרָא אָמַר בַּגְּמָרָא יְתֵירָה מִכְּבֵיצָה לֹא מַפְלְגִינַן מִנֵּיהּ מַאן דְּפָלֵיג עֲלֵיהּ. הָרַ"ן: (ז) הַשִּׁעוּר בְּאָדָם בֵּינוֹנִי הוּא פָּחוֹת מֵרְבִיעִית, אֲבָל בִּשְׁאָר אָדָם הַכֹּל לְפִי גָּדְלוֹ וּלְפִי קָטְנוֹ. וּמִיהוּ בַּאֲכִילָה לֹא מְפַלְגִינַן אֶלָּא אֲפִלּוּ לְעוֹג מֶלֶךְ הַבָּשָׁן מְחַיְּבִים בִּכְכוֹתֶבֶת, וְטַעֲמָא אָמְרִינַן בַּגְּמָרָא דְּבַאֲכִילָה קִים לְהוּ לְרַבָּנָן דִּבְהָכִי מִיַּתְבָא דַּעְתֵּיהּ. וּמִיהוּ כֻּלֵּי עָלְמָא טוּבָא וְעוֹג מֶלֶךְ הַבָּשָׁן פּוּרְתָא. וְהַשְּׁתִיָּה בְּדִידֵיהּ מִיַּתְבָא דַּעְתֵּיהּ, בִּדְחַבְרֵיהּ לֹא מִיַּתְבָא דַּעְתֵּיהּ. וּמִשּׁוּם הָכִי לְעִנְיַן דְּלָאו בִּיתוּבָא דַּעְתָּא הוּא, בְּשֶׁל עוֹלָם הֵן שָׁמִין בְּבֵינוֹנִית. הָרַ"נ: (ח) לֹא קָאִימְנָא שַׁפִּיר בְּהַאי פָּחוֹת כַּמָּה הֲוָה בָּצִיר מִנֵּיהּ. הָרַ"ן: (ט) אֵין מִצְטָרְפִין. דְּאַף עַל גַּב דְּחַד שְׁמָא נִינְהוּ כְּדִלְקַמָּן, הוֹאִיל וּמִשּׁוּם יִתּוּבָא דַּעְתָּא הוּא, הַאי לֹא קָמִיַּתְבָא דַּעְתֵּיהּ, כִּדְאָמַר בַּגְּמָרָא:

(י) שֶׁנֶּאֱמַר וְנָתַתָּ הַכֶּסֶף בְּכֹל אֲשֶׁר תְּאַוֶּה נַפְשְׁךָ כוּ' בַּיַּיִן וּבַשֵּׁכָר. שֵׁכָר שְׁתִיָּה הוּא, דְּיָלִיף שֵׁכָר שֵׁכָר מִנָּזִיר, וְקַרְיֵהּ רַחְמָנָא וְאָכַלְתָּ, דְּאִלּוּ מִיַּיִן אִיכָּא לְמֵימַר עַל יְדֵי אֲנִיגְרוֹן שֶׁהוּא מִין מַאֲכָל שֶׁנּוֹתְנִין בּוֹ יַיִן: (יא) שֶׁאֵינָן רְאוּיִין. כְּגוֹן עֲשָׂבִים הַמָּרִים אוֹ שְׂרָפִים הַבְּאוּשִׁין. הָרַמְבַּ"ם: (יב) צִיר. כְּגוֹן צִיר כוּ':

(יג) כָּךְ כָּתַב רַשִׁ"י. וְנִרְאֶה לִי שֶׁדִּקְדְּקוּ לְפָרֵשׁ כֵּן מֵהָנֵי עֻבְדֵּי דְּאִיתְּנְהוּ בַּגְּמָרָא. הַהִיא עֻבָּרָה דְּאַרְחָא, לְחִישׁוּ לְהוּ וְאִלְּחִישָׁה, קָרֵי עֲלֵיהּ בְּטֶרֶם אֱצָרְךָ בַּבֶּטֶן וְגוֹ' וְנָפֵיק מִינַהּ רַבִּי יוֹחָנָן. וְהַהִיא עֻבָּרָהּ דְּאַרְחָה וּלְחִישׁוּ לַהּ וְלֹא אִלְּחִישָׁא, וְקָרֵי עֲלֵיהּ זֹרוּ רְשָׁעִים מֵרָחֶם, וְנָפֵיק מִינַהּ שַׁבְּתַאי אוֹצֵר פְּרִי. הָא קַמָּן דְּמִצַּד הָעֻבָּר בָּא הַתַּאֲוָה, שֶׁהוּא מֵרִיחַ וּמִתְאַוֶּה, שֶׁהֲרֵי בְּיָדוֹ לַחֲזֹר בּוֹ. תּוֹסְפוֹת יוֹם טוֹב: (יד) וְאָמְרִינַן תּוּנְבָּא, פֵּרוּשׁ שְׁטוּת נָקֵיט לֵיהּ מֵחֲמַת חָלְיוֹ, אֲבָל אָמַר צָרִיךְ כוּ', דְּלֵב יוֹדֵעַ מָרַת נַפְשׁוֹ, וּמִשּׁוּם דִּסְפֵק נְפָשׁוֹת לְהָקֵל הִלְכָּךְ כִּי אָמַר צָרִיךְ תָּלִינַן בְּלֵב יוֹדֵעַ, וְלָאו מִשּׁוּם תּוּנְבָּא, אֶלָּא קִים לֵיהּ בְּגַוֵּיהּ טְפֵי, וְכִי אָמַר אֵינוֹ צָרִיךְ תָּלִינַן בְּתוּנְבָּא. תּוֹסְפוֹת יוֹם טוֹב:

(טו) דְּבָרִים טְמֵאִים. הֵם הַדְּבָרִים הָאֲסוּרִים כִּשְׁקָצִים וּרְמָשִׂים וּבְהֵמָה טְמֵאָה וְזוּלָתָם. הָרַמְבַּ"ם: (טז) זֶה אֵינוֹ מוֹעִיל אֶלָּא בְּדֶרֶךְ סְגֻלָּה, וְאֵין עוֹבְרִין עַל הַמִּצְוֹת אֶלָּא בִּרְפוּאָה בִּלְבַד, רְצוֹנוֹ לוֹמַר בִּדְבָרִים הַמְרַפְּאִים בְּטֶבַע, וְהוּא דָּבָר אֲמִתִּי הוֹצִיאוּ מִדַּעַת וְהַנִּסָּיוֹן הַקָּרוֹב לָאֱמֶת. אֲבָל לְהִתְרַפְּאוֹת בִּדְבָרִים שֶׁהֵם מְרַפְּאִים בִּסְגֻלָּתָן אָסוּר, כִּי כֹּחָם חַלָּשׁ, אֵינוֹ מִצַּד הַדַּעַת, וְנִסְיוֹנוֹ רָחוֹק, וְהִיא טַעֲנָה חֲלוּשָׁה מִן הַטּוֹעֵן: (יז) סְפֵק נְפָשׁוֹת כוּ'. דִּכְתִיב (וַיִּקְרָא יח) אֲשֶׁר יַעֲשֶׂה אוֹתָם הָאָדָם וָחַי בָּהֶם, וְלֹא שֶׁיָּמוּת בָּהֶם. וּפֵרֵשׁ רַשִׁ"י אֲשֶׁר יַעֲשֶׂה הָאָדָם הַמִּצְוֹת שֶׁיִּחְיֶה בָּהֶם וַדַּאי וְלֹא שָׁיְכָא בַּעֲשִׂיָּתָם לָבֹא לִידֵי סְפֵק מִיתָה, אַלְמָא מְחַלְּלִין עַל הַסָּפֵק. גְּמָרָא דַּף פ"ה:

עַל אֲשֶׁר קִלְקֵל נֶגְדְּכֶם, דַּהֲרֵי כֻּלְכֶם חַיָּבִים בִּכְבֹדִי הַנֹּגֵעַ לִכְבֹדְכֶם, וּבִדְבָרֵינוּ מְבֹאָר בַּאֲרִיכוּת בְּעֶזְרַת הַשֵּׁם יִתְבָּרַךְ. מֵעַתָּה עוֹלֶה שַׁפִּיר נֹפֶת דִּבְרֵי הַמִּשְׁנָה, דְּבַתְּחִלָּה קָאָמַר אֶת זוֹ דָּרַשׁ רַבִּי אֶלְעָזָר בֶּן עֲזַרְיָה, שֶׁדּוֹרֵשׁ הַכְרָעַת הַמִּקְרָא מִכָּל חַטֹּאתֵיכֶם לִפְנֵי ה', דְּנִדְרָשׁ לְאַחֲרָיו, וּמְיַשֵּׁב רַבִּי עֲקִיבָא לְהַכְרִיעַ דְּכֵן הַהֶכְרַח שֶׁלֹּא לִדְרֹשׁ לְפָנָיו כַּמְבֹאָר. וְזֶהוּ אַשְׁרֵיכֶם יִשְׂרָאֵל, דַּאֲשֶׁר תְּהִלָּה שֶׁהוּא לָכֶם לִפְנֵי מִי אַתֶּם מִטַּהֲרִים, דְּמַעֲלֶה עֲלֵיכֶם כִּבְיָכוֹל כְּאִלּוּ קִדַּשְׁתֶּם אוֹתִי הוּא רַק מִי מְטַהֵר אֶתְכֶם, דְּטָהֳרָה הוּא רַק לְעַצְמְכֶם, וְאִם קִדַּשְׁתֶּם עַצְמְכֶם מַעֲלֶה אֲנִי עֲלֵיכֶם כְּאִלּוּ קִדַּשְׁתֶּם אוֹתִי כוּ', וְהוּא רַק בְּמַדְרֵגַת אַהֲבַת הַבֵּן, וְכִי קָדוֹשׁ ה' בֵּין אַתֶּם כוּ'. וְהַשֹּׁרֶשׁ פּוֹרֶה כוּ' שֶׁאִם אֵין אַתֶּם מַקְדִּישִׁים אֵינִי מְקֻדָּשׁ, הוּא רֹאשׁ וְלַעֲנָה, וְאִם חָטָאתָ מַה תִּפְעַל לְיוֹצֵר הָאָדָם. וְעוֹלֶה שַׁפִּיר אֲבִיכֶם שֶׁבַּשָּׁמַיִם, דְּהוּא בְּמַדְרֵגַת אָב כִּבְיָכוֹל וְהַשָּׁמַיִם שָׁמַיִם לַה' כְּמוֹ שֶׁנֶּאֱמַר וְזָרַקְתִּי כוּ' וּטְהַרְתֶּם, שֶׁאַתֶּם תִּטָּהֲרוּ. וּרְאָיָה מְפֹרֶשֶׁת עוֹד מִפָּסוּק מִקְוֵה יִשְׂרָאֵל, דְּקָאֵי לְרֵישָׁא דִּקְרָא בְּיִרְמְיָה, אִם עֲוֹנֵינוּ עָנוּ בָּנוּ ה' עֲשֵׂה לְמַעַן שְׁמֶךָ כִּי רַבּוּ מְשׁוּבוֹתֵינוּ לְךָ חָטָאנוּ. דְּבָזֶה שֶׁהָלַכְנוּ שׁוֹבָב בְּדַרְכֵי לְבָבֵנוּ לְךָ חָטָאנוּ, אֲנַחְנוּ הַחוֹטְאִים וְהַחֲסֵרִים, וּלְנַפְשֵׁנוּ הֲרֵעֹנוּ וְלֹא לְךָ יוֹצֵר הָאָדָם. וְעַל זֶה מְיַעֵד אוֹתָנוּ מִקְוֵה יִשְׂרָאֵל כוּ' לְטַהֵר אוֹתָנוּ. וְכֵיוָן שֶׁכֵּן מֻכְרָח הַכְרָעַת הַפָּסוּק מִכָּל חַטֹּאתֵיכֶם לִפְנֵי ה', דַּעֲבֵרוֹת שֶׁבֵּין אָדָם לַחֲבֵרוֹ אֵין יוֹם כִּפּוּר מְכַפֵּר עַד שֶׁיְּרַצֶּה אֶת חֲבֵרוֹ. לְשׁוֹן תּוֹסְפוֹת יוֹם טוֹב. לְפִי שֶׁהַדֶּרֶךְ הַיְּשָׁרָה הִיא שֶׁיָּבוֹר הָאָדָם בְּעַצְמוֹ לְהִתְקַדֵּשׁ וְלִטָּהֵר, וְאָז כְּשֶׁבָּא לִטָּהֵר מְסַיְּעִין לוֹ, וְעַל זֶה אָמַר לִפְנֵי מִי אַתֶּם מִטַּהֲרִים. וּמִפְּנֵי שֶׁלִּפְעָמִים לְפִי שֶׁיֵּצֶר לֵב הָאָדָם רַע עָלָיו לֹא יוּכַל לְהִתְגַּבֵּר עָלָיו לְכוּפוֹ לִטָּהֵר בִּתְשׁוּבָה, וְהוּא רַחוּם יְכַפֵּר עָוֹן יְעוֹרְרֵהוּ וּבִימִין צִדְקוֹ יִתְמְכֵהוּ לְבִלְתִּי יִדַּח מִמֶּנּוּ נִדָּח. וְזֶהוּ מִי מְטַהֵר אֶתְכֶם. וְנֶאֱמַר וְזָרַקְתִּי, וְזֶהוּ בְּלִי הִתְעוֹרְרוּת הָאָדָם, אֶלָּא בְּרַחֲמָיו מְטַהֲרָם. וְהַשֵּׁם יִתְבָּרַךְ
